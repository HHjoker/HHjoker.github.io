struct User {
    username: String,
    email: String,
    active: bool,
    sign_in_count: u64,
}

fn main() {
    // 创建结构体实例
    let user1 = User {
        username: String::from("alice"),
        email: String::from("alice@example.com"),
        active: true,
        sign_in_count: 1,
    };
    
    // 访问字段
    println!("用户名：{}", user1.username);
    println!("邮箱：{}", user1.email);
    
    // 修改字段（需要可变声明）
    let mut user2 = User {
        username: String::from("bob"),
        email: String::from("bob@example.com"),
        active: false,
        sign_in_count: 0,
    };
    
    user2.active = true;
    user2.sign_in_count = 1;
    
    println!("用户 2 状态：{}", user2.active);
}

    let user2 = User {

    // println!("{}", user1.username); // ❌ 编译错误
    println!("{}", user1.active); // ✅ 可以访问（Copy 类型）
}

// 定义元组结构体
struct Color(i32, i32, i32);
struct Point(i32, i32, i32);

fn main() {
    let black = Color(0, 0, 0);
    let origin = Point(0, 0, 0);
    
    // 通过索引访问字段
    println!("黑色的 R 分量：{}", black.0);
    println!("原点的 X 坐标：{}", origin.0);
    
    // 即使内部类型相同，不同类型的元组结构体也是不同的类型
    // let p: Point = black; // ❌ 编译错误：类型不匹配
    
    // 解构元组结构体
    let Color(r, g, b) = black;
    println!("RGB: ({}, {}, {})", r, g, b);
    
    // 忽略某些字段
    let Point(x, _, _) = origin;
    println!("X 坐标：{}", x);
}

struct Point {
    x: i32,
    y: i32,
}

    let p1 = Point { x: 1, y: 2 };
    
    // Clone
    let p2 = p1.clone();
    
    // Copy (自动复制)
    let p3 = p1; // p1 仍然可用，因为 Point 实现了 Copy
    println!("p1: {:?}", p1);
    
    // PartialEq
    println!("p1 == p2: {}", p1 == p2); // true
    
    // Eq (需要 PartialEq + 所有字段实现 Eq)
    assert_eq!(p1, p2);
}

#[derive(Debug)]
struct Rectangle {
    width: u32,
    height: u32,
}

impl Rectangle {
    // 关联函数（通常用作构造函数）
    fn new(width: u32, height: u32) -> Self {
        Rectangle { width, height }
    }
    
    // 多个构造函数
    fn square(size: u32) -> Self {
        Rectangle {
            width: size,
            height: size,
        }
    }
    
    // 带默认值的构造函数
    fn with_default_width(height: u32) -> Self {
        Rectangle {
            width: 100,
            height,
        }
    }
}

fn main() {
    let rect1 = Rectangle::new(30, 50);
    let square1 = Rectangle::square(25);
    let rect2 = Rectangle::with_default_width(40);
    
    println!("rect1: {:?}", rect1);
    println!("square1: {:?}", square1);
    println!("rect2: {:?}", rect2);
}

#[derive(Debug)]
struct Rectangle {
    width: u32,
    height: u32,
}

impl Rectangle {
    // 需要获取所有权的方法
    fn consume(self) -> u32 {
        self.width * self.height
    }
    
    // 不可变借用（最常见）
    fn area(&self) -> u32 {
        self.width * self.height
    }
    
    // 可变借用
    fn scale(&mut self, factor: u32) {
        self.width *= factor;
        self.height *= factor;
    }
    
    // 检查是否能容纳另一个矩形
    fn can_hold(&self, other: &Rectangle) -> bool {
        self.width > other.width && self.height > other.height
    }
    
    // 多个参数的方法
    fn resize(&mut self, width: u32, height: u32) {
        self.width = width;
        self.height = height;
    }
}

fn main() {
    let mut rect1 = Rectangle::new(30, 50);
    let rect2 = Rectangle::new(10, 40);
    
    // 调用方法
    println!("rect1 面积：{}", rect1.area()); // 1500
    println!("rect1 能否容纳 rect2: {}", rect1.can_hold(&rect2)); // true
    
    // 可变方法
    rect1.scale(2);
    println!("放大后的 rect1: {:?}", rect1); // width: 60, height: 100
    
    // 方法链（需要返回 self）
    rect1.resize(50, 70);
    println!("调整大小后：{:?}", rect1);
}

    username: String,
    email: String,

    }

        self.email = email.to_string();

        .email("alice@example.com")
        .active(true)
        .build();
    
    println!("用户：{:?}", user);
}

#[derive(Debug)]
struct Point {
    x: f64,
    y: f64,
}

// 第一个 impl 块：基本方法
impl Point {
    fn new(x: f64, y: f64) -> Self {
        Point { x, y }
    }
    
    fn distance_from_origin(&self) -> f64 {
        (self.x.powi(2) + self.y.powi(2)).sqrt()
    }
}

// 第二个 impl 块：几何方法
impl Point {
    fn distance_to(&self, other: &Point) -> f64 {
        let dx = self.x - other.x;
        let dy = self.y - other.y;
        (dx.powi(2) + dy.powi(2)).sqrt()
    }
    
    fn midpoint(&self, other: &Point) -> Point {
        Point {
            x: (self.x + other.x) / 2.0,
            y: (self.y + other.y) / 2.0,
        }
    }
}

fn main() {
    let p1 = Point::new(0.0, 0.0);
    let p2 = Point::new(3.0, 4.0);
    
    println!("p1 到原点距离：{}", p1.distance_from_origin());
    println!("p2 到原点距离：{}", p2.distance_from_origin());
    println!("p1 到 p2 距离：{}", p1.distance_to(&p2));
    
    let mid = p1.midpoint(&p2);
    println!("中点：{:?}", mid); // (1.5, 2.0)
}

use std::fmt;

#[derive(Debug, Clone)]
struct BlogPost {
    id: u64,
    title: String,
    content: String,
    author: String,
    published: bool,
    views: u64,
}

impl BlogPost {
    // 构造函数
    fn new(id: u64, title: &str, author: &str) -> Self {
        BlogPost {
            id,
            title: title.to_string(),
            content: String::new(),
            author: author.to_string(),
            published: false,
            views: 0,
        }
    }
    
    // 添加内容
    fn add_content(&mut self, content: &str) {
        self.content.push_str(content);
    }
    
    // 发布文章
    fn publish(&mut self) {
        if !self.content.is_empty() {
            self.published = true;
            println!("文章《{}》已发布", self.title);
        } else {
            println!("错误：不能发布空内容文章");
        }
    }
    
    // 增加浏览量
    fn increment_views(&mut self) {
        self.views += 1;
    }
    
    // 获取文章摘要
    fn summary(&self) -> String {
        format!(
            "《{}》by {} - {} 次浏览 - {}",
            self.title,
            self.author,
            self.views,
            if self.published { "已发布" } else { "草稿" }
        )
    }
    
    // 内容预览（限制长度）
    fn preview(&self, max_len: usize) -> &str {
        if self.content.len() <= max_len {
            &self.content
        } else {
            &self.content[..max_len]
        }
    }
}

// 实现 Display trait
impl fmt::Display for BlogPost {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        write!(
            f,
            "【{}】{}\n作者：{}\n{}\n状态：{}",
            self.title,
            if self.published { "✓ 已发布" } else { "📝 草稿" },
            self.author,
            self.content,
            self.views
        )
    }
}

fn main() {
    // 创建文章
    let mut post1 = BlogPost::new(1, "Rust 结构体入门", "Alice");
    post1.add_content("Rust 的结构体是自定义数据类型的基础...");
    post1.add_content("\n\n结构体可以包含多个字段，每个字段有自己的类型。");
    
    let mut post2 = BlogPost::new(2, "所有权详解", "Bob");
    post2.add_content("所有权是 Rust 最独特的特性之一...");
    
    // 发布文章
    post1.publish();
    
    // 增加浏览量
    post1.increment_views();
    post1.increment_views();
    post2.increment_views();
    
    // 打印文章
    println!("{}", post1.summary());
    println!("预览：{}", post1.preview(30));
    println!("\n完整文章：\n{}", post1);
    
    println!("\n{}", post2.summary());
}

#[derive(Debug)]
struct Article {
    title: String,
    content: String,
}

#[derive(Debug)]
struct Blog {
    name: String,
    articles: Vec<Article>,
}

impl Blog {
    fn new(name: &str) -> Self {
        Blog {
            name: name.to_string(),
            articles: Vec::new(),
        }
    }
    
    // 获取文章的所有权
    fn add_article(&mut self, article: Article) {
        self.articles.push(article);
    }
    
    // 借用文章
    fn get_article(&self, index: usize) -> Option<&Article> {
        self.articles.get(index)
    }
    
    // 可变借用文章
    fn edit_article(&mut self, index: usize, new_content: &str) -> Option<&mut Article> {
        if let Some(article) = self.articles.get_mut(index) {
            article.content = new_content.to_string();
            Some(article)
        } else {
            None
        }
    }
    
    // 移除并返回文章
    fn remove_article(&mut self, index: usize) -> Option<Article> {
        if index < self.articles.len() {
            Some(self.articles.remove(index))
        } else {
            None
        }
    }
}

fn main() {
    let mut blog = Blog::new("Rust 学习博客");
    
    let article1 = Article {
        title: String::from("Day 11: 结构体"),
        content: String::from("今天学习了结构体的定义和使用..."),
    };
    
    blog.add_article(article1);
    
    // 借用访问
    if let Some(article) = blog.get_article(0) {
        println!("文章标题：{}", article.title);
    }
    
    // 可变借用修改
    if let Some(article) = blog.edit_article(0, "更新后的内容...") {
        println!("文章已更新：{}", article.content);
    }
    
    println!("博客：{:?}", blog);
}

impl Rectangle {
    // self: 获取所有权，实例被消耗
    fn consume(self) -> T { }
    
    // &self: 不可变借用，最常见
    fn read(&self) -> T { }

    // self: 不引用，用于构造函数
    fn new() -> Self { }
}

今日感悟

结构体是 Rust 数据建模的基石。今天的学习让我有几个重要收获：

  1. 结构体不只是数据容器：通过 impl 块，结构体可以封装数据和行为，这是面向对象编程的基础。但 Rust 的方式更灵活——多个 impl 块、分离关注点。

  2. 更新语法的妙用：..instance 语法在配置对象、构建器模式中非常有用，可以避免大量重复代码。

  3. 方法链的优雅：通过返回 Self，可以创建流畅的 API。这是 Rust 中常见的模式，在很多库中都能看到。

  4. Debug trait 的重要性：开发过程中 #[derive(Debug)] 几乎是必须的，它让调试和日志记录变得简单。

  5. 所有权在结构体中的体现：结构体字段遵循所有权规则，这意味着设计结构体时需要考虑数据的所有权关系。这对后续学习生命周期很重要。

明天的枚举学习应该会更有趣，特别是 Option<T> 和 match 表达式，这是 Rust 函数式编程风格的精髓。
