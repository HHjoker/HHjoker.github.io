今日内容

经过一周的 async/await 学习，我们已经能够编写异步代码了。但你有没有好奇过：这些 async 代码到底是怎么运行的？任务是怎么被调度的？IO 是怎么被等待的？今天让我们深入探索 async 运行时的内部世界。

1. 为什么需要 async 运行时？

在 Rust 中，async 只是一个语法特性，它不会自己执行。需要一个运行时（Runtime）来：

  1. 调度任务：决定哪个任务在哪个线程上运行
  2. 管理协程状态：保存和恢复协程的执行上下文
  3. 处理 IO 事件：监控 IO 状态变化，唤醒等待的任务
  4. 绑定线程池：管理线程池，执行阻塞操作

主流运行时：

  • Tokio：最流行，生产级支持
  • async-std：Rust 标准库风格的异步运行时
  • smol：轻量级、模块化的运行时

// src/runtime/executor.rs

use std::sync::{Arc, Mutex};
use std::collections::VecDeque;
use std::task::{Context, Poll, RawWaker, RawWakerVTable, Waker};

type Task = std::pin::Pin<Box<dyn Future<Output = ()> + Send>>;

/// 极简任务执行器
pub struct MiniExecutor {
    tasks: Arc<Mutex<VecDeque<Task>>>,
    workers: Vec<std::thread::JoinHandle<()>>,
}

impl MiniExecutor {
    pub fn new(threads: usize) -> Self {
        let tasks: Arc<Mutex<VecDeque<Task>>> = Arc::new(VecDeque::new().into());
        let mut workers = Vec::new();

        for i in 0..threads {
            let tasks = Arc::clone(&tasks);
            let handle = std::thread::spawn(move || {
                println!("Worker {} 启动", i);
                loop {
                    // 从队列取出任务
                    let task = {
                        let mut guard = tasks.lock().unwrap();
                        guard.pop_front()
                    };

                    if let Some(mut task) = task {
                        // 轮询任务直到完成
                        // 注意：这简化了很多，实际需要 Waker 支持
                        loop {
                            let waker = create_simple_waker();
                            let mut cx = Context::from_waker(&waker);
                            match task.as_mut().poll(&mut cx) {
                                Poll::Ready(_) => {
                                    println!("任务完成");
                                    break;
                                }
                                Poll::Pending => {
                                    // 简化版：直接继续等待
                                    // 实际运行时这里会放弃 CPU，等待被唤醒
                                    std::thread::sleep(std::time::Duration::from_micros(100));
                                }
                            }
                        }
                    } else {
                        // 没有任务，短暂休眠避免空转
                        std::thread::sleep(std::time::Duration::from_millis(1));
                    }
                }
            });
            workers.push(handle);
        }

        Self { tasks, workers }
    }

    /// 提交任务到执行器
    pub fn spawn(&self, future: impl Future<Output = ()> + Send + 'static) {
        let task = Box::pin(future);
        let mut guard = self.tasks.lock().unwrap();
        guard.push_back(task);
    }

    /// 等待所有工作线程结束
    pub fn shutdown(self) {
        drop(self.tasks);
        for handle in self.workers {
            handle.join().unwrap();
        }
    }
}

// src/runtime/waker.rs

use std::task::{Context, Poll, RawWaker, RawWakerVTable, Waker, LocalWake};
use std::sync::Arc;

/// 自定义的异步 IO 事件源
pub struct AsyncIoSource {
    /// 保存任务唤醒器
    waker: Arc<Mutex<Option<Waker>>>,
}

impl AsyncIoSource {
    pub fn new() -> Self {
        Self {
            waker: Arc::new(Mutex::new(None)),
        }
    }

    /// 注册 Waker，当 IO 完成时唤醒任务
    pub fn register_waker(&self, waker: Waker) {
        let mut guard = self.waker.lock().unwrap();
        *guard = Some(waker);
    }

    /// 模拟 IO 完成，唤醒等待的任务
    pub fn signal_ready(&self) {
        let guard = self.waker.lock().unwrap();
        if let Some(waker) = guard.as_ref() {
            println!("唤醒任务！");
            waker.wake_by_ref();
        }
    }
}

/// 为 Future 创建简化的 Waker
fn create_simple_waker() -> Waker {
    // 实际实现需要完整的 RawWaker VTable
    // 这里简化处理，返回一个不做任何事的 Waker
    unsafe { Waker::from_raw(create_raw_waker()) }
}

fn create_raw_waker() -> RawWaker {
    RawWaker::new(std::ptr::null(), &VTABLE)
}

static VTABLE: RawWakerVTable = RawWakerVTable::new(
    |_| RawWaker::new(std::ptr::null(), &VTABLE), // clone
    |_| {},                                        // wake
    |_| {},                                        // wake_by_ref
    |_| {},                                        // drop
);

// src/runtime/multi_future.rs

use std::pin::Pin;
use std::task::{Context, Poll};
use std::future::Future;
use std::sync::{Arc, Mutex};

/// 同时等待多个 Future，哪个先完成就返回哪个的结果
pub struct Select<F1, F2> {
    fut1: Option<F1>,
    fut2: Option<F2>,
}

impl<F1, F2, T1, T2> Future for Select<F1, F2>
where
    F1: Future<Output = T1>,
    F2: Future<Output = T2>,
{
    type Output = Either<T1, T2>;

    fn poll(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Self::Output> {
        if let Some(ref mut f1) = self.fut1 {
            if let Poll::Ready(v) = Pin::new(f1).poll(cx) {
                return Poll::Ready(Either::Left(v));
            }
        }
        if let Some(ref mut f2) = self.fut2 {
            if let Poll::Ready(v) = Pin::new(f2).poll(cx) {
                return Poll::Ready(Either::Right(v));
            }
        }
        Poll::Pending
    }
}

/// 二选一的结果类型
pub enum Either<L, R> {
    Left(L),
    Right(R),
}

/// 创建选择器
pub fn select<F1, F2>(fut1: F1, fut2: F2) -> Select<F1, F2> {
    Select {
        fut1: Some(fut1),
        fut2: Some(fut2),
    }
}

// examples/tokio_runtime.rs

use tokio::time::{sleep, Duration};
use tokio::sync::oneshot;

#[tokio::main]
async fn main() {
    println!("=== Tokio 运行时示例 ===");

    // 创建多个任务
    let (tx1, rx1) = oneshot::channel();
    let (tx2, rx2) = oneshot::channel();

    // 任务 1：模拟 IO 延迟
    let task1 = tokio::spawn(async {
        println!("任务1 开始");
        sleep(Duration::from_millis(50)).await;
        println!("任务1 完成");
        tx1.send("任务1结果").unwrap();
    });

    // 任务 2：更快的任务
    let task2 = tokio::spawn(async {
        println!("任务2 开始");
        sleep(Duration::from_millis(20)).await;
        println!("任务2 完成");
        tx2.send("任务2结果").unwrap();
    });

    // 使用 select! 同时等待
    tokio::select! {
        result1 = rx1 => println!("收到: {}", result1),
        result2 = rx2 => println!("收到: {}", result2),
    }

    // 等待所有任务完成
    task1.await.unwrap();
    task2.await.unwrap();

    println!("所有任务完成");
}

// 伪代码展示 work-stealing 概念

struct Worker {
    local_queue: VecDeque<Task>,  // 本地任务队列
    steal_target: Arc<Worker>,     // 可以从其他 worker 偷任务
}

impl Worker {
    fn run(&self) {
        loop {
            // 1. 先尝试从本地队列取任务
            if let Some(task) = self.local_queue.pop_front() {
                self.execute_task(task);
                continue;
            }

            // 2. 本地队列空，尝试从其他 worker 偷任务
            if let Some(task) = self.steal_target.steal() {
                self.execute_task(task);
                continue;
            }

            // 3. 都没找到，park 等待新任务
            self.park();
        }
    }

    fn steal(&self) -> Option<Task> {
        self.local_queue.pop_front()
    }
}

use std::io::{Read, Write};
use std::net::TcpStream;
use std::time::Duration;

/// 同步方式：线程在等待 IO 时被阻塞
fn sync_io_example() {
    let mut stream = TcpStream::connect("example.com:80").unwrap();
    stream.set_read_timeout(Some(Duration::from_secs(5))).unwrap();

    let mut buffer = [0u8; 1024];
    // 这会阻塞整个线程
    let _n = stream.read(&mut buffer).unwrap();
}

/// 异步方式：IO 操作立即返回，稍后检查结果
trait AsyncRead {
    fn poll_read(&mut self, buf: &mut [u8], cx: &mut Context<'_>) -> Poll<usize>;
}

1. Task 调用 poll()，返回 Pending
2. Executor 暂停任务，释放线程
3. IO Manager 注册 IO 事件 + Waker
4. IO 完成 → 调用 Waker.wake()
5. Executor 重新调度任务
6. Task 再次 poll()，这次返回 Ready

今日感悟

async 运行时是 Rust 异步编程的核心。通过今天的探索，我深刻理解了：

  1. async 只是语法糖：编译器将 async 函数转换为状态机，真正的工作由运行时完成
  2. Waker 是关键桥梁：任务和 IO 系统通过 Waker 连接，实现事件驱动
  3. work-stealing 是高效调度：避免线程空闲，充分利用 CPU

理解运行时的内部机制，不仅帮助我们更好地使用 async，也为后续学习 unsafe 和系统编程打下基础。Rust 的设计哲学再次体现：显式优于隐式，控制优于自动。
