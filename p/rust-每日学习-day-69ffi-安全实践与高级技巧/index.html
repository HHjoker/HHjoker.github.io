use std::ffi::{CStr, CString};
use std::os::raw::c_char;
use std::ptr;

#[repr(C)]
pub struct Config {
    pub api_key: *const c_char,
    pub timeout_ms: u32,
}

impl Config {
    pub fn new(api_key: &str, timeout_ms: u32) -> Self {
        Config {
            api_key: CString::new(api_key).unwrap().into_raw(),
            timeout_ms,
        }
    }
}

// 模拟外部 C 函数的错误码
extern "C" {
    fn init_config(config: *const Config) -> i32;
    fn free_config(config: *const Config);
}

// 安全的 Rust 封装
fn init_config_safe(api_key: &str, timeout_ms: u32) -> Result<(), ConfigError> {
    let config = Config::new(api_key, timeout_ms);
    
    // 验证输入参数
    if timeout_ms == 0 || timeout_ms > 300_000 {
        return Err(ConfigError::InvalidTimeout(timeout_ms));
    }
    
    if api_key.is_empty() {
        return Err(ConfigError::EmptyApiKey);
    }
    
    unsafe {
        let result = init_config(&config);
        
        if result == 0 {
            Ok(())
        } else {
            Err(ConfigError::InitializationFailed(result))
        }
    }
}

#[derive(Debug)]
pub enum ConfigError {
    InvalidTimeout(u32),
    EmptyApiKey,
    InitializationFailed(i32),
}

impl std::fmt::Display for ConfigError {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        match self {
            ConfigError::InvalidTimeout(ms) => {
                write!(f, "Invalid timeout value: {}ms (must be 1-300000)", ms)
            }
            ConfigError::EmptyApiKey => write!(f, "API key cannot be empty"),
            ConfigError::InitializationFailed(code) => {
                write!(f, "Configuration initialization failed with code: {}", code)
            }
        }
    }
}

impl std::error::Error for ConfigError {}

use std::ffi::CStr;
use std::os::raw::c_char;

extern "C" {
    fn c_read_file(path: *const c_char, buffer: *mut c_char, size: usize) -> i32;
    fn c_get_error() -> *const c_char;
}

fn read_file_content(path: &str) -> Result<String, FfiError> {
    let c_path = CString::new(path).map_err(|_| FfiError::InvalidPath)?;
    
    let mut buffer = vec![0u8; 1024];
    
    let bytes_read = unsafe {
        c_read_file(c_path.as_ptr(), buffer.as_mut_ptr() as *mut c_char, buffer.len())
    };
    
    if bytes_read < 0 {
        // 获取 C 库的错误信息并转换为 Rust 错误
        let c_error = unsafe { CStr::from_ptr(c_get_error()) };
        let error_msg = c_error.to_string_lossy().into_owned();
        return Err(FfiError::ReadFailed(error_msg));
    }
    
    if bytes_read >= buffer.len() {
        return Err(FfiError::BufferTooSmall);
    }
    
    Ok(String::from_utf8_lossy(&buffer[..bytes_read as usize]).to_string())
}

#[derive(Debug)]
enum FfiError {
    InvalidPath,
    ReadFailed(String),
    BufferTooSmall,
}

impl std::fmt::Display for FfiError {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        match self {
            FfiError::InvalidPath => write!(f, "Invalid file path"),
            FfiError::ReadFailed(msg) => write!(f, "Read failed: {}", msg),
            FfiError::BufferTooSmall => write!(f, "Buffer too small to hold file content"),
        }
    }
}

use std::ffi::{CString, CStr};
use std::os::raw::{c_char, c_void};
use std::ptr;

extern "C" {
    fn c_malloc(size: usize) -> *mut c_void;
    fn c_free(ptr: *mut c_void);
    fn c_realloc(ptr: *mut c_void, new_size: usize) -> *mut c_void;
}

/// Rust 风格的智能指针封装 C 内存分配
struct CMemoryBlock {
    ptr: *mut c_void,
    size: usize,
}

impl CMemoryBlock {
    /// 分配指定大小的内存
    fn new(size: usize) -> Option<Self> {
        if size == 0 {
            return None;
        }
        
        unsafe {
            let ptr = c_malloc(size);
            if ptr.is_null() {
                None
            } else {
                Some(CMemoryBlock { ptr, size })
            }
        }
    }
    
    /// 重新分配内存
    fn realloc(&mut self, new_size: usize) -> bool {
        if new_size == 0 {
            return false;
        }
        
        unsafe {
            let new_ptr = c_realloc(self.ptr, new_size);
            if new_ptr.is_null() {
                false
            } else {
                self.ptr = new_ptr;
                self.size = new_size;
                true
            }
        }
    }
    
    /// 获取指针用于 FFI 调用
    pub fn as_ptr(&self) -> *const c_void {
        self.ptr
    }
    
    pub fn as_mut_ptr(&mut self) -> *mut c_void {
        self.ptr
    }
}

impl Drop for CMemoryBlock {
    fn drop(&mut self) {
        if !self.ptr.is_null() {
            unsafe {
                c_free(self.ptr);
            }
        }
    }
}

// 使用示例
fn use_c_allocator() -> Result<Vec<u8>, &'static str> {
    let mut mem = CMemoryBlock::new(1024).ok_or("Allocation failed")?;
    
    // 使用这块内存...
    println!("Allocated {} bytes of C memory", mem.size);
    
    // 重新分配
    if !mem.realloc(2048) {
        return Err("Reallocation failed");
    }
    println!("Reallocated to {} bytes", mem.size);
    
    // mem 在作用域结束时自动释放
    Ok(vec![0; mem.size])
}

use std::ffi::{CStr, CString, NulError};
use std::os::raw::c_char;
use std::ptr;

/// 接收 C 字符串并安全地转换为 Rust String
/// 关键：必须小心处理指针的有效性
fn c_string_to_rust(c_ptr: *const c_char) -> Result<String, StringError> {
    if c_ptr.is_null() {
        return Err(StringError::NullPointer);
    }
    
    unsafe {
        let c_str = CStr::from_ptr(c_ptr);
        Ok(c_str.to_string_lossy().into_owned())
    }
}

/// 将 Rust String 转换为 C 字符串
/// 返回的指针必须由调用者释放
fn rust_string_to_c_string(rust_string: &str) -> Result<*mut c_char, NulError> {
    CString::new(rust_string).map(|c| c.into_raw())
}

/// 释放通过 rust_string_to_c_string 分配的内存
/// 这是一个配对操作
unsafe fn free_c_string(c_ptr: *mut c_char) {
    if !c_ptr.is_null() {
        // 重新构造 CString 并自动释放
        let _ = CString::from_raw(c_ptr);
    }
}

#[derive(Debug)]
enum StringError {
    NullPointer,
    InvalidUtf8,
}

impl std::fmt::Display for StringError {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        match self {
            StringError::NullPointer => write!(f, "Received null pointer from C"),
            StringError::InvalidUtf8 => write!(f, "C string is not valid UTF-8"),
        }
    }
}

// 自动管理 C 字符串生命周期的包装器
struct OwnedCString {
    ptr: *mut c_char,
}

impl OwnedCString {
    fn from_rust(s: &str) -> Result<Self, NulError> {
        CString::new(s).map(|c| OwnedCString {
            ptr: c.into_raw(),
        })
    }
    
    fn as_ptr(&self) -> *const c_char {
        self.ptr
    }
    
    fn as_mut_ptr(&mut self) -> *mut c_char {
        self.ptr
    }
}

impl Drop for OwnedCString {
    fn drop(&mut self) {
        if !self.ptr.is_null() {
            unsafe {
                let _ = CString::from_raw(self.ptr);
            }
        }
    }
}

use std::ffi::{CString, CStr};
use std::os::raw::c_char;
use std::ptr;
use std::sync::Mutex;

/// 如果底层 C 类型是线程安全的，我们才能实现 Send/Sync
pub struct ThreadSafeConfig {
    // 内部使用 Mutex 保护
    inner: Mutex<CConfigInner>,
}

struct CConfigInner {
    ptr: *mut c_char,
    initialized: bool,
}

unsafe impl Send for ThreadSafeConfig {}
unsafe impl Sync for ThreadSafeConfig {}

impl ThreadSafeConfig {
    pub fn new(key: &str) -> Result<Self, &'static str> {
        let c_key = CString::new(key).map_err(|_| "Invalid key")?;
        
        Ok(ThreadSafeConfig {
            inner: Mutex::new(CConfigInner {
                ptr: c_key.into_raw(),
                initialized: true,
            }),
        })
    }
    
    /// 线程安全的配置获取
    pub fn get_key(&self) -> String {
        let guard = self.inner.lock().unwrap();
        if guard.ptr.is_null() || !guard.initialized {
            return String::new();
        }
        
        unsafe {
            CStr::from_ptr(guard.ptr).to_string_lossy().into_owned()
        }
    }
}

impl Drop for ThreadSafeConfig {
    fn drop(&mut self) {
        let mut guard = self.inner.lock().unwrap();
        if !guard.ptr.is_null() {
            unsafe {
                let _ = CString::from_raw(guard.ptr);
            }
            guard.ptr = ptr::null_mut();
        }
        guard.initialized = false;
    }
}

use std::thread;
use std::sync::Arc;
use std::ffi::{CString, CStr};
use std::os::raw::c_char;

extern "C" {
    fn c_process_in_thread(data: *const c_char, callback: extern "C" fn(*const c_char));
}

// FFI 回调函数
#[no_mangle]
unsafe extern "C" fn rust_callback(result: *const c_char) {
    if !result.is_null() {
        let result_str = CStr::from_ptr(result).to_string_lossy();
        println!("[Thread] Callback received: {}", result_str);
    }
}

fn call_c_with_callback(data: &str) -> Result<(), &'static str> {
    let c_data = CString::new(data).map_err(|_| "Invalid data")?;
    
    // 在单独线程中调用 FFI，避免阻塞主线程
    let handle = thread::spawn(move || {
        unsafe {
            c_process_in_thread(c_data.as_ptr(), rust_callback);
        }
    });
    
    // 等待线程完成
    handle.join().map_err(|_| "Thread panicked")?;
    
    Ok(())
}

use bindings::*;
use std::ffi::{CStr, CString};
use std::ptr;

fn main() {
    // 创建用户列表
    let mut list = unsafe { create_user_list(10) };
    
    if list.is_null() {
        println!("Failed to create user list");
        return;
    }
    
    // 创建用户记录
    let mut user = UserRecord {
        id: 1,
        name: [0; 64],
        score: 95.5,
    };
    
    // 设置用户名
    let name = CString::new("Alice").unwrap();
    for (i, byte) in name.as_bytes_with_nul().iter().enumerate() {
        if i < 64 {
            user.name[i] = *byte;
        }
    }
    
    // 添加用户
    unsafe {
        add_user(list, &mut user);
    }
    
    println!("Added user: {} with score: {}", 
             unsafe { CStr::from_ptr(user.name.as_ptr()).to_string_lossy() },
             user.score);
    
    // 获取用户
    let retrieved = unsafe { get_user(list, 0) };
    if !retrieved.is_null() {
        let user_ref = unsafe { &*retrieved };
        println!("Retrieved user ID: {}, score: {}", user_ref.id, user_ref.score);
    }
    
    // 释放内存
    unsafe {
        free_user_list(list);
    }
}

关键概念理解

FFI 安全原则

  1. 永不信任外部输入

    • C 代码可能返回 null、无效指针或异常值
    • 始终进行边界检查和数据验证

  2. 使用 RAII 管理资源

    • 为每个 malloc 实现对应的 free
    • 使用智能指针封装裸指针

  3. 最小化 unsafe 代码

    • 将 unsafe 代码封装在安全的抽象层后面
    • 公共 API 应该是安全的

  4. 正确处理字符串编码

    • C 使用 null 终止的字节数组
    • Rust 使用 UTF-8 字符串，需要显式转换

fn validate_token(token: &str) -> Result<(), &'static str> {
    let c_token = CString::new(token).map_err(|_| "Invalid token string")?;
    
    unsafe {
        match c_validate_token(c_token.as_ptr()) {
            0 => Ok(()),
            -1 => Err("Invalid token"),
            _ => Err("Unknown error"),
        }
    }
}

struct Buffer {
    ptr: *mut c_void,
    size: usize,
}

impl Buffer {
    fn new(size: usize) -> Option<Self> {
        unsafe {
            let ptr = c_buffer_create(size);
            if ptr.is_null() {
                None
            } else {
                Some(Buffer { ptr, size })
            }
        }
    }
    
    fn append(&mut self, data: &str) -> Result<(), &'static str> {
        let c_data = CString::new(data).map_err(|_| "Invalid data")?;
        unsafe {
            c_buffer_append(self.ptr, c_data.as_ptr(), data.len());
        }
        Ok(())
    }
}

impl Drop for Buffer {
    fn drop(&mut self) {
        if !self.ptr.is_null() {
            unsafe { c_buffer_destroy(self.ptr); }
        }
    }
}

fn get_config_dir() -> String {
    use std::env;
    
    #[cfg(target_os = "windows")]
    {
        env::var("APPDATA")
            .map(|p| format!("{}/MyApp", p))
            .unwrap_or_else(|_| ".".to_string())
    }
    
    #[cfg(target_os = "macos")]
    {
        env::var("HOME")
            .map(|h| format!("{}/Library/Application Support/MyApp", h))
            .unwrap_or_else(|_| ".".to_string())
    }
    
    #[cfg(target_os = "linux")]
    {
        env::var("XDG_CONFIG_HOME")
            .or_else(|_| env::var("HOME").map(|h| format!("{}/.config", h)))
            .map(|p| format!("{}/myapp", p))
            .unwrap_or_else(|_| ".".to_string())
    }

今日感悟

FFI 是 Rust 中最复杂的领域之一，它要求我们同时理解 Rust 的安全保证和 C 的不安全现实。最大的收获是：

  1. 信任但要验证——永远不要假设外部代码会按预期工作
  2. 最小化 unsafe——unsafe 块越小，审查越容易
  3. RAII 是你的朋友——利用 Rust 的析构函数自动管理资源
  4. 使用工具——bindgen 等工具能大幅减少错误

FFI 虽然危险，但在需要与 C 代码库集成或追求极致性能时，它是不可或缺的工具。掌握这些技术，你就能在 Rust 和其他语言之间架起桥梁。
