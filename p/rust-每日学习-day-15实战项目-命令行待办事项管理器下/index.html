impl std::fmt::Display for Priority {

        match self {
            Priority::Low => write!(f, "低"),
            Priority::Medium => write!(f, "中"),
            Priority::High => write!(f, "高"),

/// 按优先级排序任务
fn sort_by_priority(tasks: &mut Vec<&mut Task>) {
    tasks.sort_by(|a, b| {
        // 高优先级在前，同优先级按 ID 排序
        b.priority.cmp(&a.priority)
            .then_with(|| a.id.cmp(&b.id))
    });
}

/// 列出任务（支持优先级排序）
fn list_tasks(filter: Option<&str>, sort: bool) -> Result<(), Box<dyn std::error::Error>> {
    let mut todo_list = load_todo_list()?;
    
    let status_filter = match filter {
        Some("pending") | Some("p") => Some(Status::Pending),
        Some("completed") | Some("c") => Some(Status::Completed),
        _ => None,
    };
    
    let mut tasks: Vec<&mut Task> = todo_list.tasks.iter_mut()
        .filter(|t| {
            match &status_filter {
                Some(s) => t.status == *s,
                None => true,
            }
        })
        .collect();
    
    if sort {
        sort_by_priority(&mut tasks);
    }
    
    if tasks.is_empty() {
        println!("📭 没有找到任务");
        return Ok(());
    }
    
    println!("📋 待办事项清单：\n");
    for task in tasks {
        let status_icon = match task.status {
            Status::Pending => "⬜",
            Status::Completed => "✅",
        };
        let priority_icon = match task.priority {
            Priority::High => "🔴",
            Priority::Medium => "🟡",
            Priority::Low => "🟢",
        };
        println!("{} {} [{}] {}", status_icon, priority_icon, task.id, task.title);
        if let Some(desc) = &task.description {
            println!("   └─ {}", desc);
        }
        println!("   └─ 优先级：{}", task.priority);
        println!();
    }
    
    Ok(())
}

/// 编辑任务
fn edit_task(
    id: u32,
    title: Option<&str>,
    description: Option<&str>,
    priority: Option<&str>,
) -> Result<(), Box<dyn std::error::Error>> {
    let mut todo_list = load_todo_list()?;
    
    let task = todo_list.tasks.iter_mut()
        .find(|t| t.id == id)
        .ok_or_else(|| format!("❌ 未找到任务 ID: {}", id))?;
    
    if let Some(new_title) = title {
        task.title = new_title.to_string();
    }
    
    if let Some(new_desc) = description {
        task.description = if new_desc.is_empty() {
            None
        } else {
            Some(new_desc.to_string())
        };
    }
    
    if let Some(new_priority) = priority {
        task.priority = match new_priority.to_lowercase().as_str() {
            "high" | "h" | "高" => Priority::High,
            "low" | "l" | "低" => Priority::Low,
            _ => Priority::Medium,
        };
    }
    
    task.updated_at = chrono::Local::now().format("%Y-%m-%d %H:%M:%S").to_string();
    
    save_todo_list(&todo_list)?;
    println!("✅ 任务 [{}] 已更新", id);
    
    Ok(())
}

// 在 main 函数中添加 edit 命令处理
"edit" => {
    if args.len() < 3 {
        eprintln!("❌ 错误：请提供任务 ID");
        eprintln!("用法：todo_cli edit <ID> [选项]");
        eprintln!("选项：--title <标题> --description <描述> --priority <高/中/低>");
        return;
    }
    
    let id = match args[2].parse::<u32>() {
        Ok(id) => id,
        Err(_) => {
            eprintln!("❌ 错误：无效的 ID");
            return;
        }
    };
    
    let mut title = None;
    let mut description = None;
    let mut priority = None;
    
    let mut i = 3;
    while i < args.len() {
        match args[i].as_str() {
            "--title" | "-t" => {
                if i + 1 < args.len() {
                    title = Some(&args[i + 1]);
                    i += 2;
                } else {
                    eprintln!("❌ 错误：--title 需要参数");
                    return;
                }
            }
            "--description" | "-d" => {
                if i + 1 < args.len() {
                    description = Some(&args[i + 1]);
                    i += 2;
                } else {
                    eprintln!("❌ 错误：--description 需要参数");
                    return;
                }
            }
            "--priority" | "-p" => {
                if i + 1 < args.len() {
                    priority = Some(&args[i + 1]);
                    i += 2;
                } else {
                    eprintln!("❌ 错误：--priority 需要参数");
                    return;
                }
            }
            _ => {
                eprintln!("❌ 未知选项：{}", args[i]);
                return;
            }
        }
    }
    
    if title.is_none() && description.is_none() && priority.is_none() {
        eprintln!("❌ 错误：请至少提供一个要修改的字段");
        return;
    }
    
    edit_task(id, title, description, priority)
}

use std::fs;
use std::path::PathBuf;

/// 备份数据文件
fn backup_data() -> Result<PathBuf, Box<dyn std::error::Error>> {
    let todo_list = load_todo_list()?;
    
    // 创建备份目录
    let backup_dir = PathBuf::from("backups");
    if !backup_dir.exists() {
        fs::create_dir_all(&backup_dir)?;
    }
    
    // 生成带时间戳的文件名
    let timestamp = chrono::Local::now().format("%Y%m%d_%H%M%S");
    let backup_file = backup_dir.join(format!("todo_backup_{}.json", timestamp));
    
    // 写入备份文件
    let json = serde_json::to_string_pretty(&todo_list)?;
    fs::write(&backup_file, json)?;
    
    println!("✅ 备份已创建：{}", backup_file.display());
    Ok(backup_file)
}

/// 从备份恢复数据
fn restore_data(backup_path: &str) -> Result<(), Box<dyn std::error::Error>> {
    let backup_file = PathBuf::from(backup_path);
    
    if !backup_file.exists() {
        return Err(format!("❌ 备份文件不存在：{}", backup_path).into());
    }
    
    // 先备份当前数据
    let current_backup = backup_data()?;
    println!("📦 当前数据已备份到：{}", current_backup.display());
    
    // 读取备份文件
    let json = fs::read_to_string(&backup_file)?;
    let todo_list: TodoList = serde_json::from_str(&json)?;
    
    // 覆盖当前数据
    save_todo_list(&todo_list)?;
    
    println!("✅ 数据已从备份恢复：{}", backup_path);
    Ok(())
}

/// 列出所有备份
fn list_backups() -> Result<(), Box<dyn std::error::Error>> {
    let backup_dir = PathBuf::from("backups");
    
    if !backup_dir.exists() {
        println!("📭 没有找到备份文件");
        return Ok(());
    }
    
    let mut backups: Vec<_> = fs::read_dir(&backup_dir)?
        .filter_map(|entry| entry.ok())
        .filter(|entry| {
            entry.path()
                .extension()
                .map(|ext| ext == "json")
                .unwrap_or(false)
        })
        .collect();
    
    // 按修改时间排序（最新的在前）
    backups.sort_by(|a, b| {
        b.metadata()
            .and_then(|m| m.modified())
            .unwrap_or(std::time::UNIX_EPOCH)
            .cmp(
                &a.metadata()
                    .and_then(|m| m.modified())
                    .unwrap_or(std::time::UNIX_EPOCH),
            )
    });
    
    println!("📦 备份文件列表：\n");
    for (i, entry) in backups.iter().enumerate().take(10) {
        let path = entry.path();
        let metadata = entry.metadata()?;
        let modified = metadata.modified()?;
        let size = metadata.len();
        
        println!("{}. {}", i + 1, path.file_name().unwrap().to_string_lossy());
        println!("   大小：{:.2} KB", size as f64 / 1024.0);
        println!("   时间：{:?}", modified);
        println!();
    }
    
    if backups.len() > 10 {
        println!("... 还有 {} 个备份文件", backups.len() - 10);
    }
    
    Ok(())
}

use crate::models::TodoList;
use serde_json;
use std::fs;
use std::path::Path;

const TODO_FILE: &str = "todo.json";

pub fn load_todo_list() -> Result<TodoList, Box<dyn std::error::Error>> {
    if !Path::new(TODO_FILE).exists() {
        return Ok(create_new_todo_list());
    }
    
    let json = fs::read_to_string(TODO_FILE)?;
    let todo_list: TodoList = serde_json::from_str(&json)?;
    Ok(todo_list)
}

pub fn save_todo_list(todo_list: &TodoList) -> Result<(), Box<dyn std::error::Error>> {
    let json = serde_json::to_string_pretty(todo_list)?;
    fs::write(TODO_FILE, json)?;
    Ok(())
}

fn create_new_todo_list() -> TodoList {
    TodoList {
        tasks: Vec::new(),
        next_id: 1,
    }
}

    fn test_create_task() {

            title: "测试任务".to_string(),

        assert!(Priority::High > Priority::Medium);
        assert!(Priority::Medium > Priority::Low);
        assert!(Priority::High > Priority::Low);

    fn test_filter_pending_tasks() {
        let tasks = vec![
            Task { id: 1, title: "任务 1".into(), description: None,

                   created_at: String::new(), updated_at: String::new() },

        assert_eq!(pending.len(), 1);
        assert_eq!(pending[0].id, 1);
    }

        let title = "学习 Rust";
