// src/errors.rs
use std::fmt;
use std::path::PathBuf;
use std::io;
use std::num::ParseIntError;

// ============ 核心错误类型 ============

/// 应用级别的错误，统一所有子系统的错误
#[derive(Debug)]
pub enum FileProcError {
    /// 配置相关错误
    Config(ConfigError),
    /// 文件处理相关错误
    Processing(ProcessingError),
    /// IO 相关错误
    Io(IoError),
    /// 命令行参数错误
    Cli(CliError),
    /// 内部错误（不应该发生）
    Internal(String),
}

/// 配置错误
#[derive(Debug)]
pub enum ConfigError {
    NotFound(PathBuf),
    InvalidFormat { path: PathBuf, line: usize, reason: String },
    MissingField { field: String, path: PathBuf },
    InvalidValue { field: String, value: String, expected: String },
}

/// 处理错误
#[derive(Debug)]
pub enum ProcessingError {
    FileNotFound { path: PathBuf },
    PermissionDenied { path: PathBuf },
    EncodingError { path: PathBuf, encoding: String },
    TransformFailed { operation: String, reason: String },
    OutputFailed { path: PathBuf, source: String },
}

/// IO 错误包装
#[derive(Debug)]
pub struct IoError {
    pub context: String,
    pub source: io::Error,
}

/// CLI 参数错误
#[derive(Debug)]
pub enum CliError {
    MissingArgument { name: String },
    InvalidArgument { name: String, value: String, reason: String },
    ConflictingArguments { arg1: String, arg2: String },
}

// ============ Display 实现 ============

impl fmt::Display for FileProcError {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        match self {
            FileProcError::Config(e) => write!(f, "配置错误：{}", e),
            FileProcError::Processing(e) => write!(f, "处理错误：{}", e),
            FileProcError::Io(e) => write!(f, "IO 错误：{}: {}", e.context, e.source),
            FileProcError::Cli(e) => write!(f, "参数错误：{}", e),
            FileProcError::Internal(msg) => write!(f, "内部错误：{}", msg),
        }
    }
}

impl fmt::Display for ConfigError {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        match self {
            ConfigError::NotFound(path) => {
                write!(f, "配置文件未找到：{}", path.display())
            }
            ConfigError::InvalidFormat { path, line, reason } => {
                write!(f, "配置文件格式错误 ({}:{}): {}", path.display(), line, reason)
            }
            ConfigError::MissingField { field, path } => {
                write!(f, "配置文件缺少必需字段 '{}': {}", field, path.display())
            }
            ConfigError::InvalidValue { field, value, expected } => {
                write!(f, "配置字段 '{}' 值无效: '{}' (期望：{})", field, value, expected)
            }
        }
    }
}

impl fmt::Display for ProcessingError {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        match self {
            ProcessingError::FileNotFound { path } => {
                write!(f, "文件未找到：{}", path.display())
            }
            ProcessingError::PermissionDenied { path } => {
                write!(f, "权限不足，无法访问：{}", path.display())
            }
            ProcessingError::EncodingError { path, encoding } => {
                write!(f, "文件编码错误 ({}): {}", encoding, path.display())
            }
            ProcessingError::TransformFailed { operation, reason } => {
                write!(f, "转换操作 '{}' 失败：{}", operation, reason)
            }
            ProcessingError::OutputFailed { path, source } => {
                write!(f, "写入输出文件失败 ({}): {}", path.display(), source)
            }
        }
    }
}

impl fmt::Display for IoError {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        write!(f, "{}", self.context)
    }
}

impl fmt::Display for CliError {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        match self {
            CliError::MissingArgument { name } => {
                write!(f, "缺少必需参数：--{}", name)
            }
            CliError::InvalidArgument { name, value, reason } => {
                write!(f, "参数 '--{}' 值无效: '{}' ({})", name, value, reason)
            }
            CliError::ConflictingArguments { arg1, arg2 } => {
                write!(f, "参数冲突：--{} 和 --{} 不能同时使用", arg1, arg2)
            }
        }
    }
}

// ============ Error trait 实现 ============

impl std::error::Error for FileProcError {
    fn source(&self) -> Option<&(dyn std::error::Error + 'static)> {
        match self {
            FileProcError::Config(e) => Some(e),
            FileProcError::Processing(e) => Some(e),
            FileProcError::Io(e) => Some(&e.source),
            FileProcError::Cli(e) => Some(e),
            FileProcError::Internal(_) => None,
        }
    }
}

impl std::error::Error for ConfigError {}
impl std::error::Error for ProcessingError {}
impl std::error::Error for IoError {
    fn source(&self) -> Option<&(dyn std::error::Error + 'static)> {
        Some(&self.source)
    }
}
impl std::error::Error for CliError {}

// ============ From 实现（自动转换） ============

impl From<io::Error> for FileProcError {
    fn from(err: io::Error) -> Self {
        match err.kind() {
            io::ErrorKind::NotFound => {
                FileProcError::Processing(ProcessingError::FileNotFound {
                    path: PathBuf::from(err.to_string()),
                })
            }
            io::ErrorKind::PermissionDenied => {
                FileProcError::Processing(ProcessingError::PermissionDenied {
                    path: PathBuf::from(err.to_string()),
                })
            }
            _ => FileProcError::Io(IoError {
                context: "IO 操作失败".to_string(),
                source: err,
            }),
        }
    }
}

impl From<ConfigError> for FileProcError {
    fn from(err: ConfigError) -> Self {
        FileProcError::Config(err)
    }
}

impl From<ProcessingError> for FileProcError {
    fn from(err: ProcessingError) -> Self {
        FileProcError::Processing(err)
    }
}

impl From<CliError> for FileProcError {
    fn from(err: CliError) -> Self {
        FileProcError::Cli(err)
    }
}

// ============ 错误上下文扩展 ============

/// 为 Result 添加上下文方法
pub trait ResultExt<T, E> {
    fn context(self, msg: impl Into<String>) -> Result<T, FileProcError>;
    fn with_context(self, f: impl FnOnce() -> String) -> Result<T, FileProcError>;
}

impl<T, E: std::error::Error + 'static> ResultExt<T, E> for Result<T, E> {
    fn context(self, msg: impl Into<String>) -> Result<T, FileProcError> {
        self.map_err(|e| FileProcError::Io(IoError {
            context: msg.into(),
            source: io::Error::new(io::ErrorKind::Other, e.to_string()),
        }))
    }
    
    fn with_context(self, f: impl FnOnce() -> String) -> Result<T, FileProcError> {
        self.map_err(|e| FileProcError::Io(IoError {
            context: f(),
            source: io::Error::new(io::ErrorKind::Other, e.to_string()),
        }))
    }
}

// ============ 错误报告生成 ============

/// 生成详细的错误报告
pub fn generate_error_report(err: &FileProcError) -> String {
    let mut report = String::new();
    
    report.push_str("❌ 错误发生\n");
    report.push_str(&"=".repeat(50));
    report.push('\n');
    report.push_str(&format!("{}\n", err));
    
    // 添加错误链
    let mut current: Option<&(dyn std::error::Error + 'static)> = err.source();
    let mut depth = 0;
    
    while let Some(source) = current {
        depth += 1;
        report.push_str(&format!("\n└─ 原因 {}: {}\n", depth, source));
        current = source.source();
    }
    
    // 添加建议
    report.push_str(&"\n".to_owned() + &"=".repeat(50));
    report.push_str("\n💡 建议:\n");
    
    match err {
        FileProcError::Config(ConfigError::NotFound(path)) => {
            report.push_str(&format!("   - 检查配置文件是否存在：{}\n", path.display()));
            report.push_str("   - 使用 --config 指定正确的配置文件路径\n");
        }
        FileProcError::Processing(ProcessingError::PermissionDenied { path }) => {
            report.push_str(&format!("   - 检查文件权限：{}\n", path.display()));
            report.push_str("   - 尝试使用 sudo 或修改文件权限\n");
        }
        FileProcError::Cli(CliError::MissingArgument { name }) => {
            report.push_str(&format!("   - 添加必需参数：--{}\n", name));
            report.push_str("   - 使用 --help 查看完整参数列表\n");
        }
        _ => {
            report.push_str("   - 查看详细错误信息\n");
            report.push_str("   - 使用 --verbose 启用详细日志\n");
        }
    }
    
    report
}

// src/config.rs
use std::fs;
use std::path::{Path, PathBuf};
use crate::errors::{ConfigError, FileProcError, ResultExt};

/// 应用配置
#[derive(Debug, Clone)]
pub struct AppConfig {
    pub input_dir: PathBuf,
    pub output_dir: PathBuf,
    pub max_file_size: u64,
    pub supported_extensions: Vec<String>,
    pub default_encoding: String,
    pub verbose: bool,
}

impl AppConfig {
    /// 从文件加载配置
    pub fn load(path: &Path) -> Result<Self, FileProcError> {
        if !path.exists() {
            return Err(ConfigError::NotFound(path.to_path_buf()).into());
        }
        
        let content = fs::read_to_string(path)
            .context(format!("读取配置文件失败：{}", path.display()))?;
        
        Self::parse(&content, path)
    }
    
    /// 解析配置内容
    fn parse(content: &str, path: &Path) -> Result<Self, FileProcError> {
        let mut config = AppConfig::default();
        
        for (line_num, line) in content.lines().enumerate() {
            let line = line.trim();
            
            // 跳过注释和空行
            if line.starts_with('#') || line.is_empty() {
                continue;
            }
            
            let parts: Vec<&str> = line.splitn(2, '=').collect();
            if parts.len() != 2 {
                return Err(ConfigError::InvalidFormat {
                    path: path.to_path_buf(),
                    line: line_num + 1,
                    reason: "期望格式：key = value".to_string(),
                }.into());
            }
            
            let key = parts[0].trim();
            let value = parts[1].trim();
            
            match key {
                "input_dir" => config.input_dir = PathBuf::from(value),
                "output_dir" => config.output_dir = PathBuf::from(value),
                "max_file_size" => {
                    config.max_file_size = value.parse()
                        .map_err(|_| ConfigError::InvalidValue {
                            field: "max_file_size".to_string(),
                            value: value.to_string(),
                            expected: "正整数".to_string(),
                        })?;
                }
                "supported_extensions" => {
                    config.supported_extensions = value
                        .split(',')
                        .map(|s| s.trim().to_string())
                        .collect();
                }
                "default_encoding" => {
                    config.default_encoding = value.to_string();
                }
                "verbose" => {
                    config.verbose = value.parse()
                        .map_err(|_| ConfigError::InvalidValue {
                            field: "verbose".to_string(),
                            value: value.to_string(),
                            expected: "true 或 false".to_string(),
                        })?;
                }
                _ => {
                    // 忽略未知字段
                }
            }
        }
        
        // 验证必需字段
        if config.input_dir.as_os_str().is_empty() {
            return Err(ConfigError::MissingField {
                field: "input_dir".to_string(),
                path: path.to_path_buf(),
            }.into());
        }
        
        Ok(config)
    }
    
    /// 使用默认配置
    pub fn default() -> Self {
        AppConfig {
            input_dir: PathBuf::from("./input"),
            output_dir: PathBuf::from("./output"),
            max_file_size: 10 * 1024 * 1024, // 10MB
            supported_extensions: vec![
                ".txt".to_string(),
                ".md".to_string(),
                ".log".to_string(),
            ],
            default_encoding: "utf-8".to_string(),
            verbose: false,
        }
    }
}

// src/processor.rs
use std::fs::{self, File};
use std::io::{BufRead, BufReader, Write};
use std::path::{Path, PathBuf};
use crate::config::AppConfig;
use crate::errors::{ProcessingError, FileProcError, ResultExt};

/// 文件处理器
pub struct FileProcessor {
    config: AppConfig,
    stats: ProcessingStats,
}

/// 处理统计信息
#[derive(Debug, Default)]
pub struct ProcessingStats {
    pub files_processed: usize,
    pub files_skipped: usize,
    pub total_lines: usize,
    pub total_bytes: u64,
    pub errors: Vec<(PathBuf, String)>,
}

impl FileProcessor {
    pub fn new(config: AppConfig) -> Self {
        FileProcessor {
            config,
            stats: ProcessingStats::default(),
        }
    }
    
    /// 处理目录中的所有文件
    pub fn process_directory(&mut self) -> Result<(), FileProcError> {
        if !self.config.input_dir.exists() {
            fs::create_dir_all(&self.config.input_dir)
                .context("创建输入目录失败")?;
        }
        
        if !self.config.output_dir.exists() {
            fs::create_dir_all(&self.config.output_dir)
                .context("创建输出目录失败")?;
        }
        
        let entries = fs::read_dir(&self.config.input_dir)
            .context(format!("读取目录失败：{}", self.config.input_dir.display()))?;
        
        for entry in entries {
            let entry = entry?;
            let path = entry.path();
            
            if path.is_file() {
                match self.process_file(&path) {
                    Ok(_) => self.stats.files_processed += 1,
                    Err(e) => {
                        self.stats.files_skipped += 1;
                        self.stats.errors.push((path.clone(), e.to_string()));
                        
                        if self.config.verbose {
                            eprintln!("⚠️  跳过文件 {}: {}", path.display(), e);
                        }
                    }
                }
            }
        }
        
        Ok(())
    }
    
    /// 处理单个文件
    fn process_file(&self, path: &Path) -> Result<(), FileProcError> {
        // 检查文件扩展名
        if let Some(ext) = path.extension() {
            let ext_str = format!(".{}", ext.to_string_lossy());
            if !self.config.supported_extensions.contains(&ext_str) {
                return Ok(()); // 跳过不支持的扩展名
            }
        }
        
        // 检查文件大小
        let metadata = fs::metadata(path)?;
        if metadata.len() > self.config.max_file_size {
            return Err(ProcessingError::TransformFailed {
                operation: "文件大小检查".to_string(),
                reason: format!("文件超过最大限制 ({} > {} bytes)", 
                    metadata.len(), self.config.max_file_size),
            }.into());
        }
        
        // 读取文件
        let file = File::open(path)?;
        let reader = BufReader::new(file);
        
        // 处理每一行
        let output_path = self.config.output_dir.join(
            format!("processed_{}", path.file_name().unwrap().to_string_lossy())
        );
        
        let mut output_file = File::create(&output_path)
            .map_err(|e| ProcessingError::OutputFailed {
                path: output_path.clone(),
                source: e.to_string(),
            })?;
        
        let mut line_count = 0;
        
        for (line_num, line_result) in reader.lines().enumerate() {
            let line = line_result?;
            line_count += 1;
            
            // 执行转换：转大写并添加行号
            let transformed = self.transform_line(&line, line_num + 1)?;
            
            writeln!(output_file, "{}", transformed)?;
        }
        
        if self.config.verbose {
            println!("✓ 处理完成：{} ({} 行)", path.display(), line_count);
        }
        
        Ok(())
    }
    
    /// 转换单行内容
    fn transform_line(&self, line: &str, line_num: usize) -> Result<String, FileProcError> {
        // 示例转换：转大写并添加行号前缀
        let transformed = format!("[{:04}] {}", line_num, line.to_uppercase());
        
        // 验证转换结果
        if transformed.contains('\0') {
            return Err(ProcessingError::TransformFailed {
                operation: "行转换".to_string(),
                reason: "转换结果包含空字符".to_string(),
            }.into());
        }
        
        Ok(transformed)
    }
    
    /// 获取统计信息
    pub fn get_stats(&self) -> &ProcessingStats {
        &self.stats
    }
    
    /// 生成处理报告
    pub fn generate_report(&self, output_path: &Path) -> Result<(), FileProcError> {
        let mut file = File::create(output_path)?;
        
        writeln!(file, "文件处理报告")?;
        writeln!(file, "===============")?;
        writeln!(file)?;
        writeln!(file, "统计信息:")?;
        writeln!(file, "  处理文件数：{}", self.stats.files_processed)?;
        writeln!(file, "  跳过文件数：{}", self.stats.files_skipped)?;
        writeln!(file, "  总行数：{}", self.stats.total_lines)?;
        writeln!(file, "  总字节数：{}", self.stats.total_bytes)?;
        writeln!(file)?;
        
        if !self.stats.errors.is_empty() {
            writeln!(file, "错误列表:")?;
            for (path, error) in &self.stats.errors {
                writeln!(file, "  - {}: {}", path.display(), error)?;
            }
        }
        
        Ok(())
    }
}

// src/main.rs
mod errors;
mod config;
mod processor;

use std::env;
use std::path::PathBuf;
use std::process;

use errors::{FileProcError, CliError, generate_error_report};
use config::AppConfig;
use processor::FileProcessor;

fn main() {
    match run() {
        Ok(_) => {
            println!("\n✅ 处理完成！");
            process::exit(0);
        }
        Err(e) => {
            eprintln!("\n{}", generate_error_report(&e));
            process::exit(1);
        }
    }
}

fn run() -> Result<(), FileProcError> {
    let args: Vec<String> = env::args().collect();
    
    // 解析命令行参数
    let config_path = parse_args(&args)?;
    
    // 加载配置
    let config = if config_path.exists() {
        AppConfig::load(&config_path)?
    } else {
        println!("⚠️  配置文件未找到，使用默认配置");
        AppConfig::default()
    };
    
    if config.verbose {
        println!("📋 配置信息:");
        println!("   输入目录：{}", config.input_dir.display());
        println!("   输出目录：{}", config.output_dir.display());
        println!("   最大文件大小：{} MB", config.max_file_size / 1024 / 1024);
        println!("   支持的扩展名：{:?}", config.supported_extensions);
        println!();
    }
    
    // 创建处理器并执行
    let mut processor = FileProcessor::new(config);
    processor.process_directory()?;
    
    // 生成报告
    let report_path = PathBuf::from("./output/report.txt");
    processor.generate_report(&report_path)?;
    println!("📊 报告已生成：{}", report_path.display());
    
    // 打印统计
    let stats = processor.get_stats();
    println!("\n📈 处理统计:");
    println!("   成功处理：{} 个文件", stats.files_processed);
    println!("   跳过：{} 个文件", stats.files_skipped);
    
    if !stats.errors.is_empty() {
        println!("\n⚠️  发生 {} 个错误:", stats.errors.len());
        for (path, error) in &stats.errors {
            println!("   - {}: {}", path.display(), error);
        }
    }
    
    Ok(())
}

fn parse_args(args: &[String]) -> Result<PathBuf, FileProcError> {
    let mut config_path = PathBuf::from("./config/app.conf");
    
    let mut i = 1;
    while i < args.len() {
        match args[i].as_str() {
            "--config" | "-c" => {
                if i + 1 >= args.len() {
                    return Err(CliError::MissingArgument {
                        name: "config 的值".to_string(),
                    }.into());
                }
                config_path = PathBuf::from(&args[i + 1]);
                i += 2;
            }
            "--verbose" | "-v" => {
                // 在配置中处理
                i += 1;
            }
            "--help" | "-h" => {
                print_help();
                process::exit(0);
            }
            arg if arg.starts_with('-') => {
                return Err(CliError::InvalidArgument {
                    name: arg.trim_start_matches('-').to_string(),
                    value: "".to_string(),
                    reason: "未知参数".to_string(),
                }.into());
            }
            _ => {
                i += 1;
            }
        }
    }
    
    Ok(config_path)
}

fn print_help() {
    println!("fileproc - 文件处理工具");
    println!();
    println!("用法:");
    println!("  fileproc [选项]");
    println!();
    println!("选项:");
    println!("  -c, --config <路径>    配置文件路径 (默认：./config/app.conf)");
    println!("  -v, --verbose          启用详细输出");
    println!("  -h, --help             显示帮助信息");
    println!();
    println!("示例:");
    println!("  fileproc --config ./my_config.conf");
    println!("  fileproc -v");
}

  • 并发错误（多线程处理时）

实现相应的 Display 和 From trait。

  • 最多重试 3 次
  • 每次重试间隔递增（1s, 2s, 4s）
  • 记录重试次数到统计信息

今日感悟

今天的综合实战让我真正理解了 Rust 错误处理系统的强大之处。

核心收获：

  1. 错误类型设计是 API 设计的一部分

    • 好的错误类型让调用者知道如何处理
    • 分层的错误类型让代码更清晰
    • 实现 From trait 让错误传播变得自然

  2. 上下文是关键

    • 裸的错误信息不够用
    • 添加路径、操作类型、状态等上下文
    • 错误链让调试更容易

  3. 用户友好的错误报告

    • 开发者需要详细的技术信息
    • 用户需要清晰的行动建议
    • generate_error_report 函数可以统一处理

遇到的挑战：

在实现 ResultExt trait 时，泛型约束让我困惑了很久。特别是 E: std::error::Error + 'static 这个约束，需要理解：

  • std::error::Error 确保可以调用 source() 方法
  • 'static 确保错误类型没有非静态引用

一个顿悟时刻：

之前我觉得 Rust 的错误处理太繁琐，要写这么多 impl。但今天写完整个项目后明白了：

  • 这些"繁琐"的代码让错误类型安全
  • 编译器帮我检查是否遗漏了错误处理
  • 运行时几乎不会有意外崩溃

代码质量提升：

对比第 1 周写的代码，今天的代码：

  • 每个可能失败的操作都有处理
  • 错误信息包含足够的上下文
  • 用户可以理解发生了什么以及如何处理
