// src/repl/mod.rs

use std::io::{self, Write};
use crate::storage::Storage;
use crate::commands::{add_task_interactive, complete_task, delete_task};
use crate::search::TaskSearch;

pub struct Repl {
    storage: Storage,
    running: bool,
}

impl Repl {
    pub fn new(storage: Storage) -> Self {
        Self {
            storage,
            running: true,
        }
    }
    
    /// 启动 REPL 主循环
    pub fn run(&mut self) {
        println!("\n🛠️  rustodo 交互式模式");
        println!("输入 'help' 查看命令，输入 'exit' 退出\n");
        
        while self.running {
            print!("rustodo> ");
            io::stdout().flush().unwrap();
            
            let mut input = String::new();
            if io::stdin().read_line(&mut input).is_err() {
                break;
            }
            
            let input = input.trim();
            if input.is_empty() {
                continue;
            }
            
            self.execute(input);
        }
        
        println!("再见！👋");
    }
    
    /// 执行命令
    fn execute(&mut self, input: &str) {
        let parts: Vec<&str> = input.split_whitespace().collect();
        if parts.is_empty() {
            return;
        }
        
        match parts[0].to_lowercase().as_str() {
            "help" => self.show_help(),
            "exit" | "quit" | "q" => {
                self.running = false;
            }
            "add" | "a" => {
                let content = parts[1..].join(" ");
                if content.is_empty() {
                    add_task_interactive(&mut self.storage);
                } else {
                    match self.storage.add_task(content) {
                        Ok(id) => println!("✅ 任务已创建，ID: {}", id),
                        Err(e) => println!("❌ 创建失败: {}", e),
                    }
                }
            }
            "list" | "ls" | "l" => {
                self.list_tasks(parts.get(1).map(|s| s.to_string()).as_deref());
            }
            "done" | "complete" | "c" => {
                if let Some(id_str) = parts.get(1) {
                    if let Ok(id) = id_str.parse::<u64>() {
                        match self.storage.complete_task(id) {
                            Ok(_) => println!("✅ 任务 {} 已完成", id),
                            Err(e) => println!("❌ 操作失败: {}", e),
                        }
                    } else {
                        println!("❌ 无效的 ID: {}", id_str);
                    }
                } else {
                    println!("用法: done <ID>");
                }
            }
            "delete" | "del" | "d" => {
                if let Some(id_str) = parts.get(1) {
                    if let Ok(id) = id_str.parse::<u64>() {
                        match self.storage.delete_task(id) {
                            Ok(_) => println!("🗑️  任务 {} 已删除", id),
                            Err(e) => println!("❌ 删除失败: {}", e),
                        }
                    } else {
                        println!("❌ 无效的 ID: {}", id_str);
                    }
                } else {
                    println!("用法: delete <ID>");
                }
            }
            "search" | "find" | "s" => {
                let query = parts[1..].join(" ");
                self.search_tasks(&query);
            }
            "stats" | "stat" => {
                self.show_stats();
            }
            "clear" | "cls" => {
                print!("\x1B[2J\x1B[1;1H");
            }
            _ => {
                println!("未知命令: {}，输入 'help' 查看帮助", parts[0]);
            }
        }
    }
    
    fn show_help(&self) {
        println("\n📖 可用命令:");
        println!("  add, a [内容]     - 添加新任务");
        println!("  list, ls, l [筛选条件] - 列出任务");
        println!("  done, c <ID>      - 完成任务");
        println!("  delete, d <ID>    - 删除任务");
        println!("  search, s <关键词> - 搜索任务");
        println!("  stats             - 显示统计");
        println!("  clear             - 清屏");
        println!("  exit, q           - 退出");
    }
    
    fn list_tasks(&self, filter: Option<&str>) {
        let tasks = self.storage.list_tasks();
        
        let filtered: Vec<_> = if let Some(f) = filter {
            TaskSearch::filter_tasks(&tasks, f)
        } else {
            tasks
        };
        
        if filtered.is_empty() {
            println!("暂无任务");
            return;
        }
        
        println!("\n📋 任务列表 (共 {} 个):", filtered.len());
        for task in &filtered {
            let status = if task.completed { "✅" } else { "⬜" };
            let due = task.due_date
                .map(|d| d.format("%m-%d").to_string())
                .unwrap_or_default();
            println!("  [{}] {} - {} {:?} {}", 
                task.id, status, task.content, task.priority, due);
        }
    }
    
    fn search_tasks(&self, query: &str) {
        if query.is_empty() {
            println!("请提供搜索关键词");
            return;
        }
        
        let tasks = self.storage.list_tasks();
        let results = TaskSearch::search(&tasks, query);
        
        if results.is_empty() {
            println!("没有找到匹配的任务");
            return;
        }
        
        println!("\n🔍 搜索结果 '{}' (共 {} 个):", query, results.len());
        for task in &results {
            let due = task.due_date
                .map(|d| d.format("%m-%d").to_string())
                .unwrap_or_default();
            println!("  [{}] {} {:?} {}", 
                task.id, task.content, task.priority, due);
        }
    }
    
    fn show_stats(&self) {
        let stats = self.storage.get_stats();
        let rate = self.storage.get_completion_rate();
        
        println!("\n📊 统计概览:");
        println!("  总任务: {}", stats.total);
        println!("  已完成: {} ({:.1}%)", stats.completed, rate);
        println!("  待完成: {}", stats.pending);
        println!("  已过期: {}", stats.overdue);
    }
}

// src/search/mod.rs

use crate::models::{Task, Priority};
use std::collections::HashSet;

#[derive(Debug, Clone)]
pub struct TaskSearch;

#[derive(Debug, Clone)]
pub enum FilterCriterion {
    Completed(bool),
    Priority(Priority),
    HasTag(String),
    Overdue,
    DueBefore(chrono::DateTime<chrono::Utc>),
    DueAfter(chrono::DateTime<chrono::Utc>),
}

impl TaskSearch {
    /// 搜索任务（全文匹配）
    pub fn search(tasks: &[Task], query: &str) -> Vec<Task> {
        let query_lower = query.to_lowercase();
        
        tasks.iter()
            .filter(|t| {
                t.content.to_lowercase().contains(&query_lower) ||
                t.tags.iter().any(|tag| tag.to_lowercase().contains(&query_lower))
            })
            .cloned()
            .collect()
    }
    
    /// 多条件组合过滤
    pub fn filter_tasks(tasks: &[Task], filter_str: &str) -> Vec<Task> {
        let parts: Vec<&str> = filter_str.split('&').collect();
        let mut result = tasks.to_vec();
        
        for part in parts {
            let part = part.trim();
            result = Self::apply_filter(&result, part);
        }
        
        result
    }
    
    fn apply_filter(tasks: &[Task], filter: &str) -> Vec<Task> {
        let filter = filter.trim();
        
        // 按优先级过滤
        if let Some(p) = Self::parse_priority(filter) {
            return tasks.iter()
                .filter(|t| t.priority == p)
                .cloned()
                .collect();
        }
        
        // 按状态过滤
        match filter.to_lowercase().as_str() {
            "done" | "completed" => {
                tasks.iter().filter(|t| t.completed).cloned().collect()
            }
            "pending" | "todo" | "undone" => {
                tasks.iter().filter(|t| !t.completed).cloned().collect()
            }
            "overdue" => {
                let now = chrono::Utc::now();
                tasks.iter()
                    .filter(|t| !t.completed && 
                        t.due_date.map_or(false, |d| d < now))
                    .cloned()
                    .collect()
            }
            _ => tasks.to_vec()
        }
    }
    
    fn parse_priority(s: &str) -> Option<Priority> {
        match s.to_lowercase().as_str() {
            "high" | "h" | "urgent" => Some(Priority::High),
            "medium" | "m" | "normal" => Some(Priority::Medium),
            "low" | "l" | "minor" => Some(Priority::Low),
            _ => None,
        }
    }
    
    /// 按标签分组
    pub fn group_by_tags(tasks: &[Task]) -> std::collections::HashMap<String, Vec<&Task>> {
        let mut groups: std::collections::HashMap<String, Vec<&Task>> = 
            std::collections::HashMap::new();
        
        for task in tasks {
            if task.tags.is_empty() {
                groups.entry("无标签".to_string()).or_default().push(task);
            } else {
                for tag in &task.tags {
                    groups.entry(tag.clone()).or_default().push(task);
                }
            }
        }
        
        groups
    }
    
    /// 按到期日期排序
    pub fn sort_by_due_date(tasks: &[Task], ascending: bool) -> Vec<Task> {
        let mut tasks = tasks.to_vec();
        
        tasks.sort_by(|a, b| {
            let comp = a.due_date.cmp(&b.due_date);
            if ascending { comp } else { comp.reverse() }
        });
        
        tasks
    }
    
    /// 模糊搜索（支持正则）
    pub fn fuzzy_search(tasks: &[Task], pattern: &str) -> Vec<Task> {
        tasks.iter()
            .filter(|t| {
                // 检查内容是否包含所有关键词字符（顺序不限）
                let content = t.content.to_lowercase();
                let pattern_lower = pattern.to_lowercase();
                pattern_lower.chars().all(|c| content.contains(c))
            })
            .cloned()
            .collect()
    }
}

// src/commands/interactive.rs

use crate::models::{Priority, Task};
use crate::storage::Storage;
use std::io::{self, Write};

pub fn add_task_interactive(storage: &mut Storage) {
    println!("\n📝 添加新任务");
    println!("=".repeat(40));
    
    // 获取任务内容
    print!("任务内容: ");
    io::stdout().flush().unwrap();
    let mut content = String::new();
    if io::stdin().read_line(&mut content).is_err() {
        return;
    }
    let content = content.trim().to_string();
    if content.is_empty() {
        println!("任务内容不能为空");
        return;
    }
    
    // 选择优先级
    println!("\n优先级:");
    println!("  1. 🔴 高 (High)");
    println!("  2. 🟡 中 (Medium) - 默认");
    println!("  3. 🟢 低 (Low)");
    print!("选择 (1-3) [2]: ");
    io::stdout().flush().unwrap();
    
    let mut priority_input = String::new();
    io::stdin().read_line(&mut priority_input).unwrap();
    let priority = match priority_input.trim() {
        "1" => Priority::High,
        "3" => Priority::Low,
        _ => Priority::Medium,
    };
    
    // 设置到期日期（可选）
    print!("到期日期 (YYYY-MM-DD，留空跳过): ");
    io::stdout().flush().unwrap();
    let mut due_input = String::new();
    io::stdin().read_line(&mut due_input).unwrap();
    let due_date = parse_date(&due_input.trim());
    
    // 添加标签
    print!("标签 (逗号分隔，留空跳过): ");
    io::stdout().flush().unwrap();
    let mut tags_input = String::new();
    io::stdin().read_line(&mut tags_input).unwrap();
    let tags: Vec<String> = tags_input.trim()
        .split(',')
        .map(|s| s.trim().to_string())
        .filter(|s| !s.is_empty())
        .collect();
    
    // 创建并保存任务
    let mut task = Task::new(storage.next_id, content);
    task.set_priority(priority);
    task.due_date = due_date;
    task.tags = tags;
    
    match storage.add_task(task.content.clone()) {
        Ok(id) => {
            println!("✅ 任务已创建，ID: {}", id);
            // 更新完整信息
            if let Some(mut t) = storage.find_task_by_id(id).cloned() {
                t.set_priority(priority);
                t.due_date = due_date;
                t.tags = tags;
                storage.update_task(&t).ok();
            }
        }
        Err(e) => println!("❌ 创建失败: {}", e),
    }
}

fn parse_date(s: &str) -> Option<chrono::DateTime<chrono::Utc>> {
    if s.is_empty() {
        return None;
    }
    
    chrono::DateTime::parse_from_str(
        &format!("{}T00:00:00+00:00", s), 
        "%Y-%m-%dT%H:%M:%S%z"
    ).ok()
    .map(|dt| dt.with_timezone(&chrono::Utc))
}

use crate::repl::Repl;
use crate::search::TaskSearch;
use crate::commands::interactive::add_task_interactive;

fn main() {
    let matches = clap::App::new("rustodo")
        .subcommand(
            clap::SubCommand::with_name("interactive")
                .aliases(&["i", "repl"])
                .about("启动交互式模式")
        )
        .subcommand(
            clap::SubCommand::with_name("search")
                .about("搜索任务")
                .arg(clap::Arg::with_name("QUERY").required(true))
                .arg(clap::Arg::with_name("filter")
                    .short("f")
                    .long("filter")
                    .takes_value(true)
                    .help("过滤条件: done|pending|overdue|high|medium|low"))
        )
        .get_matches();
    
    if let Some(_matches) = matches.subcommand_matches("interactive") {
        let storage = Storage::load();
        let mut repl = Repl::new(storage);
        repl.run();
        return;
    }
    
    if let Some(matches) = matches.subcommand_matches("search") {
        let query = matches.value_of("QUERY").unwrap();
        let filter = matches.value_of("filter");
        let storage = Storage::load();
        let tasks = storage.list_tasks();
        
        let results = if let Some(f) = filter {
            TaskSearch::filter_tasks(&tasks, f)
        } else {
            tasks
        };
        
        let results = TaskSearch::search(&results, query);
        
        println!("\n🔍 搜索结果:");
        for task in &results {
            println!("  [{}] {}", task.id, task.content);
        }
        return;
    }
}

// src/repl/history.rs

use rustyline::Editor;
use std::path::Path;

pub struct ReplHistory {
    editor: Editor<()>,
    history_file: PathBuf,
}

impl ReplHistory {
    pub fn new() -> Self {
        let history_file = dirs::data_local_dir()
            .unwrap_or_else(|| PathBuf::from("."))
            .join("rustodo")
            .join("history.txt");
        
        let mut editor = Editor::new().unwrap();
        
        // 加载历史记录
        if history_file.exists() {
            editor.load_history(&history_file).ok();
        }
        
        Self { editor, history_file }
    }
    
    pub fn read_line(&mut self, prompt: &str) -> Option<String> {
        self.editor.readline(prompt).ok()
    }
    
    pub fn add_history(&mut self, line: &str) {
        self.editor.add_history_entry(line);
        self.editor.save_history(&self.history_file).ok();
    }
}
