use chrono::{DateTime, Utc};
use serde::{Deserialize, Serialize};

/// 任务优先级
#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
#[serde(rename_all = "lowercase")]
pub enum Priority {
    Low,
    Medium,
    High,
}

impl Default for Priority {
    fn default() -> Self {
        Priority::Medium
    }
}

/// 单个任务
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct Task {
    pub id: u64,
    pub content: String,
    pub completed: bool,
    pub priority: Priority,
    pub created_at: DateTime<Utc>,
    pub due_date: Option<DateTime<Utc>>,
    pub completed_at: Option<DateTime<Utc>>,
    pub tags: Vec<String>,
}

impl Task {
    pub fn new(id: u64, content: String) -> Self {
        Task {
            id,
            content,
            completed: false,
            priority: Priority::default(),
            created_at: Utc::now(),
            due_date: None,
            completed_at: None,
            tags: Vec::new(),
        }
    }
    
    pub fn mark_done(&mut self) {
        self.completed = true;
        self.completed_at = Some(Utc::now());
    }
    
    pub fn is_overdue(&self) -> bool {
        if let Some(due) = self.due_date {
            !self.completed && due < Utc::now()
        } else {
            false
        }
    }
}

/// 任务列表（整个应用的数据）
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct TaskList {
    pub tasks: Vec<Task>,
    pub next_id: u64,
}

impl Default for TaskList {
    fn default() -> Self {
        TaskList {
            tasks: Vec::new(),
            next_id: 1,
        }
    }
}

impl TaskList {
    pub fn add_task(&mut self, content: String) -> u64 {
        let id = self.next_id;
        self.next_id += 1;
        let task = Task::new(id, content);
        self.tasks.push(task);
        id
    }
    
    pub fn get_task(&self, id: u64) -> Option<&Task> {
        self.tasks.iter().find(|t| t.id == id)
    }
    
    pub fn get_task_mut(&mut self, id: u64) -> Option<&mut Task> {
        self.tasks.iter_mut().find(|t| t.id == id)
    }
    
    pub fn remove_task(&mut self, id: u64) -> Option<Task> {
        if let Some(pos) = self.tasks.iter().position(|t| t.id == id) {
            Some(self.tasks.remove(pos))
        } else {
            None
        }
    }
    
    pub fn filter_completed(&self) -> Vec<&Task> {
        self.tasks.iter().filter(|t| t.completed).collect()
    }
    
    pub fn filter_pending(&self) -> Vec<&Task> {
        self.tasks.iter().filter(|t| !t.completed).collect()
    }
    
    pub fn search(&self, keyword: &str) -> Vec<&Task> {
        let keyword_lower = keyword.to_lowercase();
        self.tasks.iter()
            .filter(|t| t.content.to_lowercase().contains(&keyword_lower))
            .collect()
    }
}

use chrono::{DateTime, Utc};
use serde::{Deserialize, Serialize};

#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
#[serde(rename_all = "lowercase")]
pub enum Priority {
    Low,
    Medium,
    High,
}

impl Default for Priority {
    fn default() -> Self {
        Priority::Medium
    }
}

impl std::fmt::Display for Priority {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        match self {
            Priority::Low => write!(f, "low"),
            Priority::Medium => write!(f, "medium"),
            Priority::High => write!(f, "high"),
        }
    }
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct Task {
    pub id: u64,
    pub content: String,
    pub completed: bool,
    pub priority: Priority,
    pub created_at: DateTime<Utc>,
    pub due_date: Option<DateTime<Utc>>,
    pub completed_at: Option<DateTime<Utc>>,
    pub tags: Vec<String>,
}

impl Task {
    pub fn new(id: u64, content: String) -> Self {
        Task {
            id,
            content,
            completed: false,
            priority: Priority::default(),
            created_at: Utc::now(),
            due_date: None,
            completed_at: None,
            tags: Vec::new(),
        }
    }
    
    pub fn mark_done(&mut self) {
        self.completed = true;
        self.completed_at = Some(Utc::now());
    }
    
    pub fn is_overdue(&self) -> bool {
        if let Some(due) = self.due_date {
            !self.completed && due < Utc::now()
        } else {
            false
        }
    }
    
    pub fn set_priority(&mut self, priority: Priority) {
        self.priority = priority;
    }
    
    pub fn set_due_date(&mut self, due_date: DateTime<Utc>) {
        self.due_date = Some(due_date);
    }
    
    pub fn set_content(&mut self, content: String) {
        self.content = content;
    }
    
    pub fn add_tag(&mut self, tag: String) {
        if !self.tags.contains(&tag) {
            self.tags.push(tag);
        }
    }
    
    pub fn remove_tag(&mut self, tag: &str) {
        self.tags.retain(|t| t != tag);
    }
}

use super::task::Task;
use serde::{Deserialize, Serialize};

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct TaskList {
    pub tasks: Vec<Task>,
    pub next_id: u64,
}

impl Default for TaskList {
    fn default() -> Self {
        TaskList {
            tasks: Vec::new(),
            next_id: 1,
        }
    }
}

impl TaskList {
    pub fn new() -> Self {
        Self::default()
    }
    
    pub fn add_task(&mut self, content: String) -> u64 {
        let id = self.next_id;
        self.next_id += 1;
        let task = Task::new(id, content);
        self.tasks.push(task);
        id
    }
    
    pub fn get_task(&self, id: u64) -> Option<&Task> {
        self.tasks.iter().find(|t| t.id == id)
    }
    
    pub fn get_task_mut(&mut self, id: u64) -> Option<&mut Task> {
        self.tasks.iter_mut().find(|t| t.id == id)
    }
    
    pub fn remove_task(&mut self, id: u64) -> Option<Task> {
        if let Some(pos) = self.tasks.iter().position(|t| t.id == id) {
            Some(self.tasks.remove(pos))
        } else {
            None
        }
    }
    
    pub fn complete_task(&mut self, id: u64) -> Option<()> {
        let task = self.get_task_mut(id)?;
        task.mark_done();
        Some(())
    }
    
    pub fn filter_completed(&self) -> Vec<&Task> {
        self.tasks.iter().filter(|t| t.completed).collect()
    }
    
    pub fn filter_pending(&self) -> Vec<&Task> {
        self.tasks.iter().filter(|t| !t.completed).collect()
    }
    
    pub fn filter_by_priority(&self, priority: super::task::Priority) -> Vec<&Task> {
        self.tasks.iter().filter(|t| t.priority == priority).collect()
    }
    
    pub fn search(&self, keyword: &str) -> Vec<&Task> {
        let keyword_lower = keyword.to_lowercase();
        self.tasks.iter()
            .filter(|t| t.content.to_lowercase().contains(&keyword_lower))
            .collect()
    }
    
    pub fn sort_by_created(&mut self, descending: bool) {
        if descending {
            self.tasks.sort_by(|a, b| b.created_at.cmp(&a.created_at));
        } else {
            self.tasks.sort_by(|a, b| a.created_at.cmp(&b.created_at));
        }
    }
    
    pub fn sort_by_priority(&mut self) {
        self.tasks.sort_by(|a, b| b.priority.cmp(&a.priority));
    }
    
    pub fn len(&self) -> usize {
        self.tasks.len()
    }
    
    pub fn is_empty(&self) -> bool {
        self.tasks.is_empty()
    }
}

use clap::{Parser, Subcommand, ValueEnum};

#[derive(Parser)]
#[command(
    name = "rustodo",
    about = "A simple and efficient command-line task manager",
    author = "Your Name",
    version = "0.1.0"
)]
pub struct Cli {
    #[command(subcommand)]
    pub command: Commands,
    
    /// 启用 verbose 输出
    #[arg(short, long)]
    pub verbose: bool,
    
    /// 指定数据文件路径
    #[arg(short, long, value_name = "FILE")]
    pub file: Option<std::path::PathBuf>,
}

#[derive(Subcommand)]
pub enum Commands {
    /// 添加新任务
    Add {
        /// 任务内容
        content: String,
        
        /// 设置优先级
        #[arg(short, long, value_enum)]
        priority: Option<PriorityArg>,
        
        /// 设置截止日期 (YYYY-MM-DD)
        #[arg(short, long)]
        due: Option<String>,
        
        /// 添加标签
        #[arg(short, long)]
        tag: Vec<String>,
    },
    
    /// 列出所有任务
    List {
        /// 只显示未完成的任务
        #[arg(short, long)]
        pending: bool,
        
        /// 只显示已完成的任务
        #[arg(short, long)]
        completed: bool,
        
        /// 按优先级筛选
        #[arg(short, long, value_enum)]
        priority: Option<PriorityArg>,
    },
    
    /// 标记任务完成
    Done {
        /// 任务 ID
        id: u64,
    },
    
    /// 删除任务
    Delete {
        /// 任务 ID
        id: u64,
    },
    
    /// 编辑任务
    Edit {
        /// 任务 ID
        id: u64,
        
        /// 新内容
        #[arg(short, long)]
        content: Option<String>,
        
        /// 设置优先级
        #[arg(short, long, value_enum)]
        priority: Option<PriorityArg>,
        
        /// 设置截止日期
        #[arg(short, long)]
        due: Option<String>,
    },
    
    /// 搜索任务
    Search {
        /// 搜索关键字
        keyword: String,
    },
    
    /// 显示统计信息
    Stats,
}

#[derive(Copy, Clone, PartialEq, Eq, ValueEnum)]
#[arg(rename_all = "lowercase")]
pub enum PriorityArg {
    Low,
    Medium,
    High,
}

impl From<PriorityArg> for crate::models::Priority {
    fn from(p: PriorityArg) -> Self {
        match p {
            PriorityArg::Low => crate::models::Priority::Low,
            PriorityArg::Medium => crate::models::Priority::Medium,
            PriorityArg::High => crate::models::Priority::High,
        }
    }
}

use thiserror::Error;

#[derive(Error, Debug)]
pub enum AppError {
    #[error("Task not found: id={0}")]
    TaskNotFound(u64),
    
    #[error("Storage error: {0}")]
    Storage(String),
    
    #[error("Invalid date format: {0}. Expected YYYY-MM-DD")]
    InvalidDateFormat(String),
    
    #[error("IO error: {0}")]
    Io(#[from] std::io::Error),
    
    #[error("Serialization error: {0}")]
    Serialization(String),
}

pub type Result<T> = std::result::Result<T, AppError>;

impl serde::Serialize for AppError {
    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
    where
        S: serde::Serializer,
    {
        serializer.serialize_str(self.to_string().as_ref())
    }
}

use crate::errors::{AppError, Result};
use crate::models::TaskList;
use std::fs;
use std::path::Path;

/// 从文件加载任务列表
pub fn load(path: &Path) -> Result<TaskList> {
    if !path.exists() {
        return Ok(TaskList::new());
    }
    
    let content = fs::read_to_string(path)
        .map_err(|e| AppError::Storage(e.to_string()))?;
    
    // 尝试 JSON 格式
    if let Ok(list) = serde_json::from_str(&content) {
        return Ok(list);
    }
    
    // 尝试 YAML 格式
    if let Ok(list) = serde_yaml::from_str(&content) {
        return Ok(list);
    }
    
    // 格式不支持
    Err(AppError::Storage("Unknown file format".to_string()))
}

/// 保存任务列表到文件
pub fn save(path: &Path, list: &TaskList) -> Result<()> {
    // 确保目录存在
    if let Some(parent) = path.parent() {
        fs::create_dir_all(parent)
            .map_err(|e| AppError::Storage(e.to_string()))?;
    }
    
    // 默认使用 JSON 格式
    let content = serde_json::to_string_pretty(list)
        .map_err(|e| AppError::Serialization(e.to_string()))?;
    
    fs::write(path, content)
        .map_err(|e| AppError::Storage(e.to_string()))?;
    
    Ok(())
}

/// 获取默认数据文件路径
pub fn default_path() -> std::path::PathBuf {
    let home = dirs::home_dir().unwrap_or_else(|| std::path::PathBuf::from("."));
    home.join(".rustodo").join("tasks.json")
}

关键概念理解

项目结构设计原则

原则 说明 示例
模块化 按功能划分模块 commands/, models/, storage/
单一职责 每个模块只做一件事 task.rs 只处理 Task 模型
依赖注入 通过参数传递依赖而非全局状态 load(path: &Path)
错误抽象 统一错误类型，便于处理 AppError 枚举

依赖管理最佳实践

  1. 固定主版本：使用 clap = "4" 而非 "*"，确保 API 兼容性
  2. 最小化依赖：只添加真正需要的依赖
  3. 开发依赖分离：[dev-dependencies] 用于测试工具
  4. 条件编译：使用 features 按需启用功能

数据存储设计

  • 优先级：先 JSON 后 YAML 兼容
  • 路径管理：使用 dirs crate 获取跨平台路径
  • 容错处理：文件不存在时返回空列表而非报错

use anyhow::Result;
use clap::Parser;
use log::info;

fn main() -> Result<()> {
    env_logger::Builder::from_env(env_logger::Env::default().default_filter_or("info"))
        .init();
    
    let cli = rustodo::cli::Cli::parse();
    let data_path = cli.file.unwrap_or_else(rustodo::storage::file::default_path);
    let mut task_list = rustodo::storage::file::load(&data_path)?;
    
    match &cli.command {
        rustodo::cli::Commands::Add { content, priority, due, tag } => {
            let id = task_list.add_task(content.clone());
            if let Some(p) = priority {
                if let Some(task) = task_list.get_task_mut(id) {
                    task.set_priority(rustodo::models::Priority::from(*p));
                }
            }
            println!("Task added with id: {}", id);
        },
        
        rustodo::cli::Commands::List { pending, completed, priority } => {
            let tasks: Vec<_> = task_list.tasks.iter().collect();
            let filtered: Vec<_> = match (pending, completed, priority) {
                (true, _, _) => tasks.iter().filter(|t| !t.completed).copied().collect(),
                (_, true, _) => tasks.iter().filter(|t| t.completed).copied().collect(),
                (_, _, Some(p)) => tasks.iter().filter(|t| {
                    use rustodo::models::Priority;
                    t.priority == Priority::from(*p)
                }).copied().collect(),
                _ => tasks,
            };
            
            for task in filtered {
                let status = if task.completed { "✓" } else { " " };
                println!("[{}] [{}] {}", status, task.id, task.content);
            }
        },
        
        rustodo::cli::Commands::Done { id } => {
            if task_list.complete_task(*id).is_some() {
                println!("Task {} marked as done", id);
            } else {
                anyhow::bail!("Task {} not found", id);
            }
        },
        
        rustodo::cli::Commands::Delete { id } => {
            if task_list.remove_task(*id).is_some() {
                println!("Task {} deleted", id);
            } else {
                anyhow::bail!("Task {} not found", id);
            }
        },
        
        rustodo::cli::Commands::Search { keyword } => {
            let results = task_list.search(keyword);
            for task in results {
                println!("[{}] {}", task.id, task.content);
            }
        },
        
        rustodo::cli::Commands::Stats => {
            let total = task_list.len();
            let done = task_list.filter_completed().len();
            let pending = total - done;
            println!("Total: {}, Done: {}, Pending: {}", total, done, pending);
        },
        
        _ => {
            println!("Command not yet fully implemented");
        }
    }
    
    rustodo::storage::file::save(&data_path, &task_list)?;
    Ok(())
}

use chrono::{DateTime, NaiveDateTime, TimeZone, Utc};

fn parse_date(date_str: &str) -> Result<DateTime<Utc>, String> {
    let naive = NaiveDateTime::parse_from_str(date_str, "%Y-%m-%d")
        .map_err(|e| format!("Invalid date format: {}", e))?;
    Ok(Utc.from_utc_datetime(&naive))
}

impl Task {
    pub fn add_tag(&mut self, tag: String) {
        if !self.tags.contains(&tag) {
            self.tags.push(tag);
        }
    }
    
    pub fn remove_tag(&mut self, tag: &str) {
        self.tags.retain(|t| t != tag);
    }
    
    pub fn has_tag(&self, tag: &str) -> bool {
        self.tags.iter().any(|t| t == tag)
    }
}
