use serde::{Deserialize, Serialize};
use std::sync::OnceLock;
use tokio::runtime::Runtime;

// 共享测试数据结构
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct TestUser {
    pub id: u32,
    pub name: String,
    pub email: String,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct TestPost {
    pub id: u32,
    pub title: String,
    pub content: String,
    pub author_id: u32,
}

// 测试数据库配置
pub struct TestDbConfig {
    pub url: String,
    pub pool_size: u32,
}

impl Default for TestDbConfig {
    fn default() -> Self {
        TestDbConfig {
            url: "sqlite::memory:".to_string(),
            pool_size: 1,
        }
    }
}

// 共享运行时（避免每个测试都创建）
static TEST_RUNTIME: OnceLock<Runtime> = OnceLock::new();

pub fn get_test_runtime() -> &'static Runtime {
    TEST_RUNTIME.get_or_init(|| {
        Runtime::new().expect("Failed to create test runtime")
    })
}

// 测试辅助函数
pub fn create_test_user(id: u32, name: &str) -> TestUser {
    TestUser {
        id,
        name: name.to_string(),
        email: format!("{}@test.example.com", name),
    }
}

pub fn create_test_post(id: u32, title: &str, content: &str, author_id: u32) -> TestPost {
    TestPost {
        id,
        title: title.to_string(),
        content: content.to_string(),
        author_id,
    }
}

// 生成唯一 ID（避免测试间冲突）
use std::sync::atomic::{AtomicU32, Ordering};

static ID_COUNTER: AtomicU32 = AtomicU32::new(1000);

pub fn generate_unique_id() -> u32 {
    ID_COUNTER.fetch_add(1, Ordering::SeqCst)
}

pub fn generate_unique_email() -> String {
    format!("user_{}@test.example.com", generate_unique_id())
}

async fn test_create_user() {
    // 初始化测试数据库
    let db = Database::new_in_memory().await.unwrap();
    let service = UserService::new(db);

    // 创建用户
    let user = create_test_user(1, "Alice");
    let result = service.create_user(&user).await;

    assert!(result.is_ok());
    let created = result.unwrap();
    assert_eq!(created.name, "Alice");
    assert_eq!(created.email, "Alice@test.example.com");

#[tokio::test]
async fn test_get_user_by_id() {
    let db = Database::new_in_memory().await.unwrap();
    let service = UserService::new(db);

    // 先创建
    let user = create_test_user(1, "Bob");
    service.create_user(&user).await.unwrap();

    // 再查询
    let found = service.get_user_by_id(1).await.unwrap();
    assert_eq!(found.name, "Bob");

    // 查询不存在的用户
    let not_found = service.get_user_by_id(999).await;
    assert!(not_found.is_none());
}

#[tokio::test]
async fn test_update_user() {
    let db = Database::new_in_memory().await.unwrap();
    let service = UserService::new(db);

    let user = create_test_user(1, "Charlie");
    service.create_user(&user).await.unwrap();

    // 更新
    let mut update = user.clone();
    update.email = "charlie_new@test.example.com".to_string();
    
    let updated = service.update_user(1, &update).await.unwrap();
    assert_eq!(updated.email, "charlie_new@test.example.com");
}

async fn test_delete_user() {
    let db = Database::new_in_memory().await.unwrap();
    let service = UserService::new(db);

    let user = create_test_user(1, "David");
    service.create_user(&user).await.unwrap();

    // 删除
    let deleted = service.delete_user(1).await;
    assert!(deleted.is_ok());

    // 验证已删除
    let found = service.get_user_by_id(1).await;
    assert!(found.is_none());
}

// tests/e2e_tests.rs

use reqwest;
use serde::{Deserialize, Serialize};

#[derive(Debug, Serialize, Deserialize)]
struct CreateUserRequest {
    name: String,
    email: String,
}

#[derive(Debug, Serialize, Deserialize)]
struct UserResponse {
    id: u32,
    name: String,
    email: String,
    created_at: String,
}

// 测试服务器地址（本地测试服务器）
const BASE_URL: &str = "http://localhost:8080";

#[tokio::test]
async fn test_e2e_user_crud() {
    let client = reqwest::Client::new();

    // 1. 创建用户
    let create_req = CreateUserRequest {
        name: "E2E Test User".to_string(),
        email: "e2e@test.com".to_string(),
    };

    let response = client
        .post(format!("{}/api/users", BASE_URL))
        .json(&create_req)
        .send()
        .await
        .expect("Failed to send create request");

    assert!(response.status().is_success());
    let created: UserResponse = response.json().await.unwrap();
    assert_eq!(created.name, "E2E Test User");
    let user_id = created.id;

    // 2. 获取用户
    let response = client
        .get(format!("{}/api/users/{}", BASE_URL, user_id))
        .send()
        .await
        .expect("Failed to send get request");

    assert!(response.status().is_success());
    let fetched: UserResponse = response.json().await.unwrap();
    assert_eq!(fetched.id, user_id);

    // 3. 删除用户
    let response = client
        .delete(format!("{}/api/users/{}", BASE_URL, user_id))
        .send()
        .await
        .expect("Failed to send delete request");

    assert!(response.status().is_success());

    // 4. 验证已删除
    let response = client
        .get(format!("{}/api/users/{}", BASE_URL, user_id))
        .send()
        .await
        .expect("Failed to send verify delete request");

    assert_eq!(response.status(), 404);
}

#[tokio::test]
async fn test_e2e_error_handling() {
    let client = reqwest::Client::new();

    // 测试 404
    let response = client
        .get(format!("{}/api/users/99999", BASE_URL))
        .send()
        .await
        .unwrap();

    assert_eq!(response.status(), 404);

    // 测试 400（无效输入）
    let invalid_req = CreateUserRequest {
        name: "".to_string(),
        email: "invalid".to_string(),
    };

    let response = client
        .post(format!("{}/api/users", BASE_URL))
        .json(&invalid_req)
        .send()
        .await
        .unwrap();

    assert_eq!(response.status(), 400);
}

// src/calculator.rs

use std::fmt;

#[derive(Debug, Clone, PartialEq)]
pub enum CalculatorError {
    DivisionByZero,
    Overflow,
}

impl fmt::Display for CalculatorError {
    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
        match self {
            CalculatorError::DivisionByZero => write!(f, "Division by zero"),
            CalculatorError::Overflow => write!(f, "Arithmetic overflow"),
        }
    }
}

pub type CalcResult<T> = Result<T, CalculatorError>;

pub struct Calculator;

impl Calculator {
    pub fn new() -> Self {
        Calculator
    }

    pub fn add(&self, a: i32, b: i32) -> CalcResult<i32> {
        a.checked_add(b).ok_or(CalculatorError::Overflow)
    }

    pub fn subtract(&self, a: i32, b: i32) -> CalcResult<i32> {
        a.checked_sub(b).ok_or(CalculatorError::Overflow)
    }

    pub fn multiply(&self, a: i32, b: i32) -> CalcResult<i32> {
        a.checked_mul(b).ok_or(CalculatorError::Overflow)
    }

    pub fn divide(&self, a: i32, b: i32) -> CalcResult<i32> {
        if b == 0 {
            Err(CalculatorError::DivisionByZero)
        } else {
            Ok(a / b)
        }
    }
}

// src/stack.rs

use std::fmt;

pub struct Stack<T> {
    items: Vec<T>,
}

impl<T> Stack<T> {
    pub fn new() -> Self {
        Stack { items: Vec::new() }
    }

    pub fn push(&mut self, item: T) {
        self.items.push(item);
    }

    pub fn pop(&mut self) -> Option<T> {
        self.items.pop()
    }

    pub fn peek(&self) -> Option<&T> {
        self.items.last()
    }

    pub fn is_empty(&self) -> bool {
        self.items.is_empty()
    }

    pub fn len(&self) -> usize {
        self.items.len()
    }

    pub fn clear(&mut self) {
        self.items.clear();
    }
}

impl<T> Default for Stack<T> {
    fn default() -> Self {
        Self::new()
    }
}

impl<T: fmt::Debug> fmt::Debug for Stack<T> {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        f.debug_struct("Stack")
            .field("items", &self.items)
            .field("len", &self.items.len())
            .finish()
    }
}

pub struct UserBuilder {

impl UserBuilder {

            id: None,

            age: None,
        }

    pub fn id(mut self, id: u32) -> Self {

        self
    }

    }

    pub fn email(mut self, email: &str) -> Self {
        self.email = email.to_string();
        self
    }

    pub fn age(mut self, age: u32) -> Self {
        self.age = Some(age);
        self
    }

    pub fn build(self) -> User {
        User {
            id: self.id.unwrap_or(1),
            name: self.name,
            email: self.email,
            age: self.age,
        }
    }
}

// 使用
#[test]
fn test_user_creation() {
    let user = UserBuilder::new()
        .name("Alice")
        .email("alice@test.com")
        .age(25)
        .build();

    assert_eq!(user.name, "Alice");
}

  • 单元测试：测试单个函数/方法，快速，隔离
  • 集成测试：测试模块间交互
  • E2E 测试：测试完整系统流程

2. FIRST 原则

  • Fast：测试要快
  • Isolated：测试要独立
  • Repeatable：测试结果要可重复
  • Self-validating：测试自动判断成败
  • Timely：测试及时编写（TDD）

💭 今日感悟

今天的测试实战内容让我深刻体会到：

  1. 测试是设计工具，不仅仅是验证工具：TDD 强迫你在写代码前先思考 API 设计，结果往往是更清晰、更易用的接口。

  2. 好的测试结构决定项目可维护性：共享的测试工具模块、清晰的测试组织、独立的测试用例，这些看似繁琐的工作在长期维护中会带来巨大回报。

  3. 测试金字塔是实践指南：不要过度依赖 E2E 测试（慢且脆弱），也不要只写单元测试（可能错过集成问题）。找到平衡点。

  4. FIRST 原则是检验标准：每次写测试时，问问自己：这个测试够快吗？独立吗？可重复吗？自动验证吗？及时编写吗？

  5. 覆盖率是指标，不是目标：追求 100% 覆盖率可能导致过度测试，80-90% 通常是合理的目标。重要的是测试的质量，而非数量。
