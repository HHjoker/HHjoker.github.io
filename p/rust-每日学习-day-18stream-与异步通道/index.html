use std::pin::Pin;
use std::task::{Context, Poll};

/// Stream trait 定义
pub trait Stream {
    type Item;
    
    /// 轮询下一个元素
    fn poll_next(
        self: Pin<&mut Self>,
        cx: &mut Context<'_>
    ) -> Poll<Option<Self::Item>>;
}

Stream vs Iterator：

特性 Iterator Stream
执行方式 同步 异步
返回值 Option<T> Poll<Option<T>>
方法 next() poll_next()
适用场景 本地数据 I/O、网络、定时器等

            sleep(Duration::from_millis(100)).await;
            x

    let stream2 = stream::iter(vec!['a', 'b', 'c'])
        .then(|x| async move {
            sleep(Duration::from_millis(150)).await;
            x
        });
    
    // merge 会交替从两个 Stream 获取元素
    let merged = stream1.merge(stream2);
    
    merged.for_each(|item| async move {
        println!("合并流：{:?}", item);

use tokio::sync::mpsc;

#[tokio::main]
async fn main() {
    // 创建有界通道（容量 32）
    let (tx, mut rx) = mpsc::channel(32);
    
    // 启动多个生产者
    for i in 0..3 {
        let tx = tx.clone();
        tokio::spawn(async move {
            println!("📤 生产者 {} 开始", i);
            for j in 0..3 {
                tx.send(format!("生产者 {} 的消息 {}", i, j))
                    .await
                    .unwrap();
            }
            println!("✅ 生产者 {} 完成", i);
        });
    }
    
    // 丢弃原始 sender，否则 rx.recv() 永远不会返回 None
    drop(tx);
    
    // 单消费者接收所有消息
    println!("📥 消费者开始接收...");
    while let Some(msg) = rx.recv().await {
        println!("收到：{}", msg);
    }
    println!("🎉 所有消息接收完成");
}

use tokio::sync::broadcast;

#[tokio::main]
async fn main() {
    // 创建广播通道（容量 16）
    let (tx, mut rx1) = broadcast::channel(16);
    let mut rx2 = tx.subscribe();  // 创建第二个接收者
    let mut rx3 = tx.subscribe();  // 创建第三个接收者
    
    // 启动广播者
    tokio::spawn(async move {
        for i in 0..5 {
            let msg = format!("广播消息 {}", i);
            tx.send(msg).unwrap();
            println!("📢 发送：广播消息 {}", i);
            tokio::time::sleep(tokio::time::Duration::from_millis(100)).await;
        }
    });
    
    // 三个接收者同时接收
    let recv1 = tokio::spawn(async move {
        while let Ok(msg) = rx1.recv().await {
            println!("📱 接收者 1: {}", msg);
        }
    });
    
    let recv2 = tokio::spawn(async move {
        while let Ok(msg) = rx2.recv().await {
            println!("📱 接收者 2: {}", msg);
        }
    });
    
    let recv3 = tokio::spawn(async move {
        while let Ok(msg) = rx3.recv().await {
            println!("📱 接收者 3: {}", msg);
        }
    });
    
    // 等待所有任务完成
    let _ = tokio::join!(recv1, recv2, recv3);
}

  • 📡 实时通知系统
  • 🎮 游戏状态同步
  • 📊 实时数据推送

use tokio::sync::watch;

#[tokio::main]
async fn main() {
    // 创建 watch 通道（初始值为 0）
    let (tx, mut rx1) = watch::channel(0);
    let mut rx2 = tx.subscribe();
    
    // 生产者：更新状态
    tokio::spawn(async move {
        for i in 1..=5 {
            tx.send(i).unwrap();
            println!("📝 更新状态：{}", i);
            tokio::time::sleep(tokio::time::Duration::from_millis(200)).await;
        }
    });
    
    // 消费者 1
    tokio::spawn(async move {
        while rx1.changed().await.is_ok() {
            println!("👁️  观察者 1 看到：{}", *rx1.borrow());
        }
    });
    
    // 消费者 2
    tokio::spawn(async move {
        while rx2.changed().await.is_ok() {
            println!("👁️  观察者 2 看到：{}", *rx2.borrow());
        }
    });
    
    // 等待一段时间
    tokio::time::sleep(tokio::time::Duration::from_secs(2)).await;
}

watch vs broadcast：

特性 watch broadcast
存储 只保留最新值 保留所有消息（有限容量）
消费者错过消息 只看到最新值 会收到 Lagged 错误
适用场景 配置、状态同步 事件流、日志、通知

use tokio::sync::mpsc;
use tokio::time::{sleep, Duration};

/// 工作任务
#[derive(Debug)]
struct Job {
    id: u32,
    description: String,
}

/// 工作结果
#[derive(Debug)]
struct JobResult {
    job_id: u32,
    output: String,
}

/// 生产者：生成任务
async fn producer(tx: mpsc::Sender<Job>, num_jobs: u32) {
    println!("🏭 生产者启动");
    
    for i in 0..num_jobs {
        let job = Job {
            id: i,
            description: format!("任务 #{}", i),
        };
        
        println!("📤 生产：{:?}", job);
        tx.send(job).await.unwrap();
        
        // 模拟生产时间
        sleep(Duration::from_millis(100)).await;
    }
    
    println!("✅ 生产者完成，共生产 {} 个任务", num_jobs);
}

/// 消费者：处理任务
async fn consumer(mut rx: mpsc::Receiver<Job>, worker_id: u32) {
    println!("👷 工人 {} 启动", worker_id);
    
    while let Some(job) = rx.recv().await {
        println!("👷 工人 {} 处理：{:?}", worker_id, job);
        
        // 模拟处理时间
        sleep(Duration::from_millis(200)).await;
        
        let result = JobResult {
            job_id: job.id,
            output: format!("工人 {} 完成了任务 {}", worker_id, job.id),
        };
        
        println!("✅ 工人 {} 完成：{:?}", worker_id, result);
    }
    
    println!("👋 工人 {} 下班", worker_id);
}

#[tokio::main]
async fn main() {
    // 创建通道
    let (tx, rx) = mpsc::channel(10);
    
    // 启动生产者
    let producer_handle = tokio::spawn(producer(tx, 5));
    
    // 启动多个消费者
    let consumer1 = tokio::spawn(consumer(rx, 1));
    let consumer2 = tokio::spawn({
        // 需要创建新的 receiver
        let (tx2, rx2) = mpsc::channel(10);
        // 这里简化处理，实际应该使用更复杂的分发逻辑
        consumer(rx2, 2)
    });
    
    // 等待所有任务完成
    let _ = tokio::join!(producer_handle, consumer1);
    
    println!("🎉 所有工作完成！");
}

use futures::stream::{self, StreamExt};
use tokio::time::{sleep, Duration};

/// 模拟 API 响应
#[derive(Debug)]
struct ApiResponse {
    id: u32,
    data: String,
    status: u16,
}

/// 模拟异步 API 调用
async fn call_api(id: u32) -> Result<ApiResponse, String> {
    // 模拟网络延迟
    sleep(Duration::from_millis(100)).await;
    
    // 模拟可能的失败
    if id == 3 {
        return Err("API 错误".to_string());
    }
    
    Ok(ApiResponse {
        id,
        data: format!("数据 {}", id),
        status: 200,
    })
}

#[tokio::main]
async fn main() {
    let ids = vec![1, 2, 3, 4, 5];
    
    // 创建请求流
    let request_stream = stream::iter(ids);
    
    // 并发执行 API 调用（限制并发数为 2）
    let results = request_stream
        .map(|id| call_api(id))
        .buffered(2);  // 最多同时 2 个请求
    
    // 处理结果
    results
        .for_each(|result| async move {
            match result {
                Ok(response) => {
                    println!("✅ 成功：ID={}, 数据={}", response.id, response.data);
                }
                Err(e) => {
                    println!("❌ 失败：{}", e);
                }
            }
        })
        .await;
}

use tokio::sync::broadcast;
use tokio::time::{interval, Duration};
use futures::stream::StreamExt;

/// 日志条目
#[derive(Clone, Debug)]
struct LogEntry {
    timestamp: u64,
    level: String,
    message: String,
}

/// 日志生成器（模拟服务）
async fn log_generator(tx: broadcast::Sender<LogEntry>) {
    let mut interval = interval(Duration::from_millis(500));
    let mut count = 0;
    
    loop {
        interval.tick().await;
        count += 1;
        
        let level = if count % 5 == 0 {
            "ERROR"
        } else if count % 3 == 0 {
            "WARN"
        } else {
            "INFO"
        };
        
        let entry = LogEntry {
            timestamp: count,
            level: level.to_string(),
            message: format!("日志消息 #{}", count),
        };
        
        // 广播日志（忽略发送失败）
        let _ = tx.send(entry);
    }
}

/// 日志分析器
async fn log_analyzer(mut rx: broadcast::Receiver<LogEntry>) {
    let mut error_count = 0;
    let mut warn_count = 0;
    let mut info_count = 0;
    
    while let Ok(entry) = rx.recv().await {
        match entry.level.as_str() {
            "ERROR" => error_count += 1,
            "WARN" => warn_count += 1,
            "INFO" => info_count += 1,
            _ => {}
        }
        
        println!("📊 [{}] {}: {}", entry.level, entry.timestamp, entry.message);
        
        // 每 10 条日志输出统计
        if (error_count + warn_count + info_count) % 10 == 0 {
            println!("📈 统计：INFO={}, WARN={}, ERROR={}", 
                     info_count, warn_count, error_count);
        }
    }
}

/// 日志过滤器（只输出 ERROR）
async fn error_filter(mut rx: broadcast::Receiver<LogEntry>) {
    println!("🔴 错误过滤器启动");
    
    while let Ok(entry) = rx.recv().await {
        if entry.level == "ERROR" {
            println!("🚨 [ERROR] {}: {}", entry.timestamp, entry.message);
        }
    }
}

#[tokio::main]
async fn main() {
    // 创建广播通道
    let (tx, _rx) = broadcast::channel(100);
    
    // 启动日志生成器
    tokio::spawn(log_generator(tx.clone()));
    
    // 启动多个日志处理器
    tokio::spawn(log_analyzer(tx.subscribe()));
    tokio::spawn(error_filter(tx.subscribe()));
    
    // 运行 10 秒后退出
    tokio::time::sleep(Duration::from_secs(10)).await;
    
    println!("\n🎉 日志收集演示结束");
}

    let stream = stream::iter(vec![1, 2, 3])
        .map(|x| {
            println!("转换：{}", x);
            x * 2
        });
    
    println!("Stream 已创建，但还未执行");
    
    // 只有消费时才会执行
    stream.for_each(|x| async move {
        println!("消费：{}", x);
    }).await;
}

#[tokio::main]
async fn main() {
    // 有界通道：容量满时会阻塞 sender
    let (tx, mut rx) = mpsc::channel(3);
    
    tokio::spawn(async move {
        // 发送 3 个消息后，通道满了
        for i in 0..5 {
            println!("发送：{}", i);
            tx.send(i).await.unwrap();  // 第 4、5 个会等待
            println!("发送完成：{}", i);
        }
    });
    
    // 慢速消费者
    tokio::time::sleep(tokio::time::Duration::from_secs(1)).await;
    
    while let Some(msg) = rx.recv().await {
        println!("接收：{}", msg);
        tokio::time::sleep(tokio::time::Duration::from_millis(500)).await;
    }
}

// ❌ 错误：不能克隆 receiver
async fn wrong_way() {

// ✅ 正确：使用多个通道或 broadcast
async fn right_way() {
    // 方案 1：每个消费者一个通道
    let (tx1, rx1) = mpsc::channel(10);
    let (tx2, rx2) = mpsc::channel(10);
    
    // 方案 2：使用 broadcast
    let (tx, _rx) = broadcast::channel(10);
    let rx1 = tx.subscribe();
    let rx2 = tx.subscribe();
}

    tokio::spawn(async move {
        for i in 0..3 {
            tx.send(i).await.unwrap();
        }
        // tx 被丢弃，通道关闭

今日感悟

🌟 Stream 与通道：异步世界的血脉

今天深入学习了 Stream 和异步通道，这是构建复杂异步系统的基石。

最大的收获：

  1. Stream 的本质：异步版本的 Iterator，用于处理随时间到达的数据流
  2. 通道的选择：
    • mpsc：多对一，任务分发
    • broadcast：一对多，事件广播
    • watch：一对多，状态同步
  3. 背压机制：有界通道可以防止内存溢出
  4. Stream Combinator：map、filter、merge 等操作让流处理更优雅

遇到的困难：

  • 理解 Stream 的惰性求值（和 Iterator 类似但更容易混淆）
  • 多消费者场景下通道的选择（mpsc 不能克隆 receiver）
  • broadcast 通道的 Lagged 错误处理

有趣的发现：

  • buffered(n) 可以控制并发数，非常实用
  • watch 通道的 changed() 方法优雅地处理状态更新
  • Stream 可以组合出复杂的数据处理管道

实际应用场景：

  • 🌐 Web 服务器：使用 mpsc 分发请求到 worker
  • 📡 实时应用：使用 broadcast 推送更新
  • ⚙️ 配置管理：使用 watch 实现热更新
  • 📊 数据处理：使用 Stream 处理数据流

给初学者的建议：

  1. 先掌握 Iterator，再学 Stream（概念相似）
  2. 理解不同通道的适用场景
  3. 使用有界通道避免内存问题
  4. 多用 tokio::spawn 和通道构建并发系统

  • 掌握任务取消模式
  • 实践：构建完整的异步网络服务
