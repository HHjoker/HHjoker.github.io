/// 搜索任务（支持标题和描述）
fn search_tasks(query: &str) -> Result<(), Box<dyn std::error::Error>> {
    let todo_list = load_todo_list()?;
    let query_lower = query.to_lowercase();
    
    let matching_tasks: Vec<&Task> = todo_list.tasks.iter()
        .filter(|t| {
            t.title.to_lowercase().contains(&query_lower) ||
            t.description.as_ref()
                .map(|d| d.to_lowercase().contains(&query_lower))
                .unwrap_or(false)
        })
        .collect();
    
    if matching_tasks.is_empty() {
        println!("🔍 未找到匹配 \"{}\" 的任务", query);
        return Ok(());
    }
    
    println!("🔍 找到 {} 个匹配的任务：\n", matching_tasks.len());
    for task in matching_tasks {
        print_task(task);
    }
    
    Ok(())
}

/// 高级搜索（支持多条件）
#[derive(Debug, Default)]
struct SearchFilter {
    query: Option<String>,
    status: Option<Status>,
    priority: Option<Priority>,
    date_from: Option<String>,
    date_to: Option<String>,
}

fn advanced_search(filter: &SearchFilter) -> Result<(), Box<dyn std::error::Error>> {
    let todo_list = load_todo_list()?;
    
    let results: Vec<&Task> = todo_list.tasks.iter()
        .filter(|t| {
            // 文本搜索
            if let Some(q) = &filter.query {
                let q_lower = q.to_lowercase();
                if !t.title.to_lowercase().contains(&q_lower) &&
                   !t.description.as_ref()
                       .map(|d| d.to_lowercase().contains(&q_lower))
                       .unwrap_or(false) {
                    return false;
                }
            }
            
            // 状态过滤
            if let Some(status) = &filter.status {
                if &t.status != status {
                    return false;
                }
            }
            
            // 优先级过滤
            if let Some(priority) = &filter.priority {
                if &t.priority != priority {
                    return false;
                }
            }
            
            // 日期范围过滤
            if let Some(from) = &filter.date_from {
                if t.created_at < from {
                    return false;
                }
            }
            if let Some(to) = &filter.date_to {
                if t.created_at > to {
                    return false;
                }
            }
            
            true
        })
        .collect();
    
    println!("📊 找到 {} 个匹配的任务：\n", results.len());
    for task in results {
        print_task(task);
    }
    
    Ok(())
}

use std::cell::RefCell;
use std::collections::HashMap;

/// 带缓存的待办列表管理器
struct TodoManager {
    cache: RefCell<HashMap<String, TodoList>>,
    cache_hits: RefCell<u32>,
    cache_misses: RefCell<u32>,
}

impl TodoManager {
    fn new() -> Self {
        TodoManager {
            cache: RefCell::new(HashMap::new()),
            cache_hits: RefCell::new(0),
            cache_misses: RefCell::new(0),
        }
    }
    
    /// 带缓存的加载
    fn load_with_cache(&self, key: &str) -> Result<TodoList, Box<dyn std::error::Error>> {
        // 尝试从缓存获取
        if let Some(cached) = self.cache.borrow().get(key) {
            *self.cache_hits.borrow_mut() += 1;
            return Ok(cached.clone());
        }
        
        // 缓存未命中，从文件加载
        *self.cache_misses.borrow_mut() += 1;
        let todo_list = load_todo_list()?;
        
        // 写入缓存
        self.cache.borrow_mut().insert(key.to_string(), todo_list.clone());
        
        Ok(todo_list)
    }
    
    /// 使缓存失效
    fn invalidate_cache(&self) {
        self.cache.borrow_mut().clear();
    }
    
    /// 打印缓存统计
    fn print_cache_stats(&self) {
        let hits = *self.cache_hits.borrow();
        let misses = *self.cache_misses.borrow();
        let total = hits + misses;
        let hit_rate = if total > 0 {
            (hits as f64 / total as f64) * 100.0
        } else {
            0.0
        };
        println!("📈 缓存统计：命中 {} 次，未命中 {} 次，命中率 {:.1}%", 
                 hits, misses, hit_rate);
    }
}

/// 批量添加任务（减少文件 I/O 次数）
fn batch_add_tasks(tasks: Vec<NewTask>) -> Result<(), Box<dyn std::error::Error>> {
    let mut todo_list = load_todo_list()?;
    
    let start_id = todo_list.next_id;
    for (i, task) in tasks.into_iter().enumerate() {
        let new_task = Task {
            id: start_id + i as u32,
            title: task.title,
            description: task.description,
            status: Status::Pending,
            priority: task.priority.unwrap_or_default(),
            created_at: chrono::Local::now().format("%Y-%m-%d %H:%M:%S").to_string(),
            updated_at: chrono::Local::now().format("%Y-%m-%d %H:%M:%S").to_string(),
        };
        todo_list.tasks.push(new_task);
    }
    
    todo_list.next_id = start_id + tasks.len() as u32;
    save_todo_list(&todo_list)?;
    
    println!("✅ 批量添加 {} 个任务成功", tasks.len());
    Ok(())
}

/// 批量完成任务
fn batch_complete_tasks(ids: Vec<u32>) -> Result<(), Box<dyn std::error::Error>> {
    let mut todo_list = load_todo_list()?;
    let mut completed = 0;
    
    for task in todo_list.tasks.iter_mut() {
        if ids.contains(&task.id) && task.status == Status::Pending {
            task.status = Status::Completed;
            task.updated_at = chrono::Local::now().format("%Y-%m-%d %H:%M:%S").to_string();
            completed += 1;
        }
    }
    
    save_todo_list(&todo_list)?;
    println!("✅ 批量完成 {} 个任务", completed);
    Ok(())
}

/// 友好的错误类型
#[derive(Debug)]
enum TodoError {
    FileNotFound(String),
    ParseError(String),
    TaskNotFound(u32),
    InvalidInput(String),
    IoError(std::io::Error),
}

impl std::fmt::Display for TodoError {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        match self {
            TodoError::FileNotFound(path) => {
                write!(f, "❌ 文件未找到：{}", path)
            }
            TodoError::ParseError(msg) => {
                write!(f, "❌ 解析错误：{}", msg)
            }
            TodoError::TaskNotFound(id) => {
                write!(f, "❌ 任务 #{} 不存在", id)
            }
            TodoError::InvalidInput(msg) => {
                write!(f, "❌ 无效输入：{}", msg)
            }
            TodoError::IoError(e) => {
                write!(f, "❌ I/O 错误：{}", e)
            }
        }
    }
}

impl std::error::Error for TodoError {}

impl From<std::io::Error> for TodoError {
    fn from(err: std::io::Error) -> Self {
        TodoError::IoError(err)
    }
}

/// 主函数中的错误处理
fn run() -> Result<(), TodoError> {
    let args: Vec<String> = env::args().collect();
    
    if args.len() < 2 {
        print_help();
        return Ok(());
    }
    
    match args[1].as_str() {
        "add" => cmd_add(&args[2..]).map_err(|e| TodoError::InvalidInput(e.to_string()))?,
        "list" => cmd_list(&args[2..])?,
        "done" => cmd_done(&args[2..])?,
        "search" => cmd_search(&args[2..])?,
        "help" | "--help" | "-h" => print_help(),
        _ => {
            return Err(TodoError::InvalidInput(
                format!("未知命令：{}，使用 'todo help' 查看帮助", args[1])
            ));
        }
    }
    
    Ok(())
}

fn main() {
    if let Err(e) = run() {
        eprintln!("{}", e);
        std::process::exit(1);
    }
}

fn run() -> Result<()> {
    let args: Vec<String> = env::args().collect();
    
    if args.len() < 2 {
        commands::print_help();
        return Ok(());
    }
    
    match args[1].as_str() {
        "add" => commands::add::execute(&args[2..]),
        "list" => commands::list::execute(&args[2..]),
        "done" => commands::done::execute(&args[2..]),
        "edit" => commands::edit::execute(&args[2..]),
        "delete" => commands::delete::execute(&args[2..]),
        "search" => commands::search::execute(&args[2..]),
        "backup" => storage::backup::create_backup(),
        "restore" => storage::backup::restore_backup(),
        "stats" => commands::stats::execute(),
        "help" | "--help" | "-h" => {
            commands::print_help();
            Ok(())
        }
        "version" | "--version" | "-V" => {
            println!("Todo CLI v1.0.0");
            Ok(())
        }
        _ => Err(TodoError::InvalidInput(
            format!("未知命令：{}，使用 'todo help' 查看帮助", args[1])
        )),
    }
}

今日感悟

🌟 16 天的 Rust 学习之旅

从第一天对所有权概念的困惑，到今天能够独立完成一个完整的 CLI 项目，这段旅程充满了挑战与收获。

最大的收获：

  1. 所有权思维：从一开始的不适应，到现在能够自然地思考数据的归属和生命周期
  2. 类型系统：Rust 强大的类型系统在编译期就捕获了大量潜在错误
  3. 错误处理：Result 和 Option 让错误处理变得优雅且安全
  4. 模块化设计：学会了如何组织 Rust 项目结构

遇到的困难：

  • 借用检查器的"严格"要求（但现在理解了它的价值）
  • 生命周期的概念（还需要更多实践）
  • 异步编程的学习曲线（下一步的目标）

给初学者的建议：

  1. 不要害怕编译器错误，它是你的朋友
  2. 多写代码，实践是最好的老师
  3. 阅读优秀项目的源码
  4. 参与社区，提问和分享

下一步计划：

  • 深入学习 async/await
  • 探索 Web 开发（axum/actix-web）
  • 学习 Rust 宏系统
  • 参与开源项目

Rust 不仅是一门语言，更是一种思维方式。它教会我们如何在追求性能的同时，保证内存安全和并发安全。这段学习之旅只是一个开始，期待在未来的项目中继续探索 Rust 的可能性！

16 天，从 0 到 1，感谢坚持的自己。🦀
