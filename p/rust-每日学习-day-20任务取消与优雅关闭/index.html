   - 用户点击"取消"按钮

   - 释放不再需要的资源
   - 避免资源泄漏

   - 某个任务失败，取消相关任务
   - 超时后尝试备用方案
   - 竞态条件处理

4. 优雅关闭
   - 服务器接收 SIGTERM
   - 完成当前请求后退出
   - 清理状态后终止

/*
Rust 中的任务取消不是强制的：

1. 没有内置的"杀死"任务机制
   - 不能强行终止任务（不安全）
   - 任务必须自己检查取消信号
   - 协作式取消（cooperative cancellation）

2. 资源清理
   - 任务被取消时必须释放资源
   - Drop 保证会被调用
   - 但需要显式处理清理逻辑

3. 部分完成的问题
   - 任务可能已经修改了共享状态
   - 需要回滚或补偿
   - 事务性操作的挑战
*/

use tokio::time::{sleep, Duration};

#[tokio::main]
async fn main() {
    let slow = sleep(Duration::from_millis(500));
    let fast = sleep(Duration::from_millis(100));
    
    tokio::select! {
        _ = slow => {
            println!("慢操作完成");
        }
        _ = fast => {
            println!("快操作完成");
        }
    }
    
    println!("select! 结束");
}

输出：

1
2

快操作完成
select! 结束

select! 的工作原理：

  • 🏁 同时轮询所有分支
  • 🏆 第一个准备好的分支执行
  • ❌ 其他分支被取消（不再轮询）
  • 🧹 被取消分支的资源通过 Drop 清理

use tokio::time::{sleep, Duration};
use tokio::sync::oneshot;

#[tokio::main]
async fn main() {
    let (tx, rx) = oneshot::channel::<String>();
    
    tokio::select! {
        // 分支 1：超时
        _ = sleep(Duration::from_secs(2)) => {
            println!("⏰ 超时！");
        }
        
        // 分支 2：接收消息
        msg = rx => {
            match msg {
                Ok(m) => println!("📨 收到消息：{}", m),
                Err(_) => println!("❌ 发送器已丢弃"),
            }
        }
        
        // 分支 3：直接就绪的值
        _ = async { 42 } => {
            println!("🔢 直接值");
        }
        
        // 分支 4：带条件的分支
        if false => {
            println!("这个分支永远不会执行");
        }
        
        // 分支 5：带绑定的模式
        result = async { Ok::<_, ()>("成功") } => {
            println!("✅ 结果：{:?}", result);
        }
    }
}

use tokio::time::{timeout, Duration, sleep};
use tokio::sync::oneshot;

/// 方式 1：使用 timeout 函数（推荐）
async fn with_timeout_function() {
    let operation = async {
        sleep(Duration::from_millis(500)).await;
        "操作完成"
    };
    
    match timeout(Duration::from_millis(200), operation).await {
        Ok(result) => println!("✅ 操作成功：{}", result),
        Err(_) => println!("⏰ 操作超时"),
    }
}

/// 方式 2：使用 select! 手动实现
async fn with_select_timeout() {
    let operation = async {
        sleep(Duration::from_millis(500)).await;
        "操作完成"
    };
    
    let timeout_future = sleep(Duration::from_millis(200));
    
    tokio::select! {
        result = operation => {
            println!("✅ 操作成功：{}", result);
        }
        _ = timeout_future => {
            println!("⏰ 操作超时");
        }
    }
}

/// 方式 3：带超时的网络请求模拟
async fn fetch_with_timeout(url: &str, timeout_ms: u64) -> Result<String, &'static str> {
    let fetch = async {
        // 模拟网络请求
        sleep(Duration::from_millis(100)).await;
        Ok(format!("从 {} 获取的数据", url))
    };
    
    tokio::select! {
        result = fetch => {
            result
        }
        _ = sleep(Duration::from_millis(timeout_ms)) => {
            Err("请求超时")
        }
    }
}

#[tokio::main]
async fn main() {
    println!("=== 方式 1：timeout 函数 ===");
    with_timeout_function().await;
    
    println!("\n=== 方式 2：select! 手动超时 ===");
    with_select_timeout().await;
    
    println!("\n=== 方式 3：带超时的请求 ===");
    match fetch_with_timeout("https://example.com", 500).await {
        Ok(data) => println!("📄 {}", data),
        Err(e) => println!("❌ 错误：{}", e),
    }
}

use tokio::time::{interval, Duration, sleep};
use tokio::sync::oneshot;

#[tokio::main]
async fn main() {
    let (tx, rx) = oneshot::channel::<()>();
    
    let mut ticker = interval(Duration::from_millis(500));
    
    println!("🔄 开始循环，每 500ms 打印一次...");
    println!("📥 发送信号将停止循环");
    
    // 在另一个任务中发送停止信号
    let stop_handle = tokio::spawn(async move {
        sleep(Duration::from_millis(2500)).await;
        println!("\n🛑 发送停止信号...");
        let _ = tx.send(());
    });
    
    // 主循环
    loop {
        tokio::select! {
            // 定时器触发
            _ = ticker.tick() => {
                println!("⏰ 滴答...");
            }
            
            // 停止信号
            _ = &rx => {
                println!("\n✅ 收到停止信号，退出循环");
                break;
            }
        }
    }
    
    let _ = stop_handle.await;
}

use tokio::sync::watch;
use tokio::time::{sleep, Duration};
use std::sync::Arc;

/// 取消令牌
#[derive(Clone)]
struct CancellationToken {
    receiver: watch::Receiver<bool>,
}

/// 取消令牌源
struct CancellationTokenSource {
    sender: watch::Sender<bool>,
}

impl CancellationTokenSource {
    fn new() -> Self {
        let (sender, receiver) = watch::channel(false);
        CancellationTokenSource { sender }
    }
    
    /// 获取令牌
    fn token(&self) -> CancellationToken {
        CancellationToken {
            receiver: self.sender.subscribe(),
        }
    }
    
    /// 触发取消
    fn cancel(&self) {
        let _ = self.sender.send(true);
    }
    
    /// 检查是否已取消
    fn is_cancelled(&self) -> bool {
        *self.sender.borrow()
    }
}

impl CancellationToken {
    /// 检查是否已取消
    fn is_cancelled(&self) -> bool {
        *self.receiver.borrow()
    }
    
    /// 等待取消信号
    async fn cancelled(&mut self) {
        // 如果已经取消，立即返回
        if *self.receiver.borrow() {
            return;
        }
        
        // 等待取消信号
        let _ = self.receiver.changed().await;
    }
    
    /// 等待操作完成或被取消
    async fn run_until_cancelled<F, T>(&mut self, future: F) -> Option<T>
    where
        F: std::future::Future<Output = T>,
    {
        tokio::select! {
            _ = self.cancelled() => {
                None  // 被取消
            }
            result = future => {
                Some(result)  // 操作完成
            }
        }
    }
}

/// 示例：可取消的长时间任务
async fn long_running_task(mut token: CancellationToken, task_id: usize) {
    for i in 0..10 {
        // 检查取消
        if token.is_cancelled() {
            println!("❌ 任务 {} 检测到取消，提前退出", task_id);
            return;
        }
        
        println!("🔄 任务 {} 进度：{}/10", task_id, i + 1);
        
        // 使用 run_until_cancelled 包装异步操作
        match token.run_until_cancelled(sleep(Duration::from_millis(300))).await {
            Some(_) => {}  // 正常完成
            None => {
                println!("🛑 任务 {} 被取消", task_id);
                return;
            }
        }
    }
    
    println!("✅ 任务 {} 完成", task_id);
}

#[tokio::main]
async fn main() {
    println!("🚀 启动可取消任务演示\n");
    
    let source = CancellationTokenSource::new();
    let token = source.token();
    
    // 启动任务
    let task_handle = tokio::spawn(long_running_task(token, 1));
    
    // 3 秒后取消
    sleep(Duration::from_millis(1500)).await;
    println!("\n🛑 触发取消...\n");
    source.cancel();
    
    // 等待任务结束
    let _ = task_handle.await;
    
    println!("\n🎉 演示结束");
}

use tokio::net::TcpListener;
use tokio::sync::{watch, broadcast};
use tokio::time::{timeout, Duration};
use std::sync::Arc;

/// 服务器状态
#[derive(Clone, Copy, PartialEq)]
enum ServerState {
    Running,
    ShuttingDown,
    Stopped,
}

/// 服务器
struct Server {
    state: watch::Sender<ServerState>,
    shutdown_complete: broadcast::Sender<()>,
}

impl Server {
    fn new() -> Self {
        let (state_tx, _) = watch::channel(ServerState::Running);
        let (shutdown_tx, _) = broadcast::channel(16);
        
        Server {
            state: state_tx,
            shutdown_complete: shutdown_tx,
        }
    }
    
    /// 获取关闭令牌
    fn shutdown_token(&self) -> ShutdownToken {
        ShutdownToken {
            state_rx: self.state.subscribe(),
            shutdown_rx: self.shutdown_complete.subscribe(),
        }
    }
    
    /// 启动服务器
    async fn run(&self, addr: &str) -> Result<(), Box<dyn std::error::Error>> {
        let listener = TcpListener::bind(addr).await?;
        println!("🚀 服务器启动在 {}", addr);
        
        let mut state_rx = self.state.subscribe();
        
        loop {
            tokio::select! {
                // 接受新连接
                result = listener.accept() => {
                    match result {
                        Ok((socket, addr)) => {
                            println!("📡 新连接：{}", addr);
                            
                            // 克隆关闭令牌给处理程序
                            let token = self.shutdown_token();
                            
                            // 在新任务中处理连接
                            tokio::spawn(async move {
                                handle_connection(socket, token).await;
                            });
                        }
                        Err(e) => {
                            eprintln!("❌ 接受连接失败：{}", e);
                        }
                    }
                }
                
                // 关闭信号
                _ = state_rx.changed() => {
                    if *state_rx.borrow() == ServerState::ShuttingDown {
                        println!("🛑 收到关闭信号，停止接受新连接");
                        break;
                    }
                }
            }
        }
        
        // 等待所有连接处理完成
        println!("⏳ 等待现有连接完成...");
        timeout(Duration::from_secs(5), self.shutdown_complete.recv()).await.ok();
        
        *self.state.borrow_mut() = ServerState::Stopped;
        println!("✅ 服务器已停止");
        
        Ok(())
    }
    
    /// 请求关闭
    fn shutdown(&self) {
        *self.state.borrow_mut() = ServerState::ShuttingDown;
        let _ = self.state.send(ServerState::ShuttingDown);
    }
}

/// 关闭令牌
struct ShutdownToken {
    state_rx: watch::Receiver<ServerState>,
    shutdown_rx: broadcast::Receiver<()>,
}

impl ShutdownToken {
    fn is_shutting_down(&self) -> bool {
        *self.state_rx.borrow() != ServerState::Running
    }
    
    async fn wait_for_shutdown(&mut self) {
        let _ = self.shutdown_rx.recv().await;
    }
}

/// 处理连接
async fn handle_connection(
    socket: tokio::net::TcpStream,
    mut token: ShutdownToken,
) {
    println!("🔌 开始处理连接");
    
    // 模拟处理
    for i in 0..5 {
        if token.is_shutting_down() {
            println!("🛑 连接处理被中断");
            return;
        }
        
        println!("📝 处理数据块 {}/5", i + 1);
        tokio::time::sleep(Duration::from_millis(500)).await;
    }
    
    println!("✅ 连接处理完成");
    
    // 通知完成
    let _ = socket;  // 实际代码中会发送信号
}

#[tokio::main]
async fn main() -> Result<(), Box<dyn std::error::Error>> {
    let server = Arc::new(Server::new());
    let server_clone = Arc::clone(&server);
    
    // 在后台运行服务器
    let server_handle = tokio::spawn(async move {
        let _ = server_clone.run("127.0.0.1:0").await;
    });
    
    // 模拟运行 3 秒
    tokio::time::sleep(Duration::from_secs(3)).await;
    
    // 请求关闭
    println!("\n📴 请求服务器关闭...\n");
    server.shutdown();
    
    // 等待服务器停止
    let _ = server_handle.await;
    
    Ok(())
}

use tokio::time::{sleep, Duration, Instant};
use tokio::sync::watch;
use std::sync::Arc;

/// 下载进度
#[derive(Clone, Debug)]
struct DownloadProgress {
    downloaded: u64,
    total: u64,
    speed: f64,  // bytes per second
}

impl DownloadProgress {
    fn percent(&self) -> f64 {
        if self.total == 0 {
            0.0
        } else {
            (self.downloaded as f64 / self.total as f64) * 100.0
        }
    }
}

/// 下载器
struct Downloader {
    progress_tx: watch::Sender<DownloadProgress>,
}

impl Downloader {
    fn new(total: u64) -> Self {
        let (progress_tx, _) = watch::channel(DownloadProgress {
            downloaded: 0,
            total,
            speed: 0.0,
        });
        
        Downloader { progress_tx }
    }
    
    /// 获取进度订阅
    fn progress(&self) -> watch::Receiver<DownloadProgress> {
        self.progress_tx.subscribe()
    }
    
    /// 执行下载（可取消）
    async fn download(
        &self,
        url: &str,
        mut cancel_rx: watch::Receiver<bool>,
    ) -> Result<DownloadProgress, &'static str> {
        println!("📥 开始下载：{}", url);
        
        let total = 1000000u64;  // 1MB
        let chunk_size = 100000u64;  // 100KB per chunk
        let mut downloaded = 0u64;
        let start = Instant::now();
        
        loop {
            // 检查取消
            if *cancel_rx.borrow() {
                println!("\n🛑 下载被取消");
                return Err("cancelled");
            }
            
            // 模拟下载一个块
            sleep(Duration::from_millis(200)).await;
            downloaded = (downloaded + chunk_size).min(total);
            
            let elapsed = start.elapsed().as_secs_f64();
            let speed = if elapsed > 0.0 {
                downloaded as f64 / elapsed
            } else {
                0.0
            };
            
            let progress = DownloadProgress {
                downloaded,
                total,
                speed,
            };
            
            // 更新进度
            let _ = self.progress_tx.send(progress);
            
            println!(
                "\r📊 进度：{:.1}% ({:.2} KB/s) ",
                progress.percent(),
                progress.speed / 1024.0
            );
            
            // 检查是否完成
            if downloaded >= total {
                println!("\n✅ 下载完成！");
                return Ok(progress);
            }
        }
    }
}

/// 进度显示器
async fn display_progress(mut progress_rx: watch::Receiver<DownloadProgress>) {
    loop {
        match progress_rx.changed().await {
            Ok(_) => {
                let p = progress_rx.borrow();
                print!(
                    "\r📊 进度：{:5.1}% | 已下载：{:6} KB | 速度：{:6.1} KB/s",
                    p.percent(),
                    p.downloaded / 1024,
                    p.speed / 1024.0
                );
            }
            Err(_) => break,  // 发送器已关闭
        }
    }
    println!();
}

#[tokio::main]
async fn main() {
    println!("🚀 可取消下载演示\n");
    
    let downloader = Downloader::new(1000000);
    let progress_rx = downloader.progress();
    
    // 创建取消通道
    let (cancel_tx, cancel_rx) = watch::channel(false);
    
    // 启动进度显示
    let display_handle = tokio::spawn(display_progress(progress_rx));
    
    // 启动下载
    let download_handle = tokio::spawn({
        let downloader = &downloader;
        async move {
            downloader.download("https://example.com/file.zip", cancel_rx).await
        }
    });
    
    // 模拟：下载 1.5 秒后取消
    sleep(Duration::from_millis(1500)).await;
    println!("\n\n📴 用户点击取消...\n");
    let _ = cancel_tx.send(true);
    
    // 等待下载结束
    match download_handle.await.unwrap() {
        Ok(progress) => {
            println!("✅ 下载完成：{}/{} bytes", progress.downloaded, progress.total);
        }
        Err(e) => {
            println!("❌ 下载失败：{}", e);
        }
    }
    
    let _ = display_handle.await;
}

use tokio::fs::File;
use tokio::io::AsyncWriteExt;

async fn unsafe_write() {
    let mut file = File::create("data.txt").await.unwrap();
    
    file.write_all(b"Hello").await.unwrap();
    // ❌ 如果在这里被取消，文件只写了一半！
    
    file.write_all(b" World").await.unwrap();
    file.flush().await.unwrap();
}

async fn safe_write() {
    let mut file = File::create("data.txt").await.unwrap();
    
    // ✅ 使用临时文件 + 原子重命名
    let temp_path = "data.txt.tmp";
    {
        let mut temp = File::create(temp_path).await.unwrap();
        temp.write_all(b"Hello World").await.unwrap();
        temp.flush().await.unwrap();
        // temp 在这里被 drop，确保写入完成
    }
    
    // 原子重命名（要么全有，要么全无）
    std::fs::rename(temp_path, "data.txt").unwrap();
}

use tokio::sync::watch;

/// 清理守卫
struct CleanupGuard<F: FnOnce()> {
    cleanup: Option<F>,
}

impl<F: FnOnce()> CleanupGuard<F> {
    fn new(cleanup: F) -> Self {
        CleanupGuard {
            cleanup: Some(cleanup),
        }
    }
    
    /// 解除清理（操作成功完成）
    fn disarm(mut self) {
        self.cleanup = None;
    }
}

impl<F: FnOnce()> Drop for CleanupGuard<F> {
    fn drop(&mut self) {
        if let Some(cleanup) = self.cleanup.take() {
            cleanup();
        }
    }
}

async fn operation_with_cleanup() {
    println!("🔧 开始操作...");
    
    let _guard = CleanupGuard::new(|| {
        println!("🧹 清理资源...");
    });
    
    // 模拟操作
    tokio::time::sleep(tokio::time::Duration::from_millis(100)).await;
    
    // 如果成功，解除清理守卫
    _guard.disarm();
    println!("✅ 操作完成，无需清理");
}

#[tokio::main]
async fn main() {
    // 正常完成
    operation_with_cleanup().await;
    
    // 被取消的情况（清理会执行）
    tokio::select! {
        _ = operation_with_cleanup() => {}
        _ = tokio::time::sleep(tokio::time::Duration::from_millis(50)) => {
            println!("🛑 操作被取消，清理守卫触发");
        }
    }
}

/*
优雅关闭的关键点：

1. 信号传播
   ┌─────────────┐
   │  关闭信号   │
   │  (SIGTERM)  │
   └──────┬──────┘
          │
   ┌──────▼──────┐
   │  主服务器   │──▶ 停止接受新连接
   └──────┬──────┘
          │
   ┌──────▼──────┐
   │  现有连接   │──▶ 完成当前请求
   └──────┬──────┘
          │
   ┌──────▼──────┐
   │  清理资源   │──▶ 关闭文件、数据库
   └──────┬──────┘
          │
   ┌──────▼──────┐
   │    退出     │
   └─────────────┘

2. 超时保护
   - 给现有连接合理的完成时间
   - 超时后强制关闭
   - 避免无限等待

3. 状态通知
   - 使用 watch/broadcast 通道
   - 所有任务监听同一信号
   - 协调一致的关闭
*/
