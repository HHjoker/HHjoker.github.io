/// 过滤任务列表
fn filter_tasks(todo_list: &TodoList, status: Option<&Status>) -> Vec<&Task> {
    match status {
        Some(Status::Pending) => {
            todo_list.tasks.iter().filter(|t| t.status == Status::Pending).collect()
        }
        Some(Status::Completed) => {
            todo_list.tasks.iter().filter(|t| t.status == Status::Completed).collect()
        }
        None => {
            todo_list.tasks.iter().collect()
        }
    }
}

/// 列出任务（支持过滤）
fn list_tasks(filter: Option<&str>) -> Result<(), Box<dyn std::error::Error>> {
    let todo_list = load_todo_list()?;
    
    let status_filter = match filter {
        Some("pending") | Some("p") => Some(Status::Pending),
        Some("completed") | Some("c") => Some(Status::Completed),
        _ => None,
    };
    
    let tasks = filter_tasks(&todo_list, status_filter.as_ref());
    
    if tasks.is_empty() {
        println!("📭 没有找到任务");
        return Ok(());
    }
    
    println!("📋 待办事项清单：\n");
    for task in tasks {
        let status_icon = match task.status {
            Status::Pending => "⬜",
            Status::Completed => "✅",
        };
        println!("{} [{}] {}", status_icon, task.id, task.title);
        if let Some(desc) = &task.description {
            println!("   └─ {}", desc);
        }
        println!();
    }
    
    Ok(())
}

关键概念理解：

  1. Option<&Status>：使用 Option 表示可选的过滤条件
  2. iter().filter()：迭代器适配器，过滤符合条件的元素
  3. collect()：将迭代器结果收集到 Vec 中
  4. 模式匹配：使用 match 处理不同的过滤选项

/// 搜索任务
fn search_tasks(query: &str) -> Result<(), Box<dyn std::error::Error>> {
    let todo_list = load_todo_list()?;
    let query_lower = query.to_lowercase();
    
    let matching_tasks: Vec<&Task> = todo_list.tasks.iter()
        .filter(|task| {
            task.title.to_lowercase().contains(&query_lower) ||
            task.description.as_ref()
                .map(|d| d.to_lowercase().contains(&query_lower))
                .unwrap_or(false)
        })
        .collect();
    
    if matching_tasks.is_empty() {
        println!("🔍 没有找到匹配 \"{}\" 的任务", query);
        return Ok(());
    }
    
    println!("🔍 搜索结果（\"{}\"）：\n", query);
    for task in matching_tasks {
        let status_icon = match task.status {
            Status::Pending => "⬜",
            Status::Completed => "✅",
        };
        println!("{} [{}] {}", status_icon, task.id, task.title);
    }
    
    Ok(())
}

技术要点：

  1. to_lowercase()：转换为小写实现不区分大小写搜索
  2. contains()：检查字符串是否包含子串
  3. Option::map()：对 Option 中的值进行转换
  4. unwrap_or()：提供默认值

/// 显示任务统计
fn show_stats() -> Result<(), Box<dyn std::error::Error>> {
    let todo_list = load_todo_list()?;
    
    let total = todo_list.tasks.len();
    let completed = todo_list.tasks.iter()
        .filter(|t| t.status == Status::Completed)
        .count();
    let pending = total - completed;
    
    // 计算完成率
    let completion_rate = if total > 0 {
        (completed as f64 / total as f64) * 100.0
    } else {
        0.0
    };
    
    println!("📊 任务统计");
    println!("═══════════════════════");
    println!("总任务数：    {}", total);
    println!("未完成：      {}", pending);
    println!("已完成：      {}", completed);
    println!("完成率：      {:.1}%", completion_rate);
    println!("═══════════════════════");
    
    // 显示进度条
    if total > 0 {
        let bar_width = 30;
        let filled = ((completed as f64 / total as f64) * bar_width as f64) as usize;
        let empty = bar_width - filled;
        
        print!("进度：        [");
        print!("{}", "█".repeat(filled));
        print!("{}", "░".repeat(empty));
        println!("]");
    }
    
    Ok(())
}

use std::env;

fn main() {
    let args: Vec<String> = env::args().collect();
    
    if args.len() < 2 {
        print_help();
        return;
    }
    
    let command = &args[1];
    
    let result = match command.as_str() {
        "add" => {
            if args.len() < 3 {
                eprintln!("❌ 错误：请提供任务标题");
                eprintln!("用法：todo_cli add <标题> [描述]");
                return;
            }
            let title = &args[2];
            let description = args.get(3).map(|s| s.as_str());
            add_task(title, description)
        }
        "list" | "ls" => {
            let filter = args.get(2).map(|s| s.as_str());
            list_tasks(filter)
        }
        "complete" | "done" => {
            if args.len() < 3 {
                eprintln!("❌ 错误：请提供任务 ID");
                eprintln!("用法：todo_cli complete <ID>");
                return;
            }
            match args[2].parse::<u32>() {
                Ok(id) => complete_task(id),
                Err(_) => {
                    eprintln!("❌ 错误：无效的 ID");
                    return;
                }
            }
        }
        "delete" | "rm" => {
            if args.len() < 3 {
                eprintln!("❌ 错误：请提供任务 ID");
                eprintln!("用法：todo_cli delete <ID>");
                return;
            }
            match args[2].parse::<u32>() {
                Ok(id) => delete_task(id),
                Err(_) => {
                    eprintln!("❌ 错误：无效的 ID");
                    return;
                }
            }
        }
        "search" | "find" => {
            if args.len() < 3 {
                eprintln!("❌ 错误：请提供搜索关键词");
                eprintln!("用法：todo_cli search <关键词>");
                return;
            }
            search_tasks(&args[2])
        }
        "stats" => show_stats(),
        "help" | "--help" | "-h" => {
            print_help();
            Ok(())
        }
        _ => {
            eprintln!("❌ 未知命令：{}", command);
            print_help();
            return;
        }
    };
    
    if let Err(e) = result {
        eprintln!("❌ 发生错误：{}", e);
    }
}

关键概念理解

1. 迭代器适配器

Rust 的迭代器是惰性的，可以链式调用：

1
2
3
4
5

// 链式调用示例
let result = tasks.iter()
    .filter(|t| t.status == Status::Pending)  // 过滤
    .map(|t| &t.title)                         // 转换
    .collect::<Vec<_>>();                      // 收集

常用适配器：

  • filter()：过滤元素
  • map()：转换元素
  • collect()：收集到集合
  • count()：计数
  • find()：查找第一个匹配项

这种写法不仅简洁，而且类型安全，编译器会帮我们检查每一步的正确性。

遇到的挑战是处理 Option 类型的链式调用。一开始总是忘记 as_ref()，导致所有权问题。后来理解了：

  • as_ref() 将 Option<T> 转为 Option<&T>
  • 这样可以在不移动值的情况下操作内部数据
