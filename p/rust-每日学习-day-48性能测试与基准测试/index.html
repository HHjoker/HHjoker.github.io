use criterion::{black_box, criterion_group, criterion_main, Criterion};

fn build_string_v1(n: usize) -> String {
    let mut s = String::new();
    for i in 0..n {
        s.push_str(&format!("{}", i));
    }
    s
}

fn build_string_v2(n: usize) -> String {
    let mut s = String::with_capacity(n * 3);
    for i in 0..n {
        s.push_str(&format!("{}", i));
    }
    s
}

fn benchmark_build_string(c: &mut Criterion) {
    let mut group = c.benchmark_group("Build String");
    
    group.bench_function("v1_no_capacity", |b| {
        b.iter(|| build_string_v1(black_box(1000)))
    });
    
    group.bench_function("v2_with_capacity", |b| {
        b.iter(|| build_string_v2(black_box(1000)))
    });
    
    group.finish();
}

criterion_group!(benches, benchmark_build_string);
criterion_main!(benches);

use criterion::{BenchmarkId, Criterion};

fn fibonacci(n: u64) -> u64 {
    match n {
        0 => 0,
        1 => 1,
        _ => fibonacci(n - 1) + fibonacci(n - 2),
    }
}

fn fibonacci_iterative(n: u64) -> u64 {
    let mut a = 0;
    let mut b = 1;
    for _ in 0..n {
        let temp = a;
        a = b;
        b = temp + b;
    }
    a
}

fn benchmark_fibonacci(c: &mut Criterion) {
    let mut group = c.benchmark_group("Fibonacci");
    
    for i in [10, 20, 30].iter() {
        group.bench_with_input(
            BenchmarkId::new("recursive", i),
            i,
            |b, &i| b.iter(|| fibonacci(i)),
        );
        
        group.bench_with_input(
            BenchmarkId::new("iterative", i),
            i,
            |b, &i| b.iter(|| fibonacci_iterative(i)),
        );
    }
    
    group.finish();
}

fn sum_vec_v1(vec: &[i32]) -> i32 {
    let mut sum = 0;
    for &x in vec {
        sum += x;
    }
    sum
}

fn sum_vec_v2(vec: &[i32]) -> i32 {
    vec.iter().sum()
}

fn sum_vec_v3(vec: &[i32]) -> i32 {
    vec.par_iter().sum()  // 需要 rayon
}

fn benchmark_sum(c: &mut Criterion) {
    let vec: Vec<i32> = (0..10000).collect();
    
    let mut group = c.benchmark_group("Sum Vector");
    group.throughput(criterion::Throughput::Elements(vec.len() as u64));
    
    group.bench_function("for_loop", |b| b.iter(|| sum_vec_v1(black_box(&vec))));
    group.bench_function("iter_sum", |b| b.iter(|| sum_vec_v2(black_box(&vec))));
    group.bench_function("parallel", |b| b.iter(|| sum_vec_v3(black_box(&vec))));
    
    group.finish();
}

use criterion::{Bencher, BenchmarkId, Criterion, Throughput};

fn benchmark_with_setup(c: &mut Criterion) {
    let mut group = c.benchmark_group("Database Query");
    
    group.setup(|_| {
        // 设置代码，每个基准测试前运行
        println!("Setting up database connection...");
    });
    
    group.bench_function(BenchmarkId::from_parameter("query"), |b| {
        b.iter_custom(|iters| {
            let start = std::time::Instant::now();
            for _ in 0..iters {
                // 被测量的代码
                std::hint::black_box(42);
            }
            start.elapsed()
        })
    });
    
    group.finish();
}

fn process_text(text: &str) -> String {
    let mut result = String::new();
    
    for line in text.lines() {
        let trimmed = line.trim();
        if !trimmed.is_empty() {
            let upper = trimmed.to_uppercase();
            result.push_str(&upper);
            result.push('\n');
        }
    }
    
    result
}

fn process_text_v3(text: &str) -> String {
    text.lines()
        .map(|line| line.trim())
        .filter(|line| !line.is_empty())
        .map(|line| line.to_uppercase())
        .collect::<Vec<_>>()
        .join("\n")
}

fn process_text_v4(text: &str) -> String {
    let mut result = String::with_capacity(text.len());
    
    for line in text.lines() {
        let trimmed = line.trim();
        if !trimmed.is_empty() {
            for c in trimmed.chars() {
                result.push(c.to_ascii_uppercase());
            }
            result.push('\n');
        }
    }
    
    result
}

fn benchmark_all_versions(c: &mut Criterion) {
    let text = "Hello\nWorld\n\nRust\n\nProgramming\n".repeat(100);
    
    let mut group = c.benchmark_group("Process Text");

    });
    
    group.bench_function("v4_optimized", |b| {
        b.iter(|| process_text_v4(black_box(&text)))
    });
    
    group.finish();
}

use std::hint::black_box;

// 防止编译器优化掉被测量的代码
fn benchmark(b: &mut Bencher) {
    b.iter(|| {
        // 没有 black_box，编译器可能优化掉这个计算

2. 统计显著性

Criterion 使用统计方法确保结果可靠：

  • 多次采样：每个测试运行多次

  • 异常值检测：自动识别并处理异常值
  • p 值：判断变化是否显著（< 0.05）

fn bubble_sort(mut vec: Vec<i32>) -> Vec<i32> {
    let len = vec.len();
    for i in 0..len {
        for j in 0..len - i - 1 {
            if vec[j] > vec[j + 1] {
                vec.swap(j, j + 1);
            }
        }
    }
    vec
}

fn quick_sort(mut vec: Vec<i32>) -> Vec<i32> {
    if vec.len() <= 1 {
        return vec;
    }
    
    let pivot = vec.pop().unwrap();
    let (less, greater): (Vec<_>, Vec<_>) = vec
        .into_iter()
        .partition(|&x| x <= pivot);
    
    let mut result = quick_sort(less);
    result.push(pivot);
    result.extend(quick_sort(greater));
    result
}

要求：

  • 使用 Criterion 比较两种算法
  • 测试不同大小的输入（100, 1000, 10000 元素）
  • 分析结果并解释差异

fn sum_squares_serial(numbers: &[i64]) -> i64 {
    numbers.iter().map(|&x| x * x).sum()

fn sum_squares_parallel(numbers: &[i64]) -> i64 {
    use rayon::prelude::*;
    numbers.par_iter().map(|&x| x * x).sum()

今日感悟

今天的性能测试学习让我有了几个重要认识：

  1. 测量优于猜测 - 以前总觉得"这段代码应该很快"，但基准测试告诉我，直觉常常是错的。只有测量才能知道真相。

  2. 微优化的陷阱 - 有些优化看起来"应该"更快，但实际提升微乎其微。基准测试帮助我把精力放在真正有影响的地方。

  3. Criterion 的强大 - 统计显著性、自动重试、可视化报告… 这些功能让性能测试变得科学可靠，不再是"跑一次看时间"。

  4. 预分配的力量 - 今天最惊喜的发现是 with_capacity 的优化效果。在很多场景下，简单的预分配能带来 20-30% 的提升。

  5. 并行不是银弹 - 对于小数据集，并行反而更慢（线程开销）。基准测试帮助找到并行的临界点。

  6. 性能是功能的一部分 - 正确的代码但太慢，对用户来说就是"错误的代码"。性能测试应该和单元测试一样重要。

  7. 持续监控 - 性能回归检测让我意识到，性能不应该只在发布前检查，而应该持续监控。

明天的周总结将把本周所有测试知识串联起来。期待形成完整的测试方法论！
