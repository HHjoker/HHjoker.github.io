use std::sync::Arc;
use tokio::task;

#[tokio::main]
async fn main() {
    // 创建 Arc 包裹的数据
    let data = Arc::new(String::from("共享数据"));
    
    // 克隆 Arc（只增加引用计数，不复制数据）
    let data_clone = Arc::clone(&data);
    
    // 在多个任务中共享
    let task1 = task::spawn(async move {
        println!("任务 1: {}", data);
    });
    
    let task2 = task::spawn(async move {
        println!("任务 2: {}", data_clone);
    });
    
    let _ = tokio::join!(task1, task2);
    
    // 所有任务完成后，数据自动释放
    println!("引用计数：{}", Arc::strong_count(&data));
}

Arc 的特点：

  • 📊 线程安全的引用计数
  • 🔒 本身不提供互斥，只管理生命周期
  • 📦 通常与 Mutex 或 RwLock 配合使用
  • ⚡ 克隆成本低（只增加计数）

Arc vs Rc：

特性 Arc Rc
线程安全 ✅ 是 ❌ 否
性能 稍慢（原子操作） 更快
使用场景 多线程/多任务 单线程
异步编程 ✅ 推荐 ❌ 不适用

use std::sync::{Arc, Mutex};

#[tokio::main]
async fn main() {
    let counter = Arc::new(Mutex::new(0));
    
    let tasks: Vec<_> = (0..10).map(|i| {
        let counter = Arc::clone(&counter);
        tokio::spawn(async move {
            // ❌ 问题：lock() 是阻塞的！
            // 如果在 async 上下文中等待锁，会阻塞整个线程
            let mut num = counter.lock().unwrap();
            *num += 1;
            println!("任务 {} 完成，计数：{}", i, *num);
        })
    }).collect();
    
    for task in tasks {
        let _ = task.await;
    }
    
    println!("最终计数：{}", *counter.lock().unwrap());
}

问题分析：

  • std::sync::Mutex::lock() 是阻塞操作
  • 在 async 任务中阻塞会阻塞整个线程
  • 如果持有锁的任务被挂起，其他任务无法获取锁
  • 可能导致死锁或性能问题

use std::sync::Arc;
use tokio::sync::Mutex;

#[tokio::main]
async fn main() {
    let counter = Arc::new(Mutex::new(0));
    
    let tasks: Vec<_> = (0..10).map(|i| {
        let counter = Arc::clone(&counter);
        tokio::spawn(async move {
            // ✅ lock().await 是异步的，不会阻塞线程
            let mut num = counter.lock().await;
            *num += 1;
            println!("任务 {} 完成，计数：{}", i, *num);
        })
    }).collect();
    
    for task in tasks {
        let _ = task.await;
    }
    
    println!("最终计数：{}", *counter.lock().await);
}

tokio::sync::Mutex 的优势：

  • 🔓 lock().await 是异步操作
  • 🧵 等待锁时不会阻塞线程
  • 📋 公平的锁（FIFO 顺序）
  • ⚠️ 性能比标准库 Mutex 稍慢（但在 async 中更安全）

use std::sync::Arc;
use tokio::sync::RwLock;

#[tokio::main]
async fn main() {
    let data = Arc::new(RwLock::new(vec![1, 2, 3]));
    
    // 多个读任务可以并发
    let read_tasks: Vec<_> = (0..5).map(|i| {
        let data = Arc::clone(&data);
        tokio::spawn(async move {
            let read_guard = data.read().await;
            println!("读者 {} 看到：{:?}", i, *read_guard);
            // read_guard 在此自动释放
        })
    }).collect();
    
    // 写任务需要独占访问
    let write_task = tokio::spawn(async move {
        let mut write_guard = data.write().await;
        write_guard.push(4);
        write_guard.push(5);
        println!("写者完成，数据：{:?}", *write_guard);
    });
    
    // 等待所有读任务
    for task in read_tasks {
        let _ = task.await;
    }
    
    // 等待写任务
    let _ = write_task.await;
    
    println!("最终数据：{:?}", *data.read().await);
}

RwLock 的锁类型：

锁类型 方法 并发特性
读锁 read().await 多个读者可并发
写锁 write().await 独占，排斥所有其他锁

读写锁的使用场景：

  • 📖 配置数据（读多写少）
  • 🗄️ 缓存系统
  • 📊 共享状态查询
  • ✏️ 偶尔更新的共享数据

use std::sync::{Mutex as StdMutex, RwLock as StdRwLock};
use tokio::sync::{Mutex as TokioMutex, RwLock as TokioRwLock};

// ============ 标准库锁 ============

/// 标准库 Mutex
/// 适用场景：
/// - 同步代码
/// - 持有锁期间不执行 async 操作
/// - 性能敏感且确定不会死锁
fn std_mutex_example() {
    let mutex = StdMutex::new(0);
    
    // 阻塞式获取锁
    let mut guard = mutex.lock().unwrap();
    *guard += 1;
    // guard 离开作用域自动释放
}

/// 标准库 RwLock
fn std_rwlock_example() {
    let rwlock = StdRwLock::new(vec![1, 2, 3]);
    
    // 读锁（多个可并发）
    let read_guard = rwlock.read().unwrap();
    println!("{:?}", *read_guard);
    
    // 写锁（独占）
    let mut write_guard = rwlock.write().unwrap();
    write_guard.push(4);
}

// ============ Tokio 异步锁 ============

/// Tokio Mutex
/// 适用场景：
/// - async/await 代码
/// - 持有锁期间可能需要 .await
/// - 需要避免阻塞线程
async fn tokio_mutex_example() {
    let mutex = TokioMutex::new(0);
    
    // 异步获取锁
    let mut guard = mutex.lock().await;
    *guard += 1;
    // 可以在持有锁时执行其他 async 操作
    tokio::task::yield_now().await;
}

/// Tokio RwLock
async fn tokio_rwlock_example() {
    let rwlock = TokioRwLock::new(vec![1, 2, 3]);
    
    // 异步读锁
    let read_guard = rwlock.read().await;
    println!("{:?}", *read_guard);
    
    // 异步写锁
    let mut write_guard = rwlock.write().await;
    write_guard.push(4);
}

// ============ 选择指南 ============

/*
选择标准库锁当：
✅ 代码完全是同步的
✅ 持有锁的时间非常短
✅ 确定不会在持有锁时阻塞
✅ 性能是首要考虑

选择 Tokio 锁当：
✅ 代码使用 async/await
✅ 持有锁期间可能需要 .await
✅ 需要公平的锁顺序
✅ 安全性比极致性能更重要
*/

use std::sync::Arc;
use tokio::sync::Mutex;

#[tokio::main]
async fn main() {
    let resource_a = Arc::new(Mutex::new(0));
    let resource_b = Arc::new(Mutex::new(0));
    
    let task1 = tokio::spawn({
        let a = Arc::clone(&resource_a);
        let b = Arc::clone(&resource_b);
        async move {
            let _a_guard = a.lock().await;  // 先锁 A
            tokio::time::sleep(tokio::time::Duration::from_millis(10)).await;
            let _b_guard = b.lock().await;  // 再锁 B
        }
    });
    
    let task2 = tokio::spawn({
        let a = Arc::clone(&resource_a);
        let b = Arc::clone(&resource_b);
        async move {
            let _b_guard = b.lock().await;  // 先锁 B
            tokio::time::sleep(tokio::time::Duration::from_millis(10)).await;
            let _a_guard = a.lock().await;  // 再锁 A - 死锁！
        }
    });
    
    // 这两个任务会互相等待，永远无法完成
    let _ = tokio::join!(task1, task2);
}

use std::sync::Arc;
use tokio::sync::Mutex;

#[tokio::main]
async fn main() {
    let resource_a = Arc::new(Mutex::new(0));
    let resource_b = Arc::new(Mutex::new(0));
    
    // 约定：总是先锁 A，再锁 B
    let task1 = tokio::spawn({
        let a = Arc::clone(&resource_a);
        let b = Arc::clone(&resource_b);
        async move {
            let _a_guard = a.lock().await;
            tokio::time::sleep(tokio::time::Duration::from_millis(10)).await;
            let _b_guard = b.lock().await;
        }
    });
    
    let task2 = tokio::spawn({
        let a = Arc::clone(&resource_a);
        let b = Arc::clone(&resource_b);
        async move {
            let _a_guard = a.lock().await;  // 也先锁 A
            tokio::time::sleep(tokio::time::Duration::from_millis(10)).await;
            let _b_guard = b.lock().await;  // 再锁 B
        }
    });
    
    let _ = tokio::join!(task1, task2);
    println!("✅ 没有死锁！");
}

use std::sync::Arc;
use tokio::sync::Mutex;

#[tokio::main]
async fn main() {
    let counter = Arc::new(Mutex::new(0));
    
    let counter_clone = Arc::clone(&counter);
    
    // 任务 1：持有锁，等待任务 2
    let task1 = tokio::spawn(async move {
        let mut guard = counter_clone.lock().await;
        *guard = 1;
        
        // ❌ 危险：持有锁时等待另一个任务
        // 如果任务 2 也需要这个锁，就会死锁
        let task2 = tokio::spawn(async move {
            let mut guard = counter.lock().await;  // 等待锁
            *guard = 2;
        });
        
        let _ = task2.await;  // 死锁！
    });
    
    let _ = task1.await;
}

use tokio::sync::Mutex;

async fn main() {
    let counter = Arc::new(Mutex::new(0));
    
    let counter_clone = Arc::clone(&counter);
    
    let task1 = tokio::spawn(async move {
        // ✅ 锁的作用域尽可能小
        {
            let mut guard = counter_clone.lock().await;
            *guard = 1;
            // guard 在此释放
        }
        
        // 现在可以安全地等待其他任务
        let task2 = tokio::spawn(async move {
            let mut guard = counter.lock().await;
            *guard = 2;
        });
        
        let _ = task2.await;
    });
    
    let _ = task1.await;
    println!("✅ 安全！");
}

use std::sync::Arc;
use tokio::sync::Mutex;

#[tokio::main]
async fn main() {
    let data = Arc::new(Mutex::new(String::new()));
    
    // ❌ 不推荐：锁持有时间过长
    async fn bad_example(data: Arc<Mutex<String>>) {
        let mut guard = data.lock().await;
        
        // 锁持有期间做很多事
        guard.push_str("Hello");
        tokio::time::sleep(tokio::time::Duration::from_secs(1)).await;
        guard.push_str(" World");
        tokio::time::sleep(tokio::time::Duration::from_secs(1)).await;
        guard.push_str("!");
        
        // guard 在函数结束时才释放
    }
    
    // ✅ 推荐：使用代码块限制作用域
    async fn good_example(data: Arc<Mutex<String>>) {
        {
            let mut guard = data.lock().await;
            guard.push_str("Hello");
        }  // guard 立即释放
        
        tokio::time::sleep(tokio::time::Duration::from_secs(1)).await;
        
        {
            let mut guard = data.lock().await;
            guard.push_str(" World");
        }  // guard 立即释放
        
        tokio::time::sleep(tokio::time::Duration::from_secs(1)).await;
        
        {
            let mut guard = data.lock().await;
            guard.push_str("!");
        }
    }
    
    good_example(Arc::clone(&data));
    
    println!("结果：{}", *data.lock().await);
}

use tokio::sync::Mutex;
use std::sync::Arc;

#[tokio::main]
async fn main() {
    let counter = Arc::new(Mutex::new(0));
    
    // try_lock() 立即返回，不等待
    match counter.try_lock() {
        Ok(mut guard) => {
            *guard += 1;
            println!("获取锁成功，计数：{}", *guard);
        }
        Err(_) => {
            println!("锁被占用，跳过本次操作");
        }
    }
    
    // 在循环中尝试获取锁
    for i in 0..5 {
        if let Ok(mut guard) = counter.try_lock() {
            *guard += 1;
            println!("第 {} 次尝试成功", i);
            break;
        } else {
            println!("第 {} 次尝试失败，等待...", i);
            tokio::time::sleep(tokio::time::Duration::from_millis(100)).await;
        }
    }
}

use std::sync::Arc;
use tokio::sync::Mutex;
use tokio::time::{interval, Duration};

/// 计数器结构
struct Counter {
    value: i64,
    name: String,
}

impl Counter {
    fn new(name: &str) -> Self {
        Counter {
            value: 0,
            name: name.to_string(),
        }
    }
    
    fn increment(&mut self) -> i64 {
        self.value += 1;
        self.value
    }
    
    fn decrement(&mut self) -> i64 {
        self.value -= 1;
        self.value
    }
    
    fn get(&self) -> i64 {
        self.value
    }
}

/// 计数器服务（线程安全）
struct CounterService {
    counter: Arc<Mutex<Counter>>,
}

impl CounterService {
    fn new(name: &str) -> Self {
        CounterService {
            counter: Arc::new(Mutex::new(Counter::new(name))),
        }
    }
    
    /// 增加计数
    async fn increment(&self) -> i64 {
        let mut counter = self.counter.lock().await;
        counter.increment()
    }
    
    /// 减少计数
    async fn decrement(&self) -> i64 {
        let mut counter = self.counter.lock().await;
        counter.decrement()
    }
    
    /// 获取当前值
    async fn get(&self) -> i64 {
        let counter = self.counter.lock().await;
        counter.get()
    }
    
    /// 获取计数器名称
    async fn get_name(&self) -> String {
        let counter = self.counter.lock().await;
        counter.name.clone()
    }
}

/// 生产者：定期增加计数
async fn producer(service: Arc<CounterService>, count: i32) {
    let mut interval = interval(Duration::from_millis(100));
    
    for i in 0..count {
        interval.tick().await;
        let new_value = service.increment().await;
        println!("📈 生产者：增加计数，当前值 = {}", new_value);
    }
    
    println!("✅ 生产者完成");
}

/// 消费者：定期减少计数
async fn consumer(service: Arc<CounterService>, count: i32) {
    let mut interval = interval(Duration::from_millis(150));
    
    for i in 0..count {
        interval.tick().await;
        let new_value = service.decrement().await;
        println!("📉 消费者：减少计数，当前值 = {}", new_value);
    }
    
    println!("✅ 消费者完成");
}

/// 监控者：定期报告计数
async fn monitor(service: Arc<CounterService>) {
    let mut interval = interval(Duration::from_millis(500));
    
    loop {
        interval.tick().await;
        let name = service.get_name().await;
        let value = service.get().await;
        println!("📊 [{}] 当前计数 = {}", name, value);
    }
}

#[tokio::main]
async fn main() {
    println!("🚀 启动计数器服务");
    
    let service = Arc::new(CounterService::new("主计数器"));
    
    // 启动生产者
    let producer_handle = tokio::spawn(producer(Arc::clone(&service), 10));
    
    // 启动消费者
    let consumer_handle = tokio::spawn(consumer(Arc::clone(&service), 8));
    
    // 启动监控者
    let monitor_handle = tokio::spawn(monitor(Arc::clone(&service)));
    
    // 等待生产者和消费者完成
    let _ = tokio::join!(producer_handle, consumer_handle);
    
    // 给监控者一点时间输出最后的状态
    tokio::time::sleep(Duration::from_millis(600)).await;
    
    // 取消监控者
    monitor_handle.abort();
    
    // 输出最终结果
    let final_value = service.get().await;
    println!("\n🎉 服务结束，最终计数 = {}", final_value);
}

use std::sync::Arc;
use tokio::sync::RwLock;
use tokio::time::{interval, Duration};

/// 配置数据
#[derive(Clone, Debug)]
struct Config {
    max_connections: u32,
    timeout_ms: u64,
    enabled: bool,
}

impl Config {
    fn default() -> Self {
        Config {
            max_connections: 100,
            timeout_ms: 5000,
            enabled: true,
        }
    }
}

/// 配置管理器
struct ConfigManager {
    config: Arc<RwLock<Config>>,
}

impl ConfigManager {
    fn new() -> Self {
        ConfigManager {
            config: Arc::new(RwLock::new(Config::default())),
        }
    }
    
    /// 读取配置（多个读者可并发）
    async fn get_config(&self) -> Config {
        let config = self.config.read().await;
        config.clone()  // 返回副本，释放锁
    }
    
    /// 更新配置（独占写）
    async fn update_config(&self, f: impl FnOnce(&mut Config)) {
        let mut config = self.config.write().await;
        f(&mut config);
    }
    
    /// 检查是否启用
    async fn is_enabled(&self) -> bool {
        let config = self.config.read().await;
        config.enabled
    }
}

/// 多个并发读者
async fn reader(id: usize, manager: Arc<ConfigManager>) {
    let mut interval = interval(Duration::from_millis(200));
    
    for i in 0..5 {
        interval.tick().await;
        let config = manager.get_config().await;
        println!("📖 读者 {} (第 {} 次): {:?}", id, i + 1, config);
    }
}

/// 配置更新者
async fn updater(manager: Arc<ConfigManager>) {
    let mut interval = interval(Duration::from_millis(500));
    
    for i in 0..3 {
        interval.tick().await;
        
        manager.update_config(|config| {
            config.max_connections += 50;
            config.timeout_ms += 1000;
        }).await;
        
        println!("✏️  更新者：第 {} 次更新完成", i + 1);
    }
}

#[tokio::main]
async fn main() {
    println!("🚀 启动配置管理系统");
    
    let manager = Arc::new(ConfigManager::new());
    
    // 启动多个读者（可以并发）
    let readers: Vec<_> = (0..3)
        .map(|i| tokio::spawn(reader(i, Arc::clone(&manager))))
        .collect();
    
    // 启动更新者
    let updater_handle = tokio::spawn(updater(Arc::clone(&manager)));
    
    // 等待所有任务
    for reader in readers {
        let _ = reader.await;
    }
    let _ = updater_handle.await;
    
    // 输出最终配置
    let final_config = manager.get_config().await;
    println!("\n📊 最终配置：{:?}", final_config);
}

/*
标准库 Mutex 的问题：

1. 阻塞整个线程
   ┌─────────────┐
   │  线程 1     │
   │  ┌───────┐  │
   │  │ 任务 A │  │
   │  │  lock │──┼──▶ 阻塞！
   │  │  await│  │
   │  └───────┘  │
   └─────────────┘
   
   当任务 A 等待锁时，整个线程被阻塞
   同一线程上的其他任务也无法运行

2. 异步锁的解决方案：
   ┌─────────────┐
   │  线程 1     │
   │  ┌───────┐  │
   │  │ 任务 A │──┼──▶ 等待锁（不阻塞）
   │  │ lock().await │
   │  └───────┘  │
   │  ┌───────┐  │
   │  │ 任务 B │  │◀── 可以继续运行
   │  └───────┘  │
   └─────────────┘
   
   任务 A 等待锁时，任务 B 可以继续执行
*/

use tokio::sync::Mutex;

async fn process_data(data: &str) -> String {
    // 模拟异步处理
    tokio::time::sleep(tokio::time::Duration::from_millis(10)).await;
    format!("处理：{}", data)
}

#[tokio::main]
async fn main() {
    let m = Mutex::new(String::from("test"));
    
    // ❌ 错误：guard 持有锁时不能 .await
    async fn wrong(m: &Mutex<String>) {
        let mut guard = m.lock().await;
        
        // 这会导致编译错误！
        // let result = process_data(&guard).await;
        // guard 被 process_data 调用借用，同时又有 .await
        
        // 错误信息：
        // "future cannot be sent between threads safely"
        // "MutexGuard<'_, String>` cannot be sent between threads safely
    }
    
    // ✅ 正确：先复制数据，再释放锁
    async fn right(m: &Mutex<String>) {
        let data = {
            let guard = m.lock().await;
            guard.clone()  // 复制数据
        };  // guard 在此释放
        
        // 现在可以安全地 .await
        let result = process_data(&data).await;
        println!("{}", result);
    }
    
    right(&m).await;
}

🌟 锁的艺术：在安全与性能之间寻找平衡

今天深入学习了异步编程中的共享状态管理，这是构建可靠并发系统的关键。

最大的收获：

  1. Arc 是基础：在 async 中共享数据，首先用 Arc 包装
  2. 选择合适的锁：
    • 标准库 Mutex：同步代码，性能敏感
    • Tokio Mutex：async 代码，安全性优先
    • RwLock：读多写少场景
  3. 死锁避免：
    • 统一的锁获取顺序
    • 缩小锁的作用域
    • 避免持有锁时 .await
  4. 守卫管理：使用代码块明确锁的生命周期

遇到的困难：

  • 理解为什么标准库 Mutex 在 async 中不安全
  • 处理 MutexGuard 不能跨 .await 边界的限制
  • 设计合理的锁粒度（太粗影响性能，太细增加复杂度）

有趣的发现：

  • try_lock() 可以避免阻塞，适合非关键操作
  • RwLock 在读多写少场景下性能提升明显
  • 使用 drop(guard) 可以显式提前释放锁

实际应用场景：

  • 🌐 Web 服务器：共享连接池、配置
  • 🗄️ 数据库连接池：并发安全的连接管理
  • 📊 指标收集：多线程/任务共享计数器
  • ⚙️ 配置管理：热更新配置

给初学者的建议：

  1. 默认使用 Tokio 的锁（更安全）
  2. 锁的作用域越小越好
  3. 避免在持有锁时执行异步操作
  4. 多个锁时，约定统一的获取顺序
  5. 考虑使用通道替代共享状态（更 Rustacean）

下一步：

  • 学习任务取消模式
  • 掌握 tokio::select! 的使用
  • 实践：构建可优雅关闭的服务

共享状态管理是异步编程的核心技能之一。掌握了锁的艺术，就能构建更安全、高效的并发系统！🦀
