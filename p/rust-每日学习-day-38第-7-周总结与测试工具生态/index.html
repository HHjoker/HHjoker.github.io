use mockall::automock;

#[automock]
trait Database {
    fn get_user(&self, id: u32) -> Option<String>;
    fn save_user(&self, name: &str) -> Result<u32, String>;
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn test_with_mock() {
        let mut mock_db = MockDatabase::new();
        
        mock_db.expect_get_user()
            .with(mockall::predicate::eq(1))
            .times(1)
            .returning(|_| Some("Alice".to_string()));
        
        mock_db.expect_save_user()
            .with(mockall::predicate::always())
            .times(1)
            .returning(|_| Ok(1));

        // 使用 mock_db 进行测试
    }
}

适用场景：

  • 外部依赖隔离（数据库、网络、文件系统）
  • 验证调用次数和参数
  • 模拟错误场景

use proptest::prelude::*;

fn parse_date(s: &str) -> Option<(u32, u32, u32)> {
    // 实现略
    Some((2024, 1, 15))
}

proptest! {
    #[test]
    fn test_parse_date_valid_input(
        year in 1900u32..2100,
        month in 1u32..=12,
        day in 1u32..=31
    ) {
        let input = format!("{:04}-{:02}-{:02}", year, month, day);
        if let Some((y, m, d)) = parse_date(&input) {
            prop_assert!(y >= 1900 && y <= 2100);
            prop_assert!(m >= 1 && m <= 12);
            prop_assert!(d >= 1 && d <= 31);
        }
    }

    #[test]
    fn test_parse_date_invalid_input(
        s in "[^0-9-]{10}"
    ) {
        prop_assert!(parse_date(&s).is_none());
    }
}

特点：

  • 自动生成大量随机输入
  • 发现边界情况和意外输入
  • 适合纯函数测试

fn reverse_twice_is_identity(vec: Vec<i32>) -> bool {
    let reversed: Vec<i32> = vec.iter().copied().rev().collect();
    let double_reversed: Vec<i32> = reversed.iter().copied().rev().collect();
    vec == double_reversed
}

// benches/my_benchmark.rs
use criterion::{black_box, criterion_group, criterion_main, Criterion};

fn fibonacci(n: u64) -> u64 {
    match n {
        0 => 0,
        1 => 1,
        _ => fibonacci(n - 1) + fibonacci(n - 2),
    }
}

fn fibonacci_iterative(n: u64) -> u64 {
    let mut a = 0;
    let mut b = 1;
    for _ in 0..n {
        let temp = a;
        a = b;
        b = temp + b;
    }
    a
}

fn criterion_benchmark(c: &mut Criterion) {
    c.bench_function("fib_20_recursive", |b| {
        b.iter(|| fibonacci(black_box(20)))
    });

    c.bench_function("fib_20_iterative", |b| {
        b.iter(|| fibonacci_iterative(black_box(20)))
    });

    // 参数化基准
    let mut group = c.benchmark_group("fibonacci_comparison");
    for i in [10, 20, 30].iter() {
        group.bench_with_input(
            format!("recursive_{}", i),
            i,
            |b, &n| b.iter(|| fibonacci(black_box(n)))
        );
        group.bench_with_input(
            format!("iterative_{}", i),
            i,
            |b, &n| b.iter(|| fibonacci_iterative(black_box(n)))
        );
    }
    group.finish();
}

criterion_group!(benches, criterion_benchmark);
criterion_main!(benches);

覆盖率类型：

  • 行覆盖率（Line Coverage）：多少行代码被执行
  • 函数覆盖率（Function Coverage）：多少函数被调用
  • 区域覆盖率（Region Coverage）：分支覆盖情况

目标建议：

  • 核心业务逻辑：80-90%
  • 工具/辅助代码：60-80%
  • 简单 getter/setter：可接受较低

# .github/workflows/ci.yml

name: CI

on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [main]

env:
  CARGO_TERM_COLOR: always
  RUST_BACKTRACE: 1

jobs:
  # 1. 代码格式化检查
  fmt:
    name: Format Check
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: dtolnay/rust-action@stable
      - run: cargo fmt -- --check

  # 2. Clippy 静态分析
  clippy:
    name: Clippy Check
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: dtolnay/rust-action@stable
        with:
          components: clippy
      - run: cargo clippy -- -D warnings

  # 3. 单元测试 + 集成测试
  test:
    name: Test Suite
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: dtolnay/rust-action@stable
      - uses: Swatinem/rust-cache@v2
      
      - name: Run tests
        run: cargo test --all-features --verbose
      
      - name: Run doctests
        run: cargo test --doc

  # 4. 代码覆盖率
  coverage:
    name: Code Coverage
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: dtolnay/rust-action@stable
      - uses: Swatinem/rust-cache@v2
      
      - name: Install cargo-llvm-cov
        uses: taiki-e/install-action@cargo-llvm-cov
      
      - name: Generate coverage
        run: cargo llvm-cov --all-features --workspace --lcov --output-path lcov.info
      
      - name: Upload to codecov.io
        uses: codecov/codecov-action@v4
        with:
          files: lcov.info
          fail_ci_if_error: false

  # 5. 基准测试（仅 main 分支）
  benchmark:
    name: Benchmarks
    runs-on: ubuntu-latest
    if: github.ref == 'refs/heads/main'
    steps:
      - uses: actions/checkout@v4
      - uses: dtolnay/rust-action@stable
      
      - name: Run benchmarks
        run: cargo bench --bench my_benchmark
      
      - name: Store benchmark result
        uses: benchmark-action/github-action-benchmark@v1
        with:
          name: Rust Benchmark
          tool: 'cargo'
          output-file-path: target/criterion/output.json

  # 6. 多 Rust 版本测试
  matrix-test:
    name: Matrix Test
    runs-on: ubuntu-latest
    strategy:
      matrix:
        rust: [stable, beta, nightly]
    steps:
      - uses: actions/checkout@v4
      - uses: dtolnay/rust-action@master
        with:
          toolchain: ${{ matrix.rust }}
      - run: cargo test --all-features
        continue-on-error: ${{ matrix.rust == 'nightly' }}

  # 7. 跨平台测试
  cross-platform:
    name: Cross Platform
    runs-on: ${{ matrix.os }}
    strategy:
      matrix:
        os: [ubuntu-latest, windows-latest, macOS-latest]
    steps:
      - uses: actions/checkout@v4
      - uses: dtolnay/rust-action@stable
      - run: cargo test

# .gitlab-ci.yml

stages:
  - check
  - test
  - coverage

variables:
  CARGO_HOME: "${CI_PROJECT_DIR}/.cargo"

cache:
  key: "${CI_JOB_NAME}"
  paths:
    - .cargo/
    - target/

fmt:
  stage: check
  image: rust:latest
  script:
    - cargo fmt -- --check

clippy:
  stage: check
  image: rust:latest
  script:
    - rustup component add clippy
    - cargo clippy -- -D warnings

test:
  stage: test
  image: rust:latest
  script:
    - cargo test --all-features --verbose
  coverage: '/lines\.:\s+\d+\.\d+\%/'

coverage:
  stage: coverage
  image: rust:latest
  before_script:
    - cargo install cargo-llvm-cov
  script:
    - cargo llvm-cov --all-features --workspace --html
  artifacts:
    paths:
      - target/llvm-cov/html/
    reports:
      coverage_report:
        coverage_format: cobertura
        path: target/llvm-cov/cobertura.xml

📋 测试检查清单

项目启动时

  • 设置基础测试框架（#[cfg(test)] 模块）
  • 配置 CI 自动运行测试
  • 决定覆盖率目标（建议 80%+）

开发过程中

  • 新功能先写测试（TDD）
  • 修复 bug 先写复现测试
  • 复杂逻辑添加属性测试
  • 性能关键代码添加基准测试

发布前

  • 所有测试通过
  • 覆盖率达标
  • 基准测试无显著退化
  • 文档测试通过（cargo test --doc）

💭 今日感悟

第 7 周的测试学习让我对 Rust 的测试生态有了全面认识：

  1. 内置测试框架已经足够强大：对于大多数项目，cargo test 配合良好的测试组织就能满足需求。不要过早引入复杂工具。

  2. 工具选择要看场景：

    • Mock 工具适合隔离外部依赖
    • 属性测试适合纯函数和边界发现
    • 基准测试适合性能敏感代码
    • 覆盖率是质量指标，不是目标

  3. CI/CD 是测试的放大器：自动化让测试真正发挥作用。每次提交都运行测试，每次 PR 都检查覆盖率，这比手动运行可靠得多。

  4. 测试是投资，不是成本：花时间写测试、搭建测试环境、配置 CI，这些短期看似"耽误进度"的工作，长期会节省大量调试和维护时间。

  5. Rust 类型系统 + 测试 = 高可靠性：Rust 的编译器已经帮我们排除了很多错误，再加上完善的测试，可以构建非常可靠的系统。

第 8 周我们将进入函数式编程的核心——闭包与迭代器。这是 Rust 最优雅、最强大的特性之一！
