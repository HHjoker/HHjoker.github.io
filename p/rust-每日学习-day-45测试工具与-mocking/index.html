什么是 Mocking？

Mocking 是一种测试技术，用于创建"模拟对象"来替代真实的依赖项。为什么要用 Mock？

使用 Mock 的场景

  1. 外部依赖 - 数据库、API、文件系统等
  2. 耗时操作 - 网络请求、复杂计算
  3. 不稳定因素 - 随机数、时间、环境变量
  4. 难以构造的对象 - 需要复杂设置的依赖
  5. 验证交互 - 确认某个方法被正确调用

// ✅ 易于测试 - 依赖抽象 trait
trait Database {
    fn query_user(&self, id: u32) -> Result<User>;
}

struct UserService<DB: Database> {
    db: DB,
}

impl<DB: Database> UserService<DB> {
    fn new(db: DB) -> Self {
        Self { db }
    }

    fn get_user(&self, id: u32) -> Result<User> {
        self.db.query_user(id)
    }
}

// 生产环境使用真实数据库
struct PostgreSQLDatabase {
    // 真实连接
}

impl Database for PostgreSQLDatabase {
    fn query_user(&self, id: u32) -> Result<User> {
        // 真实数据库查询
        Ok(User { id, name: "Real".to_string() })
    }
}

// 测试环境使用 Mock
struct MockDatabase {
    users: std::collections::HashMap<u32, User>,
}

impl Database for MockDatabase {
    fn query_user(&self, id: u32) -> Result<User> {
        self.users.get(&id).cloned().ok_or_else(|| "User not found".into())
    }
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn test_get_user() {
        let mut mock_db = MockDatabase {
            users: std::collections::HashMap::new(),
        };
        mock_db.users.insert(1, User { id: 1, name: "Test".to_string() });

        let service = UserService::new(mock_db);
        let user = service.get_user(1).unwrap();
        assert_eq!(user.name, "Test");
    }
}

use mockall::automock;

// 1. 定义 trait 并用 #[automock] 生成 Mock
#[automock]
trait Calculator {
    fn add(&self, a: i32, b: i32) -> i32;
    fn multiply(&self, a: i32, b: i32) -> i32;
}

// 2. 使用 trait 的结构
struct MathService<C: Calculator> {
    calc: C,
}

impl<C: Calculator> MathService<C> {
    fn new(calc: C) -> Self {
        Self { calc }
    }

    fn compute(&self, x: i32, y: i32) -> i32 {
        let sum = self.calc.add(x, y);
        self.calc.multiply(sum, 2)
    }
}

// 3. 真实实现
struct RealCalculator;

impl Calculator for RealCalculator {
    fn add(&self, a: i32, b: i32) -> i32 {
        a + b
    }

    fn multiply(&self, a: i32, b: i32) -> i32 {
        a * b
    }
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn test_compute_with_mock() {
        // 创建 Mock 对象
        let mut mock_calc = MockCalculator::new();

        // 设置期望：当 add(2, 3) 被调用时，返回 5
        mock_calc
            .expect_add()
            .with(mockall::predicate::eq(2), mockall::predicate::eq(3))
            .times(1)
            .returning(|a, b| a + b);

        // 设置期望：当 multiply(5, 2) 被调用时，返回 10
        mock_calc
            .expect_multiply()
            .with(mockall::predicate::eq(5), mockall::predicate::eq(2))
            .times(1)
            .returning(|a, b| a * b);

        let service = MathService::new(mock_calc);
        let result = service.compute(2, 3);

        assert_eq!(result, 10);
        // mockall 会自动验证期望是否被满足
    }
}

use mockall::automock;

#[automock]
trait UserRepository {
    fn find_by_id(&self, id: u32) -> Option<User>;
    fn find_all(&self) -> Vec<User>;
    fn save(&self, user: User) -> Result<User, String>;
}

#[derive(Debug, Clone, PartialEq)]
struct User {
    id: u32,
    name: String,
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn test_repository_mock() {
        let mut mock_repo = MockUserRepository::new();

        // 返回固定值
        mock_repo
            .expect_find_by_id()
            .returning(|id| {
                if id == 1 {
                    Some(User { id: 1, name: "Alice".to_string() })
                } else {
                    None
                }
            });

        // 返回序列值（每次调用返回不同值）
        mock_repo
            .expect_find_all()
            .times(1)
            .return_once(|| {
                vec![
                    User { id: 1, name: "Alice".to_string() },
                    User { id: 2, name: "Bob".to_string() },
                ]
            });

        // 使用闭包处理参数
        mock_repo
            .expect_save()
            .returning(|user| {
                if user.name.is_empty() {
                    Err("Name cannot be empty".to_string())
                } else {
                    Ok(user)
                }
            });

        // 测试
        assert_eq!(mock_repo.find_by_id(1).unwrap().name, "Alice");
        assert_eq!(mock_repo.find_by_id(999), None);
        assert_eq!(mock_repo.find_all().len(), 2);

        let new_user = User { id: 3, name: "Charlie".to_string() };
        assert!(mock_repo.save(new_user).is_ok());

        let empty_user = User { id: 4, name: "".to_string() };
        assert!(mock_repo.save(empty_user).is_err());
    }
}

use mockall::automock;

#[automock]
trait Logger {
    fn info(&self, msg: &str);
    fn error(&self, msg: &str);
    fn debug(&self, msg: &str);
}

struct App<L: Logger> {
    logger: L,
}

impl<L: Logger> App<L> {
    fn run(&self) -> Result<(), String> {
        self.logger.info("Starting app");
        self.logger.debug("Loading configuration");
        
        // 模拟可能失败的操作
        let success = true;
        if !success {
            self.logger.error("App failed");
            return Err("Failed".to_string());
        }
        
        self.logger.info("App completed");
        Ok(())
    }
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn test_app_logging() {
        let mut mock_logger = MockLogger::new();

        // 验证调用顺序
        let ctx = mockall::Sequence::new();
        
        mock_logger
            .expect_info()
            .with(mockall::predicate::eq("Starting app"))
            .times(1)
            .in_sequence(&ctx);

        mock_logger
            .expect_debug()
            .with(mockall::predicate::eq("Loading configuration"))
            .times(1)
            .in_sequence(&ctx);

        mock_logger
            .expect_info()
            .with(mockall::predicate::eq("App completed"))
            .times(1)
            .in_sequence(&ctx);

        // error 不应该被调用
        mock_logger
            .expect_error()
            .times(0);

        let app = App { logger: mock_logger };
        assert!(app.run().is_ok());
    }
}

use mockito::{mock, server_address};
use reqwest;
use serde::{Deserialize, Serialize};

#[derive(Debug, Serialize, Deserialize, PartialEq)]
struct User {
    id: u32,
    name: String,
    email: String,
}

struct ApiClient {
    base_url: String,
    client: reqwest::Client,
}

impl ApiClient {
    fn new(base_url: String) -> Self {
        Self {
            base_url,
            client: reqwest::Client::new(),
        }
    }

    async fn get_user(&self, id: u32) -> Result<User, reqwest::Error> {
        let url = format!("{}/users/{}", self.base_url, id);
        let response = self.client.get(&url).send().await?;
        response.json().await
    }

    async fn create_user(&self, user: &User) -> Result<User, reqwest::Error> {
        let url = format!("{}/users", self.base_url);
        let response = self.client
            .post(&url)
            .json(user)
            .send()
            .await?;
        response.json().await
    }
}

#[cfg(test)]
mod tests {
    use super::*;

    #[tokio::test]
    async fn test_get_user() {
        // 创建 Mock HTTP 服务
        let _mock = mock("GET", "/users/1")
            .with_status(200)
            .with_header("content-type", "application/json")
            .with_body(r#"{"id": 1, "name": "Alice", "email": "alice@example.com"}"#)
            .create();

        let client = ApiClient::new(format!("http://{}", server_address()));
        let user = client.get_user(1).await.unwrap();

        assert_eq!(user, User {
            id: 1,
            name: "Alice".to_string(),
            email: "alice@example.com".to_string(),
        });
    }

    #[tokio::test]
    async fn test_create_user() {
        let _mock = mock("POST", "/users")
            .with_status(201)
            .with_header("content-type", "application/json")
            .with_body(r#"{"id": 2, "name": "Bob", "email": "bob@example.com"}"#)
            .create();

        let client = ApiClient::new(format!("http://{}", server_address()));
        let new_user = User {
            id: 0, // 会被服务器忽略
            name: "Bob".to_string(),
            email: "bob@example.com".to_string(),
        };

        let created = client.create_user(&new_user).await.unwrap();
        assert_eq!(created.id, 2);
        assert_eq!(created.name, "Bob");
    }

    #[tokio::test]
    async fn test_get_user_not_found() {
        let _mock = mock("GET", "/users/999")
            .with_status(404)
            .with_body(r#"{"error": "User not found"}"#)
            .create();

        let client = ApiClient::new(format!("http://{}", server_address()));
        let result = client.get_user(999).await;
        
        // 404 会返回错误
        assert!(result.is_err());
    }
}

// 内存中的假数据库
struct FakeDatabase {
    data: std::sync::RwLock<std::collections::HashMap<u32, User>>,
}

impl FakeDatabase {
    fn new() -> Self {
        Self {
            data: std::sync::RwLock::new(std::collections::HashMap::new()),
        }
    }

    fn seed(&self, users: Vec<User>) {
        let mut data = self.data.write().unwrap();
        for user in users {
            data.insert(user.id, user);
        }
    }
}

impl Database for FakeDatabase {
    fn query_user(&self, id: u32) -> Result<User> {
        let data = self.data.read().unwrap();
        data.get(&id).cloned().ok_or_else(|| "User not found".into())
    }

    fn save_user(&self, user: User) -> Result<User> {
        let mut data = self.data.write().unwrap();
        data.insert(user.id, user.clone());
        Ok(user)
    }
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn test_with_fake_database() {
        let fake_db = FakeDatabase::new();
        
        // 准备测试数据
        fake_db.seed(vec![
            User { id: 1, name: "Alice".to_string() },
            User { id: 2, name: "Bob".to_string() },
            User { id: 3, name: "Charlie".to_string() },
        ]);

        // 运行测试
        assert_eq!(fake_db.query_user(1).unwrap().name, "Alice");
        assert_eq!(fake_db.query_user(2).unwrap().name, "Bob");
        assert!(fake_db.query_user(999).is_err());

        // 测试保存
        let new_user = User { id: 4, name: "David".to_string() };
        fake_db.save_user(new_user.clone()).unwrap();
        assert_eq!(fake_db.query_user(4).unwrap().name, "David");
    }
}

// src/lib.rs
use mockall::automock;

#[derive(Debug, Clone, PartialEq)]
pub struct User {
    pub id: u32,
    pub name: String,
    pub email: String,
}

#[derive(Debug)]
pub enum AppError {
    NotFound,
    ValidationError(String),
    DatabaseError(String),
}

pub type Result<T> = std::result::Result<T, AppError>;

#[automock]
pub trait UserStore {
    fn get(&self, id: u32) -> Result<User>;
    fn create(&self, user: User) -> Result<User>;
    fn update(&self, id: u32, user: User) -> Result<User>;
    fn delete(&self, id: u32) -> Result<()>;
}

pub struct UserService<S: UserStore> {
    store: S,
}

impl<S: UserStore> UserService<S> {
    pub fn new(store: S) -> Self {
        Self { store }
    }

    pub fn register(&self, name: String, email: String) -> Result<User> {
        // 验证
        if name.is_empty() {
            return Err(AppError::ValidationError("Name cannot be empty".into()));
        }
        if !email.contains('@') {
            return Err(AppError::ValidationError("Invalid email".into()));
        }

        // 创建用户
        let user = User {
            id: 1, // 实际中由数据库生成
            name,
            email,
        };

        self.store.create(user)
    }

    pub fn get_user(&self, id: u32) -> Result<User> {
        self.store.get(id)
    }

    pub fn delete_user(&self, id: u32) -> Result<()> {
        // 先检查用户是否存在
        self.store.get(id)?;
        self.store.delete(id)
    }
}

关键概念理解

1. 为什么需要 Mock？

  • 隔离测试 - 只测试当前单元，不依赖外部系统
  • 快速测试 - Mock 比真实数据库/API 快得多
  • 确定性 - Mock 返回可控的数据，测试更可靠
  • 边界情况 - 轻松模拟错误、超时等场景

2. Mock vs Fake 的选择

场景 推荐 原因
验证方法调用 Mock 可以验证调用次数和顺序
简单数据返回 Fake 实现简单，维护成本低
复杂业务逻辑 Fake 更接近真实行为
HTTP API mockito 专门为此设计
Trait 抽象 mockall 自动生成 Mock 代码

3. 依赖注入的好处

  • 可测试性 - 可以轻松替换依赖
  • 灵活性 - 运行时可以切换实现
  • 解耦 - 降低模块间耦合度
  • 可维护性 - 代码更清晰

今日感悟

Mocking 是测试中的一项高级技能，但它也带来了复杂性。今天的核心收获：

  1. Mock 不是银弹 - 过度使用 Mock 会让测试变得脆弱。优先使用 Fake 或真实对象。

  2. 设计决定可测试性 - 如果代码难以 Mock，通常是设计问题。依赖注入是解决之道。

  3. 验证行为而非实现 - Mock 应该验证"做了什么"，而不是"怎么做的"。

  4. 保持测试简洁 - 一个测试只验证一个行为。复杂的 Mock 设置可能意味着测试范围太大。

  5. Mock 也是文档 - 好的 Mock 设置展示了代码的预期使用方式。
