1. 并行执行多个 Future：join!

在实际项目中，我们经常需要同时发起多个请求，等待它们全部完成后再继续处理。futures crate 提供的 join! 宏正是为此设计。

use futures::join;
use tokio::time::{sleep, Duration};

async fn fetch_user() -> String {
    sleep(Duration::from_secs(1)).await;
    "Alice".to_string()
}

async fn fetch_orders() -> Vec<String> {
    sleep(Duration::from_secs(2)).await;
    vec!["Order A".to_string(), "Order B".to_string()]
}

async fn fetch_notifications() -> i32 {
    sleep(Duration::from_millis(500)).await;
    5
}

#[tokio::main]
async fn main() {
    // 并行执行三个 Future，全部完成才继续
    let (user, orders, notif_count) = join!(
        fetch_user(),
        fetch_orders(),
        fetch_notifications()
    );

    println!("用户: {}", user);
    println!("订单: {:?}", orders);
    println!("未读通知: {} 条", notif_count);
    // 总耗时：约 2 秒（最长那个）
}

注意：join! 会并行执行所有 Future，但总耗时等于最长那个的耗时。上例中 fetch_orders() 要 2 秒，所以整个 join! 耗时约 2 秒。

use futures::race;
use tokio::time::{sleep, Duration};

async fn slow_api() -> &'static str {
    sleep(Duration::from_secs(5)).await;
    "Slow API 返回"
}

async fn fast_api() -> &'static str {
    sleep(Duration::from_secs(1)).await;
    "Fast API 返回"
}

#[tokio::main]
async fn main() {
    let result = race!(slow_api(), fast_api());
    println!("竞速结果: {}", result);
    // 输出：竞速结果: Fast API 返回
    // 只耗时约 1 秒
}

典型应用场景：

  • 多源数据获取：同时请求多个 CDN，取最快返回的
  • 超时模式：用 race!(long_task(), timeout()) 实现超时控制

use tokio::time::{sleep, Duration};
use tokio::sync::oneshot;

#[tokio::main]
async fn main() {
    let (tx1, rx1) = oneshot::channel::<&str>("一");
    let (tx2, rx2) = oneshot::channel::<&str>("二");
    let (tx3, rx3) = oneshot::channel::<&str>("三");

    // 用 tokio::spawn 模拟并发发送
    tokio::spawn(async move {
        sleep(Duration::from_millis(300)).await;
        let _ = tx1.send("消息来自通道一");
    });

    tokio::spawn(async move {
        sleep(Duration::from_millis(150)).await;
        let _ = tx2.send("消息来自通道二");
    });

    tokio::spawn(async move {
        sleep(Duration::from_millis(500)).await;
        let _ = tx3.send("消息来自通道三");
    });

    // select! 循环，处理多个消息
    for _ in 0..3 {
        tokio::select! {
            msg1 = rx1 => {
                println!("收到: {:?}", msg1);
            }
            msg2 = rx2 => {
                println!("收到: {:?}", msg2);
            }
            msg3 = rx3 => {
                println!("收到: {:?}", msg3);
            }
        }
    }

    println!("所有消息处理完毕");
}

use tokio::time::{sleep, Duration};

async fn long_task() {
    sleep(Duration::from_secs(10)).await;
    println!("任务完成");
}

#[tokio::main]
async fn main() {
    tokio::select! {
        _ = long_task() => {
            println!("任务正常完成");
        }
        _ = sleep(Duration::from_secs(2)) => {
            println!("超时了！任务被取消");
        }
    }
    // 输出：超时了！任务被取消
    // 注意：long_task 在超时后会被取消
}

use tokio::time::{sleep, Duration};
use tokio::time::timeout;

async fn api_call() -> String {
    sleep(Duration::from_secs(3)).await;
    "API 返回数据".to_string()
}

#[tokio::main]
async fn main() {
    // 最多等待 2 秒
    let result = timeout(Duration::from_secs(2), api_call()).await;

    match result {
        Ok(data) => println!("成功: {}", data),
        Err(_) => println!("请求超时！"),
    }
    // 输出：请求超时！
}

    // 取消任务

}

use futures::try_join;
use tokio::time::{sleep, Duration};

async fn fetch_data() -> Result<String, &'static str> {
    Ok("数据获取成功".to_string())
}

async fn parse_data() -> Result<i32, &'static str> {
    Err("解析失败")
}

#[tokio::main]
async fn main() {
    let result = try_join!(fetch_data(), parse_data());

    match result {
        Ok((data, count)) => println!("{} {}", data, count),
        Err(e) => println!("错误: {}", e),
        // 输出：错误: 解析失败
    }
}

关键概念理解

join! vs select! vs race!

宏 行为 等待条件 典型用途
join! 等待所有完成 全部完成 并行请求，汇总结果
race! 取最先完成的 任一完成 超时、竞速、CDN 多源
select! 依次处理完成的 任一完成 多路复用、事件循环

取消与 Drop

当 select! 选择了一个分支，其他未完成的 Future 会被自动取消（内部调用 drop）。这意味着它们的资源会被释放。

    tokio::spawn(async move {
        tx.send("已发送").unwrap();
    });

    tokio::select! {
        msg = rx => println!("收到: {:?}", msg),
        _ = sleep(Duration::from_secs(1)) => println!("超时"),
    }
    // 如果 rx 先收到消息，sleep Future 被取消
    // 反之亦然
}

今日感悟

今天深入学习了 Tokio 异步并发三大核心宏——join!、select! 和 race!。这三个宏是异步编程中最常用的工具，分别应对"全做完"、“谁先做完谁处理”、“取最快的"三种场景。

最有收获的是理解 select! 的取消语义：当一个分支完成时，其他未完成的 Future 会被自动丢弃。这要求我们在写代码时考虑清楚每个 Future 的清理逻辑。

通过练习题我发现，实际项目中往往需要组合使用这些宏：比如用 select! 实现超时，用 join! 并行获取数据，用 race! 实现竞速降级。这种组合能力才是异步编程的精髓所在。
