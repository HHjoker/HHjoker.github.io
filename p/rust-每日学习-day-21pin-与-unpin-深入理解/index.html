2. Pin 的基本概念

Pin 是一个包装器，保证被包装的值不会被移动：

use std::pin::Pin;
use std::marker::Unpin;

// Pin<P> 保证 P 指向的数据不会被移动
let mut value = String::from("Hello");
let pinned: Pin<&mut String> = Pin::new(&mut value);

// ❌ 不能直接移动 Pin 中的值
// let moved = *pinned;  // 编译错误！

// ✅ 但可以通过 Deref 读取
println!("{}", *pinned);  // Hello

Pin 的保证：

  • 📌 一旦数据被 Pin，就不能再移动它
  • 🔒 只能通过 Pin 提供的安全 API 访问
  • 🛡️ 编译器确保不会违反这个保证

/*
大多数类型都自动实现 Unpin：
- String, Vec, Box, Rc, Arc
- 所有基本类型（i32, bool, 等）
- 不包含自引用的结构体

不实现 Unpin 的类型：
- Future（在 async 块中捕获引用时）
- 显式使用 !Unpin 的类型
*/

// ✅ 这些类型是 Unpin
fn assert_unpin<T: Unpin>() {}

fn main() {
    assert_unpin::<String>();
    assert_unpin::<Vec<i32>>();
    assert_unpin::<Box<i32>>();
    
    // ❌ 这个不是 Unpin
    // assert_unpin::<std::future::Ready<i32>>();  // 编译错误
}

use std::marker::PhantomPinned;
use std::pin::Pin;

// 使用 PhantomPinned 使结构体 !Unpin
struct SelfReferential {
    data: String,
    ptr: *const String,
    _marker: PhantomPinned,  // 标记为 !Unpin
}

impl SelfReferential {
    // 必须通过 Pin 创建
    fn new() -> Pin<Box<Self>> {
        let data = String::from("Hello");
        let boxed = Box::new(SelfReferential {
            data,
            ptr: std::ptr::null(),  // 稍后设置
            _marker: PhantomPinned,
        });
        let mut pinned = Box::pin(boxed);
        
        // 安全：我们刚创建它，知道它不会移动
        let ptr = &pinned.data as *const String;
        unsafe {
            let this = self_referential::get_unchecked_mut(pinned.as_mut());
            this.ptr = ptr;
        }
        
        pinned
    }
}

// 场景 3：递归 Future
async fn recursive(n: u32) -> u32 {
    if n == 0 {
        return 0;
    }
    
    // 需要 Box::pin 来打破递归大小限制
    let future = Box::pin(recursive(n - 1));
    future.await + 1
}

use std::pin::Pin;
use std::marker::PhantomPinned;

struct SelfReferentialStruct {
    data: String,
    ptr: *const String,
    _marker: PhantomPinned,
}

impl SelfReferentialStruct {
    /// 创建固定的实例
    fn new() -> Pin<Box<Self>> {
        let data = String::from("Hello, Pin!");
        let boxed = Box::new(SelfReferentialStruct {
            data,
            ptr: std::ptr::null(),
            _marker: PhantomPinned,
        });
        
        let mut pinned = Box::pin(boxed);
        
        // 安全地设置自引用指针
        unsafe {
            let this = SelfReferentialStruct::get_unchecked_mut(pinned.as_mut());
            this.ptr = &this.data as *const String;
        }
        
        pinned
    }
    
    /// 安全地获取可变引用
    unsafe fn get_unchecked_mut(
        pinned: Pin<&mut Self>,
    ) -> &mut Self {
        // 这是安全的，因为我们知道数据不会被移动
        pinned.get_unchecked_mut()
    }
    
    /// 读取数据
    fn read(&self) -> &str {
        &self.data
    }
    
    /// 验证指针
    fn verify(&self) -> bool {
        unsafe {
            if self.ptr.is_null() {
                return false;
            }
            let data_ptr = &self.data as *const String;
            self.ptr == data_ptr
        }
    }
}

fn main() {
    let pinned = SelfReferentialStruct::new();
    
    println!("数据：{}", pinned.read());
    println!("指针有效：{}", pinned.verify());
    
    // pinned 不能移动，直到被 drop
}

use std::pin::Pin;
use std::marker::PhantomPinned;

struct Container {
    pinned_field: String,  // 假设这是 !Unpin
    normal_field: i32,
    _marker: PhantomPinned,
}

impl Container {
    // ❌ 错误：不能直接获取 pinned_field 的可变引用
    // fn get_pinned_mut(self: Pin<&mut Self>) -> &mut String {
    //     &mut self.pinned_field  // 编译错误！
    // }
    
    // ✅ 正确：使用投影
    fn get_pinned_mut(self: Pin<&mut Self>) -> Pin<&mut String> {
        unsafe {
            // 安全：我们只投影 pinned_field，不移动它
            let this = self.get_unchecked_mut();
            Pin::new_unchecked(&mut this.pinned_field)
        }
    }
    
    // ✅ normal_field 是 Unpin，可以直接访问
    fn get_normal_mut(self: Pin<&mut Self>) -> &mut i32 {
        unsafe {
            &mut self.get_unchecked_mut().normal_field
        }
    }
}

use std::pin::Pin;
use std::task::{Context, Poll, Waker};
use std::future::Future;
use std::sync::{Arc, Mutex};
use std::time::{Duration, Instant};

/// 一个简单的延迟 Future
struct Delay {
    when: Instant,
    waker: Option<Arc<Mutex<Waker>>>,
}

impl Delay {
    fn new(when: Instant) -> Self {
        Delay {
            when,
            waker: None,
        }
    }
}

impl Future for Delay {
    type Output = ();
    
    fn poll(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Self::Output> {
        // 获取可变引用（安全，因为我们没有移动 self）
        let this = self.get_mut();
        
        if Instant::now() >= this.when {
            // 时间到了，完成
            Poll::Ready(())
        } else {
            // 还没到，存储 waker 以便稍后唤醒
            *this.waker.get_or_insert_with(|| Arc::new(Mutex::new(cx.waker().clone()))) 
                = Arc::new(Mutex::new(cx.waker().clone()));
            
            Poll::Pending
        }
    }
}

/// 定时器（模拟）
struct Timer {
    callbacks: Arc<Mutex<Vec<Arc<Mutex<Waker>>>>>,
}

impl Timer {
    fn new() -> Self {
        Timer {
            callbacks: Arc::new(Mutex::new(Vec::new())),
        }
    }
    
    fn create_delay(&self, duration: Duration) -> Delay {
        let when = Instant::now() + duration;
        let mut delay = Delay::new(when);
        delay.waker = Some(Arc::new(Mutex::new(Waker::from(
            // 实际实现需要自定义 Waker
            std::task::Waker::from(std::sync::Arc::new(DummyWaker))
        ))));
        delay
    }
}

struct DummyWaker;
impl std::task::Wake for DummyWaker {
    fn wake(self: Arc<Self>) {}
}

#[tokio::main]
async fn main() {
    println!("⏱️ 等待 1 秒...");
    
    let delay = Delay::new(Instant::now() + Duration::from_secs(1));
    delay.await;
    
    println!("✅ 时间到！");
}

    let mut data = String::from("Hello");
    let data_ref = &mut data;

use std::marker::PhantomPinned;

    data: String,
    _marker: PhantomPinned,

fn trap3() {

        data: String::from("Hello"),
        _marker: PhantomPinned,

    let mut boxed = Box::pin(bad);

use std::pin::Pin;
use std::marker::PhantomPinned;
use std::ptr::NonNull;

/// 链表节点（自引用）
struct ListNode {
    value: i32,
    next: Option<NonNull<ListNode>>,
    _marker: PhantomPinned,
}

impl ListNode {
    fn new(value: i32) -> Pin<Box<Self>> {
        Box::pin(ListNode {
            value,
            next: None,
            _marker: PhantomPinned,
        })
    }
    
    fn set_next(mut self: Pin<&mut Self>, next: Pin<&mut Self>) {
        unsafe {
            let this = self.as_mut().get_unchecked_mut();
            let next_ptr = NonNull::new_unchecked(next.get_ref() as *const ListNode as *mut ListNode);
            this.next = Some(next_ptr);
        }
    }
}

/// 简单的固定链表
struct FixedList {
    head: Option<Pin<Box<ListNode>>>,
}

impl FixedList {
    fn new() -> Self {
        FixedList { head: None }
    }
    
    fn push_front(&mut self, value: i32) {
        let mut new_node = ListNode::new(value);
        
        // 安全地设置 next 指针
        if let Some(ref mut old_head) = self.head {
            // 需要特殊处理，这里简化
            // 实际实现需要更复杂的 Pin 管理
        }
        
        self.head = Some(new_node);
    }
}

fn main() {
    let mut list = FixedList::new();
    list.push_front(3);

Pin<P> 的核心保证：

如果 P: Deref<Target = T>，那么 Pin<P> 保证：
1. T 的数据不会被移动
2. 只能通过 Pin 提供的 API 访问 T
3. 这个保证在编译期强制执行

为什么重要？
- 自引用结构体依赖稳定的内存地址
- Future 在 await 点可能包含自引用
- 移动会破坏这些引用，导致未定义行为
*/

/*
Pin::get_unchecked_mut 是不安全的，因为：
1. 它允许获取可变引用
2. 如果之后移动了数据，会导致悬垂引用

安全使用的条件：
1. 你知道数据不会被移动
2. 你没有创建指向数据的其他指针
3. 你在 Pin 的"作用域"内使用

示例：
*/

use std::pin::Pin;

struct Safe {
    data: String,
}

impl Safe {
    fn method(mut self: Pin<&mut Self>) {
        // ✅ 安全：我们在 Pin 的作用域内
        let this = unsafe { self.as_mut().get_unchecked_mut() };
        this.data.push_str(" modified");
    }
}

今日感悟

Pin 和 Unpin 是 Rust 异步编程中最难理解的概念之一。今天的学习让我明白：

  1. Pin 的本质是"承诺不移动"：它不是魔法，而是一个编译期的保证。一旦你 Pin 了一个值，你就承诺不会移动它。

  2. 为什么 Future 需要 Pin：async/await 语法糖背后是状态机，状态机在挂起时可能包含自引用。Pin 保证了这些引用的安全性。

  3. 大多数时候你不需要直接处理 Pin：async/await 隐藏了大部分复杂性。只有当你需要实现自定义 Future 或处理自引用结构体时，才需要深入理解 Pin。

  4. 安全第一：get_unchecked_mut 是不安全的，但这不是因为它危险，而是因为编译器无法验证你的使用是否正确。只要你遵循规则，它就是安全的。

今天的难点在于理解"为什么"而不仅仅是"怎么用"。多画图、多思考内存布局，会有帮助。🧠
