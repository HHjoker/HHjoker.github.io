use std::collections::HashMap;

#[derive(Debug, Clone)]
struct LogEntry {
    date: String,
    time: String,
    level: String,
    ip: String,
    method: String,
    path: String,
    status: u16,
    duration_ms: u64,
}

fn parse_log(line: &str) -> Option<LogEntry> {
    let parts: Vec<&str> = line.split_whitespace().collect();
    if parts.len() < 8 {
        return None;
    }
    
    let status = parts[6].parse::<u16>().ok()?;
    let duration_str = parts[7].trim_end_matches("ms");
    let duration_ms = duration_str.parse::<u64>().ok()?;
    
    Some(LogEntry {
        date: parts[0].to_string(),
        time: parts[1].to_string(),
        level: parts[2].to_string(),
        ip: parts[3].to_string(),
        method: parts[4].to_string(),
        path: parts[5].to_string(),
        status,
        duration_ms,
    })
}

fn main() {
    let entries: Vec<LogEntry> = LOG_DATA
        .lines()
        .filter(|line| !line.trim().is_empty())
        .filter_map(parse_log)
        .collect();
    
    println!("成功解析 {} 条日志", entries.len());
    
    // 打印前 3 条验证
    for entry in entries.iter().take(3) {
        println!("{:?}", entry);
    }
}

fn main() {
    let entries: Vec<LogEntry> = LOG_DATA
        .lines()
        .filter(|line| !line.trim().is_empty())
        .filter_map(parse_log)
        .collect();
    
    // 总请求数
    let total_requests = entries.len();
    println!("总请求数：{}", total_requests);
    
    // 按级别统计
    let level_counts: HashMap<&str, usize> = entries.iter()
        .fold(HashMap::new(), |mut map, entry| {
            let level = entry.level.as_str();
            *map.entry(level).or_insert(0) += 1;
            map
        });
    
    println!("\n按级别统计：");
    for (level, count) in level_counts.iter() {
        let bar = "█".repeat(*count);
        println!("  {:>5}: {} ({})", level, bar, count);
    }
    
    // 按状态码分类
    let status_categories: HashMap<&str, Vec<&LogEntry>> = entries.iter()
        .fold(HashMap::new(), |mut map, entry| {
            let category = match entry.status {
                200..=299 => "2xx Success",
                300..=399 => "3xx Redirect",
                400..=499 => "4xx Client Error",
                500..=599 => "5xx Server Error",
                _ => "Other",
            };
            map.entry(category).or_default().push(entry);
            map
        });
    
    println!("\n按状态码分类：");
    for (category, reqs) in &status_categories {
        println!("  {}: {} 次请求", category, reqs.len());
    }
}

fn main() {
    let entries: Vec<LogEntry> = LOG_DATA
        .lines()
        .filter(|line| !line.trim().is_empty())
        .filter_map(parse_log)
        .collect();
    
    // 计算响应时间统计
    let durations: Vec<u64> = entries.iter()
        .map(|e| e.duration_ms)
        .collect();
    
    let sum: u64 = durations.iter().sum();
    let avg = sum as f64 / durations.len() as f64;
    let min = *durations.iter().min().unwrap();
    let max = *durations.iter().max().unwrap();
    
    println!("响应时间统计：");
    println!("  平均：{:.1}ms", avg);
    println!("  最小：{}ms", min);
    println!("  最大：{}ms", max);
    println!("  范围：{}ms", max - min);
    
    // 慢请求（> 1000ms）
    let slow_requests: Vec<&LogEntry> = entries.iter()
        .filter(|e| e.duration_ms > 1000)
        .collect();
    
    println!("\n慢请求（> 1000ms）：");
    for req in &slow_requests {
        println!("  [{}] {} {} {}ms", 
            req.level, req.method, req.path, req.duration_ms);
    }
    
    // 按路径统计平均响应时间
    let path_stats: HashMap<&str, (u64, usize)> = entries.iter()
        .fold(HashMap::new(), |mut map, entry| {
            let path = entry.path.as_str();
            let (total, count) = map.entry(path).or_insert((0, 0));
            *total += entry.duration_ms;
            *count += 1;
            map
        });
    
    println!("\n路径平均响应时间：");
    let mut path_avg: Vec<_> = path_stats.iter()
        .map(|(&path, &(total, count))| (path, total as f64 / count as f64))
        .collect();
    path_avg.sort_by(|a, b| b.1.partial_cmp(&a.1).unwrap());
    
    for (path, avg_ms) in path_avg.iter().take(5) {
        println!("  {:>20}: {:.1}ms", path, avg_ms);
    }
}

    let entries: Vec<LogEntry> = LOG_DATA
        .lines()
        .filter(|line| !line.trim().is_empty())
        .filter_map(parse_log)
        .collect();
    
    // 按 IP 统计请求次数
    let ip_counts: HashMap<&str, usize> = entries.iter()
        .map(|e| e.ip.as_str())
        .fold(HashMap::new(), |mut map, ip| {
            *map.entry(ip).or_insert(0) += 1;
            map
        });
    
    // 排序并取 Top 5
    let mut ip_ranking: Vec<_> = ip_counts.iter().collect();
    ip_ranking.sort_by(|a, b| b.1.cmp(a.1));
    
    println!("IP 访问 Top 5：");
    for (i, (ip, count)) in ip_ranking.iter().take(5).enumerate() {
        println!("  {}. {} - {} 次请求", i + 1, ip, count);
    }
    
    // 可疑 IP：多次 4xx/5xx 错误
    let error_ips: HashMap<&str, usize> = entries.iter()
        .filter(|e| e.status >= 400)
        .map(|e| e.ip.as_str())
        .fold(HashMap::new(), |mut map, ip| {
            *map.entry(ip).or_insert(0) += 1;
            map
        });
    
    let suspicious_ips: Vec<_> = error_ips.iter()
        .filter(|(_, count)| **count >= 2)
        .collect();
    
    println!("\n可疑 IP（多次错误请求）：");
    for (ip, count) in suspicious_ips {
        println!("  ⚠ {} - {} 次错误", ip, count);
    }
    
    // 高频访问 IP（可能爬虫）
    let frequent_ips: Vec<_> = ip_counts.iter()
        .filter(|(_, count)| **count >= 4)
        .collect();
    
    println!("\n高频访问 IP（≥4 次）：");
    for (ip, count) in frequent_ips {
        println!("  🔄 {} - {} 次请求", ip, count);
    }
}

fn main() {
    let entries: Vec<LogEntry> = LOG_DATA
        .lines()
        .filter(|line| !line.trim().is_empty())
        .filter_map(parse_log)
        .collect();
    
    // 按 API 端点分组
    let endpoint_stats: HashMap<&str, Vec<&LogEntry>> = entries.iter()
        .fold(HashMap::new(), |mut map, entry| {
            // 提取基础路径（去掉 /1, /2 等 ID）
            let base_path = entry.path.split('/')
                .take(3)
                .collect::<Vec<_>>()
                .join("/");
            map.entry(&base_path).or_default().push(entry);
            map
        });
    
    println!("API 端点统计：");
    for (endpoint, reqs) in &endpoint_stats {
        let success = reqs.iter().filter(|e| e.status < 400).count();
        let errors = reqs.len() - success;
        let avg_ms = reqs.iter()
            .map(|e| e.duration_ms)
            .sum::<u64>() as f64 / reqs.len() as f64;
        
        println!("  {}:", endpoint);
        println!("    请求：{}（成功 {}，失败 {}）", reqs.len(), success, errors);
        println!("    平均响应：{:.1}ms", avg_ms);
    }
    
    // 错误率最高的端点
    let error_rates: Vec<_> = endpoint_stats.iter()
        .map(|(&endpoint, reqs)| {
            let error_count = reqs.iter()
                .filter(|e| e.status >= 400)
                .count();
            let rate = error_count as f64 / reqs.len() as f64 * 100.0;
            (endpoint, rate, error_count, reqs.len())
        })
        .filter(|(_, rate, _, _)| *rate > 0.0)
        .collect();
    
    println!("\n错误率 > 0% 的端点：");
    for (endpoint, rate, errors, total) in error_rates {
        println!("  {}: {:.1}%（{} / {}）", endpoint, rate, errors, total);
    }
}

#[derive(Debug, Clone)]
struct LogEntry {
    level: String,
    ip: String,
    method: String,
    path: String,
    status: u16,
    duration_ms: u64,
}

fn parse_log(line: &str) -> Option<LogEntry> {
    let parts: Vec<&str> = line.split_whitespace().collect();
    if parts.len() < 8 { return None; }
    
    let status = parts[6].parse::<u16>().ok()?;
    let duration_ms = parts[7].trim_end_matches("ms").parse::<u64>().ok()?;
    
    Some(LogEntry {
        level: parts[2].to_string(),
        ip: parts[3].to_string(),
        method: parts[4].to_string(),
        path: parts[5].to_string(),
        status,
        duration_ms,

fn analyze_logs(log_data: &str) {
    let entries: Vec<LogEntry> = log_data
        .lines()
        .filter(|l| !l.trim().is_empty())
        .filter_map(parse_log)
        .collect();
    
    if entries.is_empty() {
        println!("没有可分析的日志");
        return;
    }
    
    println!("═══════════════════════════════════════════");
    println!("           日志分析报告");
    println!("═══════════════════════════════════════════");
    println!("总请求数：{}", entries.len());
    
    // 级别分布
    let level_counts: HashMap<_, _> = entries.iter()
        .map(|e| e.level.as_str())
        .fold(HashMap::new(), |mut m, l| { *m.entry(l).or_insert(0) += 1; m });
    
    println!("\n📊 级别分布：");
    for level in &["INFO", "DEBUG", "WARN", "ERROR"] {
        if let Some(&count) = level_counts.get(level) {
            let pct = count as f64 / entries.len() as f64 * 100.0;
            println!("  {:>5}: {:>3} ({:.1}%)", level, count, pct);
        }
    }
    
    // 响应时间
    let durations: Vec<_> = entries.iter().map(|e| e.duration_ms).collect();
    let avg_ms = durations.iter().sum::<u64>() as f64 / durations.len() as f64;
    let p95_idx = (durations.len() as f64 * 0.95) as usize;
    let mut sorted = durations.clone();
    sorted.sort_unstable();
    let p95 = sorted.get(p95_idx.min(sorted.len() - 1)).unwrap_or(&0);
    
    println!("\n⏱️  响应时间：");
    println!("  平均：{:.1}ms", avg_ms);
    println!("  P95：{}ms", p95);
    println!("  最小：{}ms", sorted.first().unwrap_or(&0));
    println!("  最大：{}ms", sorted.last().unwrap_or(&0));
    
    // 慢请求
    let slow: Vec<_> = entries.iter()
        .filter(|e| e.duration_ms > 1000)
        .collect();
    if !slow.is_empty() {
        println!("\n🐌 慢请求（> 1000ms）：");
        for req in &slow {
            println!("  [{}] {} {} {}ms", req.level, req.method, req.path, req.duration_ms);
        }
    }
    
    // 错误统计
    let errors: Vec<_> = entries.iter()
        .filter(|e| e.status >= 400)
        .collect();
    if !errors.is_empty() {
        println!("\n❌ 错误请求：");
        let error_by_status: HashMap<_, _> = errors.iter()
            .map(|e| e.status)
            .fold(HashMap::new(), |mut m, s| { *m.entry(s).or_insert(0) += 1; m });
        for (status, count) in error_by_status.iter() {
            println!("  HTTP {}: {} 次", status, count);
        }
    }
    
    println!("\n═══════════════════════════════════════════");
}

use std::collections::HashMap;

fn word_frequency(text: &str) -> Vec<(String, usize)> {
    // 分词 → 清理 → 统计 → 排序
    let freq: HashMap<String, usize> = text
        .to_lowercase()
        .split_whitespace()
        .map(|word| word.chars()
            .filter(|c| c.is_alphanumeric() || *c == '\'' || *c == '-')
            .collect::<String>())
        .filter(|word| !word.is_empty())
        .fold(HashMap::new(), |mut map, word| {
            *map.entry(word).or_insert(0) += 1;
            map
        });
    
    let mut freq_vec: Vec<_> = freq.into_iter().collect();
    freq_vec.sort_by(|a, b| b.1.cmp(&a.1).then_with(|| a.0.cmp(&b.0)));
    freq_vec
}

fn unique_words(text: &str) -> Vec<String> {
    text.split_whitespace()
        .map(|w| w.to_lowercase())
        .map(|w| w.chars()
            .filter(|c| c.is_alphanumeric())
            .collect::<String>())
        .filter(|w| !w.is_empty())
        .collect::<std::collections::HashSet<_>>()
        .into_iter()
        .collect()
}

fn longest_words(text: &str, n: usize) -> Vec<(String, usize)> {
    let mut words: Vec<_> = text
        .split_whitespace()
        .map(|w| {
            let clean = w.chars()
                .filter(|c| c.is_alphanumeric())
                .collect::<String>();
            (clean, clean.len())
        })
        .filter(|(w, _)| !w.is_empty())
        .collect();
    
    words.sort_by(|a, b| b.1.cmp(&a.1).then_with(|| a.0.cmp(&b.0)));
    words.into_iter().take(n).collect()
}

fn n_grams(text: &str, n: usize) -> Vec<(String, usize)> {
    let words: Vec<_> = text
        .to_lowercase()
        .split_whitespace()
        .map(|w| w.chars()
            .filter(|c| c.is_alphanumeric())
            .collect::<String>())
        .filter(|w| !w.is_empty())
        .collect();
    
    if words.len() < n {
        return Vec::new();
    }
    
    let mut freq: HashMap<String, usize> = HashMap::new();
    for window in words.windows(n) {
        let gram = window.join(" ");
        *freq.entry(gram).or_insert(0) += 1;
    }
    
    let mut result: Vec<_> = freq.into_iter().collect();
    result.sort_by(|a, b| b.1.cmp(&a.1).then_with(|| a.0.cmp(&b.0)));
    result
}

    let mut sum1 = 0i64;
    for &x in &data {

            sum1 += x as i64;

    }

    let sum2: i64 = data.iter()
        .filter(|&&x| x % 2 == 0)
        .map(|&x| x as i64)
        .sum();
    
    // 方法 3：单次遍历
    let sum3: i64 = data.iter()
        .filter_map(|&&x| {
            if x % 2 == 0 { Some(x as i64) } else { None }
        })
        .sum();
    
    println!("循环：{}", sum1);
    println!("迭代器链：{}", sum2);
    println!("filter_map：{}", sum3);
    println!("结果一致：{}", sum1 == sum2 && sum2 == sum3);

    let result3: Vec<i32> = data.iter()

use std::collections::HashMap;

struct PivotTable<K, V> {
    row_groups: HashMap<K, Vec<V>>,
    aggregator: fn(&[V]) -> f64,
}

impl<K: std::hash::Hash + Eq + std::fmt::Display, V: Copy + Default> PivotTable<K, V> {
    fn new(aggregator: fn(&[V]) -> f64) -> Self {
        PivotTable {
            row_groups: HashMap::new(),
            aggregator,
        }
    }
    
    fn add(&mut self, key: K, value: V) {
        self.row_groups.entry(key).or_default().push(value);
    }
    
    fn summarize(&self) -> Vec<(String, f64)> {
        // 使用迭代器适配器生成汇总结果
        // 提示：使用 iter + map + 排序
        let mut result: Vec<_> = self.row_groups.iter()
            .map(|(key, values)| (key.to_string(), (self.aggregator)(values)))
            .collect();
        result.sort_by(|a, b| b.1.partial_cmp(&a.1).unwrap());
        result
    }
}

fn average(values: &[i32]) -> f64 {
    values.iter().sum::<i32>() as f64 / values.len() as f64
}

fn main() {
    let mut table = PivotTable::new(average);
    let data = vec![
        ("Engineering", 85),
        ("Marketing", 78),
        ("Engineering", 92),
        ("HR", 88),
        ("Marketing", 82),
        ("Engineering", 76),
    ];
    
    for (dept, score) in data {
        table.add(dept.to_string(), score);
    }
    
    for (dept, avg) in table.summarize() {
        println!("{}: {:.1}", dept, avg);
    }
}

// 使用迭代器实现滑动窗口统计
fn window_stats(data: &[f64], window: usize) -> Vec<(f64, f64, f64)> {
    // 返回每个窗口的 (min, avg, max)
    // 提示：使用 windows() 方法
    data.windows(window)
        .map(|w| {
            let min = w.iter().cloned().fold(f64::INFINITY, f64::min);
            let max = w.iter().cloned().fold(f64::NEG_INFINITY, f64::max);
            let avg = w.iter().sum::<f64>() / w.len() as f64;
            (min, avg, max)
        })
        .collect()
}

fn main() {
    let data = vec![1.0, 3.0, 2.0, 5.0, 4.0, 6.0, 3.0];
    let stats = window_stats(&data, 3);
    
    println!("原始数据：{:?}", data);
    println!("窗口统计（窗口大小=3）：");
    for (i, (min, avg, max)) in stats.iter().enumerate() {
        println!("  窗口 {}: [{:.1}, {:.1}, {:.1}] → min={:.1}, avg={:.1}, max={:.1}",
                 i + 1, data[i], data[i+1], data[i+2], min, avg, max);
    }
}

// 模拟 SQL 的 GROUP BY + 聚合
fn sql_like_aggregate<'a>(
    records: &'a [(String, String, i32)],  // (department, position, salary)
    group_by: fn(&&(String, String, i32)) -> &str,
) -> Vec<(String, Vec<&'a (String, String, i32)>)> {
    // 使用迭代器实现 GROUP BY
    use std::collections::HashMap;
    
    let groups: HashMap<&str, Vec<&'a (String, String, i32)>> = records.iter()
        .fold(HashMap::new(), |mut map, record| {
            let key = group_by(record);
            map.entry(key).or_default().push(record);
            map
        });
    
    let mut result: Vec<_> = groups.into_iter()
        .map(|(key, records)| (key.to_string(), records))
        .collect();
    result.sort_by(|a, b| a.0.cmp(&b.0));
    result
}

fn main() {
    let records = vec![
        ("Engineering".to_string(), "Senior".to_string(), 120000),
        ("Engineering".to_string(), "Junior".to_string(), 80000),
        ("Marketing".to_string(), "Manager".to_string(), 95000),
        ("Marketing".to_string(), "Senior".to_string(), 88000),
        ("HR".to_string(), "Manager".to_string(), 85000),
        ("Engineering".to_string(), "Staff".to_string(), 145000),
    ];
    
    let by_dept = sql_like_aggregate(&records, |r| &r.0);
    
    println!("GROUP BY department:");
    for (dept, members) in by_dept {
        let total_salary: i32 = members.iter().map(|r| r.2).sum();
        let avg_salary = total_salary / members.len() as i32;
        println!("  {}: {} 人, 总薪资 ${}, 平均 ${}", 
                 dept, members.len(), total_salary, avg_salary);
    }
}

fn stream_processing() {

        .scan(Vec::new(), |window, value| {
            window.push(value);
            if window.len() > window_size {
                window.remove(0);
            }
            Some(window.iter().sum::<i32>() as f64 / window.len() as f64)
        })
        .collect();
    
    println!("生成了 {} 个随机数", moving_avg.len());
    println!("最后 5 个滑动平均：{:?}", &moving_avg[moving_avg.len()-5..]);
