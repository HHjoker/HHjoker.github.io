/// 表示 2D 空间中的向量
#[derive(Debug, Clone, Copy, PartialEq)]
struct Vector2D<T> {
    x: T,
    y: T,
}

impl<T> Vector2D<T> 
where
    T: std::ops::Add<Output = T> + Copy,
{
    /// 向量加法
    fn add(self, other: Self) -> Self {
        Vector2D {
            x: self.x + other.x,
            y: self.y + other.y,
        }
    }
}

impl<T> Vector2D<T>
where
    T: std::ops::Sub<Output = T> + Copy,
{
    /// 向量减法
    fn sub(self, other: Self) -> Self {
        Vector2D {
            x: self.x - other.x,
            y: self.y - other.y,
        }
    }
}

impl<T> Vector2D<T>
where
    T: std::ops::Mul<Output = T> + Copy,
{
    /// 标量乘法
    fn scale(self, scalar: T) -> Self {
        Vector2D {
            x: self.x * scalar,
            y: self.y * scalar,
        }
    }
}

fn main() {
    // 整数向量
    let v1 = Vector2D { x: 3, y: 4 };
    let v2 = Vector2D { x: 1, y: 2 };
    
    let sum = v1.add(v2);
    println!("向量和：{:?}", sum);  // Vector2D { x: 4, y: 6 }
    
    let scaled = v1.scale(2);
    println!("缩放后：{:?}", scaled);  // Vector2D { x: 6, y: 8 }
    
    // 浮点数向量
    let vf1 = Vector2D { x: 1.5, y: 2.5 };
    let vf2 = Vector2D { x: 0.5, y: 0.5 };
    
    let vsum = vf1.add(vf2);
    println!("浮点向量和：{:?}", vsum);  // Vector2D { x: 2.0, y: 3.0 }
}

/// 二叉树节点
#[derive(Debug)]
enum Tree<T> {
    Empty,
    Node {
        value: T,
        left: Box<Tree<T>>,
        right: Box<Tree<T>>,
    },
}

impl<T> Tree<T> {
    /// 创建空树
    fn new() -> Self {
        Tree::Empty
    }
    
    /// 创建叶子节点
    fn leaf(value: T) -> Self {
        Tree::Node {
            value,
            left: Box::new(Tree::Empty),
            right: Box::new(Tree::Empty),
        }
    }
}

impl<T: Ord> Tree<T> {
    /// 插入值（二叉搜索树）
    fn insert(self, value: T) -> Self {
        match self {
            Tree::Empty => Tree::leaf(value),
            Tree::Node { value: current, left, right } => {
                if value < current {
                    Tree::Node {
                        value: current,
                        left: Box::new(left.insert(value)),
                        right,
                    }
                } else {
                    Tree::Node {
                        value: current,
                        left,
                        right: Box::new(right.insert(value)),
                    }
                }
            }
        }
    }
    
    /// 查找值
    fn contains(&self, value: &T) -> bool {
        match self {
            Tree::Empty => false,
            Tree::Node { value: current, left, right } => {
                if value == current {
                    true
                } else if value < current {
                    left.contains(value)
                } else {
                    right.contains(value)
                }
            }
        }
    }
}

fn main() {
    // 创建整数树
    let mut tree = Tree::new();
    tree = tree.insert(5);
    tree = tree.insert(3);
    tree = tree.insert(7);
    tree = tree.insert(1);
    tree = tree.insert(9);
    
    println!("树结构：{:?}", tree);
    println!("包含 3: {}", tree.contains(&3));  // true
    println!("包含 4: {}", tree.contains(&4));  // false
    
    // 字符串树
    let mut str_tree = Tree::new();
    str_tree = str_tree.insert("banana");
    str_tree = str_tree.insert("apple");
    str_tree = str_tree.insert("cherry");
    
    println!("字符串树：{:?}", str_tree);
    println!("包含 'apple': {}", str_tree.contains(&"apple"));  // true
}

/// 单链表节点
#[derive(Debug)]
struct LinkedList<T> {
    head: Option<Box<Node<T>>>,
}

#[derive(Debug)]
struct Node<T> {
    value: T,
    next: Option<Box<Node<T>>>,
}

impl<T> LinkedList<T> {
    /// 创建空链表
    fn new() -> Self {
        LinkedList { head: None }
    }
    
    /// 在头部插入元素
    fn push_front(&mut self, value: T) {
        let new_node = Box::new(Node {
            value,
            next: self.head.take(),
        });
        self.head = Some(new_node);
    }
    
    /// 从头部弹出元素
    fn pop_front(&mut self) -> Option<T> {
        match self.head.take() {
            Some(node) => {
                self.head = node.next;
                Some(node.value)
            }
            None => None,
        }
    }
    
    /// 检查是否为空
    fn is_empty(&self) -> bool {
        self.head.is_none()
    }
    
    /// 获取链表长度
    fn len(&self) -> usize {
        let mut count = 0;
        let mut current = &self.head;
        while let Some(node) = current {
            count += 1;
            current = &node.next;
        }
        count
    }
}

impl<T: Clone> LinkedList<T> {
    /// 转换为 Vec
    fn to_vec(&self) -> Vec<T> {
        let mut result = Vec::new();
        let mut current = &self.head;
        while let Some(node) = current {
            result.push(node.value.clone());
            current = &node.next;
        }
        result
    }
}

fn main() {
    // 整数链表
    let mut list = LinkedList::new();
    list.push_front(3);
    list.push_front(2);
    list.push_front(1);
    
    println!("链表：{:?}", list);
    println!("长度：{}", list.len());  // 3
    println!("转 Vec: {:?}", list.to_vec());  // [1, 2, 3]
    
    // 弹出元素
    if let Some(val) = list.pop_front() {
        println!("弹出：{}", val);  // 1
    }
    
    // 字符串链表
    let mut str_list = LinkedList::new();
    str_list.push_front(String::from("world"));
    str_list.push_front(String::from("hello"));
    
    println!("字符串链表：{:?}", str_list.to_vec());
}

struct Rectangle<T> {
    width: T,
    height: T,
}

// 为所有 T 实现的方法（不需要额外约束）
impl<T> Rectangle<T> {
    fn new(width: T, height: T) -> Self {
        Rectangle { width, height }
    }
    
    fn get_width(&self) -> &T {
        &self.width
    }
    
    fn get_height(&self) -> &T {
        &self.height
    }
}

// 只为特定约束的 T 实现的方法
impl<T: std::ops::Mul<Output = T> + Copy> Rectangle<T> {
    fn area(&self) -> T {
        self.width * self.height
    }
}

impl<T: std::fmt::Debug> Rectangle<T> {
    fn print(&self) {
        println!("Rectangle: {:?}", self);
    }
}

fn main() {
    // 整数矩形
    let rect_int = Rectangle::new(10, 20);
    println!("宽度：{}", rect_int.get_width());
    println!("面积：{}", rect_int.area());  // 200
    
    // 浮点数矩形
    let rect_float = Rectangle::new(5.5, 3.0);
    println!("面积：{}", rect_float.area());  // 16.5
}

struct Container<T> {
    value: T,
}

// impl 块 1：无约束
impl<T> Container<T> {
    fn new(value: T) -> Self {
        Container { value }
    }
    
    fn get_value(&self) -> &T {
        &self.value
    }
}

// impl 块 2：需要 Clone
impl<T: Clone> Container<T> {
    fn clone_value(&self) -> T {
        self.value.clone()
    }
}

// impl 块 3：需要 Debug
impl<T: std::fmt::Debug> Container<T> {
    fn print(&self) {
        println!("值：{:?}", self.value);
    }
}

// impl 块 4：需要 Default
impl<T: Default> Container<T> {
    fn empty() -> Self {
        Container { value: T::default() }
    }
}

fn main() {
    let c1 = Container::new(42);
    c1.print();  // 值：42
    
    let c2 = Container::new(String::from("hello"));
    let cloned = c2.clone_value();
    println!("克隆：{}", cloned);
    
    let c3: Container<Vec<i32>> = Container::empty();
    println!("空容器：{:?}", c3.get_value());  // []
}

struct Pair<T> {
    x: T,
    y: T,
}

impl<T> Pair<T> {
    fn new(x: T, y: T) -> Self {
        Pair { x, y }
    }
}

impl<T: std::fmt::Display + PartialOrd> Pair<T> {
    /// 比较两个值并打印结果
    fn cmp_display(&self) {
        if self.x >= self.y {
            println!("较大值是：{}", self.x);
        } else {
            println!("较大值是：{}", self.y);
        }
    }
}

impl<T, U> Pair<T> 
where
    T: Clone,
{
    /// 将 x 转换为另一种类型
    fn map_x<U>(self, f: impl FnOnce(T) -> U) -> Pair<U> {
        Pair {
            x: f(self.x),
            y: self.y as U,  // 这需要 T 能转为 U
        }
    }
}

// 更好的版本
struct Wrapper<T> {
    value: T,
}

impl<T> Wrapper<T> {
    fn new(value: T) -> Self {
        Wrapper { value }
    }
    
    /// 方法有自己的泛型参数 U
    fn map<U, F>(self, f: F) -> Wrapper<U>
    where
        F: FnOnce(T) -> U,
    {
        Wrapper { value: f(self.value) }
    }
}

fn main() {
    let wrapper = Wrapper::new(5);
    
    // 转换为字符串
    let string_wrapper = wrapper.map(|x| format!("数字：{}", x));
    println!("{}", string_wrapper.value);  // 数字：5
    
    // 链式调用
    let result = Wrapper::new(10)
        .map(|x| x * 2)
        .map(|x| x + 5)
        .map(|x| format!("结果：{}", x));
    
    println!("{}", result.value);  // 结果：25
}

/// 基于 Vec 的泛型栈
#[derive(Debug)]
struct Stack<T> {
    items: Vec<T>,
}

impl<T> Stack<T> {
    /// 创建空栈
    fn new() -> Self {
        Stack { items: Vec::new() }
    }
    
    /// 压栈
    fn push(&mut self, item: T) {
        self.items.push(item);
    }
    
    /// 弹栈
    fn pop(&mut self) -> Option<T> {
        self.items.pop()
    }
    
    /// 查看栈顶元素
    fn peek(&self) -> Option<&T> {
        self.items.last()
    }
    
    /// 检查是否为空
    fn is_empty(&self) -> bool {
        self.items.is_empty()
    }
    
    /// 栈的大小
    fn size(&self) -> usize {
        self.items.len()
    }
}

impl<T: Clone> Stack<T> {
    /// 克隆栈
    fn clone_stack(&self) -> Self {
        Stack {
            items: self.items.clone(),
        }
    }
}

fn main() {
    // 整数栈
    let mut int_stack = Stack::new();
    int_stack.push(1);
    int_stack.push(2);
    int_stack.push(3);
    
    println!("栈顶：{:?}", int_stack.peek());  // Some(3)
    println!("弹出：{:?}", int_stack.pop());   // Some(3)
    println!("大小：{}", int_stack.size());    // 2
    
    // 字符串栈
    let mut str_stack = Stack::new();
    str_stack.push(String::from("first"));
    str_stack.push(String::from("second"));
    
    while let Some(item) = str_stack.pop() {
        println!("弹出：{}", item);
    }
}

use std::collections::VecDeque;

/// 基于 VecDeque 的泛型队列
#[derive(Debug)]
struct Queue<T> {
    items: VecDeque<T>,
}

impl<T> Queue<T> {
    /// 创建空队列
    fn new() -> Self {
        Queue {
            items: VecDeque::new(),
        }
    }
    
    /// 入队（尾部添加）
    fn enqueue(&mut self, item: T) {
        self.items.push_back(item);
    }
    
    /// 出队（头部移除）
    fn dequeue(&mut self) -> Option<T> {
        self.items.pop_front()
    }
    
    /// 查看队首元素
    fn front(&self) -> Option<&T> {
        self.items.front()
    }
    
    /// 查看队尾元素
    fn back(&self) -> Option<&T> {
        self.items.back()
    }
    
    /// 队列大小
    fn size(&self) -> usize {
        self.items.len()
    }
    
    /// 是否为空
    fn is_empty(&self) -> bool {
        self.items.is_empty()
    }
}

fn main() {
    let mut queue = Queue::new();
    
    queue.enqueue(1);
    queue.enqueue(2);
    queue.enqueue(3);
    
    println!("队首：{:?}", queue.front());  // Some(1)
    println!("队尾：{:?}", queue.back());   // Some(3)
    
    while let Some(item) = queue.dequeue() {
        println!("出队：{}", item);  // 1, 2, 3
    }
}

/// 泛型键值存储
#[derive(Debug)]
struct KeyValueStore<K, V> {
    data: HashMap<K, V>,
}

impl<K, V> KeyValueStore<K, V>
where
    K: Eq + Hash,
{
    /// 创建空存储
    fn new() -> Self {
        KeyValueStore {
            data: HashMap::new(),
        }
    }
    
    /// 插入键值对
    fn insert(&mut self, key: K, value: V) -> Option<V> {
        self.data.insert(key, value)
    }
    
    /// 获取值
    fn get(&self, key: &K) -> Option<&V> {
        self.data.get(key)
    }
    
    /// 获取可变引用
    fn get_mut(&mut self, key: &K) -> Option<&mut V> {
        self.data.get_mut(key)
    }
    
    /// 删除键值对
    fn remove(&mut self, key: &K) -> Option<V> {
        self.data.remove(key)
    }
    
    /// 检查是否包含键
    fn contains_key(&self, key: &K) -> bool {
        self.data.contains_key(key)
    }
    
    /// 存储大小
    fn len(&self) -> usize {
        self.data.len()
    }
    
    /// 是否为空
    fn is_empty(&self) -> bool {
        self.data.is_empty()
    }
}

impl<K, V> Default for KeyValueStore<K, V>
where
    K: Eq + Hash,
{
    fn default() -> Self {
        Self::new()
    }
}

fn main() {
    // 字符串键，整数值的存储
    let mut store: KeyValueStore<String, i32> = KeyValueStore::new();
    
    store.insert(String::from("age"), 25);
    store.insert(String::from("score"), 100);
    store.insert(String::from("level"), 5);
    
    println!("年龄：{:?}", store.get(&String::from("age")));  // Some(25)
    println!("包含 'score': {}", store.contains_key(&String::from("score")));  // true
    
    // 更新值
    if let Some(score) = store.get_mut(&String::from("score")) {
        *score = 150;

    println!("更新后分数：{:?}", store.get(&String::from("score")));  // Some(150)
    
    // 整数键，字符串值的存储

    id_store.insert(2, String::from("Bob"));

/// 自定义的 Optional 类型（类似标准库的 Option）
#[derive(Debug, Clone, Copy, PartialEq, Eq)]
enum Optional<T> {
    Some(T),
    None,
}

impl<T> Optional<T> {
    /// 检查是否有值
    fn is_some(&self) -> bool {
        matches!(self, Optional::Some(_))
    }
    
    /// 检查是否无值
    fn is_none(&self) -> bool {
        matches!(self, Optional::None)
    }
    
    /// 获取值（如果没有则 panic）
    fn unwrap(self) -> T {
        match self {
            Optional::Some(value) => value,
            Optional::None => panic!("调用了 Optional::unwrap() 但值为 None"),
        }
    }
    
    /// 获取值或默认值
    fn unwrap_or(self, default: T) -> T {
        match self {
            Optional::Some(value) => value,
            Optional::None => default,
        }
    }
    
    /// 获取值或通过函数计算默认值
    fn unwrap_or_else<F>(self, f: F) -> T
    where
        F: FnOnce() -> T,
    {
        match self {
            Optional::Some(value) => value,
            Optional::None => f(),
        }
    }
}

impl<T: Clone> Optional<T> {
    /// 获取值的引用
    fn as_ref(&self) -> Optional<&T> {
        match self {
            Optional::Some(value) => Optional::Some(value),
            Optional::None => Optional::None,
        }
    }
}

impl<T, U> Optional<T> {
    /// 映射函数
    fn map<U, F>(self, f: F) -> Optional<U>
    where
        F: FnOnce(T) -> U,
    {
        match self {
            Optional::Some(value) => Optional::Some(f(value)),
            Optional::None => Optional::None,
        }
    }
    
    /// 扁平映射
    fn and_then<U, F>(self, f: F) -> Optional<U>
    where
        F: FnOnce(T) -> Optional<U>,
    {
        match self {
            Optional::Some(value) => f(value),
            Optional::None => Optional::None,
        }
    }
}

fn main() {
    let some_val: Optional<i32> = Optional::Some(42);
    let none_val: Optional<i32> = Optional::None;
    
    println!("is_some: {}", some_val.is_some());  // true
    println!("is_none: {}", none_val.is_none());  // true
    
    println!("unwrap: {}", some_val.unwrap());  // 42
    println!("unwrap_or: {}", none_val.unwrap_or(0));  // 0
    
    // 链式调用
    let result = Optional::Some(5)
        .map(|x| x * 2)
        .and_then(|x| if x > 5 { Optional::Some(x) } else { Optional::None })
        .unwrap_or_else(|| 0);
    
    println!("结果：{}", result);  // 10
}

struct RingBuffer<T> {
    data: Vec<Option<T>>,
    capacity: usize,
    read_pos: usize,
    write_pos: usize,
    len: usize,
}

impl<T> RingBuffer<T> {
    fn new(capacity: usize) -> Self {
        RingBuffer {
            data: (0..capacity).map(|_| None).collect(),
            capacity,
            read_pos: 0,
            write_pos: 0,
            len: 0,
        }
    }
    
    fn push(&mut self, value: T) {
        if self.len < self.capacity {
            self.data[self.write_pos] = Some(value);
            self.len += 1;
        } else {
            // 覆盖最旧的数据
            self.data[self.write_pos] = Some(value);
            self.read_pos = (self.read_pos + 1) % self.capacity;
        }
        self.write_pos = (self.write_pos + 1) % self.capacity;
    }
    
    fn pop(&mut self) -> Option<T> {
        if self.len == 0 {
            return None;
        }
        let value = self.data[self.read_pos].take();
        self.read_pos = (self.read_pos + 1) % self.capacity;
        self.len -= 1;
        value
    }
}
