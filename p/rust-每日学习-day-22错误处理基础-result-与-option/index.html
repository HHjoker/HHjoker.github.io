/*

其他语言：
- null / nil（但可能导致空指针异常）
- 特殊值（如 -1，但可能冲突）

Rust 的方案：
- Option<T> 强制你处理"没有值"的情况
- 编译器确保你不会忘记检查

// ❌ 错误：不能直接将 Option<T> 当作 T 使用
fn wrong() {
    let opt: Option<i32> = Some(5);
    // let x: i32 = opt;  // 编译错误！
}

// ✅ 正确：必须显式处理
fn right() {
    let opt: Option<i32> = Some(5);
    
    // 方式 1：模式匹配
    match opt {
        Some(x) => println!("值：{}", x),
        None => println!("空"),
    }
    
    // 方式 2：使用 unwrap（可能 panic）
    // let x = opt.unwrap();  // ⚠️ 如果是 None 会 panic
    
    // 方式 3：提供默认值
    let x = opt.unwrap_or(0);  // 如果是 None，返回 0
}

fn option_methods() {
    let some = Some(5);
    let none: Option<i32> = None;
    
    // is_some / is_none
    assert!(some.is_some());
    assert!(none.is_none());
    
    // unwrap - 获取值，None 时 panic
    assert_eq!(some.unwrap(), 5);
    // none.unwrap();  // ❌ panic!
    
    // unwrap_or - 提供默认值
    assert_eq!(some.unwrap_or(0), 5);
    assert_eq!(none.unwrap_or(0), 0);
    
    // unwrap_or_else - 延迟计算默认值
    let result = none.unwrap_or_else(|| {
        println!("计算默认值...");
        42
    });
    assert_eq!(result, 42);
    
    // expect - 类似 unwrap，但可自定义错误信息
    // some.expect("期望有值");
    // none.expect("期望有值");  // panic: 期望有值
    
    // map - 转换内部的值
    let doubled = some.map(|x| x * 2);
    assert_eq!(doubled, Some(10));
    
    let none_doubled = none.map(|x| x * 2);
    assert_eq!(none_doubled, None);  // None 保持不变
    
    // map_or - map + 默认值
    let result = some.map_or(0, |x| x * 2);
    assert_eq!(result, 10);
    
    let result = none.map_or(0, |x| x * 2);
    assert_eq!(result, 0);
    
    // and_then - 链式调用，返回 Option
    let result = some.and_then(|x| Some(x * 2));
    assert_eq!(result, Some(10));
    
    let result = none.and_then(|x| Some(x * 2));
    assert_eq!(result, None);
    
    // filter - 过滤
    let result = some.filter(|&x| x > 3);
    assert_eq!(result, Some(5));
    
    let result = some.filter(|&x| x > 10);
    assert_eq!(result, None);
    
    // ok_or - 转换为 Result
    let result: Result<i32, &str> = some.ok_or("没有值");
    assert_eq!(result, Ok(5));
    
    let result: Result<i32, &str> = none.ok_or("没有值");
    assert_eq!(result, Err("没有值"));
}

/// 在 HashMap 中查找值
use std::collections::HashMap;

fn find_in_map() {
    let mut scores = HashMap::new();
    scores.insert("Alice", 95);
    scores.insert("Bob", 87);
    
    // get 返回 Option<&V>
    let alice_score: Option<&i32> = scores.get("Alice");
    
    match alice_score {
        Some(score) => println!("Alice 的分数：{}", score),
        None => println!("找不到 Alice"),
    }
    
    // 链式调用
    let bob_score = scores
        .get("Bob")
        .map(|s| s * 2)  // 分数翻倍
        .unwrap_or(0);
    
    println!("Bob 的翻倍分数：{}", bob_score);
}

/// 解析字符串为数字
fn parse_number(input: &str) -> Option<i32> {
    input.parse::<i32>().ok()  // Result -> Option
}

fn parse_example() {
    let valid = parse_number("42");
    assert_eq!(valid, Some(42));
    
    let invalid = parse_number("not a number");
    assert_eq!(invalid, None);
    
    // 提供默认值
    let number = parse_number("invalid").unwrap_or(0);
    assert_eq!(number, 0);
}

/// 查找第一个满足条件的元素
fn find_first() {
    let numbers = vec![1, 2, 3, 4, 5, 6];
    
    // find 返回 Option<&T>
    let first_even = numbers.iter().find(|&&x| x % 2 == 0);
    
    match first_even {
        Some(n) => println!("第一个偶数：{}", n),
        None => println!("没有偶数"),
    }
}

// Result 的定义（简化版）
enum Result<T, E> {
    Ok(T),   // 成功，包含值
    Err(E),  // 失败，包含错误信息
}

// 使用示例
use std::fs::File;

fn main() {
    // File::open 返回 Result<File, io::Error>
    let file_result = File::open("hello.txt");
    
    match file_result {
        Ok(file) => println!("文件打开成功"),
        Err(error) => println!("打开文件失败：{:?}", error),
    }
}

fn result_methods() {
    let ok: Result<i32, &str> = Ok(5);
    let err: Result<i32, &str> = Err("出错了");
    
    // is_ok / is_err
    assert!(ok.is_ok());
    assert!(err.is_err());
    
    // unwrap - 获取值，Err 时 panic
    assert_eq!(ok.unwrap(), 5);
    // err.unwrap();  // ❌ panic!
    
    // unwrap_or - 提供默认值
    assert_eq!(ok.unwrap_or(0), 5);
    assert_eq!(err.unwrap_or(0), 0);
    
    // unwrap_or_else - 延迟计算
    let result = err.unwrap_or_else(|e| {
        println!("错误：{}", e);
        42
    });
    assert_eq!(result, 42);
    
    // expect - 自定义错误信息
    // ok.expect("期望成功");
    // err.expect("期望成功");  // panic: 期望成功：出错了
    
    // map - 转换 Ok 值
    let doubled = ok.map(|x| x * 2);
    assert_eq!(doubled, Ok(10));
    
    let doubled = err.map(|x| x * 2);
    assert_eq!(doubled, Err("出错了"));  // Err 保持不变
    
    // map_err - 转换错误
    let converted = err.map_err(|e| format!("错误：{}", e));
    assert_eq!(converted, Err("错误：出错了".to_string()));
    
    // and_then - 链式调用
    let result = ok.and_then(|x| Ok(x * 2));
    assert_eq!(result, Ok(10));
    
    let result = err.and_then(|x| Ok(x * 2));
    assert_eq!(result, Err("出错了"));  // 短路
    
    // or_else - Err 时尝试恢复
    let result = err.or_else(|e| Ok(42));
    assert_eq!(result, Ok(42));
    
    let result = ok.or_else(|e| Ok(42));
    assert_eq!(result, Ok(5));  // Ok 保持不变
    
    // unwrap_err - 获取错误，Ok 时 panic
    assert_eq!(err.unwrap_err(), "出错了");
    // ok.unwrap_err();  // ❌ panic!
    
    // unwrap_or_default - 使用默认值
    let result: Result<i32, &str> = Err("error");
    assert_eq!(result.unwrap_or_default(), 0);
}

use std::fs::File;
use std::io::{self, Read};

// ❌ 冗长的写法
fn read_file_verbose() -> Result<String, io::Error> {
    let mut file = match File::open("hello.txt") {
        Ok(file) => file,
        Err(e) => return Err(e),
    };
    
    let mut contents = String::new();
    match file.read_to_string(&mut contents) {
        Ok(_) => Ok(contents),
        Err(e) => Err(e),
    }
}

// ✅ 使用 ? 运算符
fn read_file_concise() -> Result<String, io::Error> {
    let mut file = File::open("hello.txt")?;  // 如果 Err，直接返回
    let mut contents = String::new();
    file.read_to_string(&mut contents)?;
    Ok(contents)
}

// ✅ 更简洁的链式调用
fn read_file_chained() -> Result<String, io::Error> {
    let mut contents = String::new();
    File::open("hello.txt")?
        .read_to_string(&mut contents)?;
    Ok(contents)
}

use std::fmt;
use std::error::Error;

// 定义自定义错误类型
#[derive(Debug)]
enum AppError {
    IoError(std::io::Error),
    ParseError(std::num::ParseIntError),
    NotFound(String),
    InvalidInput(String),
}

// 实现 Display trait
impl fmt::Display for AppError {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        match self {
            AppError::IoError(e) => write!(f, "IO 错误：{}", e),
            AppError::ParseError(e) => write!(f, "解析错误：{}", e),
            AppError::NotFound(msg) => write!(f, "未找到：{}", msg),
            AppError::InvalidInput(msg) => write!(f, "无效输入：{}", msg),
        }
    }
}

// 实现 Error trait
impl Error for AppError {
    fn source(&self) -> Option<&(dyn Error + 'static)> {
        match self {
            AppError::IoError(e) => Some(e),
            AppError::ParseError(e) => Some(e),
            _ => None,
        }
    }
}

// 实现 From trait，方便使用 ?
impl From<std::io::Error> for AppError {
    fn from(err: std::io::Error) -> Self {
        AppError::IoError(err)
    }
}

impl From<std::num::ParseIntError> for AppError {
    fn from(err: std::num::ParseIntError) -> Self {
        AppError::ParseError(err)
    }
}

// 使用自定义错误
use std::fs;

fn read_and_parse(path: &str) -> Result<i32, AppError> {
    let contents = fs::read_to_string(path)?;  // 自动转换为 AppError::IoError
    let number: i32 = contents.trim().parse()?;  // 自动转换为 AppError::ParseError
    
    if number < 0 {
        return Err(AppError::InvalidInput("数字不能为负数".to_string()));
    }
    
    Ok(number)
}

fn main() {
    match read_and_parse("number.txt") {
        Ok(n) => println!("读取成功：{}", n),
        Err(e) => println!("错误：{}", e),
    }
}

impl std::fmt::Display for ConfigError {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        match self {
            ConfigError::FileNotFound(path) => {
                write!(f, "配置文件不存在：{}", path)
            }
            ConfigError::InvalidFormat(line) => {
                write!(f, "配置格式错误：{}", line)
            }
            ConfigError::MissingKey(key) => {
                write!(f, "缺少必需的配置项：{}", key)
            }
        }
    }
}

impl Error for ConfigError {}

/// 配置结构体
#[derive(Debug)]
struct Config {
    host: String,
    port: u16,
    debug: bool,
}

impl Config {
    /// 从文件加载配置
    fn from_file(path: &str) -> Result<Self, ConfigError> {
        let contents = fs::read_to_string(path)
            .map_err(|_| ConfigError::FileNotFound(path.to_string()))?;
        
        Self::from_string(&contents)
    }
    
    /// 从字符串解析配置
    fn from_string(contents: &str) -> Result<Self, ConfigError> {
        let mut map = HashMap::new();
        
        for line in contents.lines() {
            let line = line.trim();
            
            // 跳过空行和注释
            if line.is_empty() || line.starts_with('#') {
                continue;
            }
            
            // 解析 key=value
            let parts: Vec<&str> = line.splitn(2, '=').collect();
            if parts.len() != 2 {
                return Err(ConfigError::InvalidFormat(line.to_string()));
            }
            
            let key = parts[0].trim();
            let value = parts[1].trim();
            map.insert(key, value);
        }
        
        // 提取并验证配置项
        let host = map
            .get("host")
            .ok_or_else(|| ConfigError::MissingKey("host".to_string()))?
            .to_string();
        
        let port: u16 = map
            .get("port")
            .ok_or_else(|| ConfigError::MissingKey("port".to_string()))?
            .parse()
            .map_err(|_| ConfigError::InvalidFormat("port 必须是数字".to_string()))?;
        
        let debug = map
            .get("debug")
            .map(|s| s == "true")
            .unwrap_or(false);
        
        Ok(Config { host, port, debug })
    }
}

fn main() -> Result<(), Box<dyn Error>> {
    // 创建示例配置文件
    let config_content = r#"
# 服务器配置
host = localhost
port = 8080
debug = true
"#;
    
    // 写入临时文件
    fs::write("config.ini", config_content)?;
    
    // 读取配置
    match Config::from_file("config.ini") {
        Ok(config) => {
            println!("✅ 配置加载成功");
            println!("  Host: {}", config.host);
            println!("  Port: {}", config.port);
            println!("  Debug: {}", config.debug);
        }
        Err(e) => {
            println!("❌ 配置加载失败：{}", e);
        }
    }
    
    // 清理
    fs::remove_file("config.ini")?;
    
    Ok(())
}

fn process_user_data(input: Option<String>) -> Result<i32, String> {
    input
        .ok_or("没有输入数据".to_string())?  // Option -> Result
        .trim()
        .parse::<i32>()
        .map_err(|e| format!("解析失败：{}", e))?
        .checked_mul(2)
        .ok_or("计算结果溢出".to_string())
}

/// 链式处理多个可能失败的步骤
fn validate_and_process(
    username: Option<String>,
    age: Option<String>,
    email: Option<String>,
) -> Result<(), Vec<String>> {
    let mut errors = Vec::new();
    
    // 验证用户名
    let username = username
        .filter(|s| !s.is_empty())
        .filter(|s| s.len() >= 3)
        .ok_or_else(|| "用户名至少 3 个字符".to_string());
    
    match username {
        Ok(name) => println!("用户名有效：{}", name),
        Err(e) => errors.push(e),
    }
    
    // 验证年龄
    let age = age
        .and_then(|s| s.parse::<u8>().ok())
        .filter(|&a| a >= 18)
        .ok_or_else(|| "年龄必须 >= 18".to_string());
    
    match age {
        Ok(a) => println!("年龄有效：{}", a),
        Err(e) => errors.push(e),
    }
    
    // 验证邮箱
    let email_valid = email
        .filter(|s| s.contains('@'))
        .is_some();
    
    if !email_valid {
        errors.push("邮箱格式无效".to_string());
    }
    
    if errors.is_empty() {
        Ok(())
    } else {
        Err(errors)
    }
}

fn main() {
    // 测试数据转换
    match process_user_data(Some("  21  ".to_string())) {
        Ok(result) => println!("处理结果：{}", result),
        Err(e) => println!("错误：{}", e),
    }
    
    // 测试验证
    match validate_and_process(
        Some("alice".to_string()),
        Some("25".to_string()),
        Some("alice@example.com".to_string()),
    ) {
        Ok(_) => println!("✅ 所有验证通过"),
        Err(errors) => {
            println!("❌ 验证失败：");
            for error in errors {
                println!("  - {}", error);
            }
        }
    }
}
