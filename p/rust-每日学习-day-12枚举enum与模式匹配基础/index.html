#[derive(Debug)]
enum Message {
    Quit,
    Move { x: i32, y: i32 },
    Write(String),
    ChangeColor(i32, i32, i32),
}

impl Message {
    fn process(&self) {
        match self {
            Message::Quit => println!("退出程序"),
            Message::Move { x, y } => println!("移动到 ({}, {})", x, y),
            Message::Write(text) => println!("写入消息：{}", text),
            Message::ChangeColor(r, g, b) => println!("改变颜色：RGB({}, {}, {})", r, g, b),
        }
    }
}

fn main() {
    let messages = vec![
        Message::Quit,
        Message::Move { x: 100, y: 200 },
        Message::Write(String::from("Hello, Rust!")),
        Message::ChangeColor(128, 64, 32),
    ];
    
    for msg in &messages {
        msg.process();
    }
}

fn main() {
    let some_val: Option<i32> = Some(10);
    let none_val: Option<i32> = None;
    
    // is_some / is_none
    println!("some_val is_some: {}", some_val.is_some()); // true
    println!("none_val is_none: {}", none_val.is_none()); // true
    
    // unwrap - 获取值，如果是 None 则 panic
    println!("some_val unwrap: {}", some_val.unwrap()); // 10
    // println!("{}", none_val.unwrap()); // ❌ panic!
    
    // unwrap_or - 提供默认值
    println!("some_val unwrap_or: {}", some_val.unwrap_or(0)); // 10
    println!("none_val unwrap_or: {}", none_val.unwrap_or(0)); // 0
    
    // unwrap_or_else - 延迟计算默认值
    let val = none_val.unwrap_or_else(|| {
        println!("计算默认值...");
        42
    });
    println!("val: {}", val);
    
    // expect - 类似 unwrap，但可自定义 panic 消息
    // some_val.expect("期望有值");
    
    // map - 转换内部值
    let doubled = some_val.map(|x| x * 2);
    println!("doubled: {:?}", doubled); // Some(20)
    
    // and_then - 链式操作，返回 Option
    let result = some_val.and_then(|x| {
        if x > 5 {
            Some(x * 10)
        } else {
            None
        }
    });
    println!("result: {:?}", result); // Some(100)
}

fn find_user(id: u32) -> Option<String> {
    match id {
        1 => Some(String::from("Alice")),
        2 => Some(String::from("Bob")),
        _ => None,
    }
}

fn main() {
    // 场景 1：提供默认值
    let user1 = find_user(1).unwrap_or(String::from("Unknown"));
    println!("用户 1: {}", user1); // Alice
    
    let user2 = find_user(999).unwrap_or(String::from("Unknown"));
    println!("用户 999: {}", user2); // Unknown
    
    // 场景 2：链式处理
    let greeting = find_user(1)
        .map(|name| format!("Hello, {}!", name))
        .unwrap_or(String::from("Hello, Guest!"));
    println!("{}", greeting); // Hello, Alice!
    
    // 场景 3：错误传播的雏形
    match find_user(2) {
        Some(name) => println!("找到用户：{}", name),
        None => println!("用户不存在"),
    }
}

fn main() {
    let config_max = Some(3u8);
    
    // 使用 match（冗长）
    match config_max {
        Some(max) => println!("最大值是：{}", max),
        _ => (),
    }
    
    // 使用 if let（简洁）
    if let Some(max) = config_max {
        println!("最大值是：{}", max);
    }
    
    // if let else
    let coin = Some("Quarter");
    if let Some("Quarter") = coin {
        println!("是 25 分硬币");
    } else if let Some("Dime") = coin {
        println!("是 10 分硬币");
    } else {
        println!("其他硬币或没有硬币");
    }
    
    // while let 循环
    let mut stack = Vec::new();
    stack.push(1);
    stack.push(2);
    stack.push(3);
    
    while let Some(top) = stack.pop() {
        println!("{}", top); // 3, 2, 1
    }
}

#[derive(Debug, Clone)]
enum TrafficLight {
    Red,
    Yellow,
    Green,
}

impl TrafficLight {
    // 下一个状态
    fn next(&self) -> Self {
        match self {
            TrafficLight::Red => TrafficLight::Green,
            TrafficLight::Green => TrafficLight::Yellow,
            TrafficLight::Yellow => TrafficLight::Red,
        }
    }
    
    // 是否可以通行
    fn can_pass(&self) -> bool {
        matches!(self, TrafficLight::Green)
    }
    
    // 等待时间（秒）
    fn wait_time(&self) -> u8 {
        match self {
            TrafficLight::Red => 60,
            TrafficLight::Yellow => 5,
            TrafficLight::Green => 30,
        }
    }
    
    // 描述
    fn description(&self) -> &str {
        match self {
            TrafficLight::Red => "🛑 停止",
            TrafficLight::Yellow => "⚠️ 准备",
            TrafficLight::Green => "✅ 通行",
        }
    }
}

fn main() {
    let mut light = TrafficLight::Red;
    
    for i in 0..6 {
        println!("周期 {}: {} (等待 {} 秒)", 
                 i + 1, 
                 light.description(), 
                 light.wait_time());
        
        if light.can_pass() {
            println!("  → 可以通行！");
        }
        
        light = light.next();
    }
}

#[derive(Debug)]
enum Command {
    Move { x: i32, y: i32 },
    Rotate(f64),
    Draw(String),
    Clear,
    Quit,
}

impl Command {
    // 解析命令
    fn parse(input: &str) -> Option<Command> {
        let parts: Vec<&str> = input.trim().split_whitespace().collect();
        
        if parts.is_empty() {
            return None;
        }
        
        match parts[0].to_lowercase().as_str() {
            "move" | "m" => {
                if parts.len() >= 3 {
                    let x = parts[1].parse().ok()?;
                    let y = parts[2].parse().ok()?;
                    Some(Command::Move { x, y })
                } else {
                    None
                }
            },
            "rotate" | "r" => {
                if parts.len() >= 2 {
                    let angle = parts[1].parse().ok()?;
                    Some(Command::Rotate(angle))
                } else {
                    None
                }
            },
            "draw" | "d" => {
                if parts.len() >= 2 {
                    Some(Command::Draw(parts[1..].join(" ")))
                } else {
                    None
                }
            },
            "clear" | "c" => Some(Command::Clear),
            "quit" | "q" | "exit" => Some(Command::Quit),
            _ => None,
        }
    }
    
    // 执行命令
    fn execute(&self) {
        match self {
            Command::Move { x, y } => println!("移动到 ({}, {})", x, y),
            Command::Rotate(angle) => println!("旋转 {} 度", angle),
            Command::Draw(shape) => println!("绘制：{}", shape),
            Command::Clear => println!("清空画布"),
            Command::Quit => println!("退出程序"),
        }
    }
}

fn main() {
    let commands = vec![
        "move 10 20",
        "rotate 45",
        "draw circle",
        "clear",
        "quit",
        "invalid command",
    ];
    
    for cmd_str in commands {
        print!("输入：{} → ", cmd_str);
        match Command::parse(cmd_str) {
            Some(cmd) => {
                print!("解析成功，");
                cmd.execute();
            },
            None => println!("解析失败"),
        }
    }
}

// Option: 表示值可能存在或不存在
fn find_by_id(id: u32) -> Option<String> {
    match id {
        1 => Some(String::from("Alice")),
        _ => None,
    }
}

// Result: 表示操作成功或失败（带错误信息）
fn parse_number(s: &str) -> Result<i32, String> {
    s.parse::<i32>()
        .map_err(|e| format!("解析失败：{}", e))
}

fn main() {
    // Option 处理
    match find_by_id(1) {
        Some(name) => println!("找到：{}", name),
        None => println!("未找到"),
    }
    
    // Result 处理
    match parse_number("42") {
        Ok(num) => println!("数字：{}", num),
        Err(e) => println!("错误：{}", e),
    }
    
    match parse_number("not a number") {
        Ok(num) => println!("数字：{}", num),
        Err(e) => println!("错误：{}", e),
    }
}

fn main() {
    let opt = Some(5);
    
    // 传统写法
    let is_some = match opt {
        Some(_) => true,
        None => false,
    };
    
    // 使用 matches! 宏
    let is_some_macro = matches!(opt, Some(_));
    println!("is_some: {}", is_some_macro);
    
    // 复杂模式
    let point = Some((10, 20));
    println!("x=10: {}", matches!(point, Some((10, _))));
    println!("y=20: {}", matches!(point, Some((_, 20))));
    
    // 枚举变体检查
    #[derive(Debug)]
    enum Status {
        Active,
        Inactive,
        Pending { reason: String },
    }
    
    let status = Status::Pending { reason: String::from("waiting") };
    println!("是 Pending: {}", matches!(status, Status::Pending { .. }));
}

需要解构多个变体 match 更清晰

今日感悟

枚举和模式匹配是 Rust 最具表现力的特性之一，今天的学习让我有几个深刻认识：

  1. 类型安全的力量：Option<T> 强制你处理"无值"的情况，消除了 null 指针异常。这是 Rust 安全性的核心体现。

  2. match 的穷尽性检查：编译器确保你处理了所有可能的情况，这是其他语言很少有的保障。忘记处理某个变体？编译器会告诉你。

  3. 数据与行为的统一：枚举变体可以携带数据，配合 match 解构，让代码既简洁又类型安全。这比传统的"类型 + 标签"模式优雅得多。

  4. 函数式编程的影子：map、and_then、filter 这些方法让 Option 可以像管道一样处理，这是函数式编程的风格。

  5. if let 的实用性：不是所有情况都需要完整的 match，if let 在处理单一模式时更简洁，while let 在迭代中也很常用。

  6. 模式匹配的表达能力：守卫条件、@ 绑定、范围匹配… match 的能力远超简单的 switch，它是一门小型语言。

明天的模式匹配进阶应该会更深入，特别是解构复杂数据类型和引用模式。期待！
