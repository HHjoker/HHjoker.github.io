什么是 Trait？

概念理解

Trait 是 Rust 中定义共享行为的机制。你可以把它理解为：

  • 其他语言中的接口（Interface）
  • 但更强大，可以有默认实现
  • 类似于 Haskell 的类型类（Typeclass）

// Trait 定义：描述一组行为
trait Summary {
    fn summarize(&self) -> String;
}

// 为类型实现 Trait
struct Article {
    headline: String,
    location: String,
    author: String,
    content: String,
}

impl Summary for Article {
    fn summarize(&self) -> String {
        format!("{}, by {} ({})", self.headline, self.author, self.location)
    }
}

fn main() {
    let article = Article {
        headline: String::from("Rust 1.80 发布"),
        location: String::from("Beijing"),
        author: String::from("张三"),
        content: String::from("今天 Rust 发布了新版本..."),
    };
    
    println!("摘要：{}", article.summarize());
    // 输出：摘要：Rust 1.80 发布，by 张三 (Beijing)
}

fn process<T>(data: T) {
    // 但这里什么都做不了，因为不知道 T 有什么方法
}

// Trait：关注"能做什么"
trait Printable {
    fn print(&self);
}

    // 现在可以调用 print 方法了！
    data.print();
}

trait Container {
    type Item;  // 关联类型
    
    fn get(&self, index: usize) -> Option<&Self::Item>;
    fn len(&self) -> usize;
}

struct IntVec {
    data: Vec<i32>,
}

impl Container for IntVec {
    type Item = i32;  // 指定关联类型
    
    fn get(&self, index: usize) -> Option<&i32> {
        self.data.get(index)
    }
    
    fn len(&self) -> usize {
        self.data.len()
    }
}

fn main() {
    let container = IntVec { data: vec![1, 2, 3, 4, 5] };
    println!("长度：{}", container.len());
    println!("第一个元素：{:?}", container.get(0));
}

trait Display {
    fn display(&self) -> String;
}

struct Point {
    x: f64,
    y: f64,
}

impl Display for Point {
    fn display(&self) -> String {
        format!("Point({}, {})", self.x, self.y)
    }
}

struct Rectangle {
    width: f64,
    height: f64,
}

impl Display for Rectangle {
    fn display(&self) -> String {
        format!("Rectangle({}x{})", self.width, self.height)
    }
}

fn main() {
    let point = Point { x: 3.0, y: 4.0 };
    let rect = Rectangle { width: 10.0, height: 5.0 };
    
    println!("{}", point.display());   // Point(3, 4)
    println!("{}", rect.display());    // Rectangle(10x5)
}

// 可以为标准库类型实现自定义 Trait
trait Double {
    fn double(&self) -> Self;
}

impl Double for i32 {
    fn double(&self) -> i32 {
        self * 2
    }
}

impl Double for f64 {
    fn double(&self) -> f64 {
        self * 2.0
    }
}

impl Double for String {
    fn double(&self) -> String {
        format!("{}{}", self, self)
    }
}

fn main() {
    println!("{}", 5.double());           // 10
    println!("{}", 3.14.double());        // 6.28
    println!("{}", "hello".to_string().double());  // hellohello
}

use std::fmt::Display;

struct Wrapper(String);

// ✅ 可以：Wrapper 是我们定义的类型
impl Display for Wrapper {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        write!(f, "Wrapper({})", self.0)
    }
}

// ❌ 不可以：String 和 Display 都是标准库的
// impl Display for String { ... }

fn main() {
    let w = Wrapper(String::from("hello"));
    println!("{}", w);  // Wrapper(hello)
}

trait Flyable {

trait Greeting {
    fn greet(&self) -> String {
        String::from("你好！")
    }
    
    fn greet_formal(&self) -> String {
        String::from("您好！")
    }
}

struct FriendlyPerson;
struct FormalPerson;

impl Greeting for FriendlyPerson {
    // 使用默认的 greet
    // 覆盖 greet_formal
    fn greet_formal(&self) -> String {
        String::from("嘿！最近怎么样？")
    }
}

impl Greeting for FormalPerson {
    // 两个都覆盖
    fn greet(&self) -> String {
        String::from("幸会！")
    }
    
    fn greet_formal(&self) -> String {
        String::from("久仰大名！")
    }
}

fn main() {
    let friendly = FriendlyPerson;
    let formal = FormalPerson;
    
    println!("友好的人：{}", friendly.greet());         // 你好！
    println!("友好的人（正式）：{}", friendly.greet_formal());  // 嘿！最近怎么样？
    
    println!("正式的人：{}", formal.greet());           // 幸会！
    println!("正式的人（正式）：{}", formal.greet_formal());    // 久仰大名！
}

}

// 更灵活的语法，支持多个约束
fn render_multiple<T: Drawable>(items: Vec<T>) {
    for item in items {
        item.draw();
    }
}

// 多个 Trait 约束
fn process<T: Drawable + Clone>(item: T) {
    item.draw();
    let cloned = item.clone();
    cloned.draw();
}

// where 子句（复杂约束时更清晰）
fn complex_process<T>(item: T)
where
    T: Drawable + Clone + std::fmt::Debug,
{
    println!("{:?}", item);
    item.draw();
}

use std::fmt;

struct Point {
    x: i32,
    y: i32,
}

// Debug：用于调试输出 {:?}
impl fmt::Debug for Point {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        f.debug_struct("Point")
            .field("x", &self.x)
            .field("y", &self.y)
            .finish()
    }
}

// Display：用于用户输出 {}
impl fmt::Display for Point {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        write!(f, "({}, {})", self.x, self.y)
    }
}

fn main() {
    let p = Point { x: 10, y: 20 };
    
    println!("Debug: {:?}", p);   // Debug: Point { x: 10, y: 20 }
    println!("Display: {}", p);   // Display: (10, 20)
}

use std::cmp::Ordering;

#[derive(Debug, PartialEq, Eq)]
struct Version {
    major: u32,
    minor: u32,
    patch: u32,
}

impl PartialOrd for Version {
    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
        Some(self.cmp(other))
    }
}

impl Ord for Version {
    fn cmp(&self, other: &Self) -> Ordering {
        // 先比较 major，再 minor，最后 patch
        self.major.cmp(&other.major)
            .then(self.minor.cmp(&other.minor))
            .then(self.patch.cmp(&other.patch))
    }
}

fn main() {
    let v1 = Version { major: 1, minor: 2, patch: 3 };
    let v2 = Version { major: 1, minor: 2, patch: 4 };
    let v3 = Version { major: 2, minor: 0, patch: 0 };
    
    println!("v1 < v2: {}", v1 < v2);  // true
    println!("v2 < v3: {}", v2 < v3);  // true
    println!("v1 > v3: {}", v1 > v3);  // false
}

use std::cmp::Ordering;

// 定义可排序 Trait
trait Sortable {
    fn compare(&self, other: &Self) -> Ordering;
}

// 为 i32 实现
impl Sortable for i32 {
    fn compare(&self, other: &Self) -> Ordering {
        self.cmp(other)
    }
}

// 为字符串实现
impl Sortable for String {
    fn compare(&self, other: &Self) -> Ordering {
        self.cmp(other)
    }
}

// 泛型排序函数
fn bubble_sort<T: Sortable + Clone>(arr: &mut [T]) {
    let len = arr.len();
    for i in 0..len {
        for j in 0..len - i - 1 {
            if arr[j].compare(&arr[j + 1]) == Ordering::Greater {
                arr.swap(j, j + 1);
            }
        }
    }
}

fn main() {
    // 排序数字
    let mut numbers = vec![5, 2, 8, 1, 9];
    bubble_sort(&mut numbers);
    println!("排序后：{:?}", numbers);  // [1, 2, 5, 8, 9]
    
    // 排序字符串
    let mut words = vec![
        String::from("banana"),
        String::from("apple"),
        String::from("cherry"),
    ];
    bubble_sort(&mut words);
    println!("排序后：{:?}", words);  // ["apple", "banana", "cherry"]
}

/// 插件接口
trait Plugin {
    /// 插件名称
    fn name(&self) -> &'static str;
    
    /// 插件版本
    fn version(&self) -> &'static str;
    
    /// 初始化插件
    fn initialize(&mut self) -> Result<(), String>;
    
    /// 执行插件功能
    fn execute(&self, input: &str) -> String;
    
    /// 关闭插件
    fn shutdown(&mut self) {
        println!("{} 正在关闭...", self.name());
    }
}

/// 日志插件
struct LoggerPlugin {
    initialized: bool,
}

impl Plugin for LoggerPlugin {
    fn name(&self) -> &'static str {
        "Logger"
    }
    
    fn version(&self) -> &'static str {
        "1.0.0"
    }
    
    fn initialize(&mut self) -> Result<(), String> {
        self.initialized = true;
        println!("日志插件已初始化");
        Ok(())
    }
    
    fn execute(&self, input: &str) -> String {
        if self.initialized {
            format!("[LOG] {}", input)
        } else {
            String::from("插件未初始化")
        }
    }
}

/// 加密插件
struct EncryptorPlugin {
    key: u8,
}

impl Plugin for EncryptorPlugin {
    fn name(&self) -> &'static str {
        "Encryptor"
    }
    
    fn version(&self) -> &'static str {
        "2.0.0"
    }
    
    fn initialize(&mut self) -> Result<(), String> {
        self.key = 3;  // 简单的 Caesar 密码
        println!("加密插件已初始化");
        Ok(())
    }
    
    fn execute(&self, input: &str) -> String {
        input.chars()
            .map(|c| ((c as u8 - b'a' + self.key) % 26 + b'a') as char)
            .collect()
    }
    
    fn shutdown(&mut self) {
        self.key = 0;
        println!("加密插件已关闭，密钥已清除");
    }
}

/// 插件管理器
struct PluginManager {
    plugins: Vec<Box<dyn Plugin>>,
}

impl PluginManager {
    fn new() -> Self {
        PluginManager {
            plugins: Vec::new(),
        }
    }
    
    fn add_plugin(&mut self, plugin: Box<dyn Plugin>) {
        println!("添加插件：{}", plugin.name());
        self.plugins.push(plugin);
    }
    
    fn initialize_all(&mut self) -> Result<(), String> {
        for plugin in &mut self.plugins {
            plugin.initialize()?;
        }
        Ok(())
    }
    
    fn execute_all(&self, input: &str) -> Vec<String> {
        self.plugins
            .iter()
            .map(|p| p.execute(input))
            .collect()
    }
    
    fn shutdown_all(&mut self) {
        for plugin in &mut self.plugins {
            plugin.shutdown();
        }
    }
}

fn main() {
    let mut manager = PluginManager::new();
    
    // 添加插件
    manager.add_plugin(Box::new(LoggerPlugin { initialized: false }));
    manager.add_plugin(Box::new(EncryptorPlugin { key: 0 }));
    
    // 初始化所有插件
    if let Err(e) = manager.initialize_all() {
        println!("初始化失败：{}", e);
        return;
    }
    
    // 执行所有插件
    let results = manager.execute_all("hello");
    for (i, result) in results.iter().enumerate() {
        println!("插件 {} 输出：{}", i + 1, result);
    }
    
    // 关闭所有插件
    manager.shutdown_all();
}

trait Comparable {
    fn is_greater_than(&self, other: &Self) -> bool;
    fn is_less_than(&self, other: &Self) -> bool;
    fn is_equal(&self, other: &Self) -> bool;
}

impl Temperature {
    fn to_celsius(&self) -> f64 {
        if self.unit == "C" {
            self.value
        } else {
            (self.value - 32.0) * 5.0 / 9.0
        }
    }
}

impl Comparable for Temperature {
    fn is_greater_than(&self, other: &Self) -> bool {
        self.to_celsius() > other.to_celsius()
    }
    
    fn is_less_than(&self, other: &Self) -> bool {
        self.to_celsius() < other.to_celsius()
    }
    
    fn is_equal(&self, other: &Self) -> bool {
        (self.to_celsius() - other.to_celsius()).abs() < 0.01
    }
}
