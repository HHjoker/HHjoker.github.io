这是 Rust 中最容易混淆的概念之一。理解它们的区别是掌握 Rust 字符串的关键。

fn main() {
    // &str - 字符串切片（借用）
    // 通常是字符串字面量，存储在程序的二进制数据中
    let s1: &str = "Hello, world!";
    
    // String - 可增长的字符串（拥有所有权）
    // 存储在堆上，可以修改
    let s2: String = String::from("Hello, world!");
    
    // 或者使用 to_string()
    let s3: String = "Hello, world!".to_string();
    
    println!("s1: {}", s1);
    println!("s2: {}", s2);
    println!("s3: {}", s3);
}

特性 &str String
类型 引用类型 拥有所有权的类型
存储位置 通常在栈上（指向数据） 堆上
可变性 不可变 可变
大小 固定 可增长
所有权 借用 拥有

fn main() {
    let mut s = String::from("Hello");
    
    // 1. push() - 添加单个字符
    s.push('!');
    println!("{}", s); // Hello!
    
    // 2. push_str() - 添加字符串切片
    s.push_str(" How are you?");
    println!("{}", s); // Hello! How are you?
    
    // 3. 使用 + 运算符（注意：会消耗第一个字符串的所有权）
    let s1 = String::from("Hello, ");
    let s2 = String::from("world!");
    let s3 = s1 + &s2; // s1 被移动，s2 被借用
    println!("{}", s3); // Hello, world!
    // println!("{}", s1); // 错误：s1 已被移动
    
    // 4. format! 宏（不消耗所有权，推荐）
    let s4 = String::from("Hello");
    let s5 = String::from("world");
    let s6 = format!("{}, {}!", s4, s5);
    println!("{}", s6); // Hello, world!
    println!("{}", s4); // 仍然可以使用
    println!("{}", s5); // 仍然可以使用
    
    // 5. insert() - 在指定位置插入字符
    let mut s7 = String::from("Helo");
    s7.insert(2, 'l');
    println!("{}", s7); // Hello
    
    // 6. insert_str() - 在指定位置插入字符串
    let mut s8 = String::from("Hllo");
    s8.insert_str(1, "e");
    println!("{}", s8); // Hello
    
    // 7. remove() - 删除指定位置的字符
    let mut s9 = String::from("Hello!");
    let c = s9.remove(5); // 删除 '!'
    println!("删除的字符：{}", c);
    println!("{}", s9); // Hello
    
    // 8. pop() - 删除并返回最后一个字符
    let mut s10 = String::from("Hello");
    if let Some(c) = s10.pop() {
        println!("删除的最后一个字符：{}", c); // o
    }
    println!("{}", s10); // Hell
}

fn main() {
    let s = String::from("Hello");
    
    // ❌ 不能通过索引访问！
    // let c = s[0]; // 编译错误！
    
    // ✅ 方法 1: 使用 chars() 迭代器
    for c in s.chars() {
        println!("{}", c);
    }
    
    // 获取第 n 个字符
    let third_char = s.chars().nth(2);
    println!("第三个字符：{:?}", third_char); // Some('l')
    
    // ✅ 方法 2: 使用 bytes() 查看底层字节
    for b in s.bytes() {
        println!("{}", b);
    }
    
    // ✅ 方法 3: 使用切片（需要小心）
    // 只能在字符边界处切片
    let hello = "hello";
    let he = &hello[0..2]; // "he"
    println!("{}", he);
    
    // ❌ 在字符边界外切片会 panic！
    // let s_emoji = String::from("👋");
    // let slice = &s_emoji[0..1]; // panic!
}

fn main() {
    // 不同字符占用的字节数不同
    let s1 = String::from("a");      // 1 字节
    let s2 = String::from("中");      // 3 字节
    let s3 = String::from("👋");      // 4 字节
    
    println!("'a' 的字节数：{}", s1.len());     // 1
    println!("'中' 的字节数：{}", s2.len());     // 3
    println!("'👋' 的字节数：{}", s3.len());     // 4
    
    println!("'a' 的字符数：{}", s1.chars().count());     // 1
    println!("'中' 的字符数：{}", s2.chars().count());     // 1
    println!("'👋' 的字符数：{}", s3.chars().count());     // 1
    
    // 组合字符的问题
    let s4 = String::from("é");  // 可以是单个字符，也可以是 e + ́
    
    // 方式 1: 单个 Unicode 码点
    let s5 = String::from("\u{00E9}"); // é (单个字符)
    
    // 方式 2: e + 重音符号
    let s6 = String::from("e\u{0301}"); // e + ́ = é (两个字符)
    
    println!("s5 长度：{}, 字符数：{}", s5.len(), s5.chars().count());
    println!("s6 长度：{}, 字符数：{}", s6.len(), s6.chars().count());
    
    // 这就是为什么 Rust 不允许 s[0] - 它不知道你想要字节还是字符
}

fn main() {
    let s = String::from("Hello, world!");
    
    // 创建切片（借用）
    let hello = &s[0..5];   // "Hello"
    let world = &s[7..12];  // "world"
    
    println!("{}", hello);
    println!("{}", world);
    
    // 简写：省略开头或结尾
    let from_start = &s[..5];    // "Hello"
    let from_end = &s[7..];      // "world!"
    let all = &s[..];            // "Hello, world!"
    
    // 切片是 &str 类型
    let slice: &str = &s[0..5];
    
    // 函数参数通常使用 &str
    fn print_str(s: &str) {
        println!("{}", s);
    }
    
    print_str(&s);      // 传递 String 的切片
    print_str("hello"); // 传递字符串字面量
}

fn main() {
    let s = String::from("  Hello, World!  ");
    
    // 1. trim() - 去除两端空白
    println!("{}", s.trim()); // "Hello, World!"
    
    // 2. trim_start() / trim_end()
    println!("{}", s.trim_start()); // "Hello, World!  "
    println!("{}", s.trim_end());   // "  Hello, World!"
    
    // 3. to_uppercase() / to_lowercase()
    let lower = s.trim().to_lowercase();
    let upper = s.trim().to_uppercase();
    println!("{}", lower); // "hello, world!"
    println!("{}", upper); // "HELLO, WORLD!"
    
    // 4. contains() - 检查是否包含子串
    println!("{}", s.contains("World")); // true
    println!("{}", s.contains("Rust"));  // false
    
    // 5. starts_with() / ends_with()
    println!("{}", s.trim().starts_with("Hello")); // true
    println!("{}", s.trim().ends_with("!"));       // true
    
    // 6. find() - 查找子串位置
    match s.find("World") {
        Some(pos) => println!("找到 'World' 在位置 {}", pos),
        None => println!("未找到"),
    }
    
    // 7. replace() - 替换子串
    let replaced = s.replace("World", "Rust");
    println!("{}", replaced); // "  Hello, Rust!  "
    
    // 8. split() - 分割字符串
    let sentence = "Rust is awesome";
    for word in sentence.split_whitespace() {
        println!("{}", word);
    }
    
    // 9. split_at() - 在指定位置分割
    let s2 = "hello";
    let (part1, part2) = s2.split_at(2);
    println!("{} | {}", part1, part2); // he | llo
    
    // 10. lines() - 按行分割
    let multiline = "line1\nline2\nline3";
    for line in multiline.lines() {
        println!("{}", line);
    }
    
    // 11. is_empty() - 检查是否为空
    println!("{}", "".is_empty()); // true
    
    // 12. len() - 获取长度（字节数）
    println!("{}", "hello".len()); // 5
    
    // 13. chars().count() - 获取字符数
    println!("{}", "你好".chars().count()); // 2
    println!("{}", "你好".len());           // 6 (UTF-8 字节数)
}

fn main() {
    // String -> &str
    let s1 = String::from("Hello");
    let slice1: &str = &s1;
    let slice2: &str = s1.as_str();
    
    // &str -> String
    let s2: String = slice1.to_string();
    let s3: String = slice1.to_owned();
    let s4: String = String::from(slice1);
    
    // 函数参数的最佳实践
    // 使用 &str 作为参数类型，可以接受 String 和 &str
    
    fn greet(name: &str) {
        println!("Hello, {}!", name);
    }
    
    let owned = String::from("Alice");
    let borrowed = "Bob";
    
    greet(&owned);   // 传递 String 的引用（自动解引用为 &str）
    greet(borrowed); // 直接传递 &str
    greet("Carol");  // 字符串字面量也是 &str
}

fn main() {
    // ❌ 低效：每次 + 都创建新 String
    let mut s1 = String::from("Hello");
    s1 = s1 + ", ";
    s1 = s1 + "world";
    s1 = s1 + "!";
    
    // ✅ 高效：使用 push_str
    let mut s2 = String::from("Hello");
    s2.push_str(", ");
    s2.push_str("world");
    s2.push('!');
    
    // ✅ 高效：使用 format!
    let name = "Alice";
    let s3 = format!("Hello, {}!", name);
    
    // ✅ 高效：使用 write! 到 String（需要 std::fmt::Write）
    use std::fmt::Write;
    let mut s4 = String::new();
    write!(s4, "Hello, {}!", name).unwrap();
    
    // ✅ 最高效：预分配容量
    let mut s5 = String::with_capacity(100); // 预分配 100 字节
    s5.push_str("Hello");
    s5.push_str(", world!");
    
    println!("容量：{}, 长度：{}", s5.capacity(), s5.len());
}

use std::collections::HashMap;

fn main() {
    // Vec<String>
    let mut names: Vec<String> = Vec::new();
    names.push(String::from("Alice"));
    names.push(String::from("Bob"));
    
    for name in &names {
        println!("{}", name);
    }
    
    // HashMap<String, String>
    let mut phone_book: HashMap<String, String> = HashMap::new();
    phone_book.insert(String::from("Alice"), String::from("123-4567"));
    phone_book.insert(String::from("Bob"), String::from("234-5678"));
    
    for (name, phone) in &phone_book {
        println!("{}: {}", name, phone);
    }
    
    // 字符串处理实战：统计单词
    let text = "rust is awesome rust is fast";
    let mut word_count: HashMap<String, usize> = HashMap::new();
    
    for word in text.split_whitespace() {
        *word_count.entry(word.to_string()).or_insert(0) += 1;
    }
    
    println!("{:?}", word_count);
    // {"rust": 2, "is": 2, "awesome": 1, "fast": 1}
}

fn main() {
    // 解析数字
    let num_str = "42";
    let num: i32 = num_str.parse().unwrap();
    println!("{}", num + 1); // 43
    
    // 带错误处理
    let bad_num = "not a number";
    match bad_num.parse::<i32>() {
        Ok(n) => println!("数字：{}", n),
        Err(e) => println!("解析错误：{}", e),
    }
    
    // 解析浮点数
    let float_str = "3.14";
    let float: f64 = float_str.parse().unwrap();
    println!("{}", float);
    
    // 解析布尔值
    let bool_str = "true";
    let b: bool = bool_str.parse().unwrap();
    println!("{}", b);
    
    // 解析自定义类型
    #[derive(Debug)]
    struct Point {
        x: i32,
        y: i32,
    }
    
    impl std::str::FromStr for Point {
        type Err = &'static str;
        
        fn from_str(s: &str) -> Result<Self, Self::Err> {
            let parts: Vec<&str> = s.split(',').collect();
            if parts.len() != 2 {
                return Err("格式错误");
            }
            
            let x = parts[0].parse::<i32>().map_err(|_| "x 解析错误")?;
            let y = parts[1].parse::<i32>().map_err(|_| "y 解析错误")?;
            
            Ok(Point { x, y })
        }
    }
    
    let point_str = "10,20";
    match point_str.parse::<Point>() {
        Ok(p) => println!("点：{:?}", p),
        Err(e) => println!("错误：{}", e),
    }
}

操作 时间复杂度 说明

fn is_palindrome(s: &str) -> bool {
    let cleaned: String = s
        .chars()
        .filter(|c| c.is_alphanumeric())
        .map(|c| c.to_lowercase().to_string())
        .collect::<String>()
        .to_lowercase();
    
    let reversed: String = cleaned.chars().rev().collect();
    
    cleaned == reversed
}

fn main() {
    println!("{}", is_palindrome("A man, a plan, a canal: Panama")); // true
    println!("{}", is_palindrome("race a car")); // false
    println!("{}", is_palindrome("上海自来水来自海上")); // true
}

fn count_char_types(s: &str) -> (usize, usize, usize, usize) {
    let mut letters = 0;
    let mut digits = 0;
    let mut whitespace = 0;
    let mut other = 0;
    
    for c in s.chars() {
        if c.is_alphabetic() {
            letters += 1;
        } else if c.is_numeric() {
            digits += 1;
        } else if c.is_whitespace() {
            whitespace += 1;
        } else {
            other += 1;
        }
    }
    
    (letters, digits, whitespace, other)
}

fn main() {
    let s = "Hello 123! 世界";
    let (l, d, w, o) = count_char_types(s);
    println!("字母：{}, 数字：{}, 空白：{}, 其他：{}", l, d, w, o);
}

fn snake_to_camel(s: &str) -> String {
    let mut result = String::new();
    let mut capitalize_next = false;
    
    for c in s.chars() {
        if c == '_' {
            capitalize_next = true;
        } else if capitalize_next {
            result.extend(c.to_uppercase());
            capitalize_next = false;
        } else {
            result.push(c);
        }
    }
    
    result
}

fn main() {
    println!("{}", snake_to_camel("hello_world")); // helloWorld
    println!("{}", snake_to_camel("convert_to_camel_case")); // convertToCamelCase
}

struct TextEditor {
    content: String,
}

impl TextEditor {
    fn new() -> Self {
        TextEditor {
            content: String::new(),
        }
    }
    
    fn insert(&mut self, pos: usize, text: &str) {
        if pos > self.content.len() {
            self.content.push_str(text);
        } else {
            self.content.insert_str(pos, text);
        }
    }
    
    fn delete(&mut self, start: usize, end: usize) {
        if start < self.content.len() && end <= self.content.len() {
            self.content.replace_range(start..end, "");
        }
    }
    
    fn find(&self, pattern: &str) -> Option<usize> {
        self.content.find(pattern)
    }
    
    fn replace(&mut self, old: &str, new: &str) -> usize {
        let mut count = 0;
        while let Some(pos) = self.content.find(old) {
            self.content.replace_range(pos..pos + old.len(), new);
            count += 1;
        }
        count
    }
    
    fn get_content(&self) -> &str {
        &self.content
    }
}

fn main() {
    let mut editor = TextEditor::new();
    
    editor.insert(0, "Hello, World!");
    println!("{}", editor.get_content());
    
    editor.insert(7, "Rust ");
    println!("{}", editor.get_content());
    
    editor.delete(0, 7);
    println!("{}", editor.get_content());
    
    if let Some(pos) = editor.find("Rust") {
        println!("找到 'Rust' 在位置 {}", pos);
    }
    
    let count = editor.replace("Rust", "Rustlang");
    println!("替换了 {} 处", count);
    println!("{}", editor.get_content());
}

💭 今日感悟

今天的 String 学习让我深刻体会到 Rust 对正确性的坚持：

  1. UTF-8 的复杂性：以前在其他语言中，字符串索引是很自然的操作。但在 Rust 中，这种"不方便"实际上是在保护我们免受编码错误的困扰。

  2. String vs &str：这个区分初看很烦，但想明白后就觉得很合理。拥有所有权 vs 借用，这是 Rust 核心哲学的体现。

  3. API 设计的美感：String 的 API 设计非常一致。push/push_str、insert/insert_str、trim/trim_start/trim_end，这种命名模式让学习曲线变得平缓。

  4. 性能意识：Rust 让你清楚地知道每个操作的成本。预分配容量、避免不必要的复制，这些优化在其他语言中可能是隐式的，但在 Rust 中是显式的。

  5. format! 宏的便利：虽然 + 运算符存在，但 format! 才是日常使用的王者。它不消耗所有权，语法也更清晰。
