    Some(val) => println!("值：{}", val),
    None => println("无值"),

// 方式 3: unwrap (可能 panic)
let val = opt.unwrap();  // None 时 panic!

// 方式 4: unwrap_or (提供默认值)
let val = opt.unwrap_or(0);  // None 时返回 0

// 方式 5: unwrap_or_else (延迟计算默认值)
let val = opt.unwrap_or_else(|| {
    println!("使用默认值");
    0
});

// 方式 1: match (最完整，推荐)

    Ok(val) => println!("成功：{}", val),
    Err(e) => println!("错误：{}", e),
}

// 方式 2: if let (只处理成功)
if let Ok(val) = res {
    println!("成功：{}", val);
}

// 方式 3: unwrap (可能 panic)
let val = res.unwrap();  // Err 时 panic!

// 方式 4: expect (带消息的 panic)
let val = res.expect("期望操作成功");

// 方式 5: unwrap_or (提供默认值)
let val = res.unwrap_or(0);  // Err 时返回 0

// 方式 6: ? 运算符 (传播错误)
fn caller() -> Result<i32, &str> {
    let val = res?;  // Err 时直接返回
    Ok(val * 2)
}

// 在返回 Result 的函数中
fn read_config() -> Result<Config, io::Error> {
    let file = File::open("config.txt")?;  // Err 时直接返回
    let content = read_to_string(file)?;   // Err 时直接返回
    let config = parse(content)?;          // Err 时直接返回
    Ok(config)
}

// 等价于
fn read_config_verbose() -> Result<Config, io::Error> {
    let file = match File::open("config.txt") {
        Ok(f) => f,
        Err(e) => return Err(e),
    };
    
    let content = match read_to_string(file) {
        Ok(c) => c,
        Err(e) => return Err(e),
    };
    
    let config = match parse(content) {
        Ok(c) => c,
        Err(e) => return Err(e),
    };
    
    Ok(config)
}

// ? 会自动调用 From::from 进行类型转换
fn process() -> Result<(), AppError> {
    let file = File::open("data.txt")?;  // io::Error -> AppError
    let content = read_to_string(file)?; // io::Error -> AppError
    let parsed = content.parse::<i32>()?; // ParseIntError -> AppError
    Ok(())
}

// 需要实现 From trait
impl From<io::Error> for AppError {
    fn from(err: io::Error) -> Self {
        AppError::Io(err)
    }
}

impl From<ParseIntError> for AppError {
    fn from(err: ParseIntError) -> Self {
        AppError::Parse(err)
    }
}

#[derive(Debug)]
pub enum AppError {
    // 按子系统分类
    Config(ConfigError),
    Database(DatabaseError),
    Network(NetworkError),

    // 包含足够的上下文

    Io(io::Error),

    String(String), // 类型擦除，丢失信息
}

use std::fmt;
use std::error::Error;

#[derive(Debug)]
pub enum AppError {
    Config(String),
    Io(io::Error),
}

// Display 实现（必需）
impl fmt::Display for AppError {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        match self {
            AppError::Config(msg) => write!(f, "配置错误：{}", msg),
            AppError::Io(err) => write!(f, "IO 错误：{}", err),
        }
    }
}

// Error trait 实现（推荐）
impl Error for AppError {
    fn source(&self) -> Option<&(dyn Error + 'static)> {
        match self {
            AppError::Config(_) => None,
            AppError::Io(err) => Some(err),
        }
    }
}

// From 实现（方便转换）
impl From<io::Error> for AppError {
    fn from(err: io::Error) -> Self {
        AppError::Io(err)
    }
}

use thiserror::Error;

#[derive(Error, Debug)]
pub enum AppError {
    #[error("配置错误：{0}")]
    Config(String),
    
    #[error("IO 错误")]
    Io(#[from] io::Error),  // 自动实现 From
    
    #[error("文件 {path} 未找到")]
    FileNotFound { path: String },
    
    #[error("操作 {operation} 失败：{source}")]
    OperationFailed {
        operation: String,
        #[source]
        source: Box<dyn Error + 'static>,
    },
}

// ✅ 好：添加上下文
trait ResultExt<T, E> {
    fn context(self, msg: &str) -> Result<T, AppError>;
}

impl<T, E: std::error::Error> ResultExt<T, E> for Result<T, E> {
    fn context(self, msg: &str) -> Result<T, AppError> {
        self.map_err(|e| AppError::WithContext {
            message: msg.to_string(),
            source: Box::new(e),
        })
    }
}

// 使用
let content = fs::read_to_string("config.txt")
    .context("读取配置文件失败")?;

pub fn get_item(vec: &Vec<i32>, index: usize) -> Option<i32> {
    vec.get(index).copied()
}

pub fn get_item_result(vec: &Vec<i32>, index: usize) -> Result<i32, IndexError> {
    vec.get(index)
        .copied()
        .ok_or(IndexError::OutOfBounds { index, len: vec.len() })
}

Q1: 什么时候用 panic!，什么时候用 Result？

文件/网络错误 Result 外部环境问题，可处理
配置错误 Result 可回退到默认值
除零 panic! (Release) 逻辑错误
数组越界 panic! 逻辑错误

fn main() -> Result<(), Box<dyn std::error::Error>> {

点击查看答案

fn main() -> Option<()> {

    let result = value?;  // 提前返回 None
    println!("{}", result);
    Some(())
}

    let config = serde_json::from_str(&content)?;

        Ok(config) => run_app(config),
