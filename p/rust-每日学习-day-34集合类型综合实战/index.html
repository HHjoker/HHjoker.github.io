// main.rs
use std::collections::HashMap;
use std::env;
use std::fs;

mod task;
mod store;
mod commands;

use task::{Task, TaskStatus, Priority};
use store::TodoStore;
use commands::Command;

fn main() {
    let args: Vec<String> = env::args().collect();
    
    // 加载存储
    let mut store = TodoStore::load("todos.json").unwrap_or_else(|_| {
        println!("创建新的待办事项存储...");
        TodoStore::new()
    });
    
    // 解析命令
    let command = Command::parse(&args[1..]);
    
    // 执行命令
    match command {
        Command::Add { title, priority, tags } => {
            commands::add_task(&mut store, title, priority, tags);
        }
        Command::List { status, tag } => {
            commands::list_tasks(&store, status, tag);
        }
        Command::Complete { id } => {
            commands::complete_task(&mut store, id);
        }
        Command::Delete { id } => {
            commands::delete_task(&mut store, id);
        }
        Command::Search { query } => {
            commands::search_tasks(&store, query);
        }
        Command::Stats => {
            commands::show_stats(&store);
        }
        Command::Help => {
            print_help();
        }
        Command::Unknown => {
            eprintln!("未知命令。使用 'todo help' 查看帮助。");
        }
    }
    
    // 保存存储
    store.save("todos.json").expect("保存失败");
}

use serde::{Deserialize, Serialize};

/// 任务优先级
#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
pub enum Priority {
    Low,
    Medium,
    High,
}

impl Default for Priority {
    fn default() -> Self {
        Priority::Medium
    }
}

impl std::fmt::Display for Priority {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        match self {
            Priority::Low => write!(f, "🟢 低"),
            Priority::Medium => write!(f, "🟡 中"),
            Priority::High => write!(f, "🔴 高"),
        }
    }
}

/// 任务状态
#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
pub enum TaskStatus {
    Pending,
    Completed,
}

impl Default for TaskStatus {
    fn default() -> Self {
        TaskStatus::Pending
    }
}

impl std::fmt::Display for TaskStatus {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        match self {
            TaskStatus::Pending => write!(f, "⏳ 待处理"),
            TaskStatus::Completed => write!(f, "✅ 已完成"),
        }
    }
}

/// 任务结构
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct Task {
    pub id: u64,
    pub title: String,
    pub description: Option<String>,
    pub status: TaskStatus,
    pub priority: Priority,
    pub tags: Vec<String>,
    pub created_at: String,
    pub completed_at: Option<String>,
}

impl Task {
    pub fn new(id: u64, title: String) -> Self {
        Task {
            id,
            title,
            description: None,
            status: TaskStatus::default(),
            priority: Priority::default(),
            tags: Vec::new(),
            created_at: chrono::Local::now().format("%Y-%m-%d %H:%M:%S").to_string(),
            completed_at: None,
        }
    }
    
    pub fn with_priority(mut self, priority: Priority) -> Self {
        self.priority = priority;
        self
    }
    
    pub fn with_tags(mut self, tags: Vec<String>) -> Self {
        self.tags = tags;
        self
    }
    
    pub fn with_description(mut self, description: String) -> Self {
        self.description = Some(description);
        self
    }
    
    pub fn complete(&mut self) {
        self.status = TaskStatus::Completed;
        self.completed_at = Some(
            chrono::Local::now().format("%Y-%m-%d %H:%M:%S").to_string()
        );
    }
    
    pub fn has_tag(&self, tag: &str) -> bool {
        self.tags.iter().any(|t| t == tag)
    }
    
    pub fn matches_query(&self, query: &str) -> bool {
        let query_lower = query.to_lowercase();
        self.title.to_lowercase().contains(&query_lower)
            || self.description.as_ref()
                .map(|d| d.to_lowercase().contains(&query_lower))
                .unwrap_or(false)
            || self.tags.iter().any(|t| t.to_lowercase().contains(&query_lower))
    }
}

impl std::fmt::Display for Task {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        let status_icon = match self.status {
            TaskStatus::Pending => "⬜",
            TaskStatus::Completed => "✅",
        };
        
        write!(
            f,
            "[{}] {} {} | {} | {}",
            self.id,
            status_icon,
            self.title,
            self.priority,
            self.status
        )?;
        
        if !self.tags.is_empty() {
            write!(f, " | #{}", self.tags.join(", #"))?;
        }
        
        Ok(())
    }
}

// store.rs
use crate::task::Task;
use serde::{Deserialize, Serialize};
use std::collections::HashMap;
use std::fs;

/// 待办事项存储
#[derive(Debug, Serialize, Deserialize, Default)]
pub struct TodoStore {
    tasks: HashMap<u64, Task>,
    next_id: u64,
}

impl TodoStore {
    pub fn new() -> Self {
        TodoStore {
            tasks: HashMap::new(),
            next_id: 1,
        }
    }
    
    pub fn load(path: &str) -> Result<Self, Box<dyn std::error::Error>> {
        let content = fs::read_to_string(path)?;
        let store: TodoStore = serde_json::from_str(&content)?;
        Ok(store)
    }
    
    pub fn save(&self, path: &str) -> Result<(), Box<dyn std::error::Error>> {
        let content = serde_json::to_string_pretty(self)?;
        fs::write(path, content)?;
        Ok(())
    }
    
    pub fn add_task(&mut self, task: Task) -> u64 {
        let id = self.next_id;
        self.tasks.insert(id, task);
        self.next_id += 1;
        id
    }
    
    pub fn get_task(&self, id: u64) -> Option<&Task> {
        self.tasks.get(&id)
    }
    
    pub fn get_task_mut(&mut self, id: u64) -> Option<&mut Task> {
        self.tasks.get_mut(&id)
    }
    
    pub fn remove_task(&mut self, id: u64) -> Option<Task> {
        self.tasks.remove(&id)
    }
    
    pub fn all_tasks(&self) -> Vec<&Task> {
        let mut tasks: Vec<&Task> = self.tasks.values().collect();
        tasks.sort_by(|a, b| b.id.cmp(&a.id)); // 按 ID 降序
        tasks
    }
    
    pub fn tasks_by_status(&self, status: &crate::task::TaskStatus) -> Vec<&Task> {
        let mut tasks: Vec<&Task> = self.tasks
            .values()
            .filter(|t| &t.status == status)
            .collect();
        tasks.sort_by(|a, b| b.id.cmp(&a.id));
        tasks
    }
    
    pub fn tasks_by_tag(&self, tag: &str) -> Vec<&Task> {
        let mut tasks: Vec<&Task> = self.tasks
            .values()
            .filter(|t| t.has_tag(tag))
            .collect();
        tasks.sort_by(|a, b| b.id.cmp(&a.id));
        tasks
    }
    
    pub fn search_tasks(&self, query: &str) -> Vec<&Task> {
        let mut tasks: Vec<&Task> = self.tasks
            .values()
            .filter(|t| t.matches_query(query))
            .collect();
        tasks.sort_by(|a, b| b.id.cmp(&a.id));
        tasks
    }
    
    pub fn count_by_status(&self, status: &crate::task::TaskStatus) -> usize {
        self.tasks.values().filter(|t| &t.status == status).count()
    }
    
    pub fn count_by_priority(&self, priority: &crate::task::Priority) -> usize {
        self.tasks.values().filter(|t| &t.priority == priority).count()
    }
    
    pub fn total_tasks(&self) -> usize {
        self.tasks.len()
    }
    
    pub fn get_tags(&self) -> Vec<String> {
        let mut tags: Vec<String> = self.tasks
            .values()
            .flat_map(|t| t.tags.iter())
            .cloned()
            .collect();
        tags.sort();
        tags.dedup();
        tags
    }
}

// commands.rs
use crate::task::{Task, Priority, TaskStatus};
use crate::store::TodoStore;
use std::collections::HashMap;

/// 命令行命令
#[derive(Debug)]
pub enum Command {
    Add {
        title: String,
        priority: Priority,
        tags: Vec<String>,
    },
    List {
        status: Option<TaskStatus>,
        tag: Option<String>,
    },
    Complete {
        id: u64,
    },
    Delete {
        id: u64,
    },
    Search {
        query: String,
    },
    Stats,
    Help,
    Unknown,
}

impl Command {
    pub fn parse(args: &[String]) -> Self {
        if args.is_empty() {
            return Command::Unknown;
        }
        
        match args[0].as_str() {
            "add" => Self::parse_add(&args[1..]),
            "list" => Self::parse_list(&args[1..]),
            "complete" => {
                if args.len() < 2 {
                    Command::Unknown
                } else {
                    args[1].parse::<u64>()
                        .map(|id| Command::Complete { id })
                        .unwrap_or(Command::Unknown)
                }
            }
            "delete" => {
                if args.len() < 2 {
                    Command::Unknown
                } else {
                    args[1].parse::<u64>()
                        .map(|id| Command::Delete { id })
                        .unwrap_or(Command::Unknown)
                }
            }
            "search" => {
                if args.len() < 2 {
                    Command::Unknown
                } else {
                    Command::Search {
                        query: args[1..].join(" "),
                    }
                }
            }
            "stats" => Command::Stats,
            "help" => Command::Help,
            _ => Command::Unknown,
        }
    }
    
    fn parse_add(args: &[String]) -> Command {
        if args.is_empty() {
            return Command::Unknown;
        }
        
        let mut title = String::new();
        let mut priority = Priority::Medium;
        let mut tags = Vec::new();
        
        let mut i = 0;
        while i < args.len() {
            match args[i].as_str() {
                "--priority" => {
                    if i + 1 < args.len() {
                        priority = match args[i + 1].as_str() {
                            "high" => Priority::High,
                            "low" => Priority::Low,
                            _ => Priority::Medium,
                        };
                        i += 2;
                    } else {
                        i += 1;
                    }
                }
                "--tags" => {
                    if i + 1 < args.len() {
                        tags = args[i + 1]
                            .split(',')
                            .map(|s| s.trim().to_string())
                            .collect();
                        i += 2;
                    } else {
                        i += 1;
                    }
                }
                _ => {
                    if title.is_empty() {
                        title = args[i].clone();
                    } else {
                        title.push(' ');
                        title.push_str(&args[i]);
                    }
                    i += 1;
                }
            }
        }
        
        if title.is_empty() {
            Command::Unknown
        } else {
            Command::Add {
                title,
                priority,
                tags,
            }
        }
    }
    
    fn parse_list(args: &[String]) -> Command {
        let mut status = None;
        let mut tag = None;
        
        let mut i = 0;
        while i < args.len() {
            match args[i].as_str() {
                "--status" => {
                    if i + 1 < args.len() {
                        status = match args[i + 1].as_str() {
                            "pending" => Some(TaskStatus::Pending),
                            "completed" => Some(TaskStatus::Completed),
                            _ => None,
                        };
                        i += 2;
                    } else {
                        i += 1;
                    }
                }
                "--tag" => {
                    if i + 1 < args.len() {
                        tag = Some(args[i + 1].clone());
                        i += 2;
                    } else {
                        i += 1;
                    }
                }
                _ => i += 1,
            }
        }
        
        Command::List { status, tag }
    }
}

pub fn add_task(store: &mut TodoStore, title: String, priority: Priority, tags: Vec<String>) {
    let task = Task::new(store.next_id, title)
        .with_priority(priority)
        .with_tags(tags);
    
    let id = store.add_task(task);
    println!("✅ 任务已添加：ID = {}", id);
}

pub fn list_tasks(store: &TodoStore, status: Option<TaskStatus>, tag: Option<String>) {
    let tasks = match (status, tag) {
        (Some(s), None) => store.tasks_by_status(&s),
        (None, Some(t)) => store.tasks_by_tag(&t),
        (None, None) => store.all_tasks(),
        (Some(_), Some(_)) => {
            // 同时按状态和标签筛选
            store.all_tasks()
                .into_iter()
                .filter(|t| {
                    let status_match = status.as_ref().map_or(true, |s| &t.status == s);
                    let tag_match = tag.as_ref().map_or(true, |t_tag| t.has_tag(t_tag));
                    status_match && tag_match
                })
                .collect()
        }
    };
    
    if tasks.is_empty() {
        println!("📭 没有找到任务");
        return;
    }
    
    println!("📋 待办事项 (共 {} 个):\n", tasks.len());
    for task in tasks {
        println!("{}", task);
    }
}

pub fn complete_task(store: &mut TodoStore, id: u64) {
    match store.get_task_mut(id) {
        Some(task) => {
            task.complete();
            println!("✅ 任务 {} 已完成：{}", id, task.title);
        }
        None => {
            eprintln!("❌ 未找到任务 ID: {}", id);
        }
    }
}

pub fn delete_task(store: &mut TodoStore, id: u64) {
    match store.remove_task(id) {
        Some(task) => {
            println!("🗑️  任务已删除：{}", task.title);
        }
        None => {
            eprintln!("❌ 未找到任务 ID: {}", id);
        }
    }
}

pub fn search_tasks(store: &TodoStore, query: String) {
    let tasks = store.search_tasks(&query);
    
    if tasks.is_empty() {
        println!("🔍 未找到匹配 \"{}\" 的任务", query);
        return;
    }
    
    println!("🔍 搜索结果 (\"{}\"): {}\n", query, tasks.len());
    for task in tasks {

pub fn matches_query(&self, query: &str) -> bool {
    let query_lower = query.to_lowercase();
    self.title.to_lowercase().contains(&query_lower)
        || self.description.as_ref()
            .map(|d| d.to_lowercase().contains(&query_lower))
            .unwrap_or(false)
        || self.tags.iter().any(|t| t.to_lowercase().contains(&query_lower))
}

技术要点：

  • 使用 to_lowercase() 实现不区分大小写搜索
  • 使用 Option::map() 处理可选字段
  • 使用 iter().any() 检查标签匹配

// task.rs
pub struct Task {
    // ... 其他字段
    pub due_date: Option<String>,  // 格式："YYYY-MM-DD"
}

impl Task {
    pub fn with_due_date(mut self, due_date: String) -> Self {
        self.due_date = Some(due_date);
        self
    }
    
    pub fn is_overdue(&self) -> bool {
        self.due_date.as_ref().map_or(false, |due| {
            let today = chrono::Local::now().format("%Y-%m-%d").to_string();
            due < &today && self.status == TaskStatus::Pending
        })
    }
    
    pub fn is_due_soon(&self, days: u32) -> bool {
        self.due_date.as_ref().map_or(false, |due| {
            let today = chrono::Local::now().date_naive();
            let due_date = chrono::NaiveDate::parse_from_str(due, "%Y-%m-%d").ok()?;
            let diff = due_date.signed_duration_since(today).num_days();
            diff >= 0 && diff <= days as i64 && self.status == TaskStatus::Pending
        })
    }
}

// commands.rs
pub fn list_overdue_tasks(store: &TodoStore) {
    let overdue: Vec<&Task> = store.all_tasks()
        .into_iter()
        .filter(|t| t.is_overdue())
        .collect();
    
    if overdue.is_empty() {
        println!("✅ 没有过期任务");
    } else {
        println!("⚠️  过期任务 ({} 个):\n", overdue.len());
        for task in overdue {
            println!("{} | 截止：{}", task, task.due_date.as_ref().unwrap());
        }
    }
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct SubTask {
    pub title: String,
    pub completed: bool,
}

pub struct Task {
    // ... 其他字段
    pub subtasks: Vec<SubTask>,
}

impl Task {
    pub fn add_subtask(&mut self, title: String) {
        self.subtasks.push(SubTask {
            title,
            completed: false,
        });
    }
    
    pub fn complete_subtask(&mut self, index: usize) -> Result<(), String> {
        if index >= self.subtasks.len() {
            return Err("子任务索引超出范围".to_string());
        }
        self.subtasks[index].completed = true;
        Ok(())
    }
    
    pub fn subtask_progress(&self) -> (usize, usize) {
        let total = self.subtasks.len();
        let completed = self.subtasks.iter().filter(|s| s.completed).count();
        (completed, total)
    }
}

// 显示时
impl std::fmt::Display for Task {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        // ... 原有显示逻辑
        
        if !self.subtasks.is_empty() {
            let (completed, total) = self.subtask_progress();
            write!(f, "\n    子任务进度：{}/{}", completed, total)?;
        }
        
        Ok(())
    }
}
