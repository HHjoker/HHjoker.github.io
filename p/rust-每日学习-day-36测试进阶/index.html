// src/lib.rs

mod internal {
    // 私有函数
    fn secret_calculation(x: i32) -> i32 {
        x * 42
    }

    // 私有结构体
    struct InternalData {
        value: i32,
    }

    impl InternalData {
        fn new(v: i32) -> Self {
            InternalData { value: v }
        }

        fn get_value(&self) -> i32 {
            self.value
        }
    }

    // 公开 API
    pub fn process(input: i32) -> i32 {
        let data = InternalData::new(input);
        secret_calculation(data.get_value())
    }

    // 测试模块可以访问所有私有项！
    #[cfg(test)]
    mod tests {
        use super::*;

        #[test]
        fn test_secret_calculation() {
            assert_eq!(secret_calculation(1), 42);
            assert_eq!(secret_calculation(2), 84);
        }

        #[test]
        fn test_internal_data() {
            let data = InternalData::new(100);
            assert_eq!(data.get_value(), 100);
        }

        #[test]
        fn test_process_public_api() {
            assert_eq!(process(1), 42);
            assert_eq!(process(2), 84);
        }
    }
}

为什么需要 Mock？

  • 隔离被测代码
  • 控制测试环境
  • 加快测试速度
  • 避免副作用

// 定义数据库操作的 Trait
pub trait Database {
    fn get_user(&self, id: u32) -> Option<String>;
    fn save_user(&mut self, id: u32, name: &str) -> Result<(), String>;
}

// 真实实现
pub struct RealDatabase {
    connection_string: String,
}

impl Database for RealDatabase {
    fn get_user(&self, id: u32) -> Option<String> {
        // 真实的数据库查询
        unimplemented!()
    }

    fn save_user(&mut self, id: u32, name: &str) -> Result<(), String> {
        // 真实的数据库写入
        unimplemented!()
    }
}

// 业务逻辑
pub struct UserService {
    db: Box<dyn Database>,
}

impl UserService {
    pub fn new(db: Box<dyn Database>) -> Self {
        UserService { db }
    }

    pub fn register_user(&mut self, id: u32, name: &str) -> Result<(), String> {
        // 检查用户是否已存在
        if self.db.get_user(id).is_some() {
            return Err("用户已存在".to_string());
        }
        // 保存新用户
        self.db.save_user(id, name)
    }
}

// Mock 实现
pub struct MockDatabase {
    users: std::collections::HashMap<u32, String>,
}

impl MockDatabase {
    pub fn new() -> Self {
        MockDatabase {
            users: std::collections::HashMap::new(),
        }
    }

    pub fn with_existing_user(id: u32, name: &str) -> Self {
        let mut mock = MockDatabase::new();
        mock.users.insert(id, name.to_string());
        mock
    }
}

impl Database for MockDatabase {
    fn get_user(&self, id: u32) -> Option<String> {
        self.users.get(&id).cloned()
    }

    fn save_user(&mut self, id: u32, name: &str) -> Result<(), String> {
        self.users.insert(id, name.to_string());
        Ok(())
    }
}

// 测试
#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn test_register_new_user() {
        let mock_db = MockDatabase::new();
        let mut service = UserService::new(Box::new(mock_db));

        let result = service.register_user(1, "Alice");
        assert!(result.is_ok());
    }

    #[test]
    fn test_register_existing_user() {
        let mock_db = MockDatabase::with_existing_user(1, "Alice");
        let mut service = UserService::new(Box::new(mock_db));

        let result = service.register_user(1, "Bob");
        assert!(result.is_err());
        assert_eq!(result.unwrap_err(), "用户已存在");
    }
}

// src/payment.rs

use mockall::automock;

// 使用 automock 自动生成 Mock
#[automock]
pub trait PaymentGateway {
    fn charge(&self, amount: f64, card: &str) -> Result<String, String>;
    fn refund(&self, transaction_id: &str) -> Result<(), String>;
}

// 业务逻辑
pub struct OrderProcessor {
    gateway: Box<dyn PaymentGateway>,
}

impl OrderProcessor {
    pub fn new(gateway: Box<dyn PaymentGateway>) -> Self {
        OrderProcessor { gateway }
    }

    pub fn process_order(&self, amount: f64, card: &str) -> Result<String, String> {
        if amount <= 0.0 {
            return Err("金额必须大于零".to_string());
        }
        self.gateway.charge(amount, card)
    }
}

// 测试
#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn test_process_order_success() {
        let mut mock_gateway = MockPaymentGateway::new();
        
        // 设置期望
        mock_gateway
            .expect_charge()
            .with(mockall::predicate::eq(100.0), mockall::predicate::always())
            .times(1)
            .returning(|_, _| Ok("txn_123".to_string()));

        let processor = OrderProcessor::new(Box::new(mock_gateway));
        let result = processor.process_order(100.0, "4111-1111-1111-1111");
        
        assert!(result.is_ok());
        assert_eq!(result.unwrap(), "txn_123");
    }

    #[test]
    fn test_process_order_invalid_amount() {
        let mock_gateway = MockPaymentGateway::new();
        let processor = OrderProcessor::new(Box::new(mock_gateway));
        
        let result = processor.process_order(-50.0, "4111-1111-1111-1111");
        
        assert!(result.is_err());
        assert_eq!(result.unwrap_err(), "金额必须大于零");
    }

    #[test]
    fn test_process_order_payment_failed() {
        let mut mock_gateway = MockPaymentGateway::new();
        
        mock_gateway
            .expect_charge()
            .returning(|_, _| Err("支付失败".to_string()));

        let processor = OrderProcessor::new(Box::new(mock_gateway));
        let result = processor.process_order(100.0, "4111-1111-1111-1111");
        
        assert!(result.is_err());
        assert_eq!(result.unwrap_err(), "支付失败");
    }
}

// 简单的桩实现
pub struct StubFileSystem {
    files: std::collections::HashMap<String, String>,
}

impl StubFileSystem {
    pub fn new() -> Self {
        StubFileSystem {
            files: std::collections::HashMap::new(),
        }
    }

    pub fn with_file(path: &str, content: &str) -> Self {
        let mut stub = StubFileSystem::new();
        stub.files.insert(path.to_string(), content.to_string());
        stub
    }
}

pub trait FileSystem {
    fn read_file(&self, path: &str) -> Result<String, String>;
    fn write_file(&mut self, path: &str, content: &str) -> Result<(), String>;
}

impl FileSystem for StubFileSystem {
    fn read_file(&self, path: &str) -> Result<String, String> {
        self.files
            .get(path)
            .cloned()
            .ok_or_else(|| format!("文件不存在：{}", path))
    }

    fn write_file(&mut self, path: &str, content: &str) -> Result<(), String> {
        self.files.insert(path.to_string(), content.to_string());
        Ok(())
    }
}

// 属性测试：测试属性

    for s in generate_strings() {

        let r2 = reverse(&r1);
        assert_eq!(r2, s);

// src/properties.rs

use proptest::prelude::*;

// 被测试的函数
pub fn add(a: i32, b: i32) -> i32 {
    a + b
}

pub fn multiply(a: i32, b: i32) -> i32 {
    a * b
}

pub fn reverse_string(s: String) -> String {
    s.chars().rev().collect()
}

pub fn sort_vector(mut v: Vec<i32>) -> Vec<i32> {
    v.sort();
    v
}

// 属性测试
proptest! {
    // 属性 1：加法交换律
    #[test]
    fn test_add_commutative(a in any::<i32>(), b in any::<i32>()) {
        prop_assert_eq!(add(a, b), add(b, a));
    }

    // 属性 2：加法结合律
    #[test]
    fn test_add_associative(
        a in any::<i32>(),
        b in any::<i32>(),
        c in any::<i32>()
    ) {
        prop_assert_eq!(add(add(a, b), c), add(a, add(b, c)));
    }

    // 属性 3：加零不变
    #[test]
    fn test_add_zero_identity(a in any::<i32>()) {
        prop_assert_eq!(add(a, 0), a);
        prop_assert_eq!(add(0, a), a);
    }

    // 属性 4：乘法交换律
    #[test]
    fn test_multiply_commutative(a in any::<i32>(), b in any::<i32>()) {
        prop_assert_eq!(multiply(a, b), multiply(b, a));
    }

    // 属性 5：字符串反转两次等于原字符串
    #[test]
    fn test_reverse_inverse(s in ".*") {
        let r1 = reverse_string(s.clone());
        let r2 = reverse_string(r1.clone());
        prop_assert_eq!(r2, s);
    }

    // 属性 6：排序后的向量长度不变
    #[test]
    fn test_sort_preserves_length(v in prop::collection::vec(any::<i32>(), 0..100)) {
        let sorted = sort_vector(v.clone());
        prop_assert_eq!(sorted.len(), v.len());
    }

    // 属性 7：排序后的向量是有序的
    #[test]
    fn test_sort_is_sorted(v in prop::collection::vec(any::<i32>(), 0..100)) {
        let sorted = sort_vector(v);
        for i in 1..sorted.len() {
            prop_assert!(sorted[i - 1] <= sorted[i]);
        }
    }

    // 属性 8：排序不改变元素总和
    #[test]
    fn test_sort_preserves_sum(v in prop::collection::vec(any::<i32>(), 0..100)) {
        let original_sum: i32 = v.iter().sum();
        let sorted_sum: i32 = sort_vector(v.clone()).iter().sum();
        prop_assert_eq!(original_sum, sorted_sum);
    }
}

use proptest::prelude::*;

// 自定义非空字符串策略
fn non_empty_string() -> impl Strategy<Value = String> {
    ".+".prop_map(|s| s.to_string())
}

// 自定义正整数策略
fn positive_i32() -> impl Strategy<Value = i32> {
    1..i32::MAX
}

// 自定义非零整数策略
fn non_zero_i32() -> impl Strategy<Value = i32> {
    prop_oneof![
        1..i32::MAX,
        i32::MIN..-1,
    ]
}

// 使用自定义策略
proptest! {
    #[test]
    fn test_divide_with_non_zero(s in non_empty_string()) {
        prop_assert!(!s.is_empty());
    }

    #[test]
    fn test_multiply_positive(a in positive_i32(), b in positive_i32()) {
        // 可能溢出，但这是属性测试要发现的！
        let _ = a * b;
    }
}

// benches/my_benchmark.rs

use criterion::{black_box, criterion_group, criterion_main, Criterion, BenchmarkId};

// 被测试的函数
fn fibonacci_recursive(n: u64) -> u64 {
    match n {
        0 => 0,
        1 => 1,
        _ => fibonacci_recursive(n - 1) + fibonacci_recursive(n - 2),
    }
}

fn fibonacci_iterative(n: u64) -> u64 {
    match n {
        0 => 0,
        1 => 1,
        _ => {
            let mut a = 0;
            let mut b = 1;
            for _ in 2..=n {
                let temp = a + b;
                a = b;
                b = temp;
            }
            b
        }
    }
}

fn fibonacci_memo(n: u64, memo: &mut Vec<u64>) -> u64 {
    if n <= 1 {
        return n;
    }
    if memo[n as usize] != 0 {
        return memo[n as usize];
    }
    memo[n as usize] = fibonacci_memo(n - 1, memo) + fibonacci_memo(n - 2, memo);
    memo[n as usize]
}

// 基准测试组
fn fibonacci_benchmark(c: &mut Criterion) {
    let mut group = c.benchmark_group("Fibonacci");

    for i in &[10, 20, 30] {
        group.bench_with_input(
            BenchmarkId::new("recursive", i),
            i,
            |b, &i| b.iter(|| fibonacci_recursive(black_box(i))),
        );
    }

    for i in &[10, 20, 30, 100, 1000] {
        group.bench_with_input(
            BenchmarkId::new("iterative", i),
            i,
            |b, &i| b.iter(|| fibonacci_iterative(black_box(i))),
        );
    }

    for i in &[10, 20, 30, 100, 1000] {
        group.bench_with_input(
            BenchmarkId::new("memo", i),
            i,
            |b, &i| {
                b.iter(|| {
                    let mut memo = vec![0; (i + 1) as usize];
                    fibonacci_memo(black_box(i), &mut memo)
                })
            },
        );
    }
}

fn string_benchmark(c: &mut Criterion) {
    let mut group = c.benchmark_group("String Operations");

    let test_string = "Hello, World! ".repeat(100);

    group.bench_function("chars_iter", |b| {
        b.iter(|| {

            s.chars().count()
        })
    });

    group.bench_function("bytes_iter", |b| {
        b.iter(|| {
            let s = black_box(&test_string);
            s.bytes().count()
        })
    });

    group.bench_function("split_whitespace", |b| {
        b.iter(|| {

        })
    });
}

// 集合操作基准测试
fn collection_benchmark(c: &mut Criterion) {
    let mut group = c.benchmark_group("Collection Operations");

    let numbers: Vec<i32> = (0..10000).collect();

    group.bench_function("vec_push", |b| {
        b.iter(|| {
            let mut v = Vec::new();
            for i in black_box(&numbers) {
                v.push(i);
            }
            v
        })
    });

    group.bench_function("vec_with_capacity", |b| {
        b.iter(|| {
            let mut v = Vec::with_capacity(numbers.len());
            for i in black_box(&numbers) {
                v.push(i);
            }
            v
        })
    });

    group.bench_function("vec_iter_sum", |b| {
        b.iter(|| {
            black_box(&numbers).iter().sum::<i32>()
        })
    });
}

criterion_group!(

    fibonacci_benchmark,
    string_benchmark,
    collection_benchmark

criterion_main!(benches);

4. 覆盖率解读

  • 行覆盖率（Line Coverage）：多少行代码被执行
  • 分支覆盖率（Branch Coverage）：多少分支被测试
  • 函数覆盖率（Function Coverage）：多少函数被调用
  • 区域覆盖率（Region Coverage）：更细粒度的代码块覆盖

目标：80%+ 覆盖率通常是合理的，100% 可能过度测试。

pub trait Cache {
    fn get(&self, key: &str) -> Option<String>;
    fn set(&mut self, key: &str, value: &str, ttl_secs: u64);
    fn delete(&mut self, key: &str) -> bool;
    fn clear(&mut self);
}

pub struct CacheService {
    cache: Box<dyn Cache>,
}

impl CacheService {
    pub fn new(cache: Box<dyn Cache>) -> Self {
        CacheService { cache }
    }

    pub fn get_or_fetch(&mut self, key: &str, fetch: impl FnOnce() -> String) -> String {
        match self.cache.get(key) {
            Some(value) => value,
            None => {
                let value = fetch();
                self.cache.set(key, &value, 3600);
                value
            }
        }
    }
}

pub fn clamp(value: i32, min: i32, max: i32) -> i32 {
    if value < min {
        min
    } else if value > max {
        max
    } else {
        value
    }
}

pub fn binary_search(arr: &[i32], target: i32) -> Option<usize> {
    let mut left = 0;
    let mut right = arr.len();
    
    while left < right {
        let mid = left + (right - left) / 2;
        if arr[mid] == target {
            return Some(mid);
        } else if arr[mid] < target {
            left = mid + 1;
        } else {
            right = mid;
        }
    }
    None
}

要求：

  1. clamp 的结果总是在 [min, max] 范围内
  2. binary_search 找到的索引对应的值等于 target
  3. 如果 binary_search 返回 None，则 target 不在数组中
