use std::fs::File;
use std::io::{self, Read};

fn read_file_content(path: &str) -> io::Result<String> {
    let mut file = File::open(path)?;  // 如果出错，直接返回
    let mut content = String::new();
    file.read_to_string(&mut content)?;
    Ok(content)
}

// ? 运算符等价于：
fn read_file_content_explicit(path: &str) -> io::Result<String> {
    let mut file = match File::open(path) {
        Ok(f) => f,
        Err(e) => return Err(e),  // 提前返回
    };
    let mut content = String::new();
    match file.read_to_string(&mut content) {
        Ok(_) => Ok(content),
        Err(e) => Err(e),
    }
}

关键点：

  • ? 只能在返回 Result 或 Option 的函数中使用
  • 成功时解包值，失败时提前返回
  • 自动进行错误类型转换（通过 From trait）

use std::fs::File;
use std::io::{self, Read};
use std::num::ParseIntError;

// 自定义错误类型
#[derive(Debug)]
enum AppError {
    Io(io::Error),
    Parse(ParseIntError),
    EmptyFile,
}

// 实现 From trait 实现自动转换
impl From<io::Error> for AppError {
    fn from(err: io::Error) -> Self {
        AppError::Io(err)
    }
}

impl From<ParseIntError> for AppError {
    fn from(err: ParseIntError) -> Self {
        AppError::Parse(err)
    }
}

fn read_and_parse(path: &str) -> Result<i32, AppError> {
    let mut content = String::new();
    File::open(path)?.read_to_string(&mut content)?;  // io::Error 自动转为 AppError
    
    if content.trim().is_empty() {
        return Err(AppError::EmptyFile);
    }
    
    let number: i32 = content.trim().parse()?;  // ParseIntError 自动转为 AppError
    Ok(number)
}

use std::fs::File;
use std::io::{self, Read};

#[derive(Debug)]
enum DataError {
    FileNotFound(String),
    ReadFailed(String),
    ParseFailed(String),
}

fn read_config(path: &str) -> Result<String, DataError> {
    let mut file = File::open(path)
        .map_err(|e| DataError::FileNotFound(format!("无法打开 {}: {}", path, e)))?;
    
    let mut content = String::new();
    file.read_to_string(&mut content)
        .map_err(|e| DataError::ReadFailed(format!("读取失败：{}", e)))?;
    
    Ok(content)
}

fn process_numbers(strings: Vec<&str>) -> Vec<i32> {
    strings
        .into_iter()
        .filter_map(|s| s.parse::<i32>().ok())  // 失败的解析被过滤掉
        .collect()
}

fn process_numbers_with_errors(strings: Vec<&str>) -> Result<Vec<i32>, Vec<String>> {
    let (successes, errors): (Vec<_>, Vec<_>) = strings
        .into_iter()
        .map(|s| s.parse::<i32>()
            .map_err(|e| format!("'{}' 解析失败：{}", s, e)))
        .partition(|r| r.is_ok());
    
    if !errors.is_empty() {
        let error_msgs: Vec<String> = errors.into_iter()
            .filter_map(|r| r.err())
            .collect();
        return Err(error_msgs);
    }
    
    Ok(successes.into_iter().filter_map(|r| r.ok()).collect())
}

use std::fs;

#[derive(Debug)]
enum ConfigError {
    MissingFile,
    InvalidFormat(String),
    MissingField(String),
}

fn load_config(path: &str) -> Result<Config, ConfigError> {
    fs::read_to_string(path)
        .map_err(|_| ConfigError::MissingFile)?
        .lines()
        .find(|line| line.starts_with("version="))
        .ok_or_else(|| ConfigError::MissingField("version".to_string()))?
        .split('=')
        .nth(1)
        .ok_or_else(|| ConfigError::InvalidFormat("version 格式错误".to_string()))
        .map(|v| Config { version: v.to_string() })
}

#[derive(Debug)]
struct Config {
    version: String,
}

use std::fmt;

// 带上下文的错误包装器
#[derive(Debug)]
struct ContextError<E> {
    context: String,
    source: E,
}

impl<E> ContextError<E> {
    fn new(context: impl Into<String>, source: E) -> Self {
        ContextError {
            context: context.into(),
            source,
        }
    }
}

impl<E: fmt::Display> fmt::Display for ContextError<E> {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        write!(f, "{}: {}", self.context, self.source)
    }
}

impl<E: std::error::Error + 'static> std::error::Error for ContextError<E> {
    fn source(&self) -> Option<&(dyn std::error::Error + 'static)> {
        Some(&self.source)
    }
}

// 扩展 Result 添加 context 方法
trait Context<T, E> {
    fn context(self, msg: impl Into<String>) -> Result<T, ContextError<E>>;
    fn with_context(self, f: impl FnOnce() -> String) -> Result<T, ContextError<E>>;
}

impl<T, E> Context<T, E> for Result<T, E> {
    fn context(self, msg: impl Into<String>) -> Result<T, ContextError<E>> {
        self.map_err(|e| ContextError::new(msg, e))
    }
    
    fn with_context(self, f: impl FnOnce() -> String) -> Result<T, ContextError<E>> {
        self.map_err(|e| ContextError::new(f(), e))
    }
}

// 使用示例
use std::fs::File;

fn load_data(path: &str) -> Result<String, ContextError<io::Error>> {
    File::open(path)
        .context(format!("无法打开文件 '{}'", path))?
        .bytes()
        .collect::<Result<Vec<u8>, _>>()
        .context("读取文件内容失败")
        .map(|bytes| String::from_utf8_lossy(&bytes).to_string())
}

use std::fs::{self, File};
use std::io::{self, BufRead, BufReader, Write};
use std::path::{Path, PathBuf};

// ============ 错误定义 ============

#[derive(Debug)]
pub enum AppError {
    Config(ConfigError),
    Data(DataError),
    Io(io::Error),
}

#[derive(Debug)]
pub enum ConfigError {
    NotFound(PathBuf),
    InvalidFormat(String),
    MissingField(String),
}

#[derive(Debug)]
pub enum DataError {
    ValidationFailed(String),
    ProcessingFailed(String),
}

// ============ From 实现 ============

impl From<io::Error> for AppError {
    fn from(err: io::Error) -> Self {
        AppError::Io(err)
    }
}

impl From<ConfigError> for AppError {
    fn from(err: ConfigError) -> Self {
        AppError::Config(err)
    }
}

impl From<DataError> for AppError {
    fn from(err: DataError) -> Self {
        AppError::Data(err)
    }
}

// ============ Display 实现 ============

impl std::fmt::Display for AppError {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        match self {
            AppError::Config(e) => write!(f, "配置错误：{}", e),
            AppError::Data(e) => write!(f, "数据错误：{}", e),
            AppError::Io(e) => write!(f, "IO 错误：{}", e),
        }
    }
}

impl std::fmt::Display for ConfigError {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        match self {
            ConfigError::NotFound(path) => write!(f, "配置文件未找到：{}", path.display()),
            ConfigError::InvalidFormat(msg) => write!(f, "配置格式无效：{}", msg),
            ConfigError::MissingField(field) => write!(f, "缺少必需字段：{}", field),
        }
    }
}

impl std::fmt::Display for DataError {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        match self {
            DataError::ValidationFailed(msg) => write!(f, "数据验证失败：{}", msg),
            DataError::ProcessingFailed(msg) => write!(f, "数据处理失败：{}", msg),
        }
    }
}

impl std::error::Error for AppError {
    fn source(&self) -> Option<&(dyn std::error::Error + 'static)> {
        match self {
            AppError::Config(e) => Some(e),
            AppError::Data(e) => Some(e),
            AppError::Io(e) => Some(e),
        }
    }
}

// ============ 业务逻辑 ============

#[derive(Debug)]
pub struct AppConfig {
    pub data_dir: PathBuf,
    pub max_records: usize,
}

impl AppConfig {
    pub fn load(path: &Path) -> Result<Self, AppError> {
        if !path.exists() {
            return Err(ConfigError::NotFound(path.to_path_buf()).into());
        }
        
        let content = fs::read_to_string(path)?;
        let mut data_dir = None;
        let mut max_records = None;
        
        for line in content.lines() {
            let line = line.trim();
            if line.starts_with('#') || line.is_empty() {
                continue;
            }
            
            let parts: Vec<&str> = line.splitn(2, '=').collect();
            if parts.len() != 2 {
                return Err(ConfigError::InvalidFormat(format!("无效行：{}", line)).into());
            }
            
            match parts[0].trim() {
                "data_dir" => data_dir = Some(PathBuf::from(parts[1].trim())),
                "max_records" => {
                    max_records = Some(
                        parts[1].trim().parse().map_err(|_| {
                            ConfigError::InvalidFormat("max_records 必须是数字".to_string())
                        })?
                    );
                }
                _ => {}
            }
        }
        
        Ok(AppConfig {
            data_dir: data_dir.ok_or_else(|| {
                ConfigError::MissingField("data_dir".to_string())
            })?,
            max_records: max_records.ok_or_else(|| {
                ConfigError::MissingField("max_records".to_string())
            })?,
        })
    }
}

pub struct DataProcessor {
    config: AppConfig,
}

impl DataProcessor {
    pub fn new(config: AppConfig) -> Self {
        DataProcessor { config }
    }
    
    pub fn process_file(&self, filename: &str) -> Result<Vec<String>, AppError> {
        let path = self.config.data_dir.join(filename);
        
        let file = File::open(&path)
            .map_err(|e| {
                AppError::Io(io::Error::new(
                    e.kind(),
                    format!("无法打开 {}: {}", path.display(), e)
                ))
            })?;
        
        let reader = BufReader::new(file);
        let mut records = Vec::new();
        
        for (line_num, line_result) in reader.lines().enumerate() {
            if records.len() >= self.config.max_records {
                break;
            }
            
            let line = line_result?;
            let validated = self.validate_line(&line, line_num + 1)?;
            records.push(validated);
        }
        
        Ok(records)
    }
    
    fn validate_line(&self, line: &str, line_num: usize) -> Result<String, AppError> {
        if line.trim().is_empty() {
            return Err(DataError::ValidationFailed(
                format!("第 {} 行为空", line_num)
            ).into());
        }
        
        if line.contains('\0') {
            return Err(DataError::ValidationFailed(
                format!("第 {} 行包含空字符", line_num)
            ).into());
        }
        
        Ok(line.to_string())
    }
    
    pub fn write_report(&self, output_path: &Path, records: &[String]) -> Result<(), AppError> {
        let mut file = File::create(output_path)?;
        
        writeln!(file, "数据处理报告")?;
        writeln!(file, "===============")?;
        writeln!(file, "记录数：{}", records.len())?;
        writeln!(file)?;
        
        for (i, record) in records.iter().enumerate() {
            writeln!(file, "{}. {}", i + 1, record)?;
        }
        
        Ok(())
    }
}

// ============ 主函数 ============

fn main() -> Result<(), AppError> {
    let config_path = Path::new("config.txt");
    
    // 创建示例配置
    let mut config_file = File::create(config_path)?;
    writeln!(config_file, "# 应用配置")?;
    writeln!(config_file, "data_dir = ./data")?;
    writeln!(config_file, "max_records = 100")?;
    
    // 加载配置
    let config = AppConfig::load(config_path)?;
    println!("✓ 配置加载成功");
    println!("  数据目录：{}", config.data_dir.display());
    println!("  最大记录数：{}", config.max_records);
    
    // 创建数据目录和示例数据
    fs::create_dir_all(&config.data_dir)?;
    let data_path = config.data_dir.join("sample.txt");
    let mut data_file = File::create(&data_path)?;
    writeln!(data_file, "记录 1")?;
    writeln!(data_file, "记录 2")?;
    writeln!(data_file, "记录 3")?;
    
    // 处理数据
    let processor = DataProcessor::new(config);
    let records = processor.process_file("sample.txt")?;
    println!("✓ 处理了 {} 条记录", records.len());
    
    // 生成报告
    let report_path = Path::new("report.txt");
    processor.write_report(report_path, &records)?;
    println!("✓ 报告已生成：{}", report_path.display());
    
    Ok(())
}

#[cfg(test)]
mod tests {
    use super::*;
    
    #[test]
    fn test_config_load_missing() {
        let result = AppConfig::load(Path::new("nonexistent.txt"));
        assert!(matches!(result, Err(AppError::Config(ConfigError::NotFound(_)))));
    }
    
    #[test]
    fn test_validation_empty_line() {
        let config = AppConfig {
            data_dir: PathBuf::from("./data"),
            max_records: 100,
        };
        let processor = DataProcessor::new(config);
        let result = processor.validate_line("", 1);
        assert!(matches!(result, Err(AppError::Data(DataError::ValidationFailed(_)))));
    }
}

│     - 大部分错误应该用 Result 处理                       │
│                                                         │
│  2. 提供有意义的错误信息                                 │
│     ❌ return Err("failed".into());                    │
│     ✅ return Err(format!("无法连接数据库：{}", e));    │
│                                                         │
│  3. 使用 ? 运算符简化传播                               │
│     - 避免嵌套 match                                     │
│     - 保持代码线性可读                                   │
│                                                         │
│  4. 在边界处转换错误                                     │
│     - 库内部：使用具体错误类型                           │
│     - 库边界：转换为统一的错误类型                       │
│     - 应用层：使用 anyhow 添加上下文                     │
│                                                         │
│  5. 实现 Display 和 Error trait                         │
│     - 让错误可以被打印和链式处理                         │
│     - 提供 source() 方法保留原始错误                     │
│                                                         │
│  6. 使用 thiserror 减少样板代码                         │
│     #[derive(thiserror::Error, Debug)]                 │
│     enum MyError {                                     │
│         #[error("IO 错误：{0}")]                       │

? 运算符展开：

expr?  等价于：

match expr {
    Ok(val) => val,
    Err(err) => return Err(err.into()),  // 注意 .into() 转换
}

对于 Option：

match expr {
    Some(val) => val,
    None => return None,
}

今日感悟

今天的错误处理实战让我深刻理解了 Rust 错误处理系统的优雅之处。

最大的收获：

  1. ? 运算符不仅仅是语法糖，它强制我思考每个操作可能失败的情况

  • 强制处理错误 → 减少运行时崩溃
  • 类型安全的错误 → 编译器帮我检查是否遗漏
  • 错误上下文 → 调试时快速定位问题
