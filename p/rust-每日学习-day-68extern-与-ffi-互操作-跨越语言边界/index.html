今日内容

昨天我们了解了 unsafe 的核心能力：解引用原始指针、调用 unsafe 函数等。今天我们继续探索 unsafe 的高阶用法——通过 extern 关键字与 C 语言等外部代码 进行互操作，这就是 FFI（Foreign Function Interface，外部函数接口）。

1. extern 块基础

extern 块用于声明外部库中的函数。Rust 代码可以调用任何外部语言编写的函数，只要它们遵循特定的 ABI（Application Binary Interface，应用程序二进制接口）。

// extern 块声明外部 C 函数
extern "C" {
    // C 标准库中的 puts 函数
    fn puts(s: *const c_char);
    
    // 内存分配/释放
    fn malloc(size: usize) -> *mut c_void;
    fn free(ptr: *mut c_void);
}

fn main() {
    // 调用外部 C 函数
    unsafe {
        // 创建一个 C 字符串
        let msg = CString::new("Hello from Rust via FFI!").unwrap();
        
        // 调用 C 的 puts 函数
        puts(msg.as_ptr());
    }
}

注意：上面的代码需要引入 std::ffi 中的 CString 和 c_char 等类型。

use std::ffi::{CString, CStr};
use std::os::raw::c_char;

extern "C" {
    fn printf(format: *const c_char, ...);
    fn strcmp(s1: *const c_char, s2: *const c_char) -> i32;
}

fn rust_to_c() {
    unsafe {
        // 创建 C 字符串（CString 会追加 \0 并处理 UTF-8）
        let rust_string = "Hello from Rust!";
        let c_string = CString::new(rust_string).unwrap();
        
        // 传递指针给 C 函数
        printf(c_string.as_ptr() as *const c_char);
        
        // CString 在离开作用域时会自动释放内存
    }
}

fn comparing_strings() {
    let s1 = CString::new("hello").unwrap();
    let s2 = CString::new("hello").unwrap();
    let s3 = CString::new("world").unwrap();
    
    unsafe {
        // strcmp 返回 0 表示相等
        assert_eq!(strcmp(s1.as_ptr(), s2.as_ptr()), 0);
        assert!(strcmp(s1.as_ptr(), s3.as_ptr()) != 0);
    }
}

fn main() {
    rust_to_c();
    comparing_strings();
}

use std::ffi::{CStr, CString};
use std::os::raw::c_char;

extern "C" {
    // 假设这是一个返回 const char* 的 C 函数
    fn get_c_string() -> *const c_char;
    
    // 假设这是一个修改字符串的 C 函数
    fn modify_string(s: *mut c_char);
}

fn c_to_rust() {
    unsafe {
        let c_str_ptr = get_c_string();
        
        if c_str_ptr.is_null() {
            println!("C 函数返回了空指针");
            return;
        }
        
        // 将 C 字符串转换为 Rust 的 &str
        let rust_str = CStr::from_ptr(c_str_ptr)
            .to_string_lossy()
            .into_owned();
        
        println!("从 C 获取的字符串: {}", rust_str);
    }
}

fn rust_string_to_c() {
    let mut s = CString::new("original").unwrap();
    
    unsafe {
        modify_string(s.as_mut_ptr());
        
        // C 函数可能修改了字符串
        let result = CStr::from_ptr(s.as_ptr())
            .to_string_lossy()
            .into_owned();
        
        println!("修改后的字符串: {}", result);
    }
}

fn main() {
    c_to_rust();
}

let bad = CString::new("hello").unwrap();
// C 期望以 \0 结尾的字符串，CString 会自动添加

// ❌ 常见错误 2：传入包含 \0 的字符串
let with_nul = CString::new(vec![b'H', b'i', b'\0', b't', b'h', b'e', b'r', b'e']);
// CString::new 会失败，因为 \0 在中间
// 改用 from_vec_unchecked：
let safe = unsafe { CString::from_vec_unchecked(vec![b'H', b'i', b'\0', b't', b'h', b'e', b'r', b'e']) };

// ❌ 常见错误 3：忘记指针生命周期
// 确保 CString 的生命周期覆盖指针的使用时间

// ✅ 最佳实践：使用 Arc<CString> 在多个地方共享
use std::sync::Arc;
use std::ffi::CString;

fn shared_cstring() {
    let shared = Arc::new(CString::new("shared string").unwrap());
    
    // 克隆 Arc 不会复制底层数据，只是增加引用计数
    let clone1 = Arc::clone(&shared);
    
    // 在某个线程中使用
    std::thread::spawn(move || {
        unsafe {
            // 使用 clone1
            call_c_function(clone1.as_ptr());
        }
    });
    
    // 原始 shared 可以继续使用
}

fn pass_pointers() {
    let mut value = 42i32;
    
    unsafe {
        // 传递 &mut i32 作为 *mut c_void
        process_ptr(&mut value as *mut i32 as *mut c_void);
        
        // 传递数组
        let arr = [1i32, 2, 3, 4, 5];
        process_array(arr.as_ptr() as *const c_void, arr.len());
    }
}

fn receive_pointer() {

        let ptr = allocate_buffer(1024);
        
        if ptr.is_null() {
            panic!("分配失败");
        }
        
        // 使用 ptr 中的数据...
        
        // 使用完毕后释放

    }
}

// 处理可空指针
fn handle_nullable() {
    unsafe {
        let ptr = get_optional_pointer();
        
        if let Some(p) = NonNull::new(ptr) {
            // 安全使用 NonNull<T>
            let value = p.as_ref();
            // ...

    }
}

type Callback = extern "C" fn(data: *const c_char, len: usize) -> i32;

extern "C" {
    // 注册回调
    fn register_callback(cb: Callback);
    
    // 触发回调（假设 C 端会调用这个）
    fn trigger_callback();
}

static mut CALLBACK_STORAGE: Option<Callback> = None;

// 回调实现
extern "C" fn rust_callback(data: *const c_char, len: usize) -> i32 {
    unsafe {
        if let Some(cb) = CALLBACK_STORAGE {
            // 调用存储的回调
            cb(data, len)
        } else {
            -1  // 没有注册回调
        }
    }
}

fn setup_callback() {
    unsafe {
        CALLBACK_STORAGE = Some(rust_callback);
        register_callback(rust_callback);
    }
}

// 使用 thread-local 存储让回调更安全
use std::cell::RefCell;

thread_local! {
    static CALLBACK: RefCell<Option<extern "C" fn(*const c_char, usize) -> i32>> = RefCell::new(None);
}

extern "C" fn safe_callback(data: *const c_char, len: usize) -> i32 {
    CALLBACK.with(|cell| {
        if let Some(cb) = *cell.borrow() {
            cb(data, len)
        } else {
            -1
        }
    })
}

// typedef struct {
//     int id;
//     char name[64];
//     double score;
// } Student;
// 
// void student_init(Student* s, int id, const char* name);
// double student_get_score(const Student* s);
// void student_set_score(Student* s, double score);
// const char* student_get_name(const Student* s);
// void student_destroy(Student* s);
// ============================================================

use std::ffi::{CStr, CString};
use std::os::raw::{c_char, c_double, c_int};
use std::ptr;
use std::mem::size_of;

// 模拟 C 的 Student 结构体内存布局
#[repr(C)]
struct Student {
    id: c_int,
    name: [c_char; 64],
    score: c_double,
}

impl Student {
    fn new(id: c_int, name: &str) -> Self {
        let mut s = Student {
            id,
            name: [0; 64],
            score: 0.0,
        };
        
        // 复制名字到 name 数组
        let name_bytes = name.as_bytes();
        let len = name_bytes.len().min(63);
        for (i, &byte) in name_bytes.iter().take(len).enumerate() {
            s.name[i] = byte as c_char;
        }
        
        s
    }
    
    fn id(&self) -> i32 { self.id }
    
    fn score(&self) -> f64 { self.score }
    
    fn set_score(&mut self, score: f64) { self.score = score; }
    
    fn name(&self) -> String {
        unsafe {
            CStr::from_ptr(self.name.as_ptr())
                .to_string_lossy()
                .into_owned()
        }
    }
}

fn main() {
    println!("=== Rust FFI 完整示例：模拟学生结构体 ===\n");
    
    // 模拟 C 库的初始化
    let mut student = Student::new(2026001, "Rust Learner");
    
    println!("📋 创建学生:");
    println!("   学号: {}", student.id());
    println!("   姓名: {}", student.name());
    println!("   分数: {:.2}", student.score());
    
    // 修改分数（模拟 C 的 student_set_score）
    student.set_score(95.5);
    println!("\n✏️ 更新分数为: {:.2}", student.score());
    
    // 批量创建学生数组（模拟 C 的 student 数组操作）
    let mut students: Vec<Student> = (1..=3)
        .map(|i| Student::new(i, &format!("Student {}", i)))
        .collect();
    
    println!("\n📚 学生列表:");
    for s in &students {
        println!("   [{}] {} - score: {:.2}", s.id(), s.name(), s.score());
    }
    
    // 模拟 C 的 sizeof 操作
    println!("\n📐 结构体大小:");
    println!("   Student 大小: {} bytes", size_of::<Student>());
    println!("   (C 中 sizeof(Student) 也应该是这个值)");
    
    // 模拟内存布局检查
    println!("\n🔍 内存布局验证:");
    println!("   id 偏移: {} bytes", memoffset::offset_of!(Student, id));
    println!("   name 偏移: {} bytes", memoffset::offset_of!(Student, name));
    println!("   score 偏移: {} bytes", memoffset::offset_of!(Student, score));
}

// ========== 原则 1：最小化 unsafe 范围 ==========
fn wrapper_function() {
    // 非 unsafe 代码在安全的包装函数内
    let data = prepare_data();
    
    // 只在最必要时使用 unsafe 块
    let result = unsafe {
        call_external_function(data.as_ptr())
    };
    
    // 其余逻辑继续在安全范围内
    process_result(result);
}

// ========== 原则 2：使用 safe 抽象层 ==========
mod ffi_wrapper {
    use super::*;
    use std::ptr;
    
    /// Safe 抽象层：将 extern 函数封装为 safe Rust 函数
    pub fn compute(input: &[f32]) -> Result<Vec<f32>, FfiError> {
        unsafe {
            let input_ptr = input.as_ptr();
            let input_len = input.len();
            
            // 调用外部函数
            let result_ptr = raw_compute(input_ptr, input_len)?;
            
            // 将结果复制到 Rust 管理的内存
            let result_len = raw_get_result_len();
            let result = slice::from_raw_parts(result_ptr, result_len).to_vec();
            
            // 释放外部分配的内存
            raw_free(result_ptr);
            
            Ok(result)
        }
    }
    
    // ... 声明 extern 函数 ...
}

// 使用者只需要调用 safe 抽象
fn user_code() {
    let input = vec![1.0, 2.0, 3.0];
    match ffi_wrapper::compute(&input) {
        Ok(output) => println!("结果: {:?}", output),
        Err(e) => eprintln!("FFI 错误: {:?}", e),
    }
}

// ========== 原则 3：验证外部数据 ==========
extern "C" {
    fn get_buffer_info() -> *const BufferInfo;
}

fn validate_external_data() {
    unsafe {
        let info_ptr = get_buffer_info();
        
        // 检查空指针
        if info_ptr.is_null() {
            return Err("空指针");
        }
        
        // 检查对齐（某些架构要求特定对齐）
        if !info_ptr.is_aligned() {
            return Err("对齐错误");
        }
        
        // 读取并验证数据
        let info = &*info_ptr;
        
        // 验证范围
        if info.length > MAX_ALLOWED_LENGTH {
            return Err("长度超限");
        }
        
        // ... 安全使用数据 ...
    }
}

// ========== 原则 4：使用 #[repr(C)] 确保布局 ==========
// 如果要与 C 结构体互操作，必须使用 #[repr(C)]
// #[repr(C)] 保证字段顺序和 C 一样

#[repr(C)]
struct CompatibleStruct {
    field_a: i32,
    field_b: f64,
    field_c: *mut c_void,
}

// ========== 原则 5：使用测试验证 FFI ==========
#[cfg(test)]
mod ffi_tests {
    use super::*;
    
    #[test]
    fn test_roundtrip() {
        let original = vec![1.0f32, 2.0, 3.0];
        let result = ffi_wrapper::compute(&original).unwrap();
        
        // 验证结果
        assert_eq!(result.len(), original.len());
        for (o, r) in original.iter().zip(result.iter()) {
            assert!((o - r).abs() < 1e-6);
        }
    }
}

今日感悟

extern 和 FFI 是 Rust 与外部世界沟通的"桥梁"。学习这一章给我的最大感受是：unsafe 不是乱来，反而需要更严谨的设计。

在 safe Rust 中，编译器替我们做所有检查，我们可以相对随意地写代码。而在 FFI 中，每一个指针、每一块内存都需要我们亲手验证。“最小化 unsafe 范围"这一原则尤为深刻——把 unsafe 当作一座一座的孤岛，用 safe 代码的桥梁将它们连接起来，整个程序依然可以保持良好的安全性。

C 字符串和 Rust 字符串之间的转换是 FFI 中最频繁的操作之一。理解 CString（拥有者）和 CStr（借用者）的区别，以及为什么要小心 \0` 结尾、内存所有权转移这些问题，是在 FFI 世界中游刃有余的基础。

回调函数的部分让我意识到，FFI 不只是"Rust 调用 C"那么简单——还包括"C 调用 Rust”。这种双向互操作让 Rust 可以作为一等公民参与到任何 C-based 生态系统中去。
