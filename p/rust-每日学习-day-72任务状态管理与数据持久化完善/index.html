// src/models.rs

use chrono::{DateTime, Utc};
use serde::{Deserialize, Serialize};

#[derive(Debug, Clone, Serialize, Deserialize)]
pub enum Priority {
    Low,
    Medium,
    High,
}

impl Priority {
    pub fn from_str(s: &str) -> Option<Self> {
        match s.to_lowercase().as_str() {
            "low" => Some(Priority::Low),
            "medium" => Some(Priority::Medium),
            "high" => Some(Priority::High),
            _ => None,
        }
    }

    pub fn as_u8(&self) -> u8 {
        match self {
            Priority::Low => 1,
            Priority::Medium => 2,
            Priority::High => 3,
        }
    }
}

impl Default for Priority {
    fn default() -> Self {
        Priority::Medium
    }
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct Task {
    pub id: u64,
    pub content: String,
    pub completed: bool,
    pub priority: Priority,
    pub created_at: DateTime<Utc>,
    pub updated_at: DateTime<Utc>,
    pub completed_at: Option<DateTime<Utc>>,
    pub due_date: Option<DateTime<Utc>>,
    pub tags: Vec<String>,
}

impl Task {
    pub fn new(id: u64, content: String) -> Self {
        let now = Utc::now();
        Self {
            id,
            content,
            completed: false,
            priority: Priority::default(),
            created_at: now,
            updated_at: now,
            completed_at: None,
            due_date: None,
            tags: Vec::new(),
        }
    }

    pub fn mark_completed(&mut self) {
        self.completed = true;
        self.completed_at = Some(Utc::now());
        self.updated_at = Utc::now();
    }

    pub fn mark_active(&mut self) {
        self.completed = false;
        self.completed_at = None;
        self.updated_at = Utc::now();
    }

    pub fn set_priority(&mut self, priority: Priority) {
        self.priority = priority;
        self.updated_at = Utc::now();
    }

    pub fn set_due_date(&mut self, due_date: DateTime<Utc>) {
        self.due_date = Some(due_date);
        self.updated_at = Utc::now();
    }

    pub fn add_tag(&mut self, tag: String) {
        if !self.tags.contains(&tag) {
            self.tags.push(tag);
            self.updated_at = Utc::now();
        }
    }

    pub fn remove_tag(&mut self, tag: &str) {
        self.tags.retain(|t| t != tag);
        self.updated_at = Utc::now();
    }

    pub fn is_overdue(&self) -> bool {
        if let Some(due) = self.due_date {
            return !self.completed && due < Utc::now();
        }
        false
    }
}

// src/task_manager.rs

use crate::models::{Priority, Task};
use crate::storage::Storage;
use anyhow::{Context, Result};
use std::collections::HashMap;

#[derive(Debug, Clone)]
pub struct FilterOptions {
    pub show_active: bool,
    pub show_done: bool,
    pub search: Option<String>,
    pub priority: Option<Priority>,
    pub overdue_only: bool,
}

impl Default for FilterOptions {
    fn default() -> Self {
        Self {
            show_active: false,
            show_done: false,
            search: None,
            priority: None,
            overdue_only: false,
        }
    }
}

#[derive(Debug, Clone)]
pub enum SortField {
    CreatedAt,
    UpdatedAt,
    Priority,
    DueDate,
}

#[derive(Debug, Clone)]
pub enum SortOrder {
    Asc,
    Desc,
}

pub struct TaskManager {
    storage: Storage,
}

impl TaskManager {
    pub fn new(storage: Storage) -> Self {
        Self { storage }
    }

    pub fn add_task(&mut self, content: String, priority: Option<Priority>) -> Result<Task> {
        let mut tasks = self.storage.load_tasks()?;
        let max_id = tasks.iter().map(|t| t.id).max().unwrap_or(0);
        let mut task = Task::new(max_id + 1, content);
        
        if let Some(p) = priority {
            task.set_priority(p);
        }
        
        tasks.push(task.clone());
        self.storage.save_tasks(&tasks)?;
        
        Ok(task)
    }

    pub fn complete_task(&mut self, id: u64) -> Result<Option<Task>> {
        let mut tasks = self.storage.load_tasks()?;
        
        if let Some(task) = tasks.iter_mut().find(|t| t.id == id) {
            task.mark_completed();
            let result = task.clone();
            self.storage.save_tasks(&tasks)?;
            return Ok(Some(result));
        }
        
        Ok(None)
    }

    pub fn delete_task(&mut self, id: u64) -> Result<bool> {
        let mut tasks = self.storage.load_tasks()?;
        let original_len = tasks.len();
        tasks.retain(|t| t.id != id);
        
        if tasks.len() < original_len {
            self.storage.save_tasks(&tasks)?;
            Ok(true)
        } else {
            Ok(false)
        }
    }

    pub fn update_task(&mut self, id: u64, content: String) -> Result<Option<Task>> {
        let mut tasks = self.storage.load_tasks()?;
        
        if let Some(task) = tasks.iter_mut().find(|t| t.id == id) {
            task.content = content;
            task.updated_at = chrono::Utc::now();
            let result = task.clone();
            self.storage.save_tasks(&tasks)?;
            return Ok(Some(result));
        }
        
        Ok(None)
    }

    pub fn set_priority(&mut self, id: u64, priority: Priority) -> Result<Option<Task>> {
        let mut tasks = self.storage.load_tasks()?;
        
        if let Some(task) = tasks.iter_mut().find(|t| t.id == id) {
            task.set_priority(priority);
            let result = task.clone();
            self.storage.save_tasks(&tasks)?;
            return Ok(Some(result));
        }
        
        Ok(None)
    }

    pub fn list_tasks(
        &self,
        filter: FilterOptions,
        sort_by: SortField,
        sort_order: SortOrder,
    ) -> Result<Vec<Task>> {
        let tasks = self.storage.load_tasks()?;
        let mut filtered: Vec<Task> = tasks.into_iter().filter(|task| {
            // 状态过滤
            if filter.show_active && task.completed {
                return false;
            }
            if filter.show_done && !task.completed {
                return false;
            }
            
            // 搜索过滤
            if let Some(ref search) = filter.search {
                if !task.content.to_lowercase().contains(&search.to_lowercase()) {
                    return false;
                }
            }
            
            // 优先级过滤
            if let Some(ref pri) = filter.priority {
                if &task.priority != pri {
                    return false;
                }
            }
            
            // 逾期过滤
            if filter.overdue_only && !task.is_overdue() {
                return false;
            }
            
            true
        }).collect();

        // 排序
        match sort_by {
            SortField::CreatedAt => {
                filtered.sort_by(|a, b| match sort_order {
                    SortOrder::Asc => a.created_at.cmp(&b.created_at),
                    SortOrder::Desc => b.created_at.cmp(&a.created_at),
                });
            }
            SortField::UpdatedAt => {
                filtered.sort_by(|a, b| match sort_order {
                    SortOrder::Asc => a.updated_at.cmp(&b.updated_at),
                    SortOrder::Desc => b.updated_at.cmp(&a.updated_at),
                });
            }
            SortField::Priority => {
                filtered.sort_by(|a, b| match sort_order {
                    SortOrder::Asc => a.priority.as_u8().cmp(&b.priority.as_u8()),
                    SortOrder::Desc => b.priority.as_u8().cmp(&a.priority.as_u8()),
                });
            }
            SortField::DueDate => {
                filtered.sort_by(|a, b| match (a.due_date, b.due_date) {
                    (Some(d1), Some(d2)) => match sort_order {
                        SortOrder::Asc => d1.cmp(&d2),
                        SortOrder::Desc => d2.cmp(&d1),
                    },
                    (Some(_), None) => std::cmp::Ordering::Less,
                    (None, Some(_)) => std::cmp::Ordering::Greater,
                    (None, None) => std::cmp::Ordering::Equal,
                });
            }
        }

        Ok(filtered)
    }

    pub fn get_task_stats(&self) -> Result<TaskStats> {
        let tasks = self.storage.load_tasks()?;
        
        let total = tasks.len();
        let completed = tasks.iter().filter(|t| t.completed).count();
        let active = total - completed;
        let overdue = tasks.iter().filter(|t| t.is_overdue()).count();
        
        let mut priority_counts = HashMap::new();
        priority_counts.insert("high", tasks.iter().filter(|t| matches!(t.priority, Priority::High)).count());
        priority_counts.insert("medium", tasks.iter().filter(|t| matches!(t.priority, Priority::Medium)).count());
        priority_counts.insert("low", tasks.iter().filter(|t| matches!(t.priority, Priority::Low)).count());
        
        Ok(TaskStats {
            total,
            completed,
            active,
            overdue,
            priority_counts,
        })
    }
}

#[derive(Debug, Serialize)]
pub struct TaskStats {
    pub total: usize,
    pub completed: usize,
    pub active: usize,
    pub overdue: usize,
    pub priority_counts: HashMap<&str, usize>,
}

// src/storage.rs

use crate::models::Task;
use anyhow::{Context, Result};
use std::fs;
use std::path::Path;

#[derive(Clone)]
pub struct Storage {
    file_path: String,
}

impl Storage {
    pub fn new(file_path: String) -> Result<Self> {
        // 确保目录存在
        if let Some(parent) = Path::new(&file_path).parent() {
            if !parent.as_os_str().is_empty() {
                fs::create_dir_all(parent).context("无法创建数据目录")?;
            }
        }
        Ok(Self { file_path })
    }

    pub fn load_tasks(&self) -> Result<Vec<Task>> {
        let path = Path::new(&self.file_path);
        
        if !path.exists() {
            return Ok(Vec::new());
        }
        
        let content = fs::read_to_string(path)
            .context("无法读取数据文件")?;
        
        if content.trim().is_empty() {
            return Ok(Vec::new());
        }
        
        let tasks: Vec<Task> = serde_json::from_str(&content)
            .context("无法解析数据文件")?;
        
        Ok(tasks)
    }

    pub fn save_tasks(&self, tasks: &[Task]) -> Result<()> {
        let content = serde_json::to_string_pretty(tasks)
            .context("无法序列化任务数据")?;
        
        fs::write(&self.file_path, content)
            .context("无法写入数据文件")?;
        
        Ok(())
    }

    pub fn backup(&self) -> Result<String> {
        let tasks = self.load_tasks()?;
        let timestamp = chrono::Utc::now().format("%Y%m%d_%H%M%S");
        let backup_path = format!("{}.backup.{}", self.file_path, timestamp);
        
        let content = serde_json::to_string_pretty(&tasks)
            .context("无法序列化备份数据")?;
        
        fs::write(&backup_path, content)
            .context("无法创建备份文件")?;
        
        Ok(backup_path)
    }

    pub fn restore(&self, backup_path: &str) -> Result<Vec<Task>> {
        let content = fs::read_to_string(backup_path)
            .context("无法读取备份文件")?;
        
        let tasks: Vec<Task> = serde_json::from_str(&content)
            .context("无法解析备份文件")?;
        
        self.save_tasks(&tasks)?;
        Ok(tasks)
    }

    pub fn export_csv(&self, csv_path: &str) -> Result<()> {
        let tasks = self.load_tasks()?;
        
        let mut csv_content = String::from("ID,内容,已完成,优先级,创建时间,到期时间\n");
        
        for task in tasks {
            let due = task.due_date
                .map(|d| d.format("%Y-%m-%d").to_string())
                .unwrap_or_default();
            
            csv_content.push_str(&format!(
                "{},{},{},{:?},{},{}\n",
                task.id,
                task.content.replace(',', ";"),
                task.completed,
                task.priority,
                task.created_at.format("%Y-%m-%d"),
                due
            ));
        }
        
        fs::write(csv_path, csv_content)
            .context("无法导出 CSV 文件")?;
        
        Ok(())
    }
}

// src/main.rs

mod cli;
mod models;
mod storage;
mod task_manager;

use anyhow::Result;
use cli::{Cli, Commands};
use storage::Storage;
use task_manager::{FilterOptions, SortField, SortOrder, TaskManager};

fn main() -> Result<()> {
    let cli = Cli::parse();
    
    // 初始化存储和管理器
    let storage = Storage::new(
        cli.file.unwrap_or_else(|| "rustodo.json".to_string())
    )?;
    let mut manager = TaskManager::new(storage);
    
    match &cli.command {
        Commands::Add { content, priority } => {
            let pri = priority.as_ref().and_then(|p| 
                models::Priority::from_str(p)
            );
            let task = manager.add_task(content.clone(), pri)?;
            println!("✅ 已添加任务 [#{id}]", id = task.id);
        }
        
        Commands::List { active, done, search } => {
            let mut filter = FilterOptions::default();
            filter.show_active = *active;
            filter.show_done = *done;
            filter.search = search.clone();
            
            let tasks = manager.list_tasks(
                filter,
                SortField::Priority,
                SortOrder::Desc
            )?;
            
            if tasks.is_empty() {
                println!("📋 暂无任务");
            } else {
                for task in tasks {
                    let status = if task.completed { "✓" } else { "○" };
                    let overdue = if task.is_overdue() { " ⚠️逾期" } else { "" };
                    let pri_mark = match task.priority {
                        models::Priority::High => "🔴",
                        models::Priority::Medium => "🟡",
                        models::Priority::Low => "🟢",
                    };
                    println!(
                        "{status} [{id}] {pri_mark} {content}{overdue}",
                        id = task.id,
                        content = task.content,
                        status = status,
                        pri_mark = pri_mark,
                        overdue = overdue
                    );
                }
            }
        }
        
        Commands::Done { id } => {
            if let Some(task) = manager.complete_task(*id)? {
                println!("✅ 已完成任务 [#{id}]", id = task.id);
            } else {
                println!("❌ 未找到任务 [#{id}]", id = id);
            }
        }
        
        Commands::Delete { id } => {
            if manager.delete_task(*id)? {
                println!("🗑️ 已删除任务 [#{id}]", id = id);
            } else {
                println!("❌ 未找到任务 [#{id}]", id = id);
            }
        }
        
        Commands::Edit { id, content } => {
            if let Some(task) = manager.update_task(*id, content.clone())? {
                println!("✏️ 已更新任务 [#{id}]", id = task.id);
            } else {
                println!("❌ 未找到任务 [#{id}]", id = id);
            }
        }
        
        Commands::SetPriority { id, level } => {
            if let Some(pri) = models::Priority::from_str(level) {
                if let Some(task) = manager.set_priority(*id, pri)? {
                    println!("📌 已设置优先级 [#{id}]", id = task.id);
                } else {
                    println!("❌ 未找到任务 [#{id}]", id = id);
                }
            } else {
                println!("❌ 无效的优先级级别: {level}", level = level);
                println!("有效的优先级: low / medium / high");
            }
        }
        
        Commands::Stats => {
            let stats = manager.get_task_stats()?;
            println!("📊 任务统计");
            println!("  总数: {}", stats.total);
            println!("  进行中: {}", stats.active);
            println!("  已完成: {}", stats.completed);
            println!("  逾期: {}", stats.overdue);
            println!("  🔴高优先级: {}", stats.priority_counts.get("high").unwrap_or(&0));
            println!("  🟡中优先级: {}", stats.priority_counts.get("medium").unwrap_or(&0));
            println!("  🟢低优先级: {}", stats.priority_counts.get("low").unwrap_or(&0));
        }
    }
    
    Ok(())
}

#[derive(Subcommand)]

关键概念理解

1. 状态模式在任务管理中的应用

任务有多种状态（进行中/已完成/逾期），通过 completed 布尔值和 completed_at 时间戳来追踪。mark_completed() 和 mark_active() 方法确保状态转换时更新相关字段。

2. 过滤器模式

FilterOptions 结构体支持多种过滤条件的组合：

  • 可以同时指定 show_active 和搜索关键词
  • 逾期过滤独立于状态过滤

3. 排序策略

使用 SortField 和 SortOrder 枚举实现灵活的排序，支持：

  • 按创建时间
  • 按更新时间
  • 按优先级
  • 按到期日

4. 数据备份与恢复

存储层实现了 backup() 方法，在修改数据前创建时间戳备份，这是数据安全的重要实践。
