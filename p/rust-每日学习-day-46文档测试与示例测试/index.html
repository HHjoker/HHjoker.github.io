什么是文档测试？

文档测试（Doc Tests）是 Rust 独有的强大功能：写在文档注释中的代码示例，可以直接作为测试运行。

}

运行 cargo test 时，Rust 会：

  1. 提取所有文档注释中的代码块
  2. 编译并运行它们
  3. 确保示例代码始终有效

/// 
/// # Examples
/// 
/// ```
/// # fn main() {
/// use my_crate::factorial;
/// 
/// let result = factorial(5);
/// assert_eq!(result, 120);
/// # }
/// ```
pub fn factorial(n: u32) -> u32 {
    if n <= 1 {
        1
    } else {
        n * factorial(n - 1)
    }
}

/// 计算圆的面积
/// 
/// # Formula
/// 
/// `area = π × r²`
/// 
/// # Examples
/// 
/// ```
/// use my_crate::circle_area;
/// 
/// let area = circle_area(5.0);
/// assert!((area - 78.53981633974483).abs() < 0.0001);
/// ```
/// 
/// # Panics
/// 
/// 当半径为负数时会 panic
/// 
/// ```should_panic
/// circle_area(-1.0);  // 半径不能为负
/// ```
pub fn circle_area(radius: f64) -> f64 {
    if radius < 0.0 {
        panic!("Radius cannot be negative");
    }
    std::f64::consts::PI * radius * radius
}

/// 安全除法
/// 
/// # Examples
/// 
/// 成功的情况：
/// 
/// ```
/// use my_crate::safe_divide;
/// 
/// let result = safe_divide(10, 2);
/// assert_eq!(result, Ok(5));
/// ```
/// 
/// 失败的情况：
/// 
/// ```
/// use my_crate::safe_divide;
/// 
/// let result = safe_divide(10, 0);
/// assert!(result.is_err());
/// ```
pub fn safe_divide(a: i32, b: i32) -> Result<i32, &'static str> {
    if b == 0 {
        Err("Division by zero")
    } else {
        Ok(a / b)
    }
}

/// 表示一个计数器
/// 
/// # Examples
/// 
/// ```
/// use my_crate::Counter;
/// 
/// let mut counter = Counter::new();
/// counter.increment();
/// counter.increment();
/// assert_eq!(counter.value(), 2);
/// 
/// counter.reset();
/// assert_eq!(counter.value(), 0);
/// ```
#[derive(Debug, Default)]
pub struct Counter {
    value: u32,
}

impl Counter {
    /// 创建新计数器
    /// 
    /// ```
    /// use my_crate::Counter;
    /// 
    /// let counter = Counter::new();
    /// assert_eq!(counter.value(), 0);
    /// ```
    pub fn new() -> Self {
        Self::default()
    }

    /// 增加计数
    /// 
    /// ```
    /// use my_crate::Counter;
    /// 
    /// let mut counter = Counter::new();
    /// counter.increment();
    /// assert_eq!(counter.value(), 1);
    /// ```
    pub fn increment(&mut self) {
        self.value += 1;
    }

    /// 重置计数器
    /// 
    /// ```
    /// use my_crate::Counter;
    /// 
    /// let mut counter = Counter::new();
    /// counter.increment();
    /// counter.increment();
    /// counter.reset();
    /// assert_eq!(counter.value(), 0);
    /// ```
    pub fn reset(&mut self) {
        self.value = 0;
    }

    /// 获取当前值
    /// 
    /// ```
    /// use my_crate::Counter;
    /// 
    /// let counter = Counter::new();
    /// assert_eq!(counter.value(), 0);
    /// ```
    pub fn value(&self) -> u32 {
        self.value
    }
}

//! # 一个简单的事件系统
//! 
//! 这个 crate 提供了一个轻量级的事件发布/订阅系统。
//! 
//! ## 快速开始
//! 
//! ```
//! use event_system::{EventBus, Event};
//! 
//! // 创建事件总线
//! let mut bus = EventBus::new();
//! 
//! // 订阅事件
//! bus.subscribe("user_login", |event| {
//!     println!("User logged in: {:?}", event);
//! });
//! 
//! // 发布事件
//! bus.publish("user_login", Event::new("alice"));
//! ```

/// 表示一个事件
/// 
/// # Examples
/// 
/// ```
/// use event_system::Event;
/// 
/// let event = Event::new("user_login");
/// assert_eq!(event.name(), "user_login");
/// ```
#[derive(Debug, Clone)]
pub struct Event {
    name: String,
    data: Option<String>,
}

impl Event {
    /// 创建新事件
    /// 
    /// ```
    /// use event_system::Event;
    /// 
    /// let event = Event::new("click");
    /// assert_eq!(event.name(), "click");
    /// ```
    pub fn new(name: &str) -> Self {
        Self {
            name: name.to_string(),
            data: None,
        }
    }

    /// 添加事件数据
    /// 
    /// ```
    /// use event_system::Event;
    /// 
    /// let event = Event::new("user_action")
    ///     .with_data("clicked button");
    /// 
    /// assert_eq!(event.data(), Some("clicked button"));
    /// ```
    pub fn with_data(mut self, data: &str) -> Self {
        self.data = Some(data.to_string());
        self
    }

    /// 获取事件名称
    pub fn name(&self) -> &str {
        &self.name
    }

    /// 获取事件数据
    pub fn data(&self) -> Option<&str> {
        self.data.as_deref()
    }
}

/// 事件处理器类型别名
pub type EventHandler = Box<dyn Fn(&Event) + Send + Sync>;

/// 事件总线
/// 
/// # Examples
/// 
/// ```
/// use event_system::{EventBus, Event};
/// 
/// let mut bus = EventBus::new();
/// 
/// // 订阅
/// let handler_id = bus.subscribe("test", |event| {
///     println!("Received: {}", event.name());
/// });
/// 
/// // 发布
/// bus.publish("test", Event::new("hello"));
/// 
/// // 取消订阅
/// bus.unsubscribe(handler_id);
/// ```
pub struct EventBus {
    handlers: std::collections::HashMap<String, Vec<(usize, EventHandler)>>,
    next_id: usize,
}

impl EventBus {
    /// 创建新的事件总线
    /// 
    /// ```
    /// use event_system::EventBus;
    /// 
    /// let bus = EventBus::new();
    /// ```
    pub fn new() -> Self {
        Self {
            handlers: std::collections::HashMap::new(),
            next_id: 0,
        }
    }

    /// 订阅事件
    /// 
    /// 返回订阅 ID，可用于取消订阅
    /// 
    /// ```
    /// use event_system::{EventBus, Event};
    /// 
    /// let mut bus = EventBus::new();
    /// 
    /// let id = bus.subscribe("user_login", |event| {
    ///     println!("User {} logged in", event.data().unwrap_or("unknown"));
    /// });
    /// 
    /// bus.publish("user_login", Event::new("").with_data("alice"));
    /// ```
    pub fn subscribe<F>(&mut self, event_name: &str, handler: F) -> usize
    where
        F: Fn(&Event) + Send + Sync + 'static,
    {
        let id = self.next_id;
        self.next_id += 1;
        
        self.handlers
            .entry(event_name.to_string())
            .or_insert_with(Vec::new)
            .push((id, Box::new(handler)));
        
        id
    }

    /// 取消订阅
    /// 
    /// ```
    /// use event_system::{EventBus, Event};
    /// 
    /// let mut bus = EventBus::new();
    /// 
    /// let id = bus.subscribe("test", |_| {});
    /// bus.unsubscribe(id);
    /// 
    /// // 取消后发布不会调用处理器
    /// bus.publish("test", Event::new(""));  // 不会 panic
    /// ```
    pub fn unsubscribe(&mut self, id: usize) {
        for handlers in self.handlers.values_mut() {
            handlers.retain(|(handler_id, _)| *handler_id != id);
        }
    }

    /// 发布事件
    /// 
    /// ```
    /// use event_system::{EventBus, Event};
    /// 
    /// let mut bus = EventBus::new();
    /// let mut received = Vec::new();
    /// 
    /// bus.subscribe("test", |event| {
    ///     received.push(event.name().to_string());
    /// });
    /// 
    /// bus.publish("test", Event::new("hello"));
    /// 
    /// assert_eq!(received, vec!["hello"]);
    /// ```
    pub fn publish(&self, event_name: &str, event: Event) {
        if let Some(handlers) = self.handlers.get(event_name) {
            for (_, handler) in handlers {
                handler(&event);
            }
        }
    }

    /// 获取订阅者数量
    /// 
    /// ```
    /// use event_system::EventBus;
    /// 
    /// let mut bus = EventBus::new();
    /// assert_eq!(bus.subscriber_count("test"), 0);
    /// 
    /// bus.subscribe("test", |_| {});
    /// assert_eq!(bus.subscriber_count("test"), 1);
    /// ```
    pub fn subscriber_count(&self, event_name: &str) -> usize {
        self.handlers
            .get(event_name)
            .map(|handlers| handlers.len())
            .unwrap_or(0)
    }
}

impl Default for EventBus {
    fn default() -> Self {
        Self::new()
    }
}

关键概念理解

1. 文档测试 vs 单元测试

特性 文档测试 单元测试
位置 文档注释中 #[cfg(test)] 模块
目的 展示用法 + 测试 验证逻辑
可见性 用户可见 内部使用
运行命令 cargo test --doc cargo test --lib
独立性 每个代码块独立 可以共享状态

2. 为什么文档测试重要？

  • 文档即测试 - 文档不会过时
  • 测试即文档 - 测试展示正确用法
  • 用户友好 - 用户可以直接复制示例
  • 质量保证 - 示例代码始终可运行

3. 文档测试的局限性

  • 每个代码块独立运行，不能共享状态
  • 不适合复杂设置
  • 运行速度比单元测试慢
  • 调试相对困难

pub fn fibonacci(n: u32) -> u32 {
    match n {
        0 => 0,
        1 => 1,
        _ => {
            let mut a = 0;
            let mut b = 1;
            for _ in 2..=n {
                let temp = a + b;
                a = b;
                b = temp;
            }
            b
        }
    }
}

今日感悟

文档测试是 Rust 最优雅的设计之一。今天的核心收获：

  1. 文档不应该撒谎 - 文档测试确保示例代码始终可运行，这是对其他语言的降维打击。

  2. 好文档 = 好 API - 当你知道示例会被测试时，你会更认真地设计 API。文档测试倒逼 API 设计质量。

  3. 用户视角 - 写文档测试时，你被迫从用户角度思考：“这个 API 怎么用？示例清晰吗？”

  4. 维护成本降低 - 代码变更时，文档测试会立即告诉你示例是否还有效。不再有"文档已过期"的问题。

  5. 适度原则 - 不是每个函数都需要详细的文档测试。公共 API 必须有，内部函数可以简略。

文档测试让"文档与代码同步"从口号变成了默认行为。这是 Rust 对开发者体验的真正贡献。
