参数化测试

参数化测试允许我们用不同的输入数据运行同一个测试逻辑，避免重复代码。

// 定义一个参数化测试宏
macro_rules! test_case {
    ($name:ident, $input:expr, $expected:expr) => {
        #[test]
        fn $name() {
            let result = $input;
            assert_eq!(result, $expected);
        }
    };
}

// 使用宏
test_case!(test_add_2_3, 2 + 3, 5);
test_case!(test_add_0_0, 0 + 0, 0);
test_case!(test_add_neg_1_1, -1 + 1, 0);

#[derive(Debug)]
struct TestCase {
    name: &'static str,
    input: i32,
    expected: i32,
}

fn run_test_case(test_case: &TestCase) {
    let result = square(test_case.input);
    assert_eq!(
        result, test_case.expected,
        "测试失败：{} (input={}, expected={})",
        test_case.name, test_case.input, test_case.expected
    );
}

pub fn square(n: i32) -> i32 {
    n * n
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn test_square_cases() {
        let test_cases = vec![
            TestCase { name: "positive", input: 2, expected: 4 },
            TestCase { name: "zero", input: 0, expected: 0 },
            TestCase { name: "negative", input: -3, expected: 9 },
            TestCase { name: "one", input: 1, expected: 1 },
            TestCase { name: "large", input: 100, expected: 10000 },
        ];

        for test_case in test_cases {
            run_test_case(&test_case);
        }
    }
}

use rstest::rstest;

#[rstest]
#[case(2, 4)]
#[case(0, 0)]
#[case(-3, 9)]
#[case(1, 1)]
#[case(100, 10000)]
fn test_square(#[case] input: i32, #[case] expected: i32) {
    let result = square(input);
    assert_eq!(result, expected);
}

// 支持多个参数
#[rstest]
#[case(2, 3, 5)]
#[case(0, 0, 0)]
#[case(-1, 1, 0)]
#[case(10, 20, 30)]
#[case(-5, -3, -8)]
fn test_add(#[case] a: i32, #[case] b: i32, #[case] expected: i32) {
    assert_eq!(a + b, expected);
}

// 支持自定义测试名称
#[rstest]
#[case::positive(2, 4)]
#[case::zero(0, 0)]
#[case::negative(-3, 9)]
fn test_square_named(#[case] input: i32, #[case] expected: i32) {
    assert_eq!(square(input), expected);
}

// 运行特定测试用例
// cargo test test_square_named::case::positive

use rstest::{rstest, fixture, timeout};
use std::time::Duration;

// 使用 fixture 提供测试数据
#[fixture]
fn sample_data() -> Vec<i32> {
    vec![1, 2, 3, 4, 5]
}

#[fixture]
fn empty_data() -> Vec<i32> {
    vec![]
}

#[rstest]
fn test_sum_with_fixture(sample_data: Vec<i32>) {
    let sum: i32 = sample_data.iter().sum();
    assert_eq!(sum, 15);
}

#[rstest]
fn test_sum_empty(empty_data: Vec<i32>) {
    let sum: i32 = empty_data.iter().sum();
    assert_eq!(sum, 0);
}

// 设置超时
#[rstest]
#[timeout(Duration::from_secs(1))]
fn test_with_timeout() {
    // 如果超过 1 秒，测试会自动失败
    std::thread::sleep(Duration::from_millis(500));
}

// 组合多个 fixture
#[fixture]
fn user_name() -> String {
    "Alice".to_string()
}

#[fixture]
fn user_age() -> u32 {
    25
}

#[rstest]
fn test_user_data(user_name: String, user_age: u32) {
    assert_eq!(user_name, "Alice");
    assert_eq!(user_age, 25);
}

    #[test]
    fn test_with_env_var() {
        // 设置环境变量
        std::env::set_var("TEST_MODE", "true");
        
        // 读取环境变量
        let test_mode = std::env::var("TEST_MODE")
            .unwrap_or_else(|_| "false".to_string());
        
        assert_eq!(test_mode, "true");
        
        // 清理环境变量
        std::env::remove_var("TEST_MODE");

            .unwrap_or_else(|_| "sqlite::memory:".to_string());

    }
}

// src/user.rs

#[derive(Debug, Clone, PartialEq)]
pub struct User {
    pub id: u32,
    pub name: String,
    pub email: String,
}

impl User {
    pub fn new(name: &str, email: &str) -> Self {
        User {
            id: 0,
            name: name.to_string(),
            email: email.to_string(),
        }
    }

    pub fn validate_email(&self) -> Result<(), String> {
        if !self.email.contains('@') {
            return Err("邮箱必须包含 @".to_string());
        }
        if !self.email.contains('.') {
            return Err("邮箱必须包含 .".to_string());
        }
        Ok(())
    }
}

#[cfg(test)]
mod tests {
    use super::*;

    mod user_creation {
        use super::*;

        #[test]
        fn new_creates_user_with_given_name_and_email() {
            let user = User::new("Alice", "alice@example.com");
            assert_eq!(user.name, "Alice");
            assert_eq!(user.email, "alice@example.com");
            assert_eq!(user.id, 0);
        }
    }

    mod email_validation {
        use super::*;

        #[test]
        fn validate_email_succeeds_for_valid_email() {
            let user = User::new("Bob", "bob@test.com");
            assert!(user.validate_email().is_ok());
        }

        #[test]
        fn validate_email_fails_without_at_symbol() {

今日感悟

今天的测试进阶学习让我对 Rust 测试生态有了更深的理解：

参数化测试的价值：之前写测试时，经常要写很多相似的测试用例，只是输入不同。rstest 库让这个问题迎刃而解。用 #[case] 属性标记参数，一个测试函数就能覆盖多个场景。而且测试失败时，能清楚看到是哪个 case 失败了。

测试组织的重要性：随着项目变大，测试代码也会膨胀。良好的组织能让测试更容易维护。按功能模块分文件、使用辅助函数、定义 fixture，这些模式让测试代码和生产代码一样需要精心设计。

覆盖率不是目标，是工具：100% 覆盖率听起来很美好，但不代表没有 bug。有些边界情况可能永远触发不了。覆盖率应该用来发现没测试到的代码，而不是追求数字。80% 的有意义覆盖比 100% 的机械覆盖更有价值。

CI 集成的必要性：测试只有自动运行才有价值。每次提交都运行测试、检查覆盖率，能及时发现回归问题。GitHub Actions + cargo-llvm-cov + Codecov 的组合很成熟，值得每个 Rust 项目采用。

Mock 的必要性：测试涉及外部依赖（数据库、API）时，Mock 能让测试更快、更可靠。明天要学习的 mockall 库应该能解决这个问题。

测试是保证代码质量的重要手段。今天学的这些进阶技巧，能让测试写得更好、维护更容易。
