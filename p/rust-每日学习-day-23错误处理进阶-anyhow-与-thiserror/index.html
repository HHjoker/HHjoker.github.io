// ❌ 手动实现很繁琐
#[derive(Debug)]
enum AppError {
    IoError(std::io::Error),
    ParseError(std::num::ParseIntError),
    NotFound(String),
}

impl std::fmt::Display for AppError {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        match self {
            AppError::IoError(e) => write!(f, "IO 错误：{}", e),
            AppError::ParseError(e) => write!(f, "解析错误：{}", e),
            AppError::NotFound(msg) => write!(f, "未找到：{}", msg),
        }
    }
}

impl std::error::Error for AppError {
    fn source(&self) -> Option<&(dyn std::error::Error + 'static)> {
        match self {
            AppError::IoError(e) => Some(e),
            AppError::ParseError(e) => Some(e),
            _ => None,
        }
    }
}

impl From<std::io::Error> for AppError {
    fn from(err: std::io::Error) -> Self {
        AppError::IoError(err)
    }
}

impl From<std::num::ParseIntError> for AppError {
    fn from(err: std::num::ParseIntError) -> Self {
        AppError::ParseError(err)
    }
}

┌─────────────────────────────────────────────────────────┐
│  anyhow vs thiserror                                    │
├─────────────────────────────────────────────────────────┤
│  anyhow                                                 │
│  - 用于应用程序（二进制 crate）                         │
│  - 动态错误类型，类似 Box<dyn Error>                    │
│  - 重点：快速添加上下文，方便调试                       │
│  - 使用场景：main 函数、CLI 工具、服务端应用            │
│                                                         │
│  thiserror                                              │
│  - 用于库（库 crate）                                   │
│  - 静态错误类型，通过宏生成代码                         │
│  - 重点：定义清晰的错误 API，供调用者处理               │
│  - 使用场景：公开库、需要精细错误分类的场景             │
│                                                         │
│  选择原则：                                             │
│  - 写应用？用 anyhow                                    │
│  - 写库？用 thiserror                                   │
│  - 两者结合？库用 thiserror，应用用 anyhow 包装         │
└─────────────────────────────────────────────────────────┘

use anyhow::{Result, Context, anyhow, bail};

// Result 是 anyhow::Result<T> 的别名，等价于 Result<T, anyhow::Error>
fn read_file(path: &str) -> Result<String> {
    std::fs::read_to_string(path)
        .with_context(|| format!("无法读取文件：{}", path))
}

fn main() -> Result<()> {
    // 方式 1：使用 context 添加上下文
    let contents = read_file("config.txt")?;
    println!("{}", contents);
    
    // 方式 2：使用 anyhow! 宏创建错误
    if contents.is_empty() {
        return Err(anyhow!("文件内容为空"));
    }
    
    // 方式 3：使用 bail! 宏提前返回错误
    if !contents.contains("key") {
        bail!("文件缺少必需的 key");
    }
    
    Ok(())
}

use anyhow::{Result, Context, Chain};

fn demonstrate_anyhow() -> Result<()> {
    // 1. 错误上下文链
    let result = std::fs::read_to_string("nonexistent.txt")
        .with_context(|| "读取配置文件失败")
        .with_context(|| "应用启动失败");
    
    if let Err(err) = result {
        // 打印完整错误链
        println!("错误：{}", err);
        
        // 遍历错误链
        for (i, cause) in err.chain().enumerate() {
            println!("  层级 {}: {}", i, cause);
        }
        
        // 获取根本原因
        println!("根本原因：{}", err.root_cause());
    }
    
    // 2. 条件错误
    let path = "data.txt";
    let contents = std::fs::read_to_string(path)
        .with_context(|| format!("读取 {}", path))?;
    
    // 3. 转换其他错误类型
    let number: i32 = contents
        .trim()
        .parse()
        .context("解析数字失败")?;
    
    Ok(())
}

use anyhow::{Result, Context, bail, ensure};

/// 使用 ensure! 进行条件检查
fn validate_config(host: &str, port: u16) -> Result<()> {
    // ensure! 类似 if + bail!，但更简洁
    ensure!(!host.is_empty(), "主机名不能为空");
    ensure!(host.len() <= 255, "主机名过长");
    ensure!(port > 0, "端口号必须大于 0");
    ensure!(port <= 65535, "端口号超出范围");
    
    Ok(())
}

/// 包装底层错误
fn process_data() -> Result<Vec<String>> {
    let data = std::fs::read_to_string("data.json")
        .context("读取数据文件失败")?;
    
    let parsed: Vec<String> = serde_json::from_str(&data)
        .context("解析 JSON 失败")?;
    
    Ok(parsed)
}

/// 提供默认值
fn get_config_value() -> Result<String> {
    std::env::var("APP_CONFIG")
        .context("环境变量 APP_CONFIG 未设置")
        .or_else(|e| {
            // 尝试从文件读取
            std::fs::read_to_string("config.default")
                .context("默认配置文件也不存在")
                .map(|s| s.trim().to_string())
        })
}

/// 主函数示例
fn main() -> Result<()> {
    // 使用 ? 传播错误
    validate_config("localhost", 8080)?;
    
    // 捕获并处理错误
    match process_data() {
        Ok(data) => println!("处理了 {} 条数据", data.len()),
        Err(e) => {
            eprintln!("错误：{:?}", e);  // 使用 {:?} 打印完整调试信息
            std::process::exit(1);
        }
    }
    
    Ok(())
}

use thiserror::Error;

// 使用 #[error] 宏自动生成 Display 实现
#[derive(Error, Debug)]
enum DataError {
    #[error("文件未找到：{0}")]
    FileNotFound(String),
    
    #[error("IO 错误")]
    Io(#[from] std::io::Error),
    
    #[error("解析失败：{source}")]
    Parse {
        #[source]
        source: std::num::ParseIntError,
    },
    
    #[error("无效的配置：{field} - {message}")]
    InvalidConfig { field: String, message: String },
}

// 自动实现了：
// - Display (通过 #[error] 属性)
// - Error trait (自动实现 source() 方法)
// - From<std::io::Error> (通过 #[from] 属性)

fn read_data(path: &str) -> Result<i32, DataError> {
    // io::Error 自动转换为 DataError::Io
    let contents = std::fs::read_to_string(path)?;
    
    // 手动创建错误
    if contents.is_empty() {
        return Err(DataError::FileNotFound(path.to_string()));
    }
    
    // 命名参数错误
    let number: i32 = contents.trim().parse().map_err(|e| {
        DataError::Parse { source: e }
    })?;
    
    Ok(number)
}

use thiserror::Error;

/// 复杂的错误类型
#[derive(Error, Debug)]
enum ApiError {
    // 简单消息
    #[error("请求失败：{0}")]
    RequestFailed(String),
    
    // 包含多个字段
    #[error("验证失败：字段'{field}'，原因'{reason}'")]
    ValidationError { field: String, reason: String },
    
    // 包装其他错误（自动 From）
    #[error("网络错误")]
    Network(#[from] reqwest::Error),
    
    // 包装其他错误（手动 From）
    #[error("序列化错误：{0}")]
    Serialization(serde_json::Error),
    
    // 不显示错误详情（安全考虑）
    #[error("认证失败")]
    AuthFailed(#[source] std::io::Error),
    
    // 透明包装（保留原始错误类型）
    #[error(transparent)]
    Io(#[from] std::io::Error),
    
    // 带错误码
    #[error("错误 {code}: {message}")]
    WithCode { code: u16, message: String },
}

/// 使用透明包装
#[derive(Error, Debug)]
enum AppError {
    #[error(transparent)]
    Database(#[from] sqlx::Error),
    
    #[error(transparent)]
    Redis(#[from] redis::RedisError),
}

/// 错误类型作为公共 API
pub type Result<T> = std::result::Result<T, ApiError>;

pub fn fetch_user(id: u32) -> Result<User> {
    // 网络错误自动转换
    let response = reqwest::blocking::get(&format!("/users/{}", id))?;
    
    if !response.status().is_success() {
        return Err(ApiError::RequestFailed(
            format!("HTTP {}", response.status())
        ));
    }
    
    // 序列化错误手动转换
    let user: User = response.json()
        .map_err(ApiError::Serialization)?;
    
    Ok(user)
}

// Cargo.toml
// [dependencies]
// anyhow = "1.0"
// thiserror = "1.0"
// serde = { version = "1.0", features = ["derive"] }
// serde_json = "1.0"

use anyhow::{Context, Result, bail, ensure};
use thiserror::Error;
use serde::{Deserialize, Serialize};
use std::fs;
use std::path::Path;

// ============ 库层错误定义（使用 thiserror）============

#[derive(Error, Debug)]
pub enum StorageError {
    #[error("文件不存在：{path}")]
    FileNotFound { path: String },
    
    #[error("权限不足：{path}")]
    PermissionDenied { path: String },
    
    #[error("数据格式错误：{message}")]
    InvalidFormat { message: String },
    
    #[error("IO 错误")]
    Io(#[from] std::io::Error),
    
    #[error("JSON 错误")]
    Json(#[from] serde_json::Error),
}

pub type StorageResult<T> = std::result::Result<T, StorageError>;

// ============ 数据存储结构 ============

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct Task {
    pub id: u64,
    pub title: String,
    pub completed: bool,
}

#[derive(Debug, Clone, Serialize, Deserialize, Default)]
pub struct TaskStore {
    pub tasks: Vec<Task>,
    pub next_id: u64,
}

// ============ 存储实现 ============

pub struct TaskStorage {
    path: String,
}

impl TaskStorage {
    pub fn new(path: &str) -> Self {
        TaskStorage {
            path: path.to_string(),
        }
    }
    
    /// 加载存储
    pub fn load(&self) -> StorageResult<TaskStore> {
        if !Path::new(&self.path).exists() {
            return Ok(TaskStore::default());
        }
        
        let contents = fs::read_to_string(&self.path)
            .map_err(|e| {
                if e.kind() == std::io::ErrorKind::PermissionDenied {
                    StorageError::PermissionDenied { 
                        path: self.path.clone() 
                    }
                } else {
                    StorageError::from(e)
                }
            })?;
        
        if contents.trim().is_empty() {
            return Ok(TaskStore::default());
        }
        
        let store: TaskStore = serde_json::from_str(&contents)?;
        Ok(store)
    }
    
    /// 保存存储
    pub fn save(&self, store: &TaskStore) -> StorageResult<()> {
        let contents = serde_json::to_string_pretty(store)?;
        fs::write(&self.path, contents)?;
        Ok(())
    }
    
    /// 添加任务
    pub fn add_task(&self, title: &str) -> StorageResult<Task> {
        let mut store = self.load()?;
        
        let task = Task {
            id: store.next_id,
            title: title.to_string(),
            completed: false,
        };
        
        store.tasks.push(task.clone());
        store.next_id += 1;
        
        self.save(&store)?;
        Ok(task)
    }
    
    /// 完成任务
    pub fn complete_task(&self, id: u64) -> StorageResult<()> {
        let mut store = self.load()?;
        
        let task = store.tasks
            .iter_mut()
            .find(|t| t.id == id)
            .ok_or_else(|| StorageError::InvalidFormat {
                message: format!("任务 {} 不存在", id),
            })?;
        
        task.completed = true;
        self.save(&store)?;
        Ok(())
    }
    
    /// 列出所有任务
    pub fn list_tasks(&self) -> StorageResult<Vec<Task>> {
        let store = self.load()?;
        Ok(store.tasks)
    }
}

// ============ 应用层（使用 anyhow）============

/// 运行 CLI 命令
fn run_cli(args: &[String]) -> Result<()> {
    let storage = TaskStorage::new("tasks.json");
    
    match args.get(1).map(|s| s.as_str()) {
        Some("add") => {
            let title = args.get(2)
                .ok_or_else(|| anyhow::anyhow!("请提供任务标题"))?;
            
            let task = storage.add_task(title)
                .context("添加任务失败")?;
            
            println!("✅ 任务已添加 #{}: {}", task.id, task.title);
        }
        
        Some("list") => {
            let tasks = storage.list_tasks()
                .context("列出任务失败")?;
            
            if tasks.is_empty() {
                println!("📭 暂无任务");
            } else {
                println!("📋 任务列表:");
                for task in &tasks {
                    let status = if task.completed { "✅" } else { "⏳" };
                    println!("  {} #{}: {}", status, task.id, task.title);
                }
            }
        }
        
        Some("complete") => {
            let id: u64 = args.get(2)
                .ok_or_else(|| anyhow::anyhow!("请提供任务 ID"))?
                .parse()
                .context("任务 ID 必须是数字")?;
            
            storage.complete_task(id)
                .with_context(|| format!("完成任务 {} 失败", id))?;
            
            println!("✅ 任务 #{} 已完成", id);
        }
        
        Some(cmd) => {
            bail!("未知命令：{}", cmd);
        }
        
        None => {
            println!("用法：todo <add|list|complete> [args]");
        }
    }
    
    Ok(())
}

// ============ 主函数 ============

fn main() -> Result<()> {
    let args: Vec<String> = std::env::args().collect();
    
    run_cli(&args)
        .context("待办事项管理器运行失败")

最佳实践 2：提供有意义的上下文

        .with_context(|| "处理用户数据时需要读取输入文件")?;

最佳实践 3：在边界处转换错误

    // ...
}

// 边界：转换为友好错误
fn handle_request() -> HttpResponse {
    match internal() {
        Ok(data) => HttpResponse::Ok(data),
        Err(LibError::NotFound) => HttpResponse::NotFound(),
        Err(e) => HttpResponse::Error(format!("内部错误：{}", e)),
    }
}
*/

/*
最佳实践 4：使用 ensure! 简化验证

❌ 冗长的验证
fn validate(user: &User) -> Result<()> {
    if user.name.is_empty() {
        bail!("用户名不能为空");
    }
    if user.age < 18 {
        bail!("用户必须年满 18 岁");
    }
    Ok(())
}

✅ 简洁的验证
fn validate(user: &User) -> Result<()> {
    ensure!(!user.name.is_empty(), "用户名不能为空");
    ensure!(user.age >= 18, "用户必须年满 18 岁");
    Ok(())
}
*/

        .context("中间层错误")
        .context("顶层错误");
    
    if let Err(e) = result {
        // 打印完整错误链
        eprintln!("错误：{}", e);
        eprintln!("\n错误链:");
        for (i, cause) in e.chain().enumerate() {
            eprintln!("  {}: {}", i, cause);

/*
选择错误类型的决策树：

1. 是库还是应用？
   ├─ 库 → 用 thiserror 定义明确的错误类型
   └─ 应用 → 用 anyhow 快速处理
   
2. 调用者需要区分错误吗？
   ├─ 需要 → 用 thiserror + 枚举
   └─ 不需要 → 用 anyhow
   
3. 需要错误链吗？
   ├─ 需要 → anyhow 内置支持
   └─ 不需要 → 简单枚举即可
   
4. 性能敏感吗？
   ├─ 是 → 避免 Box，用静态错误类型
   └─ 否 → anyhow 足够好
*/

💡 错误处理是 Rust 最强大的特性之一

今天深入学习了 anyhow 和 thiserror，深刻体会到：

1. 工具选择很重要
   - 应用层追求开发效率 → anyhow
   - 库层追求 API 清晰 → thiserror
   
2. 上下文是关键
   - 原始错误往往不够用
   - 添加上下文让调试更容易
   
3. 错误链的力量
   - 从表面错误追溯到根本原因
   - 生产环境 debugging 的利器

Rust 的错误处理不是负担，而是保障。
类型系统确保你不会忽略错误，
工具链让你优雅地处理错误。
