// src/stats.rs

use crate::models::{Task, Priority};
use crate::storage::Storage;
use std::collections::HashMap;

#[derive(Debug, Clone, Default)]
pub struct TaskStats {
    pub total: u64,
    pub completed: u64,
    pub pending: u64,
    pub overdue: u64,
    pub by_priority: HashMap<Priority, u64>,
}

impl Storage {
    /// 获取任务统计信息
    pub fn get_stats(&self) -> TaskStats {
        let tasks = self.list_tasks();
        let now = chrono::Utc::now();
        
        let mut stats = TaskStats::default();
        stats.total = tasks.len() as u64;
        
        for task in &tasks {
            if task.completed {
                stats.completed += 1;
            } else {
                stats.pending += 1;
                
                // 检查是否过期
                if let Some(due) = task.due_date {
                    if due < now {
                        stats.overdue += 1;
                    }
                }
            }
            
            // 按优先级统计
            *stats.by_priority.entry(task.priority.clone()).or_insert(0) += 1;
        }
        
        stats
    }
    
    /// 获取完成率
    pub fn get_completion_rate(&self) -> f64 {
        let stats = self.get_stats();
        if stats.total == 0 {
            return 0.0;
        }
        stats.completed as f64 / stats.total as f64 * 100.0
    }
    
    /// 获取本周完成任务数
    pub fn get_weekly_completed(&self) -> u64 {
        let tasks = self.list_tasks();
        let now = chrono::Utc::now();
        let week_ago = now - chrono::Duration::days(7);
        
        tasks.iter()
            .filter(|task| {
                task.completed && 
                task.completed_at.map_or(false, |t| t >= week_ago)
            })
            .count() as u64
    }
}

// src/commands/batch.rs

use crate::models::{Task, Priority};
use crate::storage::Storage;

#[derive(Debug, Clone)]
pub enum BatchOp {
    Complete(Vec<u64>),           // 批量完成
    Delete(Vec<u64>),              // 批量删除
    SetPriority(Vec<u64>, Priority), // 批量设置优先级
    AddTag(Vec<u64>, String),      // 批量添加标签
    RemoveTag(Vec<u64>, String),   // 批量移除标签
}

impl Storage {
    /// 批量完成任务
    pub fn batch_complete(&mut self, ids: &[u64]) -> Result<Vec<u64>, String> {
        let mut completed_ids = Vec::new();
        
        for &id in ids {
            if let Some(task) = self.find_task_by_id(id) {
                let mut task = task.clone();
                task.mark_completed();
                if self.update_task(&task).is_ok() {
                    completed_ids.push(id);
                }
            }
        }
        
        if completed_ids.is_empty() {
            Err("没有任务被完成".to_string())
        } else {
            Ok(completed_ids)
        }
    }
    
    /// 批量删除任务
    pub fn batch_delete(&mut self, ids: &[u64]) -> Result<u64, String> {
        let mut deleted = 0;
        
        for &id in ids {
            if self.delete_task(id).is_ok() {
                deleted += 1;
            }
        }
        
        if deleted == 0 {
            Err("没有任务被删除".to_string())
        } else {
            Ok(deleted)
        }
    }
    
    /// 批量设置优先级
    pub fn batch_set_priority(&mut self, ids: &[u64], priority: Priority) -> Result<u64, String> {
        let mut updated = 0;
        
        for &id in ids {
            if let Some(mut task) = self.find_task_by_id(id).cloned() {
                task.set_priority(priority.clone());
                if self.update_task(&task).is_ok() {
                    updated += 1;
                }
            }
        }
        
        if updated == 0 {
            Err("没有任务的优先级被更新".to_string())
        } else {
            Ok(updated)
        }
    }
}

// src/commands/export.rs

use crate::models::Task;
use crate::storage::Storage;
use std::fs::File;
use std::io::{self, BufWriter, Write};
use std::path::Path;

#[derive(Debug, Clone)]
pub enum ExportFormat {
    Json,
    Csv,
    Tsv,
}

impl Storage {
    /// 导出所有任务到文件
    pub fn export(&self, path: &Path, format: ExportFormat) -> Result<(), String> {
        let tasks = self.list_tasks();
        
        match format {
            ExportFormat::Json => self.export_json(path, &tasks),
            ExportFormat::Csv => self.export_csv(path, &tasks),
            ExportFormat::Tsv => self.export_tsv(path, &tasks),
        }
    }
    
    fn export_json(&self, path: &Path, tasks: &[Task]) -> Result<(), String> {
        let json = serde_json::to_string_pretty(tasks)
            .map_err(|e| format!("JSON 序列化失败: {}", e))?;
        
        let mut file = File::create(path)
            .map_err(|e| format!("无法创建文件: {}", e))?;
        
        file.write_all(json.as_bytes())
            .map_err(|e| format!("写入文件失败: {}", e))?;
        
        Ok(())
    }
    
    fn export_csv(&self, path: &Path, tasks: &[Task]) -> Result<(), String> {
        let file = File::create(path)
            .map_err(|e| format!("无法创建文件: {}", e))?;
        let mut writer = BufWriter::new(file);
        
        // 写入 CSV 头部
        writeln!(writer, "id,content,completed,priority,created_at,due_date,tags")
            .map_err(|e| format!("写入失败: {}", e))?;
        
        // 写入数据行
        for task in tasks {
            let tags = task.tags.join(";");
            let due = task.due_date
                .map(|d| d.format("%Y-%m-%d").to_string())
                .unwrap_or_default();
            
            writeln!(writer, "{},{},{},{:?},{},{},{}",
                task.id,
                Self::escape_csv(&task.content),
                task.completed,
                task.priority,
                task.created_at.format("%Y-%m-%d"),
                due,
                tags
            ).map_err(|e| format!("写入失败: {}", e))?;
        }
        
        Ok(())
    }
    
    fn escape_csv(&self, s: &str) -> String {
        if s.contains(',') || s.contains('"') || s.contains('\n') {
            format!("\"{}\"", s.replace('"', "\"\""))
        } else {
            s.to_string()
        }
    }
    
    fn export_tsv(&self, path: &Path, tasks: &[Task]) -> Result<(), String> {
        let file = File::create(path)
            .map_err(|e| format!("无法创建文件: {}", e))?;
        let mut writer = BufWriter::new(file);
        
        // 写入 TSV 头部
        writeln!(writer, "id\tcontent\tcompleted\tpriority\tcreated_at\tdue_date\ttags")
            .map_err(|e| format!("写入失败: {}", e))?;
        
        // 写入数据行
        for task in tasks {
            let tags = task.tags.join(";");
            let due = task.due_date
                .map(|d| d.format("%Y-%m-%d").to_string())
                .unwrap_or_default();
            
            writeln!(writer, "{}\t{}\t{}\t{:?}\t{}\t{}\t{}",
                task.id,
                task.content,
                task.completed,
                task.priority,
                task.created_at.format("%Y-%m-%d"),
                due,
                tags
            ).map_err(|e| format!("写入失败: {}", e))?;
        }
        
        Ok(())
    }
}

// src/commands/import.rs

use crate::models::Task;
use crate::storage::Storage;
use std::fs::File;
use std::io::{self, BufRead, BufReader};
use std::path::Path;

impl Storage {
    /// 从文件导入任务
    pub fn import(&mut self, path: &Path) -> Result<u64, String> {
        let extension = path.extension()
            .and_then(|e| e.to_str())
            .unwrap_or("");
        
        match extension.to_lowercase().as_str() {
            "json" => self.import_json(path),
            "csv" => self.import_csv(path),
            _ => Err(format!("不支持的文件格式: {}", extension)),
        }
    }
    
    fn import_json(&mut self, path: &Path) -> Result<u64, String> {
        let file = File::open(path)
            .map_err(|e| format!("无法打开文件: {}", e))?;
        
        let tasks: Vec<Task> = serde_json::from_reader(file)
            .map_err(|e| format!("JSON 解析失败: {}", e))?;
        
        let count = tasks.len() as u64;
        for task in tasks {
            self.add_task(task.content.clone())
                .ok();
        }
        
        Ok(count)
    }
    
    fn import_csv(&mut self, path: &Path) -> Result<u64, String> {
        let file = File::open(path)
            .map_err(|e| format!("无法打开文件: {}", e))?;
        let reader = BufReader::new(file);
        let mut lines = reader.lines();
        
        // 跳过表头
        if let Some(Ok(header)) = lines.next() {
            println!("导入 CSV，表头: {}", header);
        }
        
        let mut count = 0u64;
        
        for line in lines {
            let line = line.map_err(|e| format!("读取行失败: {}", e))?;
            if let Some(task) = Self::parse_csv_task(&line, &mut self.next_id) {
                self.add_task(task.content.clone()).ok();
                count += 1;
            }
        }
        
        Ok(count)
    }
    
    fn parse_csv_task(line: &str, id_counter: &mut u64) -> Option<Task> {
        let fields: Vec<&str> = line.split(',').collect();
        if fields.len() < 2 {
            return None;
        }
        
        let content = fields[1].trim_matches('"').to_string();
        if content.is_empty() {
            return None;
        }
        
        Some(Task::new(*id_counter, content))
    }
}

// src/commands/stats.rs

use crate::storage::Storage;
use crate::stats::TaskStats;
use std::collections::HashMap;

pub fn show_stats(storage: &Storage) {
    let stats = storage.get_stats();
    let rate = storage.get_completion_rate();
    let weekly = storage.get_weekly_completed();
    
    println!("\n📊 任务统计报表");
    println!("{}", "=".repeat(40));
    println!("总任务数: {}", stats.total);
    println!("已完成:   {} ({:.1}%)", stats.completed, rate);
    println!("待完成:   {}", stats.pending);
    println!("已过期:   {}", stats.overdue);
    println!("本周完成: {}", weekly);
    
    println!("\n📈 按优先级分布:");
    for (priority, count) in &stats.by_priority {
        let bar = "█".repeat(*count as usize);
        println!("  {:?}: {} {}", priority, count, bar);
    }
}
